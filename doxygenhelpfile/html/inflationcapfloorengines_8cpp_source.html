00047     void YoYInflationCapFloorEngine::calculate() const {
00048 
00049         // copy black version then adapt to others
00050 
00051         Real value = 0.0;
00052         Size optionlets = arguments_.startDates.size();
00053         std::vector<Real> values(optionlets, 0.0);
00054         std::vector<Real> stdDevs(optionlets, 0.0);
00055         std::vector<Real> forwards(optionlets, 0.0);
00056         YoYInflationCapFloor::Type type = arguments_.type;
00057 
00058         Handle<YoYInflationTermStructure> yoyTS
00059         = index()->yoyInflationTermStructure();
00060         Handle<YieldTermStructure> nominalTS
00061         = yoyTS->nominalTermStructure();
00062         Date settlement = nominalTS->referenceDate();
00063 
00064 
00065         for (Size i=0; i<optionlets; ++i) {
00066             Date paymentDate = arguments_.payDates[i];
00067             if (paymentDate > settlement) { // discard expired caplets
00068                 DiscountFactor d = arguments_.nominals[i] *
00069                     arguments_.gearings[i] *
00070                     nominalTS->discount(paymentDate) *
00071                 arguments_.accrualTimes[i];
00072 
00073                 // We explicitly have the index and assume that
00074                 // the fixing is natural, i.e. no convexity adjustment.
00075                 // If that was required then we would also need
00076                 // nominal vols in the pricing engine, i.e. a different engine.
00077                 // This also means that we do not need the coupon to have
00078                 // a pricing engine to return the swaplet rate and then
00079                 // the adjusted fixing in the instrument.
00080                 forwards[i] = yoyTS->yoyRate(arguments_.fixingDates[i],Period(0,Days));
00081                 Rate forward = forwards[i];
00082 
00083                 Date fixingDate = arguments_.fixingDates[i];
00084                 Time sqrtTime = 0.0;
00085                 if (fixingDate > volatility_->baseDate()){
00086                     sqrtTime = std::sqrt(
00087                         volatility_->timeFromBase(fixingDate));
00088                 }
00089 
00090                 if (type == YoYInflationCapFloor::Cap || type == YoYInflationCapFloor::Collar) {
00091                     Rate strike = arguments_.capRates[i];
00092                     if (sqrtTime>0.0) {
00093                         stdDevs[i] = std::sqrt(
00094                             volatility_->totalVariance(fixingDate, strike, Period(0,Days)));
00095 
00096                     }
00097 
00098                     // sttDev=0 for already-fixed dates so everything on forward
00099                     values[i] = optionletImpl(Option::Call, strike,
00100                                               forward, stdDevs[i], d);
00101                 }
00102                 if (type == YoYInflationCapFloor::Floor || type == YoYInflationCapFloor::Collar) {
00103                     Rate strike = arguments_.floorRates[i];
00104                     if (sqrtTime>0.0) {
00105                         stdDevs[i] = std::sqrt(
00106                             volatility_->totalVariance(fixingDate, strike, Period(0,Days)));
00107                     }
00108                     Real floorlet = optionletImpl(Option::Put, strike,
00109                                                   forward, stdDevs[i], d);
00110                     if (type == YoYInflationCapFloor::Floor) {
00111                         values[i] = floorlet;
00112                     } else {
00113                         // a collar is long a cap and short a floor
00114                         values[i] -= floorlet;
00115                     }
00116 
00117                 }
00118                 value += values[i];
00119             }
00120         }
00121         results_.value = value;
00122 
00123         results_.additionalResults["optionletsPrice"] = values;
00124         results_.additionalResults["optionletsAtmForward"] = forwards;
00125         if (type != YoYInflationCapFloor::Collar)
00126             results_.additionalResults["optionletsStdDev"] = stdDevs;
00127     }
