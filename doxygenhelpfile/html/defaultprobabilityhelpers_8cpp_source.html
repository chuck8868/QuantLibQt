00096     void CdsHelper::initializeDates() {
00097         protectionStart_ = evaluationDate_ + settlementDays_;
00098         Date startDate = calendar_.adjust(protectionStart_,
00099                                           paymentConvention_);
00100         Date endDate = evaluationDate_ + tenor_;
00101 
00102         schedule_ =
00103             MakeSchedule().from(startDate)
00104                           .to(endDate)
00105                           .withFrequency(frequency_)
00106                           .withCalendar(calendar_)
00107                           .withConvention(paymentConvention_)
00108                           .withTerminationDateConvention(Unadjusted)
00109                           .withRule(rule_);
00110         earliestDate_ = schedule_.dates().front();
00111         latestDate_   = calendar_.adjust(schedule_.dates().back(),
00112                                          paymentConvention_);
