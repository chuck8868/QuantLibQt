00132     inline Spread
00133     PiecewiseZeroSpreadedTermStructure::calcSpread(Time t) const {
00134         if (t <= times_.front()) {
00135             return spreads_.front()->value();
00136         } else if (t >= times_.back()) {
00137             return spreads_.back()->value();
00138         } else {
00139             Size i;
00140             for (i = 0; i < times_.size(); i++)
00141                 if (times_[i] > t) break;
00142             Time dt = times_[i] - times_[i-1];
00143             return spreads_[i]->value() * (t - times_[i-1]) / dt
00144                 + spreads_[i-1]->value() * (times_[i] - t) / dt;
00145         }
00146     }
