00360           case MoneyMarketPlus:
00361             result = moneyMarketPlusMeasure(evolution, measureOffset_);
00362             if (!isInMoneyMarketPlusMeasure(evolution, result, measureOffset_)) {
00363                 Array a(result.size());
00364                 std::copy(result.begin(), result.end(), a.begin());

00366                             measureOffset_ << ") measure:\n" << a);
00367             }
00368             break;
00369           default:
00370             QL_FAIL("unknown measure type");

00372         checkCompatibility(evolution, result);
