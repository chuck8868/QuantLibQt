00055             // row normalization
00056             for (Size i=0; i<size; ++i) {
00057                 Real norm = 0.0;
00058                 for (Size j=0; j<pseudoCols; ++j)
00059                     norm += pseudo[i][j]*pseudo[i][j];

00061                     Real normAdj = std::sqrt(matrix[i][i]/norm);
00062                     for (Size j=0; j<pseudoCols; ++j)
00063                         pseudo[i][j] *= normAdj;
00064                 }

00384                     std::sqrt(std::max<Real>(jd.eigenvalues()[i], 0.0));

00388             break;

00390             // negative eigenvalues set to zero
