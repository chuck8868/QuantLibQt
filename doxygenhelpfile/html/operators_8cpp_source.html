00126     BOOST_MESSAGE("Testing differential operators...");

00131     Real xMin = average - 4*sigma,
00132          xMax = average + 4*sigma;

00134     // FLOATING_POINT_EXCEPTION
00135     Real h = (xMax-xMin)/(N-1);
00136 
00137     Array x(N), y(N), yi(N), yd(N), temp(N), diff(N);
00138 
00139     Size i;
00140     for (i=0; i<N; i++)
00141         x[i] = xMin+h*i;
00142     std::transform(x.begin(),x.end(),y.begin(),normal);
00143     std::transform(x.begin(),x.end(),yi.begin(),cum);
00144     for (i=0; i<x.size(); i++)
00145         yd[i] = normal.derivative(x[i]);
00146 
00147     // define the differential operators
00148     DZero D(N,h);
00149     DPlusDMinus D2(N,h);
00150 
00151     // check that the derivative of cum is Gaussian
00152     temp = D.applyTo(yi);
00153     std::transform(y.begin(),y.end(),temp.begin(),diff.begin(),
00154                    std::minus<Real>());
00155     Real e = norm(diff.begin(),diff.end(),h);
00156     if (e > 1.0e-6) {
00157         BOOST_FAIL("norm of 1st derivative of cum minus Gaussian: " << e
00158                    << "\ntolerance exceeded");
00159     }
00160 
00161     // check that the second derivative of cum is normal.derivative
00162     temp = D2.applyTo(yi);
00163     std::transform(yd.begin(),yd.end(),temp.begin(),diff.begin(),
00164                    std::minus<Real>());
00165     e = norm(diff.begin(),diff.end(),h);
00166     if (e > 1.0e-4) {
00167         BOOST_FAIL("norm of 2nd derivative of cum minus Gaussian derivative: "
