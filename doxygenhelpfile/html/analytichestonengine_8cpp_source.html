00171     Real AnalyticHestonEngine::Fj_Helper::operator()(Real phi) const
00172     {
00173         const Real rpsig(rsigma_*phi);
00174 
00175         const std::complex<Real> t1 = t0_+std::complex<Real>(0, -rpsig);
00176         const std::complex<Real> d =
00177             std::sqrt(t1*t1 - sigma2_*phi
00178                       *std::complex<Real>(-phi, (j_== 1)? 1 : -1));
00179         const std::complex<Real> ex = std::exp(-d*term_);
00180         const std::complex<Real> addOnTerm
00181             = engine_ > 0 ? engine_->addOnTerm(phi, term_, j_) : 0.0;
00182 
00183         if (cpxLog_ == Gatheral) {
00184             if (phi != 0.0) {
00185                 if (sigma_ > 1e-5) {
00186                     const std::complex<Real> p = (t1-d)/(t1+d);
00187                     const std::complex<Real> g
00188                                             = std::log((1.0 - p*ex)/(1.0 - p));
00189 
00190                     return
00191                         std::exp(v0_*(t1-d)*(1.0-ex)/(sigma2_*(1.0-ex*p))
00192                                  + (kappa_*theta_)/sigma2_*((t1-d)*term_-2.0*g)
00193                                  + std::complex<Real>(0.0, phi*(dd_-sx_))
00194                                  + addOnTerm
00195                                  ).imag()/phi;
00196                 }
00197                 else {
00198                     const std::complex<Real> td = phi/(2.0*t1)
00199                                    *std::complex<Real>(-phi, (j_== 1)? 1 : -1);
00200                     const std::complex<Real> p = td*sigma2_/(t1+d);
00201                     const std::complex<Real> g = p*(1.0-ex);
00202 
00203                     return
00204                         std::exp(v0_*td*(1.0-ex)/(1.0-p*ex)
00205                                  + (kappa_*theta_)*(td*term_-2.0*g/sigma2_)
00206                                  + std::complex<Real>(0.0, phi*(dd_-sx_))
00207                                  + addOnTerm
00208                                  ).imag()/phi;
00209                 }
00210             }
00211             else {
00212                 // use l'Hospital's rule to get lim_{phi->0}
00213                 if (j_ == 1) {
00214                     const Real kmr = rsigma_-kappa_;
00215                     if (std::fabs(kmr) > 1e-7) {
00216                         return dd_-sx_
00217                             + (std::exp(kmr*term_)*kappa_*theta_
00218                                -kappa_*theta_*(kmr*term_+1.0) ) / (2*kmr*kmr)
00219                             - v0_*(1.0-std::exp(kmr*term_)) / (2.0*kmr);
00220                     }
00221                     else
00222                         // \kappa = \rho * \sigma
00223                         return dd_-sx_ + 0.25*kappa_*theta_*term_*term_
00224                                        + 0.5*v0_*term_;
00225                 }
00226                 else {
00227                     return dd_-sx_
00228                         - (std::exp(-kappa_*term_)*kappa_*theta_
00229                            +kappa_*theta_*(kappa_*term_-1.0))/(2*kappa_*kappa_)
00230                         - v0_*(1.0-std::exp(-kappa_*term_))/(2*kappa_);
00231                 }
00232             }
00233         }
00234         else if (cpxLog_ == BranchCorrection) {
00235             const std::complex<Real> p  = (t1+d)/(t1 - d);
00236 
00237             // next term: g = std::log((1.0 - p*std::exp(d*term_))/(1.0 - p))
00238             std::complex<Real> g;
00239 
00240             // the exp of the following expression is needed.
00241             const std::complex<Real> e = std::log(p)+d*term_;
00242 
00243             // does it fit to the machine precision?
00244             if (std::exp(-e.real()) > QL_EPSILON) {
00245                 g = std::log((1.0 - p/ex)/(1.0 - p));
00246             } else {
00247                 // use a "big phi" approximation
00248                 g = d*term_ + std::log(p/(p - 1.0));
00249 
00250                 if (g.imag() > M_PI || g.imag() <= -M_PI) {
00251                     // get back to principal branch of the complex logarithm
00252                     Real im = std::fmod(g.imag(), 2*M_PI);
00253                     if (im > M_PI)
00254                         im -= 2*M_PI;
00255                     else if (im <= -M_PI)
00256                         im += 2*M_PI;
00257 
00258                     g = std::complex<Real>(g.real(), im);
00259                 }
00260             }
00261 
00262             // be careful here as we have to use a log branch correction
00263             // to deal with the discontinuities of the complex logarithm.
00264             // the principal branch is not always the correct one.
00265             // (s. A. Sepp, chapter 4)
00266             // remark: there is still the change that we miss a branch
00267             // if the order of the integration is not high enough.
00268             const Real tmp = g.imag() - g_km1_;
00269             if (tmp <= -M_PI)
00270                 ++b_;
00271             else if (tmp > M_PI)
00272                 --b_;
00273 
00274             g_km1_ = g.imag();
00275             g += std::complex<Real>(0, 2*b_*M_PI);
00276 
00277             return std::exp(v0_*(t1+d)*(ex-1.0)/(sigma2_*(ex-p))
00278                             + (kappa_*theta_)/sigma2_*((t1+d)*term_-2.0*g)
00279                             + std::complex<Real>(0,phi*(dd_-sx_))
00280                             + addOnTerm
00281                             ).imag()/phi;
00282         }
00283         else {
00284             QL_FAIL("unknown complex logarithm formula");
00285         }
00286     }

00330                                              Real dividendDiscount,
00331                                              Real spotPrice,
00332                                              Real strikePrice,
00333                                              Real term,
00334                                              Real kappa, Real theta, Real sigma, Real v0, Real rho,
00335                                              const TypePayoff& type,
00336                                              const Integration& integration,
00337                                              const ComplexLogFormula cpxLog,
00338                                              const AnalyticHestonEngine* const enginePtr,
00339                                              Real& value,
00340                                              Size& evaluations)
00341     {
00342 
00343         const Real ratio = riskFreeDiscount/dividendDiscount;
00344 
00345         const Real c_inf = std::min(10.0, std::max(0.0001,
00346                 std::sqrt(1.0-square<Real>()(rho))/sigma))
00347                 *(v0 + kappa*theta*term);
00348 
00349         evaluations = 0;
00350         const Real p1 = integration.calculate(c_inf,
00351 
00352             Fj_Helper(kappa,theta,sigma,v0,spotPrice,rho, enginePtr, cpxLog, term, strikePrice, ratio, 1))
00353             //Fj_Helper(arguments_, model_, this, cpxLog_, term, ratio, 1))
00354             /M_PI;
00355         evaluations+= integration.numberOfEvaluations();
00356 
00357         const Real p2 = integration.calculate(c_inf,
00358                   Fj_Helper(kappa,theta,sigma,v0,spotPrice,rho, enginePtr, cpxLog, term, strikePrice, ratio, 2))
00359             //Fj_Helper(arguments_, model_, this, cpxLog_, term, ratio, 2))
00360             /M_PI;
00361         evaluations+= integration.numberOfEvaluations();
00362 
00363         switch (type.optionType())
00364         {
00365           case Option::Call:
00366             value = spotPrice*dividendDiscount*(p1+0.5)
00367                            - strikePrice*riskFreeDiscount*(p2+0.5);
00368             break;
00369           case Option::Put:
00370             value = spotPrice*dividendDiscount*(p1-0.5)
00371                            - strikePrice*riskFreeDiscount*(p2-0.5);
00372             break;
00373           default:
00374             QL_FAIL("unknown option type");
00375         }

00521     Real AnalyticHestonEngine::Integration::calculate(
00522                                Real c_inf,
00523                                const boost::function1<Real, Real>& f) const {
00524         Real retVal;
00525 
00526         switch(intAlgo_) {
00527           case GaussLaguerre:
00528             retVal = gaussianQuadrature_->operator()(f);
00529             break;
00530           case GaussLegendre:
00531           case GaussChebyshev:
00532           case GaussChebyshev2nd:
00533             retVal = gaussianQuadrature_->operator()(
00534                 boost::function1<Real, Real>(
00535                     if_then_else_return ( (boost::lambda::_1+1.0)*c_inf 
00536                                           > QL_EPSILON,
00537                         boost::lambda::bind(f, -boost::lambda::bind(
00538                             std::ptr_fun<Real,Real>(std::log),
00539                             0.5*boost::lambda::_1+0.5 )/c_inf )
00540                                           /((boost::lambda::_1+1.0)*c_inf),
00541                         boost::lambda::bind(constant<Real, Real>(0.0), 
00542                                             boost::lambda::_1))));
00543             break;
00544           case Simpson:
00545           case Trapezoid:
00546           case GaussLobatto:
00547           case GaussKronrod:
00548             retVal = integrator_->operator()(
00549                 boost::function1<Real, Real>(
00550                     if_then_else_return ( boost::lambda::_1*c_inf > QL_EPSILON,
00551                         boost::lambda::bind(f,-boost::lambda::bind(
00552                             std::ptr_fun<Real,Real>(std::log), 
00553                             boost::lambda::_1)/c_inf) /(boost::lambda::_1*c_inf),
00554                         boost::lambda::bind(constant<Real, Real>(0.0), 
00555                                             boost::lambda::_1))),
00556                 0.0, 1.0);
00557             break;
00558           default:
00559               QL_FAIL("unknwon integration algorithm");
00560         }
00561
