00031     Real flatVolCovariance(Time t1, Time t2, Time T, Time S, Volatility v1, Volatility v2) {
00032         QL_REQUIRE(t1<=t2,
00033                    "integrations bounds (" << t1 <<
00034                    "," << t2 << ") are in reverse order");
00035         Time cutOff = std::min(S,T);
00036         if (t1>=cutOff) {
00037             return 0.0;
00038         } else {
00039             cutOff = std::min(t2, cutOff);
00040             return (cutOff - t1)*v1*v2;
00041         }
00042     }

00046             const boost::shared_ptr<PiecewiseConstantCorrelation>& corr,

00050             const std::vector<Spread>& displacements)

00052       numberOfRates_(initialRates.size()),

00055       displacements_(displacements),
00056       evolution_(evolution),
00057       pseudoRoots_(numberOfSteps_, Matrix(numberOfRates_, numberOfFactors_))
00058     {
00059         const std::vector<Time>& rateTimes = evolution.rateTimes();
00060         QL_REQUIRE(numberOfRates_==rateTimes.size()-1,
00061                    "mismatch between number of rates (" << numberOfRates_ <<
00062                    ") and rate times");
00063         QL_REQUIRE(numberOfRates_==displacements.size(),
00064                    "mismatch between number of rates (" << numberOfRates_ <<
00065                    ") and displacements (" << displacements.size() << ")");
00066         QL_REQUIRE(numberOfRates_==volatilities.size(),
00067                    "mismatch between number of rates (" << numberOfRates_ <<
00068                    ") and volatilities (" << volatilities.size() << ")");
00069         QL_REQUIRE(numberOfRates_<=numberOfFactors_*numberOfSteps_,
00070                    "number of rates (" << numberOfRates_ <<
00071                    ") greater than number of factors (" << numberOfFactors_
00072                    << ") times number of steps (" << numberOfSteps_ << ")");
00073         QL_REQUIRE(numberOfFactors<=numberOfRates_,
00074                    "number of factors (" << numberOfFactors <<
00075                    ") cannot be greater than numberOfRates (" <<
00076                    numberOfRates_ << ")");
00077         QL_REQUIRE(numberOfFactors>0,
00078                    "number of factors (" << numberOfFactors <<
00079                    ") must be greater than zero");
00080 
00081         std::vector<Volatility> stdDev(numberOfRates_);
00082 
00083         Real covar;
00084         Time effStartTime, effStopTime;
00085         Real correlation;
00086         const std::vector<Time>& corrTimes = corr->times();
00087         const std::vector<Time>& evolTimes = evolution.evolutionTimes();
00088         for (Size k=0, kk=0; k<numberOfSteps_; ++k) {
00089             // one covariance per evolution step
00090             Matrix covariance(numberOfRates_, numberOfRates_, 0.0);
00091 
00092             // there might be more than one correlation matrix
00093             // in a single evolution step
00094             Matrix correlations;
00095 
00096             for (; corrTimes[kk]<evolTimes[k]; ++kk) {
00097                 effStartTime = kk==0 ? 0.0 : corrTimes[kk-1];
00098                 effStopTime = corrTimes[kk];
00099                 correlations = corr->correlation(kk);
00100                 for (Size i=0; i<numberOfRates_; ++i) {
00101                     for (Size j=i; j<numberOfRates_; ++j) {
00102                         covar = flatVolCovariance(effStartTime,
00103                                                   effStopTime,
00104                                                   rateTimes[i], rateTimes[j],
00105                                                   volatilities[i], volatilities[j]);
00106                         correlation = correlations[i][j];
00107                         covariance[i][j] += covar * correlations[i][j];
00108                      }
00109                 }
00110             }
00111             // last part in the evolution step
00112             effStartTime = kk==0 ? 0.0 : corrTimes[kk-1];
00113             effStopTime = evolTimes[k];
00114             correlations = corr->correlation(kk);
00115             for (Size i=0; i<numberOfRates_; ++i) {
00116                 for (Size j=i; j<numberOfRates_; ++j) {
00117                     covar = flatVolCovariance(effStartTime,
00118                                               effStopTime,
00119                                               rateTimes[i], rateTimes[j],
00120                                               volatilities[i], volatilities[j]);
00121                     correlation = correlations[i][j];
00122                     covariance[i][j] += covar * correlation;
00123                  }
00124             }
00125             // no more use for the kk-th correlation matrix
00126             while (kk<corrTimes.size() && corrTimes[kk]<=evolTimes[k])
00127                 ++kk;
00128 
00129             // make it symmetric
00130             for (Size i=0; i<numberOfRates_; ++i) {
00131                 for (Size j=i+1; j<numberOfRates_; ++j) {
00132                      covariance[j][i] = covariance[i][j];
00133                  }
00134             }
00135 
00136             pseudoRoots_[k] = rankReducedSqrt(covariance,
00137                                               numberOfFactors, 1.0,
00138                                               SalvagingAlgorithm::None);
00139 
00140             QL_ENSURE(pseudoRoots_[k].rows()==numberOfRates_,
00141                       "step " << k
00142                       << " flat vol wrong number of rows: "
00143                       << pseudoRoots_[k].rows()
00144                       << " instead of " << numberOfRates_);
00145             QL_ENSURE(pseudoRoots_[k].columns()==numberOfFactors,
00146                       "step " << k
00147                       << " flat vol wrong number of columns: "
00148                       << pseudoRoots_[k].columns()
00149                       << " instead of " << numberOfFactors_);
00150         }
00151     }
