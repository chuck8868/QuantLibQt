00282     std::vector<Size> makeMeasure(const MarketModelMultiProduct& product,
00283                                   MeasureType measureType) {
00284         std::vector<Size> result;
00285         EvolutionDescription evolution(product.evolution());
00286         switch (measureType) {
00287           case ProductSuggested:
00288             result = product.suggestedNumeraires();
00289             break;
00290           case Terminal:
00291             result = terminalMeasure(evolution);
00292             if (!isInTerminalMeasure(evolution, result)) {
00293                 Array a(result.size());
00294                 std::copy(result.begin(), result.end(), a.begin());
00295                 BOOST_ERROR("\nfailure in verifying Terminal measure:\n" << a);
00296             }
00297             break;
00298           case MoneyMarket:
00299             result = moneyMarketMeasure(evolution);
00300             if (!isInMoneyMarketMeasure(evolution, result)) {
00301                 Array a(result.size());
00302                 std::copy(result.begin(), result.end(), a.begin());
00303                 BOOST_ERROR("\nfailure in verifying MoneyMarket measure:\n" << a);
00304             }
00305             break;
00306           case MoneyMarketPlus:
00307             result = moneyMarketPlusMeasure(evolution, measureOffset_);
00308             if (!isInMoneyMarketPlusMeasure(evolution, result, measureOffset_)) {
00309                 Array a(result.size());
00310                 std::copy(result.begin(), result.end(), a.begin());
00311                 BOOST_ERROR("\nfailure in verifying MoneyMarketPlus(" <<
00312                             measureOffset_ << ") measure:\n" << a);
00313             }
00314             break;
00315           default:
00316             QL_FAIL("unknown measure type");
00317         }
00318         checkCompatibility(evolution, result);
00319         if (printReport_) {
00320             Array num(result.size());
00321             std::copy(result.begin(), result.end(), num.begin());
00322             BOOST_MESSAGE("    " << measureTypeToString(measureType) << ": " << num);
00323         }
00324         return result;

00359     void checkCoterminalSwapsAndSwaptions(const SequenceStatisticsInc& stats,
00360                                           const Rate fixedRate,
00361                                           const std::vector<boost::shared_ptr<StrikedTypePayoff> >& displacedPayoff,
00362                                           const boost::shared_ptr<MarketModel>, //marketModel,
00363                                           const std::string& config) {
00364 
00365         std::vector<Real> results = stats.mean();
00366         std::vector<Real> errors = stats.errorEstimate();
00367         std::vector<Real> discrepancies(todaysForwards.size());
00368 
00369         Size N = todaysForwards.size();
00370 
00371         // check Swaps
00372         Real maxError = QL_MIN_REAL;
00373         LMMCurveState curveState_lmm(rateTimes);
00374         curveState_lmm.setOnForwardRates(todaysForwards);
00375 
00376         std::vector<Real> expectedNPVs(todaysSwaps.size());
00377         Real errorThreshold = 0.5;
00378         for (Size i=0; i<N; ++i) {
00379             Real expectedNPV = curveState_lmm.coterminalSwapAnnuity(i, i)
00380                 * (todaysSwaps[i]-fixedRate) * todaysDiscounts[i];
00381             expectedNPVs[i] = expectedNPV;
00382             discrepancies[i] = (results[i]-expectedNPVs[i])/errors[i];
00383             maxError = std::max(std::fabs(discrepancies[i]), maxError);

00386             BOOST_MESSAGE(config);

00388                 BOOST_MESSAGE(io::ordinal(i+1) << " coterminal swap NPV: "
00389                               << io::rate(results[i])
00390                               << " +- " << io::rate(errors[i])
00391                               << "; expected: " << io::rate(expectedNPVs[i])
00392                               << "; discrepancy/error = "
00393                               << discrepancies[N-1-i]
00394                               << " standard errors");
00395             }
00396             BOOST_ERROR("test failed");
00397         }
00398 
00399         // check Swaptions
00400         maxError = 0;
00401         std::vector<Rate> expectedSwaptions(N);
00402         std::vector<Real> stdDevSwaptions(N);
00403         for (Size i=0; i<N; ++i) {
00404             Real expectedSwaption =
00405                 BlackCalculator(displacedPayoff[i],
00406                                 todaysSwaps[i]+displacement,
00407                                 volatilities[i]*std::sqrt(rateTimes[i]),
00408                                 curveState_lmm.coterminalSwapAnnuity(i,i) *
00409                                 todaysDiscounts[i]).value();
00410             expectedSwaptions[i] = expectedSwaption;
00411             discrepancies[i] = (results[N+i]-expectedSwaptions[i])/errors[N+i];

00413         }
00414         errorThreshold = 2.0;
00415 
00416         if (maxError > errorThreshold) {
00417             BOOST_MESSAGE(config);
00418             for (Size i=1; i<=N; ++i) {
00419                 BOOST_MESSAGE(io::ordinal(i) << " Swaption: "
00420                               << io::rate(results[2*N-i])
00421                               << " +- " << io::rate(errors[2*N-i])
