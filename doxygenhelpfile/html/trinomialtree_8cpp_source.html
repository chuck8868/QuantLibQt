00026     TrinomialTree::TrinomialTree(
00027                         const boost::shared_ptr<StochasticProcess1D>& process,
00028                         const TimeGrid& timeGrid,
00029                         bool isPositive)
00030     : Tree<TrinomialTree>(timeGrid.size()), dx_(1, 0.0), timeGrid_(timeGrid) {
00031         x0_ = process->x0();
00032 
00033         Size nTimeSteps = timeGrid.size() - 1;
00034         Integer jMin = 0;
00035         Integer jMax = 0;
00036 
00037         for (Size i=0; i<nTimeSteps; i++) {
00038             Time t = timeGrid[i];
00039             Time dt = timeGrid.dt(i);
00040 
00041             //Variance must be independent of x
00042             Real v2 = process->variance(t, 0.0, dt);
00043             Volatility v = std::sqrt(v2);
00044             dx_.push_back(v*std::sqrt(3.0));
00045 
00046             Branching branching;
00047             for (Integer j=jMin; j<=jMax; j++) {
00048                 Real x = x0_ + j*dx_[i];
00049                 Real m = process->expectation(t, x, dt);
00050                 Integer temp = Integer(std::floor((m-x0_)/dx_[i+1] + 0.5));
00051 
00052                 if (isPositive) {
00053                     while (x0_+(temp-1)*dx_[i+1]<=0) {
00054                         temp++;
00055                     }
00056                 }
00057 
00058                 Real e = m - (x0_ + temp*dx_[i+1]);
00059                 Real e2 = e*e;
00060                 Real e3 = e*std::sqrt(3.0);
00061 
00062                 Real p1 = (1.0 + e2/v2 - e3/v)/6.0;
00063                 Real p2 = (2.0 - e2/v2)/3.0;
00064                 Real p3 = (1.0 + e2/v2 + e3/v)/6.0;
00065 
00066                 branching.add(temp, p1, p2, p3);
00067             }
00068             branchings_.push_back(branching);
00069 
00070             jMin = branching.jMin();
00071             jMax = branching.jMax();
00072         }
00073     }
