00027     CDO::CDO (Real attachment,
00028               Real detachment,
00029               const vector<Real>& nominals,
00030               const vector<Handle<DefaultProbabilityTermStructure> >& basket,
00031               const Handle<OneFactorCopula>& copula,
00032               bool protectionSeller,
00033               const Schedule& premiumSchedule,
00034               Rate premiumRate,
00035               const DayCounter& dayCounter,
00036               Rate recoveryRate,
00037               Rate upfrontPremiumRate,
00038               const Handle<YieldTermStructure>& yieldTS,
00039               Size nBuckets,
00040               const Period& integrationStep)
00041     : attachment_(attachment), detachment_(detachment),
00042       nominals_(nominals), basket_(basket),
00043       copula_(copula),
00044       protectionSeller_(protectionSeller),
00045       premiumSchedule_(premiumSchedule),
00046       premiumRate_(premiumRate),
00047       dayCounter_(dayCounter),
00048       recoveryRate_(recoveryRate),
00049       upfrontPremiumRate_(upfrontPremiumRate),
00050       yieldTS_(yieldTS),
00051       nBuckets_(nBuckets),
00052       integrationStep_(integrationStep) {
00053 
00054         QL_REQUIRE (!basket.empty(), "basket is empty");
00055         QL_REQUIRE (attachment_ >= 0 && attachment_ < detachment_
00056                     && detachment_ <= 1,
00057                     "illegal attachment/detachment point");
00058 
00059         registerWith (yieldTS_);
00060         registerWith (copula_);
00061         for (Size i = 0; i < basket_.size(); i++)
00062             registerWith (basket_[i]);
00063 
00064         QL_REQUIRE (nominals_.size() <= basket_.size(),
00065                     "nominal vector size too large");
00066 
00067         if (nominals_.size() < basket_.size()) {
00068             Size n = basket_.size() - nominals_.size();
00069             Real back = nominals_.back();
00070             for (Size i = 0; i < n; i++)
00071                 nominals_.push_back(back);
00072         }
00073 
00074         QL_REQUIRE (nominals_.size() == basket_.size(),
00075                     "nominal size " << nominals_.size()
00076                     << " != basket size " << basket_.size());
00077 
00078         nominal_ = 0;
00079         for (Size i = 0; i < nominals_.size(); i++) {
00080             lgds_.push_back (nominals_[i] * (1.0 - recoveryRate_));
00081             nominal_ += nominals_[i];
00082             lgd_ += lgds_[i];
00083         }
00084         xMax_ = detachment_ * nominal_;
00085         xMin_ = attachment_ * nominal_;
00086     }

00100     Real CDO::expectedTrancheLoss (Date d) const {
00101         if (d <= basket_.front()->referenceDate())
00102             return 0;
00103 
00104         vector<Real> defProb (basket_.size());
00105         for (Size j = 0; j < basket_.size(); j++)
00106             defProb[j] = basket_[j]->defaultProbability (d);
00107 
00108         LossDistBucketing op (nBuckets_, xMax_);
00109         Distribution dist = copula_->integral (op, lgds_, defProb);
00110 
00111         return dist.trancheExpectedValue (xMin_, xMax_);

00123     void CDO::performCalculations() const {
00124 
00125         QL_REQUIRE(!yieldTS_.empty(), "no yield term structure set");
00126 
00127         errorEstimate_ = Null<Real>();
00128 
00129         NPV_ = 0.0;
00130         premiumValue_ = 0;
00131         protectionValue_ = 0;
00132         error_ = 0;
00133 
00134         /* Expectations e1 and e2 are portfolio loss given default,
00135            i.e.  with recovery already "bult in". Multiplication by
00136            (1-r) is therefore not necessary, neither in premium nor
00137            protection value calculation.
00138         */
00139 
00140         Real e1 = 0;
00141         Date today = yieldTS_->referenceDate();
00142         if (premiumSchedule_[0] > today)
00143             e1 = expectedTrancheLoss (premiumSchedule_[0]);
00144 
00145         for (Size i = 1; i < premiumSchedule_.size(); i++) {
00146             Date d2 = premiumSchedule_[i];
00147             if (d2 < today)
00148                 continue;
00149 
00150             Date d1 = premiumSchedule_[i-1];
00151 
00152             Date d, d0 = d1;
00153             do {
00154                 d = NullCalendar().advance (d0 > today ? d0 : today,
00155                                             integrationStep_);
00156                 if (d > d2)
00157                     d = d2;
00158 
00159                 Real e2 = expectedTrancheLoss (d);
00160 
00161                 premiumValue_ += (xMax_ - xMin_ - e2)
00162                     * premiumRate_ * dayCounter_.yearFraction (d0, d)
00163                     * yieldTS_->discount (d);
00164 
00165                 if (e2 < e1) {
00166                     error_ ++;
00167                 }
00168 
00169                 protectionValue_ -= (e2 - e1) * yieldTS_->discount (d);
00170 
00171                 d0 = d;
00172                 e1 = e2;
00173             }
00174             while (d < d2);
00175         }
00176 
00177         if (premiumSchedule_[0] >= today)
00178             upfrontPremiumValue_ = (xMax_ - xMin_) * upfrontPremiumRate_ *
00179                 yieldTS_->discount(premiumSchedule_[0]);
00180         else
00181             upfrontPremiumValue_ = 0.0;
00182 
00183         if (!protectionSeller_) {
00184             premiumValue_ *= -1;
00185             upfrontPremiumValue_ *= -1;
00186             protectionValue_ *= -1;
00187         }
00188 
00189         NPV_ = premiumValue_ + protectionValue_ + upfrontPremiumValue_;
00190     }
