00070         Real ImpliedVolHelper::operator()(Volatility x) const {
00071             if (x!=vol_->value()) {
00072                 vol_->setValue(x);
00073                 engine_->calculate();
00074             }
00075             return results_->value-targetValue_;

00123         Leg::const_iterator i;
00124         for (i = floatingLeg_.begin(); i != floatingLeg_.end(); ++i)
00125             registerWith(*i);

00193     void CapFloor::setupArguments(PricingEngine::arguments* args) const {
00194         CapFloor::arguments* arguments =
00195             dynamic_cast<CapFloor::arguments*>(args);
00196         QL_REQUIRE(arguments != 0, "wrong argument type");
00197 
00198         Size n = floatingLeg_.size();
00199 
00200         arguments->startDates.resize(n);
00201         arguments->fixingDates.resize(n);
00202         arguments->endDates.resize(n);
00203         arguments->accrualTimes.resize(n);
00204         arguments->forwards.resize(n);
00205         arguments->nominals.resize(n);
00206         arguments->gearings.resize(n);
00207         arguments->capRates.resize(n);
00208         arguments->floorRates.resize(n);
00209         arguments->spreads.resize(n);
00210 
00211         arguments->type = type_;
00212 
00213         Date today = Settings::instance().evaluationDate();
00214 
00215         for (Size i=0; i<n; ++i) {
00216             shared_ptr<FloatingRateCoupon> coupon =
00217                 boost::dynamic_pointer_cast<FloatingRateCoupon>(
00218                                                              floatingLeg_[i]);
00219             QL_REQUIRE(coupon, "non-FloatingRateCoupon given");
00220             arguments->startDates[i] = coupon->accrualStartDate();
00221             arguments->fixingDates[i] = coupon->fixingDate();
00222             arguments->endDates[i] = coupon->date();
00223 
00224             // this is passed explicitly for precision
00225             arguments->accrualTimes[i] = coupon->accrualPeriod();
00226 
00227             // this is passed explicitly for precision...
00228             if (arguments->endDates[i] >= today) { // ...but only if needed
00229                 arguments->forwards[i] = coupon->adjustedFixing();
00230             } else {
00231                 arguments->forwards[i] = Null<Rate>();
00232             }
00233 
00234             arguments->nominals[i] = coupon->nominal();
00235             Spread spread = coupon->spread();
00236             Real gearing = coupon->gearing();
00237             arguments->gearings[i] = gearing;
00238             arguments->spreads[i] = spread;
00239 
00240             if (type_ == Cap || type_ == Collar)
00241                 arguments->capRates[i] = (capRates_[i]-spread)/gearing;
00242             else
00243                 arguments->capRates[i] = Null<Rate>();
00244 
00245             if (type_ == Floor || type_ == Collar)
00246                 arguments->floorRates[i] = (floorRates_[i]-spread)/gearing;
00247             else
00248                 arguments->floorRates[i] = Null<Rate>();
00249         }
00250     }

00262                    capRates.size() == startDates.size(),
00263                    "number of start dates (" << startDates.size()

00265                    << capRates.size() << ")");
00266         QL_REQUIRE(type == CapFloor::Cap ||
00267                    floorRates.size() == startDates.size(),
00268                    "number of start dates (" << startDates.size()
00269                    << ") different from that of floor rates ("
00270                    << floorRates.size() << ")");
00271         QL_REQUIRE(gearings.size() == startDates.size(),
00272                    "number of start dates (" << startDates.size()
00273                    << ") different from that of gearings ("

00276                    "number of start dates (" << startDates.size()
00277                    << ") different from that of spreads ("

00279         QL_REQUIRE(nominals.size() == startDates.size(),
00280                    "number of start dates (" << startDates.size()
