00104     template <template <class> class MC, class RNG, class S>
00105     inline typename McSimulation<MC,RNG,S>::result_type
00106         McSimulation<MC,RNG,S>::value(Real tolerance,
00107                                               Size maxSamples,
00108                                               Size minSamples) const {
00109         Size sampleNumber =
00110             mcModel_->sampleAccumulator().samples();
00111         if (sampleNumber<minSamples) {
00112             mcModel_->addSamples(minSamples-sampleNumber);
00113             sampleNumber = mcModel_->sampleAccumulator().samples();
00114         }
00115 
00116         Size nextBatch;
00117         Real order;
00118         result_type error(mcModel_->sampleAccumulator().errorEstimate());
00119         while (maxError(error) > tolerance) {
00120             QL_REQUIRE(sampleNumber<maxSamples,
00121                        "max number of samples (" << maxSamples
00122                        << ") reached, while error (" << error
00123                        << ") is still above tolerance (" << tolerance << ")");
00124 
00125             // conservative estimate of how many samples are needed
00126             order = maxError(error*error)/tolerance/tolerance;
00127             nextBatch =
00128                 Size(std::max<Real>(static_cast<Real>(sampleNumber)*order*0.8-static_cast<Real>(sampleNumber),
00129                                     static_cast<Real>(minSamples)));
00130 
00131             // do not exceed maxSamples
00132             nextBatch = std::min(nextBatch, maxSamples-sampleNumber);
00133             sampleNumber += nextBatch;
00134             mcModel_->addSamples(nextBatch);
00135             error = result_type(mcModel_->sampleAccumulator().errorEstimate());
00136         }
00137 
00138         return result_type(mcModel_->sampleAccumulator().mean());
00139     }
