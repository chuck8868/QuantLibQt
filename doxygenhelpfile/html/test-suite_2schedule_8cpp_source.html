00031     void check_dates(const Schedule& s,
00032                      const std::vector<Date>& expected) {
00033         if (s.size() != expected.size()) {
00034             BOOST_FAIL("expected " << expected.size() << " dates, "
00035                        << "found " << s.size());

00039                 BOOST_ERROR("expected " << expected[i]
