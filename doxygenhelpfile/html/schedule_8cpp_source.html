00031             Date result = Date(20, d.month(), d.year());
00032             if (result < d)
00033                 result += 1*Months;
00034             if (rule == DateGeneration::TwentiethIMM ||
00035                 rule == DateGeneration::OldCDS ||

00039                     Integer skip = 3 - m%3;

00389     Schedule Schedule::until(const Date& truncationDate) const {
00390         Schedule result = *this;
00391 
00392         if (truncationDate<result.dates_.back()) {
00393             // remove later dates
00394             while (result.dates_.back()>truncationDate) {
00395                 result.dates_.pop_back();
00396                 result.isRegular_.pop_back();
00397             }
00398 
00399             // add truncationDate if missing
00400             if (truncationDate!=result.dates_.back()) {
00401                 result.dates_.push_back(truncationDate);
00402                 result.isRegular_.push_back(false);
00403                 result.terminationDateConvention_ = Unadjusted;
00404             } else {
00405                 result.terminationDateConvention_ = convention_;
00406             }
00407 
00408             if (result.nextToLastDate_>=truncationDate)
00409                 result.nextToLastDate_ = Date();
00410             if (result.firstDate_>=truncationDate)
00411                 result.firstDate_ = Date();
00412         }
00413 
00414         return result;
00415     }
