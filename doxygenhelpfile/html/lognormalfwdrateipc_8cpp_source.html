00035       numeraires_(numeraires),

00038       numberOfFactors_(marketModel->numberOfFactors()),
00039       curveState_(marketModel->evolution().rateTimes()),
00040       forwards_(marketModel->initialRates()),
00041       displacements_(marketModel->displacements()),
00042       logForwards_(numberOfRates_), initialLogForwards_(numberOfRates_),
00043       drifts1_(numberOfRates_), initialDrifts_(numberOfRates_),
00044       g_(numberOfRates_), brownians_(numberOfFactors_),
00045       correlatedBrownians_(numberOfRates_),
00046       rateTaus_(marketModel->evolution().rateTaus()),
00047       alive_(marketModel->evolution().firstAliveRate())
00048     {
00049         checkCompatibility(marketModel->evolution(), numeraires);
00050         QL_REQUIRE(isInTerminalMeasure(marketModel->evolution(), numeraires),
00051                    "terminal measure required for ipc ");
00052 
00053         Size steps = marketModel->evolution().numberOfSteps();
00054 
00055         generator_ = factory.create(numberOfFactors_, steps-initialStep_);
00056 
00057         currentStep_ = initialStep_;
00058 
00059         calculators_.reserve(steps);
00060         fixedDrifts_.reserve(steps);
00061         for (Size j=0; j<steps; ++j) {
00062             const Matrix& A = marketModel->pseudoRoot(j);
00063             calculators_.push_back(LMMDriftCalculator(A,
00064                                                    displacements_,
00065                                                    marketModel->evolution().rateTaus(),
00066                                                    numeraires[j],
00067                                                    alive_[j]));
00068             const Matrix& C = marketModel->covariance(j);
00069             std::vector<Real> fixed(numberOfRates_);
00070             for (Size k=0; k<numberOfRates_; ++k) {
00071                 Real variance = C[k][k];
00072                 fixed[k] = -0.5*variance;
00073             }
00074             fixedDrifts_.push_back(fixed);
00075         }
00076 
00077         setForwards(marketModel_->initialRates());
00078     }

00105     Real LogNormalFwdRateIpc::advanceStep()
00106     {
00107         // we're going from T1 to T2:
00108 
00109         // a) compute drifts D1 at T1;
00110         if (currentStep_ > initialStep_) {
00111             calculators_[currentStep_].computePlain(forwards_, drifts1_);
00112         } else {
00113             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00114                       drifts1_.begin());
00115         }
00116 
00117         Real weight = generator_->nextStep(brownians_);
00118         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00119         const Matrix& C = marketModel_->covariance(currentStep_);
00120         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00121 
00122         Integer alive = alive_[currentStep_];
00123         Real drifts2;
00124         for (Integer i=numberOfRates_-1; i>=alive; --i) {
00125             drifts2 = 0.0;
00126             for (Size j=i+1; j<numberOfRates_; ++j) {
00127                 drifts2 -= g_[j]*C[i][j];
00128             }
00129             logForwards_[i] += 0.5*(drifts1_[i]+drifts2) + fixedDrift[i];
00130             logForwards_[i] +=
00131                 std::inner_product(A.row_begin(i), A.row_end(i),
00132                                    brownians_.begin(), 0.0);
00133             forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
00134             g_[i] = rateTaus_[i]*(forwards_[i]+displacements_[i])/
00135                 (1.0+rateTaus_[i]*forwards_[i]);
00136         }
00137 
00138         // update curve state
00139         curveState_.setOnForwardRates(forwards_);
00140 
00141         ++currentStep_;
00142 
00143         return weight;
00144     }
