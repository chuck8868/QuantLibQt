00042     Real aggregateNPV(const vector<shared_ptr<Instrument> >& instruments,
00043                       const vector<Real>& quant) {
00044         Size n = instruments.size();
00045         Real npv = 0.0;
00046         if (quant.empty() || (quant.size()==1 && quant[0]==1.0)) {
00047             for (Size k=0; k<n; ++k)
00048                 npv += instruments[k]->NPV();
00049         } else {
00050             QL_REQUIRE(quant.size()==n,
00051                        "dimension mismatch between instruments (" << n <<
00052                        ") and quantities (" << quant.size() << ")");
00053             for (Size k=0; k<n; ++k)
00054                 npv += quant[k] * instruments[k]->NPV();
00055         }
00056         return npv;
