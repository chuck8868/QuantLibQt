00078     Real FdmBatesOp::IntegroIntegrand::operator()(Real y) const {
00079         const Real x = x_ + M_SQRT2*delta_*y + nu_;
00080         Real valueOfDerivative = interpl_->operator()(x, true);
00081         
00082         for (FdmBoundaryConditionSet::const_iterator iter=bcSet_.begin();
00083             iter < bcSet_.end(); ++iter) {
00084 
00085             const boost::shared_ptr<FdmDirichletBoundary> dirichlet
00086                 = boost::dynamic_pointer_cast<FdmDirichletBoundary>(*iter);
00087 
00088             QL_REQUIRE(dirichlet, "FdmBatesOp can only deal with Dirichlet "
00089                                   "boundary conditions.")
00090 
00091             valueOfDerivative
00092                 = dirichlet->applyAfterApplying(x, valueOfDerivative);
00093         }
00094         
00095         return std::exp(-y*y)*valueOfDerivative;
