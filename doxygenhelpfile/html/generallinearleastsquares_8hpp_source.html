00109     void GeneralLinearLeastSquares::calculate(xIterator xBegin, xIterator xEnd,
00110                                               yIterator yBegin, yIterator yEnd,
00111                                               vIterator vBegin, vIterator vEnd) {
00112 
00113         const Size n = residuals_.size();
00114         const Size m = err_.size();
00115 
00116         QL_REQUIRE( n == Size(std::distance(yBegin, yEnd)),
00117             "sample set need to be of the same size");
00118         QL_REQUIRE(n >= m, "sample set is too small");
00119 
00120         Size i;
00121 
00122         Matrix A(n, m);
00123         for (i=0; i<m; ++i)
00124             std::transform(xBegin, xEnd, A.column_begin(i), *vBegin++);
00125 
00126         const SVD svd(A);
00127         const Matrix& V = svd.V();
00128         const Matrix& U = svd.U();
00129         const Array& w = svd.singularValues();
00130         const Real threshold = n*QL_EPSILON;
00131 
00132         for (i=0; i<m; ++i) {
00133             if (w[i] > threshold) {
00134                 const Real u = std::inner_product(U.column_begin(i),
00135                     U.column_end(i),
00136                     yBegin, 0.0)/w[i];
00137 
00138                 for (Size j=0; j<m; ++j) {
00139                     a_[j]  +=u*V[j][i];
00140                     err_[j]+=V[j][i]*V[j][i]/(w[i]*w[i]);
00141                 }
00142             }
00143         }
00144         err_      = Sqrt(err_);
00145         Array tmp = A*a_;
00146         std::transform(tmp.begin(), tmp.end(),
00147                        yBegin, residuals_.begin(), std::minus<Real>());
00148 
00149         const Real chiSq
00150             = std::inner_product(residuals_.begin(), residuals_.end(),
00151             residuals_.begin(), 0.0);
00152         std::transform(err_.begin(), err_.end(), standardErrors_.begin(),
00153             std::bind1st(std::multiplies<Real>(),
00154             std::sqrt(chiSq/(n-2))));
00155     }
