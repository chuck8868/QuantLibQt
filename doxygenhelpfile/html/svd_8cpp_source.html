00042         /*  returns hypotenuse of real (non-complex) scalars a and b by
00043             avoiding underflow/overflow
00044             using (a * sqrt( 1 + (b/a) * (b/a))), rather than
00045             sqrt(a*a + b*b).
00046         */
00047         Real hypot(const Real &a, const Real &b) {
00048             if (a == 0) {
00049                 return std::fabs(b);
00050             } else {
00051                 Real c = b/a;
00052                 return std::fabs(a) * std::sqrt(1 + c*c);
00053             }
00054         }
