00081 double enorm(int n,double* x)
00082 {
00083 /*
00084 *     **********
00085 *
00086 *     function enorm
00087 *
00088 *     given an n-vector x, this function calculates the
00089 *     euclidean norm of x.
00090 *
00091 *     the euclidean norm is computed by accumulating the sum of
00092 *     squares in three different sums. the sums of squares for the
00093 *     small and large components are scaled so that no overflows
00094 *     occur. non-destructive underflows are permitted. underflows
00095 *     and overflows do not occur in the computation of the unscaled
00096 *     sum of squares for the intermediate components.
00097 *     the definitions of small, intermediate and large components
00098 *     depend on two constants, rdwarf and rgiant. the main
00099 *     restrictions on these constants are that rdwarf**2 not
00100 *     underflow and rgiant**2 not overflow. the constants
00101 *     given here are suitable for every known computer.
00102 *
00103 *     the function statement is
00104 *
00105 *   double precision function enorm(n,x)
00106 *
00107 *     where
00108 *
00109 *   n is a positive integer input variable.
00110 *
00111 *   x is an input array of length n.
00112 *
00113 *     subprograms called
00114 *
00115 *   fortran-supplied ... dabs,dsqrt
00116 *
00117 *     argonne national laboratory. minpack project. march 1980.
00118 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
00119 *
00120 *     **********
00121 */
00122 int i;
00123 double agiant,floatn,s1,s2,s3,xabs,x1max,x3max;
00124 double ans, temp;
00125 static double rdwarf = 3.834e-20;
00126 static double rgiant = 1.304e19;
00127 static double zero = 0.0;
00128 static double one = 1.0;
00129 
00130 s1 = zero;
00131 s2 = zero;
00132 s3 = zero;
00133 x1max = zero;
00134 x3max = zero;
00135 floatn = n;
00136 agiant = rgiant/floatn;
00137 
00138 for( i=0; i<n; i++ )
00139 {
00140 xabs = std::fabs(x[i]);
00141 if( (xabs > rdwarf) && (xabs < agiant) )
00142     {
00143 /*
00144 *       sum for intermediate components.
00145 */
00146     s2 += xabs*xabs;
00147     continue;
00148     }
00149 
00150 if(xabs > rdwarf)
00151     {
00152 /*
00153 *          sum for large components.
00154 */
00155     if(xabs > x1max)
00156         {
00157         temp = x1max/xabs;
00158         s1 = one + s1*temp*temp;
00159         x1max = xabs;
00160         }
00161     else
00162         {
00163         temp = xabs/x1max;
00164         s1 += temp*temp;
00165         }
00166     continue;
00167     }
00168 /*
00169 *          sum for small components.
00170 */
00171 if(xabs > x3max)
00172     {
00173     temp = x3max/xabs;
00174     s3 = one + s3*temp*temp;
00175     x3max = xabs;
00176     }
00177 else
00178     {
00179     if(xabs != zero)
00180         {
00181         temp = xabs/x3max;
00182         s3 += temp*temp;
00183         }
00184     }
00185 }
00186 /*
00187 *     calculation of norm.
00188 */
00189 if(s1 != zero)
00190     {
00191     temp = s1 + (s2/x1max)/x1max;
00192     ans = x1max*std::sqrt(temp);
00193     return(ans);
00194     }
00195 if(s2 != zero)
00196     {
00197     if(s2 >= x3max)
00198         temp = s2*(one+(x3max/s2)*(x3max*s3));
00199     else
00200         temp = x3max*((s2/x3max)+(x3max*s3));
00201     ans = std::sqrt(temp);
00202     }
00203 else
00204     {
00205     ans = x3max*std::sqrt(s3);
00206     }
00207 return(ans);
00208 /*
00209 *     last card of function enorm.
00210 */
00211 }

00257 void
00258 fdjac2(int m,int n,double* x,double* fvec,double* fjac,int,
00259        int* iflag,double epsfcn,double* wa,
00260        const QuantLib::MINPACK::LmdifCostFunction& fcn)
00261 {
00262 /*
00263 *     **********
00264 *
00265 *     subroutine fdjac2
00266 *
00267 *     this subroutine computes a forward-difference approximation
00268 *     to the m by n jacobian matrix associated with a specified
00269 *     problem of m functions in n variables.
00270 *
00271 *     the subroutine statement is
00272 *
00273 *   subroutine fdjac2(fcn,m,n,x,fvec,fjac,ldfjac,iflag,epsfcn,wa)
00274 *
00275 *     where
00276 *
00277 *   fcn is the name of the user-supplied subroutine which
00278 *     calculates the functions. fcn must be declared
00279 *     in an external statement in the user calling
00280 *     program, and should be written as follows.
00281 *
00282 *     subroutine fcn(m,n,x,fvec,iflag)
00283 *     integer m,n,iflag
00284 *     double precision x(n),fvec(m)
00285 *     ----------
00286 *     calculate the functions at x and
00287 *     return this vector in fvec.
00288 *     ----------
00289 *     return
00290 *     end
00291 *
00292 *     the value of iflag should not be changed by fcn unless
00293 *     the user wants to terminate execution of fdjac2.
00294 *     in this case set iflag to a negative integer.
00295 *
00296 *   m is a positive integer input variable set to the number
00297 *     of functions.
00298 *
00299 *   n is a positive integer input variable set to the number
00300 *     of variables. n must not exceed m.
00301 *
00302 *   x is an input array of length n.
00303 *
00304 *   fvec is an input array of length m which must contain the
00305 *     functions evaluated at x.
00306 *
00307 *   fjac is an output m by n array which contains the
00308 *     approximation to the jacobian matrix evaluated at x.
00309 *
00310 *   ldfjac is a positive integer input variable not less than m
00311 *     which specifies the leading dimension of the array fjac.
00312 *
00313 *   iflag is an integer variable which can be used to terminate
00314 *     the execution of fdjac2. see description of fcn.
00315 *
00316 *   epsfcn is an input variable used in determining a suitable
00317 *     step length for the forward-difference approximation. this
00318 *     approximation assumes that the relative errors in the
00319 *     functions are of the order of epsfcn. if epsfcn is less
00320 *     than the machine precision, it is assumed that the relative
00321 *     errors in the functions are of the order of the machine
00322 *     precision.
00323 *
00324 *   wa is a work array of length m.
00325 *
00326 *     subprograms called
00327 *
00328 *   user-supplied ...... fcn
00329 *
00330 *   minpack-supplied ... dpmpar
00331 *
00332 *   fortran-supplied ... dabs,dmax1,dsqrt
00333 *
00334 *     argonne national laboratory. minpack project. march 1980.
00335 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
00336 *
00337       **********
00338 */
00339 int i,j,ij;
00340 double eps,h,temp;
00341 static double zero = 0.0;
00342 
00343 
00344 temp = dmax1(epsfcn,MACHEP);
00345 eps = std::sqrt(temp);
00346 ij = 0;
00347 for( j=0; j<n; j++ )
00348     {
00349     temp = x[j];
00350     h = eps * std::fabs(temp);
00351     if(h == zero)
00352         h = eps;
00353     x[j] = temp + h;
00354     fcn(m,n,x,wa,iflag);
00355     if( *iflag < 0)
00356         return;
00357     x[j] = temp;
00358     for( i=0; i<m; i++ )
00359         {
00360         fjac[ij] = (wa[i] - fvec[i])/h;
00361         ij += 1;    /* fjac[i+m*j] */
00362         }
00363     }
00364 /*
00365 *     last card of subroutine fdjac2.
00366 */
00367 }

00371 void
00372 qrfac(int m,int n,double* a,int,int pivot,int* ipvt,
00373       int,double* rdiag,double* acnorm,double* wa)
00374 {
00375 /*
00376 *     **********
00377 *
00378 *     subroutine qrfac
00379 *
00380 *     this subroutine uses householder transformations with column
00381 *     pivoting (optional) to compute a qr factorization of the
00382 *     m by n matrix a. that is, qrfac determines an orthogonal
00383 *     matrix q, a permutation matrix p, and an upper trapezoidal
00384 *     matrix r with diagonal elements of nonincreasing magnitude,
00385 *     such that a*p = q*r. the householder transformation for
00386 *     column k, k = 1,2,...,min(m,n), is of the form
00387 *
00388 *               t
00389 *       i - (1/u(k))*u*u
00390 *
00391 *     where u has zeros in the first k-1 positions. the form of
00392 *     this transformation and the method of pivoting first
00393 *     appeared in the corresponding linpack subroutine.
00394 *
00395 *     the subroutine statement is
00396 *
00397 *   subroutine qrfac(m,n,a,lda,pivot,ipvt,lipvt,rdiag,acnorm,wa)
00398 *
00399 *     where
00400 *
00401 *   m is a positive integer input variable set to the number
00402 *     of rows of a.
00403 *
00404 *   n is a positive integer input variable set to the number
00405 *     of columns of a.
00406 *
00407 *   a is an m by n array. on input a contains the matrix for
00408 *     which the qr factorization is to be computed. on output
00409 *     the strict upper trapezoidal part of a contains the strict
00410 *     upper trapezoidal part of r, and the lower trapezoidal
00411 *     part of a contains a factored form of q (the non-trivial
00412 *     elements of the u vectors described above).
00413 *
00414 *   lda is a positive integer input variable not less than m
00415 *     which specifies the leading dimension of the array a.
00416 *
00417 *   pivot is a logical input variable. if pivot is set true,
00418 *     then column pivoting is enforced. if pivot is set false,
00419 *     then no column pivoting is done.
00420 *
00421 *   ipvt is an integer output array of length lipvt. ipvt
00422 *     defines the permutation matrix p such that a*p = q*r.
00423 *     column j of p is column ipvt(j) of the identity matrix.
00424 *     if pivot is false, ipvt is not referenced.
00425 *
00426 *   lipvt is a positive integer input variable. if pivot is false,
00427 *     then lipvt may be as small as 1. if pivot is true, then
00428 *     lipvt must be at least n.
00429 *
00430 *   rdiag is an output array of length n which contains the
00431 *     diagonal elements of r.
00432 *
00433 *   acnorm is an output array of length n which contains the
00434 *     norms of the corresponding columns of the input matrix a.
00435 *     if this information is not needed, then acnorm can coincide
00436 *     with rdiag.
00437 *
00438 *   wa is a work array of length n. if pivot is false, then wa
00439 *     can coincide with rdiag.
00440 *
00441 *     subprograms called
00442 *
00443 *   minpack-supplied ... dpmpar,enorm
00444 *
00445 *   fortran-supplied ... dmax1,dsqrt,min0
00446 *
00447 *     argonne national laboratory. minpack project. march 1980.
00448 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
00449 *
00450 *     **********
00451 */
00452 int i,ij,jj,j,jp1,k,kmax,minmn;
00453 double ajnorm,sum,temp;
00454 static double zero = 0.0;
00455 static double one = 1.0;
00456 static double p05 = 0.05;
00457 
00458 /*
00459 *     compute the initial column norms and initialize several arrays.
00460 */
00461 ij = 0;
00462 for( j=0; j<n; j++ )
00463     {
00464     acnorm[j] = enorm(m,&a[ij]);
00465     rdiag[j] = acnorm[j];
00466     wa[j] = rdiag[j];
00467     if(pivot != 0)
00468         ipvt[j] = j;
00469     ij += m; /* m*j */
00470     }
00471 /*
00472 *     reduce a to r with householder transformations.
00473 */
00474 minmn = min0(m,n);
00475 for( j=0; j<minmn; j++ )
00476 {
00477 if(pivot == 0)
00478     goto L40;
00479 /*
00480 *    bring the column of largest norm into the pivot position.
00481 */
00482 kmax = j;
00483 for( k=j; k<n; k++ )
00484     {
00485     if(rdiag[k] > rdiag[kmax])
00486         kmax = k;
00487     }
00488 if(kmax == j)
00489     goto L40;
00490 
00491 ij = m * j;
00492 jj = m * kmax;
00493 for( i=0; i<m; i++ )
00494     {
00495     temp = a[ij]; /* [i+m*j] */
00496     a[ij] = a[jj]; /* [i+m*kmax] */
00497     a[jj] = temp;
00498     ij += 1;
00499     jj += 1;
00500     }
00501 rdiag[kmax] = rdiag[j];
00502 wa[kmax] = wa[j];
00503 k = ipvt[j];
00504 ipvt[j] = ipvt[kmax];
00505 ipvt[kmax] = k;
00506 
00507 L40:
00508 /*
00509 *    compute the householder transformation to reduce the
00510 *    j-th column of a to a multiple of the j-th unit vector.
00511 */
00512 jj = j + m*j;
00513 ajnorm = enorm(m-j,&a[jj]);
00514 if(ajnorm == zero)
00515     goto L100;
00516 if(a[jj] < zero)
00517     ajnorm = -ajnorm;
00518 ij = jj;
00519 for( i=j; i<m; i++ )
00520     {
00521     a[ij] /= ajnorm;
00522     ij += 1; /* [i+m*j] */
00523     }
00524 a[jj] += one;
00525 /*
00526 *    apply the transformation to the remaining columns
00527 *    and update the norms.
00528 */
00529 jp1 = j + 1;
00530 if(jp1 < n )
00531 {
00532 for( k=jp1; k<n; k++ )
00533     {
00534     sum = zero;
00535     ij = j + m*k;
00536     jj = j + m*j;
00537     for( i=j; i<m; i++ )
00538         {
00539         sum += a[jj]*a[ij];
00540         ij += 1; /* [i+m*k] */
00541         jj += 1; /* [i+m*j] */
00542         }
00543     temp = sum/a[j+m*j];
00544     ij = j + m*k;
00545     jj = j + m*j;
00546     for( i=j; i<m; i++ )
00547         {
00548         a[ij] -= temp*a[jj];
00549         ij += 1; /* [i+m*k] */
00550         jj += 1; /* [i+m*j] */
00551         }
00552     if( (pivot != 0) && (rdiag[k] != zero) )
00553         {
00554         temp = a[j+m*k]/rdiag[k];
00555         temp = dmax1( zero, one-temp*temp );
00556         rdiag[k] *= std::sqrt(temp);
00557         temp = rdiag[k]/wa[k];
00558         if( (p05*temp*temp) <= MACHEP)
00559             {
00560             rdiag[k] = enorm(m-j-1,&a[jp1+m*k]);
00561             wa[k] = rdiag[k];
00562             }
00563         }
00564     }
00565 }
00566 
00567 L100:
00568     rdiag[j] = -ajnorm;
00569 }
00570 /*
00571 *     last card of subroutine qrfac.
00572 */
00573 }

00578 void
00579 qrsolv(int n,double* r,int ldr,int* ipvt,double* diag,double* qtb,
00580        double* x,double* sdiag,double* wa)
00581 {
00582 /*
00583 *     **********
00584 *
00585 *     subroutine qrsolv
00586 *
00587 *     given an m by n matrix a, an n by n diagonal matrix d,
00588 *     and an m-vector b, the problem is to determine an x which
00589 *     solves the system
00590 *
00591 *       a*x = b ,     d*x = 0 ,
00592 *
00593 *     in the least squares sense.
00594 *
00595 *     this subroutine completes the solution of the problem
00596 *     if it is provided with the necessary information from the
00597 *     qr factorization, with column pivoting, of a. that is, if
00598 *     a*p = q*r, where p is a permutation matrix, q has orthogonal
00599 *     columns, and r is an upper triangular matrix with diagonal
00600 *     elements of nonincreasing magnitude, then qrsolv expects
00601 *     the full upper triangle of r, the permutation matrix p,
00602 *     and the first n components of (q transpose)*b. the system
00603 *     a*x = b, d*x = 0, is then equivalent to
00604 *
00605 *          t       t
00606 *       r*z = q *b ,  p *d*p*z = 0 ,
00607 *
00608 *     where x = p*z. if this system does not have full rank,
00609 *     then a least squares solution is obtained. on output qrsolv
00610 *     also provides an upper triangular matrix s such that
00611 *
00612 *        t   t       t
00613 *       p *(a *a + d*d)*p = s *s .
00614 *
00615 *     s is computed within qrsolv and may be of separate interest.
00616 *
00617 *     the subroutine statement is
00618 *
00619 *   subroutine qrsolv(n,r,ldr,ipvt,diag,qtb,x,sdiag,wa)
00620 *
00621 *     where
00622 *
00623 *   n is a positive integer input variable set to the order of r.
00624 *
00625 *   r is an n by n array. on input the full upper triangle
00626 *     must contain the full upper triangle of the matrix r.
00627 *     on output the full upper triangle is unaltered, and the
00628 *     strict lower triangle contains the strict upper triangle
00629 *     (transposed) of the upper triangular matrix s.
00630 *
00631 *   ldr is a positive integer input variable not less than n
00632 *     which specifies the leading dimension of the array r.
00633 *
00634 *   ipvt is an integer input array of length n which defines the
00635 *     permutation matrix p such that a*p = q*r. column j of p
00636 *     is column ipvt(j) of the identity matrix.
00637 *
00638 *   diag is an input array of length n which must contain the
00639 *     diagonal elements of the matrix d.
00640 *
00641 *   qtb is an input array of length n which must contain the first
00642 *     n elements of the vector (q transpose)*b.
00643 *
00644 *   x is an output array of length n which contains the least
00645 *     squares solution of the system a*x = b, d*x = 0.
00646 *
00647 *   sdiag is an output array of length n which contains the
00648 *     diagonal elements of the upper triangular matrix s.
00649 *
00650 *   wa is a work array of length n.
00651 *
00652 *     subprograms called
00653 *
00654 *   fortran-supplied ... dabs,dsqrt
00655 *
00656 *     argonne national laboratory. minpack project. march 1980.
00657 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
00658 *
00659 *     **********
00660 */
00661 int i,ij,ik,kk,j,jp1,k,kp1,l,nsing;
00662 double cos,cotan,qtbpj,sin,sum,tan,temp;
00663 static double zero = 0.0;
00664 static double p25 = 0.25;
00665 static double p5 = 0.5;
00666 
00667 /*
00668 *     copy r and (q transpose)*b to preserve input and initialize s.
00669 *     in particular, save the diagonal elements of r in x.
00670 */
00671 kk = 0;
00672 for( j=0; j<n; j++ )
00673     {
00674     ij = kk;
00675     ik = kk;
00676     for( i=j; i<n; i++ )
00677         {
00678         r[ij] = r[ik];
00679         ij += 1;   /* [i+ldr*j] */
00680         ik += ldr; /* [j+ldr*i] */
00681         }
00682     x[j] = r[kk];
00683     wa[j] = qtb[j];
00684     kk += ldr+1; /* j+ldr*j */
00685     }
00686 /*
00687 *     eliminate the diagonal matrix d using a givens rotation.
00688 */
00689 for( j=0; j<n; j++ )
00690 {
00691 /*
00692 *    prepare the row of d to be eliminated, locating the
00693 *    diagonal element using p from the qr factorization.
00694 */
00695 l = ipvt[j];
00696 if(diag[l] == zero)
00697     goto L90;
00698 for( k=j; k<n; k++ )
00699     sdiag[k] = zero;
00700 sdiag[j] = diag[l];
00701 /*
00702 *    the transformations to eliminate the row of d
00703 *    modify only a single element of (q transpose)*b
00704 *    beyond the first n, which is initially zero.
00705 */
00706 qtbpj = zero;
00707 for( k=j; k<n; k++ )
00708     {
00709 /*
00710 *       determine a givens rotation which eliminates the
00711 *       appropriate element in the current row of d.
00712 */
00713     if(sdiag[k] == zero)
00714         continue;
00715     kk = k + ldr * k;
00716     if(std::fabs(r[kk]) < std::fabs(sdiag[k]))
00717         {
00718         cotan = r[kk]/sdiag[k];
00719         sin = p5/std::sqrt(p25+p25*cotan*cotan);
00720         cos = sin*cotan;
00721         }
00722     else
00723         {
00724         tan = sdiag[k]/r[kk];
00725         cos = p5/std::sqrt(p25+p25*tan*tan);
00726         sin = cos*tan;
00727         }
00728 /*
00729 *       compute the modified diagonal element of r and
00730 *       the modified element of ((q transpose)*b,0).
00731 */
00732     r[kk] = cos*r[kk] + sin*sdiag[k];
00733     temp = cos*wa[k] + sin*qtbpj;
00734     qtbpj = -sin*wa[k] + cos*qtbpj;
00735     wa[k] = temp;
00736 /*
00737 *       accumulate the tranformation in the row of s.
00738 */
00739     kp1 = k + 1;
00740     if( n > kp1 )
00741         {
00742         ik = kk + 1;
00743         for( i=kp1; i<n; i++ )
00744             {
00745             temp = cos*r[ik] + sin*sdiag[i];
00746             sdiag[i] = -sin*r[ik] + cos*sdiag[i];
00747             r[ik] = temp;
00748             ik += 1; /* [i+ldr*k] */
00749             }
00750         }
00751     }
00752 L90:
00753 /*
00754 *    store the diagonal element of s and restore
00755 *    the corresponding diagonal element of r.
00756 */
00757     kk = j + ldr*j;
00758     sdiag[j] = r[kk];
00759     r[kk] = x[j];
00760 }
00761 /*
00762 *     solve the triangular system for z. if the system is
00763 *     singular, then obtain a least squares solution.
00764 */
00765 nsing = n;
00766 for( j=0; j<n; j++ )
00767     {
00768     if( (sdiag[j] == zero) && (nsing == n) )
00769         nsing = j;
00770     if(nsing < n)
00771         wa[j] = zero;
00772     }
00773 if(nsing < 1)
00774     goto L150;
00775 
00776 for( k=0; k<nsing; k++ )
00777     {
00778     j = nsing - k - 1;
00779     sum = zero;
00780     jp1 = j + 1;
00781     if(nsing > jp1)
00782         {
00783         ij = jp1 + ldr * j;
00784         for( i=jp1; i<nsing; i++ )
00785             {
00786             sum += r[ij]*wa[i];
00787             ij += 1; /* [i+ldr*j] */
00788             }
00789         }
00790     wa[j] = (wa[j] - sum)/sdiag[j];
00791     }
00792 L150:
00793 /*
00794 *     permute the components of z back to components of x.
00795 */
00796 for( j=0; j<n; j++ )
00797     {
00798     l = ipvt[j];
00799     x[l] = wa[j];
00800     }
00801 /*
00802 *     last card of subroutine qrsolv.
00803 */
00804 }

00808 
00809 void
00810 lmpar(int n,double* r,int ldr,int* ipvt,double* diag,
00811       double* qtb,double delta,double* par,double* x,double* sdiag,
00812       double* wa1,double* wa2)
00813 {
00814 /*     **********
00815 *
00816 *     subroutine lmpar
00817 *
00818 *     given an m by n matrix a, an n by n nonsingular diagonal
00819 *     matrix d, an m-vector b, and a positive number delta,
00820 *     the problem is to determine a value for the parameter
00821 *     par such that if x solves the system
00822 *
00823 *       a*x = b ,     sqrt(par)*d*x = 0 ,
00824 *
00825 *     in the least squares sense, and dxnorm is the euclidean
00826 *     norm of d*x, then either par is zero and
00827 *
00828 *       (dxnorm-delta) .le. 0.1*delta ,
00829 *
00830 *     or par is positive and
00831 *
00832 *       abs(dxnorm-delta) .le. 0.1*delta .
00833 *
00834 *     this subroutine completes the solution of the problem
00835 *     if it is provided with the necessary information from the
00836 *     qr factorization, with column pivoting, of a. that is, if
00837 *     a*p = q*r, where p is a permutation matrix, q has orthogonal
00838 *     columns, and r is an upper triangular matrix with diagonal
00839 *     elements of nonincreasing magnitude, then lmpar expects
00840 *     the full upper triangle of r, the permutation matrix p,
00841 *     and the first n components of (q transpose)*b. on output
00842 *     lmpar also provides an upper triangular matrix s such that
00843 *
00844 *        t   t           t
00845 *       p *(a *a + par*d*d)*p = s *s .
00846 *
00847 *     s is employed within lmpar and may be of separate interest.
00848 *
00849 *     only a few iterations are generally needed for convergence
00850 *     of the algorithm. if, however, the limit of 10 iterations
00851 *     is reached, then the output par will contain the best
00852 *     value obtained so far.
00853 *
00854 *     the subroutine statement is
00855 *
00856 *   subroutine lmpar(n,r,ldr,ipvt,diag,qtb,delta,par,x,sdiag,
00857 *            wa1,wa2)
00858 *
00859 *     where
00860 *
00861 *   n is a positive integer input variable set to the order of r.
00862 *
00863 *   r is an n by n array. on input the full upper triangle
00864 *     must contain the full upper triangle of the matrix r.
00865 *     on output the full upper triangle is unaltered, and the
00866 *     strict lower triangle contains the strict upper triangle
00867 *     (transposed) of the upper triangular matrix s.
00868 *
00869 *   ldr is a positive integer input variable not less than n
00870 *     which specifies the leading dimension of the array r.
00871 *
00872 *   ipvt is an integer input array of length n which defines the
00873 *     permutation matrix p such that a*p = q*r. column j of p
00874 *     is column ipvt(j) of the identity matrix.
00875 *
00876 *   diag is an input array of length n which must contain the
00877 *     diagonal elements of the matrix d.
00878 *
00879 *   qtb is an input array of length n which must contain the first
00880 *     n elements of the vector (q transpose)*b.
00881 *
00882 *   delta is a positive input variable which specifies an upper
00883 *     bound on the euclidean norm of d*x.
00884 *
00885 *   par is a nonnegative variable. on input par contains an
00886 *     initial estimate of the levenberg-marquardt parameter.
00887 *     on output par contains the final estimate.
00888 *
00889 *   x is an output array of length n which contains the least
00890 *     squares solution of the system a*x = b, sqrt(par)*d*x = 0,
00891 *     for the output par.
00892 *
00893 *   sdiag is an output array of length n which contains the
00894 *     diagonal elements of the upper triangular matrix s.
00895 *
00896 *   wa1 and wa2 are work arrays of length n.
00897 *
00898 *     subprograms called
00899 *
00900 *   minpack-supplied ... dpmpar,enorm,qrsolv
00901 *
00902 *   fortran-supplied ... dabs,dmax1,dmin1,dsqrt
00903 *
00904 *     argonne national laboratory. minpack project. march 1980.
00905 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
00906 *
00907 *     **********
00908 */
00909 int i,iter,ij,jj,j,jm1,jp1,k,l,nsing;
00910 double dxnorm,fp,gnorm,parc,parl,paru;
00911 double sum,temp;
00912 static double zero = 0.0;
00913 static double p1 = 0.1;
00914 static double p001 = 0.001;
00915 
00916 extern double DWARF;
00917 
00918 /*
00919 *     compute and store in x the gauss-newton direction. if the
00920 *     jacobian is rank-deficient, obtain a least squares solution.
00921 */
00922 nsing = n;
00923 jj = 0;
00924 for( j=0; j<n; j++ )
00925     {
00926     wa1[j] = qtb[j];
00927     if( (r[jj] == zero) && (nsing == n) )
00928         nsing = j;
00929     if(nsing < n)
00930         wa1[j] = zero;
00931     jj += ldr+1; /* [j+ldr*j] */
00932     }
00933 if(nsing >= 1)
00934     {
00935     for( k=0; k<nsing; k++ )
00936         {
00937         j = nsing - k - 1;
00938         wa1[j] = wa1[j]/r[j+ldr*j];
00939         temp = wa1[j];
00940         jm1 = j - 1;
00941         if(jm1 >= 0)
00942             {
00943             ij = ldr * j;
00944             for( i=0; i<=jm1; i++ )
00945                 {
00946                 wa1[i] -= r[ij]*temp;
00947                 ij += 1;
00948                 }
00949             }
00950         }
00951     }
00952 
00953 for( j=0; j<n; j++ )
00954     {
00955     l = ipvt[j];
00956     x[l] = wa1[j];
00957     }
00958 /*
00959 *     initialize the iteration counter.
00960 *     evaluate the function at the origin, and test
00961 *     for acceptance of the gauss-newton direction.
00962 */
00963 iter = 0;
00964 for( j=0; j<n; j++ )
00965     wa2[j] = diag[j]*x[j];
00966 dxnorm = enorm(n,wa2);
00967 fp = dxnorm - delta;
00968 if(fp <= p1*delta)
00969     {
00970     goto L220;
00971     }
00972 /*
00973 *     if the jacobian is not rank deficient, the newton
00974 *     step provides a lower bound, parl, for the zero of
00975 *     the function. otherwise set this bound to zero.
00976 */
00977 parl = zero;
00978 if(nsing >= n)
00979     {
00980     for( j=0; j<n; j++ )
00981         {
00982         l = ipvt[j];
00983         wa1[j] = diag[l]*(wa2[l]/dxnorm);
00984         }
00985     jj = 0;
00986     for( j=0; j<n; j++ )
00987         {
00988         sum = zero;
00989         jm1 = j - 1;
00990         if(jm1 >= 0)
00991             {
00992             ij = jj;
00993             for( i=0; i<=jm1; i++ )
00994                 {
00995                 sum += r[ij]*wa1[i];
00996                 ij += 1;
00997                 }
00998             }
00999         wa1[j] = (wa1[j] - sum)/r[j+ldr*j];
01000         jj += ldr; /* [i+ldr*j] */
01001         }
01002     temp = enorm(n,wa1);
01003     parl = ((fp/delta)/temp)/temp;
01004     }
01005 /*
01006 *     calculate an upper bound, paru, for the zero of the function.
01007 */
01008 jj = 0;
01009 for( j=0; j<n; j++ )
01010     {
01011     sum = zero;
01012     ij = jj;
01013     for( i=0; i<=j; i++ )
01014         {
01015         sum += r[ij]*qtb[i];
01016         ij += 1;
01017         }
01018     l = ipvt[j];
01019     wa1[j] = sum/diag[l];
01020     jj += ldr; /* [i+ldr*j] */
01021     }
01022 gnorm = enorm(n,wa1);
01023 paru = gnorm/delta;
01024 if(paru == zero)
01025     paru = DWARF/dmin1(delta,p1);
01026 /*
01027 *     if the input par lies outside of the interval (parl,paru),
01028 *     set par to the closer endpoint.
01029 */
01030 *par = dmax1( *par,parl);
01031 *par = dmin1( *par,paru);
01032 if( *par == zero)
01033     *par = gnorm/dxnorm;
01034 /*
01035 *     beginning of an iteration.
01036 */
01037 L150:
01038 iter += 1;
01039 /*
01040 *    evaluate the function at the current value of par.
01041 */
01042 if( *par == zero)
01043     *par = dmax1(DWARF,p001*paru);
01044 temp = std::sqrt( *par );
01045 for( j=0; j<n; j++ )
01046     wa1[j] = temp*diag[j];
01047 qrsolv(n,r,ldr,ipvt,wa1,qtb,x,sdiag,wa2);
01048 for( j=0; j<n; j++ )
01049     wa2[j] = diag[j]*x[j];
01050 dxnorm = enorm(n,wa2);
01051 temp = fp;
01052 fp = dxnorm - delta;
01053 /*
01054 *    if the function is small enough, accept the current value
01055 *    of par. also test for the exceptional cases where parl
01056 *    is zero or the number of iterations has reached 10.
01057 */
01058 if( (std::fabs(fp) <= p1*delta)
01059  || ((parl == zero) && (fp <= temp) && (temp < zero))
01060  || (iter == 10) )
01061     goto L220;
01062 /*
01063 *    compute the newton correction.
01064 */
01065 for( j=0; j<n; j++ )
01066     {
01067     l = ipvt[j];
01068     wa1[j] = diag[l]*(wa2[l]/dxnorm);
01069     }
01070 jj = 0;
01071 for( j=0; j<n; j++ )
01072     {
01073     wa1[j] = wa1[j]/sdiag[j];
01074     temp = wa1[j];
01075     jp1 = j + 1;
01076     if(jp1 < n)
01077         {
01078         ij = jp1 + jj;
01079         for( i=jp1; i<n; i++ )
01080             {
01081             wa1[i] -= r[ij]*temp;
01082             ij += 1; /* [i+ldr*j] */
01083             }
01084         }
01085     jj += ldr; /* ldr*j */
01086     }
01087 temp = enorm(n,wa1);
01088 parc = ((fp/delta)/temp)/temp;
01089 /*
01090 *    depending on the sign of the function, update parl or paru.
01091 */
01092 if(fp > zero)
01093     parl = dmax1(parl, *par);
01094 if(fp < zero)
01095     paru = dmin1(paru, *par);
01096 /*
01097 *    compute an improved estimate for par.
01098 */
01099 *par = dmax1(parl, *par + parc);
01100 /*
01101 *    end of an iteration.
01102 */
01103 goto L150;
01104 
01105 L220:
01106 /*
01107 *     termination.
01108 */
01109 if(iter == 0)
01110     *par = zero;
01111 /*
01112 *     last card of subroutine lmpar.
01113 */
01114 }

01121 void lmdif(int m,int n,double* x,double* fvec,double ftol,
01122       double xtol,double gtol,int maxfev,double epsfcn,
01123       double* diag, int mode, double factor,
01124       int nprint, int* info,int* nfev,double* fjac,
01125       int ldfjac,int* ipvt,double* qtf,
01126       double* wa1,double* wa2,double* wa3,double* wa4,
01127       const QuantLib::MINPACK::LmdifCostFunction& fcn)
01128 {
01129 /*
01130 *     **********
01131 *
01132 *     subroutine lmdif
01133 *
01134 *     the purpose of lmdif is to minimize the sum of the squares of
01135 *     m nonlinear functions in n variables by a modification of
01136 *     the levenberg-marquardt algorithm. the user must provide a
01137 *     subroutine which calculates the functions. the jacobian is
01138 *     then calculated by a forward-difference approximation.
01139 *
01140 *     the subroutine statement is
01141 *
01142 *   subroutine lmdif(fcn,m,n,x,fvec,ftol,xtol,gtol,maxfev,epsfcn,
01143 *            diag,mode,factor,nprint,info,nfev,fjac,
01144 *            ldfjac,ipvt,qtf,wa1,wa2,wa3,wa4)
01145 *
01146 *     where
01147 *
01148 *   fcn is the name of the user-supplied subroutine which
01149 *     calculates the functions. fcn must be declared
01150 *     in an external statement in the user calling
01151 *     program, and should be written as follows.
01152 *
01153 *     subroutine fcn(m,n,x,fvec,iflag)
01154 *     integer m,n,iflag
01155 *     double precision x(n),fvec(m)
01156 *     ----------
01157 *     calculate the functions at x and
01158 *     return this vector in fvec.
01159 *     ----------
01160 *     return
01161 *     end
01162 *
01163 *     the value of iflag should not be changed by fcn unless
01164 *     the user wants to terminate execution of lmdif.
01165 *     in this case set iflag to a negative integer.
01166 *
01167 *   m is a positive integer input variable set to the number
01168 *     of functions.
01169 *
01170 *   n is a positive integer input variable set to the number
01171 *     of variables. n must not exceed m.
01172 *
01173 *   x is an array of length n. on input x must contain
01174 *     an initial estimate of the solution vector. on output x
01175 *     contains the final estimate of the solution vector.
01176 *
01177 *   fvec is an output array of length m which contains
01178 *     the functions evaluated at the output x.
01179 *
01180 *   ftol is a nonnegative input variable. termination
01181 *     occurs when both the actual and predicted relative
01182 *     reductions in the sum of squares are at most ftol.
01183 *     therefore, ftol measures the relative error desired
01184 *     in the sum of squares.
01185 *
01186 *   xtol is a nonnegative input variable. termination
01187 *     occurs when the relative error between two consecutive
01188 *     iterates is at most xtol. therefore, xtol measures the
01189 *     relative error desired in the approximate solution.
01190 *
01191 *   gtol is a nonnegative input variable. termination
01192 *     occurs when the cosine of the angle between fvec and
01193 *     any column of the jacobian is at most gtol in absolute
01194 *     value. therefore, gtol measures the orthogonality
01195 *     desired between the function vector and the columns
01196 *     of the jacobian.
01197 *
01198 *   maxfev is a positive integer input variable. termination
01199 *     occurs when the number of calls to fcn is at least
01200 *     maxfev by the end of an iteration.
01201 *
01202 *   epsfcn is an input variable used in determining a suitable
01203 *     step length for the forward-difference approximation. this
01204 *     approximation assumes that the relative errors in the
01205 *     functions are of the order of epsfcn. if epsfcn is less
01206 *     than the machine precision, it is assumed that the relative
01207 *     errors in the functions are of the order of the machine
01208 *     precision.
01209 *
01210 *   diag is an array of length n. if mode = 1 (see
01211 *     below), diag is internally set. if mode = 2, diag
01212 *     must contain positive entries that serve as
01213 *     multiplicative scale factors for the variables.
01214 *
01215 *   mode is an integer input variable. if mode = 1, the
01216 *     variables will be scaled internally. if mode = 2,
01217 *     the scaling is specified by the input diag. other
01218 *     values of mode are equivalent to mode = 1.
01219 *
01220 *   factor is a positive input variable used in determining the
01221 *     initial step bound. this bound is set to the product of
01222 *     factor and the euclidean norm of diag*x if nonzero, or else
01223 *     to factor itself. in most cases factor should lie in the
01224 *     interval (.1,100.). 100. is a generally recommended value.
01225 *
01226 *   nprint is an integer input variable that enables controlled
01227 *     printing of iterates if it is positive. in this case,
01228 *     fcn is called with iflag = 0 at the beginning of the first
01229 *     iteration and every nprint iterations thereafter and
01230 *     immediately prior to return, with x and fvec available
01231 *     for printing. if nprint is not positive, no special calls
01232 *     of fcn with iflag = 0 are made.
01233 *
01234 *   info is an integer output variable. if the user has
01235 *     terminated execution, info is set to the (negative)
01236 *     value of iflag. see description of fcn. otherwise,
01237 *     info is set as follows.
01238 *
01239 *     info = 0  improper input parameters.
01240 *
01241 *     info = 1  both actual and predicted relative reductions
01242 *           in the sum of squares are at most ftol.
01243 *
01244 *     info = 2  relative error between two consecutive iterates
01245 *           is at most xtol.
01246 *
01247 *     info = 3  conditions for info = 1 and info = 2 both hold.
01248 *
01249 *     info = 4  the cosine of the angle between fvec and any
01250 *           column of the jacobian is at most gtol in
01251 *           absolute value.
01252 *
01253 *     info = 5  number of calls to fcn has reached or
01254 *           exceeded maxfev.
01255 *
01256 *     info = 6  ftol is too small. no further reduction in
01257 *           the sum of squares is possible.
01258 *
01259 *     info = 7  xtol is too small. no further improvement in
01260 *           the approximate solution x is possible.
01261 *
01262 *     info = 8  gtol is too small. fvec is orthogonal to the
01263 *           columns of the jacobian to machine precision.
01264 *
01265 *   nfev is an integer output variable set to the number of
01266 *     calls to fcn.
01267 *
01268 *   fjac is an output m by n array. the upper n by n submatrix
01269 *     of fjac contains an upper triangular matrix r with
01270 *     diagonal elements of nonincreasing magnitude such that
01271 *
01272 *        t     t       t
01273 *       p *(jac *jac)*p = r *r,
01274 *
01275 *     where p is a permutation matrix and jac is the final
01276 *     calculated jacobian. column j of p is column ipvt(j)
01277 *     (see below) of the identity matrix. the lower trapezoidal
01278 *     part of fjac contains information generated during
01279 *     the computation of r.
01280 *
01281 *   ldfjac is a positive integer input variable not less than m
01282 *     which specifies the leading dimension of the array fjac.
01283 *
01284 *   ipvt is an integer output array of length n. ipvt
01285 *     defines a permutation matrix p such that jac*p = q*r,
01286 *     where jac is the final calculated jacobian, q is
01287 *     orthogonal (not stored), and r is upper triangular
01288 *     with diagonal elements of nonincreasing magnitude.
01289 *     column j of p is column ipvt(j) of the identity matrix.
01290 *
01291 *   qtf is an output array of length n which contains
01292 *     the first n elements of the vector (q transpose)*fvec.
01293 *
01294 *   wa1, wa2, and wa3 are work arrays of length n.
01295 *
01296 *   wa4 is a work array of length m.
01297 *
01298 *     subprograms called
01299 *
01300 *   user-supplied ...... fcn
01301 *
01302 *   minpack-supplied ... dpmpar,enorm,fdjac2,lmpar,qrfac
01303 *
01304 *   fortran-supplied ... dabs,dmax1,dmin1,dsqrt,mod
01305 *
01306 *     argonne national laboratory. minpack project. march 1980.
01307 *     burton s. garbow, kenneth e. hillstrom, jorge j. more
01308 *
01309 *     **********
01310 */
01311 int i,iflag,ij,jj,iter,j,l;
01312 double actred,delta=0,dirder,fnorm,fnorm1,gnorm;
01313 double par,pnorm,prered,ratio;
01314 double sum,temp,temp1,temp2,temp3,xnorm=0;
01315 static double one = 1.0;
01316 static double p1 = 0.1;
01317 static double p5 = 0.5;
01318 static double p25 = 0.25;
01319 static double p75 = 0.75;
01320 static double p0001 = 1.0e-4;
01321 static double zero = 0.0;
01322 
01323 *info = 0;
01324 iflag = 0;
01325 *nfev = 0;
01326 /*
01327 *     check the input parameters for errors.
