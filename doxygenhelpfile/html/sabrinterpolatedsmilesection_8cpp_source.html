00100     void SabrInterpolatedSmileSection::performCalculations() const {
00101         forwardValue_ = forward_->value();
00102         vols_.clear();
00103         actualStrikes_.clear();
00104         // we populate the volatilities, skipping the invalid ones
00105         for (Size i=0; i<volHandles_.size(); ++i) {
00106             if (volHandles_[i]->isValid()) {
00107                 if (hasFloatingStrikes_) {
00108                     actualStrikes_.push_back(forwardValue_ + strikes_[i]);
00109                     vols_.push_back(atmVolatility_->value() + volHandles_[i]->value());
00110                 } else {
00111                     actualStrikes_.push_back(strikes_[i]);
00112                     vols_.push_back(volHandles_[i]->value());
00113                 }
00114             }
00115         }
00116         // we are recreating the sabrinterpolation object unconditionnaly to
00117         // avoid iterator invalidation
00118         createInterpolation();
00119         sabrInterpolation_->update();
00120     }
