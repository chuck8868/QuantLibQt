00072     void SMMDriftCalculator::compute(const CoterminalSwapCurveState& cs,
00073                                      std::vector<Real>& drifts) const {
00074         #if defined(QL_EXTRA_SAFETY_CHECKS)
00075         QL_REQUIRE(drifts.size()==cs.numberOfRates(),
00076                    "drifts.size() <> numberOfRates");
00077         #endif
00078 
00079         // Compute drifts with factor reduction,
00080         // using the pseudo square root of the covariance matrix.
00081 
00082         const std::vector<Rate>& SR=cs.coterminalSwapRates();
00083         // calculates and stores wkaj_, wkpj1_
00084         // assuming terminal bond measure
00085         // eq 5.4-5.7
00086         const std::vector<Time>& taus=cs.rateTaus();
00087         for (Size k=0; k<numberOfFactors_; ++k) {
00088                 // taken care in the constructor
00089                 // wkpj1_[k][numberOfRates_-1]= 0.0;
00090                 // wkaj_[k][numberOfRates_-1] = 0.0;
00091             for (Integer j=numberOfRates_-2; j>=static_cast<Integer>(alive_)-1; --j) {
00092                  // < W(k) | P(j+1)/P(n) > =
00093                  // = SR(j+1) a(j+1,k) A(j+1) / P(n) + SR(j+1) < W(k) | A(j+1)/P(n) >
00094                 Real annuity = cs.coterminalSwapAnnuity(numberOfRates_,j+1);
00095                 wkpj_[k][j+1]= SR[j+1] *
00096                             ( pseudo_[j+1][k] * annuity +  wkaj_[k][j+1] )+
00097                             pseudo_[j+1][k]*displacements_[j+1]* annuity;
00098 
00099                 if (j >=static_cast<Integer>(alive_))
00100                     wkaj_[k][j] = wkpj_[k][j+1]*taus[j ]+wkaj_[k][j+1];
00101             }
00102           }
00103 
00104 
00105         double numeraireRatio = cs.discountRatio(numberOfRates_,numeraire_);
00106 
00107 // change to work for general numeraire
00108         for (Size k=0; k<numberOfFactors_; ++k) {
00109             // compute < Wk, PN/pn>
00110             for (Size j=alive_; j<numberOfRates_; ++j)
00111             {
00112                 wkajshifted_[k][j] = -wkaj_[k][j]/cs.coterminalSwapAnnuity(numberOfRates_,j)
00113                                     + wkpj_[k][numeraire_]
00114                                                 *numeraireRatio;
00115             }
00116         }
00117 
00118         // eq 5.3 (in log coordinates)
00119         for (Size j=alive_; j<numberOfRates_; ++j) {
00120             drifts[j] = 0.0;
00121             for (Size k=0; k<numberOfFactors_; ++k) {
00122                 drifts[j] += wkajshifted_[k][j]*pseudo_[j][k];
00123             }
00124         }

00126     }
