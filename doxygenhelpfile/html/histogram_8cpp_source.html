00029         /* The discontinuous quantiles use the method (type 8) as
00030            recommended by Hyndman and Fan (1996). The resulting
00031            quantile estimates are approximately median-unbiased
00032            regardless of the distribution of 'samples'.
00033 
00034            If quantile function is called multiple times for the same
00035            dataset, it is recommended to pre-sort the sample vector.
00036         */
00037         Real quantile(const std::vector<Real>& samples, Real prob) {
00038             Size nsample = samples.size();
00039             QL_REQUIRE(prob >= 0.0 && prob <= 1.0,
00040                        "Probability has to be in [0,1].");
00041             QL_REQUIRE(nsample > 0, "The sample size has to be positive." );
00042 
00043             if (nsample == 1)
00044                 return samples[0];
00045 
00046             // two special cases: close to boundaries
00047             const Real a = 1. / 3, b = 2*a / (nsample+a);
00048             if (prob < b)
00049                 return *std::min_element(samples.begin(), samples.end());
00050             else if (prob > 1-b)
00051                 return *std::max_element(samples.begin(), samples.end());
00052 
00053             // general situation: middle region and nsample >= 2
00054             Size index = static_cast<Size>(std::floor((nsample+a)*prob+a));
00055             std::vector<Real> sorted(index+1);
00056             std::partial_sort_copy(samples.begin(), samples.end(),
00057                                    sorted.begin(), sorted.end());
00058 
00059             // use "index & index+1"th elements to interpolate the quantile
00060             Real weight = nsample*prob + a - index;
00061             return (1-weight) * sorted[index-1] + weight * sorted[index];
00062         }

00099     void Histogram::calculate() {
00100         QL_REQUIRE(!data_.empty(), "no data given");
00101 
00102         Real min = *std::min_element(data_.begin(), data_.end());
00103         Real max = *std::max_element(data_.begin(), data_.end());
00104 
00105         // calculate number of bins if necessary
00106         if (bins_ == Null<Size>()) {
00107             switch (algorithm_) {
00108               case Sturges: {
00109                   bins_ = static_cast<Size>(
00110                            std::ceil(std::log(static_cast<Real>(data_.size()))
00111                                      /std::log(2.0) + 1));
00112                   break;
00113               }
00114               case FD: {
00115                   Real r1 = quantile(data_, 0.25);
00116                   Real r2 = quantile(data_, 0.75);
00117                   Real h = 2.0 * (r2-r1) * std::pow(static_cast<Real>(data_.size()), -1.0/3.0);
00118                   bins_ = static_cast<Size>(std::ceil((max-min)/h));
00119                   break;
00120               }
00121               case Scott: {
00122                   IncrementalStatistics summary;
00123                   summary.addSequence(data_.begin(), data_.end());
00124                   Real variance = summary.variance();
00125                   Real h = 3.5 * std::sqrt(variance)
00126                          * std::pow(static_cast<Real>(data_.size()), -1.0/3.0);
00127                   bins_ = static_cast<Size>(std::ceil((max-min)/h));
00128                   break;
00129               }
00130               case None:
00131                 QL_FAIL("a bin-partition algorithm is required");
00132               default:
00133                 QL_FAIL("unknown bin-partition algorithm");
00134             };
00135             bins_ = std::max<Size>(bins_,1);
00136         }
00137 
00138         if (breaks_.empty()) {
00139             // set breaks if not provided
00140             breaks_.resize(bins_-1);
00141 
00142             // ensure breaks_ evenly span over the range of data_
00143             // TODO: borrow the idea of pretty in R.
00144             Real h = (max-min)/bins_;
00145             for (Size i=0; i<breaks_.size(); ++i) {
00146                 breaks_[i] = min + (i+1)*h;
00147             }
00148         } else {
00149             // or ensure they're sorted if given
00150             std::sort(breaks_.begin(), breaks_.end());
00151             std::vector<Real>::iterator end =
00152                 std::unique(breaks_.begin(),breaks_.end(),
00153                             std::ptr_fun(close_enough));
00154             breaks_.resize(end - breaks_.begin());
00155         }
00156 
00157         // finally, calculate counts and frequencies
00158         counts_.resize(bins_);
00159         std::fill(counts_.begin(), counts_.end(), 0);
00160 
00161         for (std::vector<Real>::const_iterator p = data_.begin();
00162              p != data_.end(); ++p) {
00163             bool processed = false;
00164             for (Size i=0; i<breaks_.size(); ++i) {
00165                 if (*p < breaks_[i]) {
00166                     ++counts_[i];
00167                     processed = true;
00168                     break;
00169                 }
00170             }
00171             if (!processed)
00172                 ++counts_[bins_-1];
00173         }
00174 
00175         frequency_.resize(bins_);
00176 
00177         Size totalCounts = data_.size();
00178         for (Size i=0; i<bins_; ++i)
00179             frequency_[i] = static_cast<Real>(counts_[i])/totalCounts;
00180     }
