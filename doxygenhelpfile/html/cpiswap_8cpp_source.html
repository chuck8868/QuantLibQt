00185         static const Spread basisPoint = 1.0e-4;
00186 
00187         // copy from VanillaSwap
00188         // works because similarly simple instrument

00190

00193         const CPISwap::results* results = dynamic_cast<const CPISwap::results*>(r);
00194         if (results) { // might be a swap engine, so no error is thrown
00195             fairRate_ = results->fairRate;
00196             fairSpread_ = results->fairSpread;
00197         } else {
00198             fairRate_ = Null<Rate>();
00199             fairSpread_ = Null<Spread>();
00200         }
00201 
00202         if (fairRate_ == Null<Rate>()) {
00203             // calculate it from other results
00204             if (legBPS_[0] != Null<Real>())
00205                 fairRate_ = fixedRate_ - NPV_/(legBPS_[0]/basisPoint);

00207         if (fairSpread_ == Null<Spread>()) {
00208             // ditto
00209             if (legBPS_[1] != Null<Real>())
00210                 fairSpread_ = spread_ - NPV_/(legBPS_[1]/basisPoint);
