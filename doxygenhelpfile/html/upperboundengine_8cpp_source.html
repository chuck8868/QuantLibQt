00225                                  rebateOffset_+rebateSize_);

00229                                  principalInNumerairePortfolio,
00230                                   hedgeRebateOffset_,
00231                                  hedgeRebateOffset_+hedgeRebateSize_);

00240 
00241                 if (k != numberOfSteps_-1) {

00249                         innerEvolvers_[exercise++];

00276                 if (strategy.exercise(evolver_->currentState())) {

00278                     portfolioValue +=

00311         }
00312 
00313         // finally, we update the maximum with the total accumulated

00315         maximumValue = std::max(maximumValue, numerairesHeld);
00316 
00317 
00318         // all done; we just convert the result back to cash
00319         maximumValue *= initialNumeraireValue_;
00320 
00321         return std::make_pair(maximumValue, weight);
