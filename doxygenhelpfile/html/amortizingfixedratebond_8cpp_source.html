00077         bool isSubPeriod(const Period& subPeriod,
00078                          const Period& superPeriod,
00079                          Integer& numSubPeriods) {
00080 
00081             std::pair<Integer, Integer> superDays(daysMinMax(superPeriod));
00082             std::pair<Integer, Integer> subDays(daysMinMax(subPeriod));
00083 
00084             //obtain the approximate time ratio
00085             Real minPeriodRatio =
00086                 ((Real)superDays.first)/((Real)subDays.second);
00087             Real maxPeriodRatio =
00088                 ((Real)superDays.second)/((Real)subDays.first);
00089             Integer lowRatio = static_cast<Integer>(std::floor(minPeriodRatio));
00090             Integer highRatio = static_cast<Integer>(std::ceil(maxPeriodRatio));
00091 
00092             try {
00093                 for(Integer i=lowRatio; i <= highRatio; ++i) {
00094                     Period testPeriod = subPeriod * i;
00095                     if(testPeriod == superPeriod) {
00096                         numSubPeriods = i;
00097                         return true;
00098                     }
00099                 }
00100             } catch(Error e) {
00101                 return false;
00102             }
00103 
00104             return false;

00119         std::vector<Real> sinkingNotionals(const Period& maturityTenor,
00120                                            const Frequency& sinkingFrequency,
00121                                            Rate couponRate,
00122                                            Real initialNotional) {
00123             Period freqPeriod(sinkingFrequency);
00124             Integer nPeriods;
00125             QL_REQUIRE(isSubPeriod(freqPeriod, maturityTenor, nPeriods),
00126                        "Bond frequency is incompatible with the maturity tenor");
00127 
00128             std::vector<Real> notionals(nPeriods+1);
00129             notionals.front() = initialNotional;
00130             Real coupon = couponRate / static_cast<Real>(sinkingFrequency);
00131             Real compoundedInterest = 1.0;
00132             Real totalValue = std::pow(1.0+coupon, nPeriods);
00133             for(Size i = 0; i < (Size)nPeriods-1; ++i) {
00134                 compoundedInterest *= (1.0 + coupon);
00135                 Real currentNotional =
00136                     initialNotional*(compoundedInterest - (compoundedInterest-1.0)/(1.0 - 1.0/totalValue));
00137                 notionals[i+1] = currentNotional;
00138             }
00139             notionals.back() = 0.0;
00140             return notionals;
00141         }
