00153     template<class Iterator>
00154     Real norm(const Iterator& begin, const Iterator& end, Real h) {
00155         // squared values
00156         std::vector<Real> f2(end-begin);
00157         std::transform(begin,end,begin,f2.begin(),
00158                        std::multiplies<Real>());
00159         // numeric integral of f^2
00160         Real I = h * (std::accumulate(f2.begin(),f2.end(),0.0)
00161                       - 0.5*f2.front() - 0.5*f2.back());
00162         return std::sqrt(I);
00163     }
