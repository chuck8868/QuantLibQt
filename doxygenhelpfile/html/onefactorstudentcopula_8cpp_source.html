00064     //-------------------------------------------------------------------------

00072 
00073         StudentDistribution dz (nz_);

00075 
00076         // FIXME:
00077         // Find a sensitive way of setting these parameters,
00078         // e.g. depending on nm and nz, and the desired table range
00079         Real minimum = -10; // -15

00081         int steps = 400;
00082 
00083         Real delta = (maximum - minimum) / steps;
00084         Real cumulated = 0;
00085 
00086         if (c < 0.5) {
00087             // outer integral -> 1 for c -> 0
00088             // inner integral -> cumulativeStudent(nz)(y) for c-> 0
00089             for (Real m = minimum + delta/2; m < maximum; m += delta)
