00088     void SwaptionVolatilityCube::registerWithVolatilitySpread()
00089     {
00090         for (Size i=0; i<nStrikes_; i++)
00091             for (Size j=0; j<nOptionTenors_; j++)
00092                 for (Size k=0; k<nSwapTenors_; k++)
00093                     registerWith(volSpreads_[j*nSwapTenors_+k][i]);
