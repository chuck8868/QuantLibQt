00059     Real FixedRateBondForward::spotIncome(const Handle<YieldTermStructure>&
00060                                                   incomeDiscountCurve) const {
00061 
00062         Real income = 0.0;
00063         Date settlement = settlementDate();
00064         Leg cf =
00065             fixedCouponBond_->cashflows();
00066 
00067         /*
00068           the following assumes
00069           1. cashflows are in ascending order !
00070           2. considers as income: all coupons paid between settlementDate()
00071           and contract delivery/maturity date
00072         */
00073         for (Size i = 0; i < cf.size(); ++i) {
00074             if (!cf[i]->hasOccurred(settlement, false)) {
00075                 if (cf[i]->hasOccurred(maturityDate_, false)) {
00076                     income += cf[i]->amount() *
00077                               incomeDiscountCurve->discount(cf[i]->date()) ;
00078                 } else {
00079                     break;
00080                 }
00081             }
00082         }
00083 
00084         return income;
00085     }
