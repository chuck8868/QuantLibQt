00108         bool firstDateIsZero = false;

00111         Real dateTolerance = 1e-6;

00113         if (dateNumber > 0) {

00118                 firstDateIsZero = true;

00120                 if(dateNumber >= 2)
00121                     firstNonZeroDate = getDividendTime(1);
00122             }
00123 
00124             if (std::fabs(getDividendTime(lastIndex) - getResidualTime())
00125                 < dateTolerance) {
00126                 lastDateIsResTime = true;
00127                 lastIndex = Integer(dateNumber) - 2;
00128             }
00129 
00130             if (!firstDateIsZero)
00131                 firstNonZeroDate = getDividendTime(0);
00132 
00133             if (dateNumber >= 2) {
00134                 for (Size j = 1; j < dateNumber; j++)
00135                     QL_REQUIRE(getDividendTime(j-1) < getDividendTime(j),
00136                                "dates must be in increasing order: "
00137                                << getDividendTime(j-1)
00138                                << " is not strictly smaller than "
00139                                << getDividendTime(j));
00140             }
00141         }
00142 
00143         Time dt = getResidualTime()/(timeStepPerPeriod_*(dateNumber+1));
00144 
00145         // Ensure that dt is always smaller than the first non-zero date
00146         if (firstNonZeroDate <= dt)

00148

00151         initializeOperator();
00152         initializeBoundaryConditions();
00153         initializeModel();
00154         initializeStepCondition();
00155 
00156         prices_ = intrinsicValues_;

00158             executeIntermediateStep(dateNumber - 1);
00159 
00160         Integer j = lastIndex;
00161         do {
00162             if (j == Integer(dateNumber) - 1)
00163                 beginDate = getResidualTime();
00164             else
00165                 beginDate = getDividendTime(j+1);
00166 
00167             if (j >= 0)
00168                 endDate = getDividendTime(j);
00169             else
00170                 endDate = dt;
00171

00173                              beginDate, endDate,
00174                              timeStepPerPeriod_, *stepCondition_);

00176                 executeIntermediateStep(j);

00179         model_->rollback(prices_.values(), dt, 0, 1, *stepCondition_);
