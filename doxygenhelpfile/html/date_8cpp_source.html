00061     Month Date::month() const {
00062         Day d = dayOfYear(); // dayOfYear is 1 based
00063         Integer m = d/30 + 1;
00064         bool leap = isLeap(year());
00065         while (d <= monthOffset(Month(m),leap))
00066             --m;
00067         while (d > monthOffset(Month(m+1),leap))
00068             ++m;
00069         return Month(m);

00151     Date Date::advance(const Date& date, Integer n, TimeUnit units) {
00152         switch (units) {
00153           case Days:
00154             return date + n;
00155           case Weeks:
00156             return date + 7*n;
00157           case Months: {
00158             Day d = date.dayOfMonth();
00159             Integer m = Integer(date.month())+n;
00160             Year y = date.year();
00161             while (m > 12) {
00162                 m -= 12;
00163                 y += 1;
00164             }
00165             while (m < 1) {
00166                 m += 12;
00167                 y -= 1;
00168             }
00169 
00170             QL_ENSURE(y >= 1900 && y <= 2199,
00171                       "year " << y << " out of bounds. "
00172                       << "It must be in [1901,2199]");
00173 
00174             Integer length = monthLength(Month(m), isLeap(y));
00175             if (d > length)
00176                 d = length;
00177 
00178             return Date(d, Month(m), y);
00179           }
00180           case Years: {
00181               Day d = date.dayOfMonth();
00182               Month m = date.month();
00183               Year y = date.year()+n;
00184 
00185               QL_ENSURE(y >= 1900 && y <= 2199,
00186                         "year " << y << " out of bounds. "
00187                         << "It must be in [1901,2199]");
00188 
00189               if (d == 29 && m == February && !isLeap(y))
00190                   d = 28;
00191 
00192               return Date(d,m,y);
00193           }
00194           default:
00195             QL_FAIL("undefined time units");
00196         }
00197     }
