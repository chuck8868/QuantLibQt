00112     inline Real BinomialDistribution::operator()(BigNatural k) const {
00113 
00114         if (k > n_) return 0.0;
00115 
00116         // p==1.0
00117         if (logP_==0.0)
00118             return (k==n_ ? 1.0 : 0.0);
00119         // p==0.0
00120         else if (logOneMinusP_==0.0)
00121             return (k==0 ? 1.0 : 0.0);
00122         else
00123             return std::exp(binomialCoefficientLn(n_, k) +
00124                             k * logP_ + (n_-k) * logOneMinusP_);
