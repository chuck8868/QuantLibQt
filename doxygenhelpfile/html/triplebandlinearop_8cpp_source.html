00259     TripleBandLinearOp::solve_splitting(const Array& r, Real a, Real b) const {
00260         const boost::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
00261         QL_REQUIRE(r.size() == layout->size(), "inconsistent size of rhs");
00262 
00263 #ifdef QL_EXTRA_SAFETY_CHECKS
00264         for (FdmLinearOpIterator iter = layout->begin();
00265              iter!=layout->end(); ++iter) {
00266             const std::vector<Size>& coordinates = iter.coordinates();
00267             QL_REQUIRE(   coordinates[direction_] != 0
00268                        || lower_[iter.index()] == 0,"removing non zero entry!");
00269             QL_REQUIRE(   coordinates[direction_] != layout->dim()[direction_]-1
00270                        || upper_[iter.index()] == 0,"removing non zero entry!");
00271         }
00272 #endif
00273 
00274         Array retVal(r.size()), tmp(r.size());
00275 
00276         const Real* lptr = lower_.get();
00277         const Real* dptr = diag_.get();
00278         const Real* uptr = upper_.get();
00279 
00280         // Thomson algorithm to solve a tridiagonal system.
00281         // Example code taken from Tridiagonalopertor and
00282         // changed to fit for the triple band operator.
00283         Size rim1 = reverseIndex_[0];
00284         Real bet=1.0/(a*dptr[rim1]+b);
00285         QL_REQUIRE(bet != 0.0, "division by zero");
00286         retVal[reverseIndex_[0]] = r[rim1]*bet;
00287 
00288         for (Size j=1; j<=layout->size()-1; j++){
00289             const Size ri = reverseIndex_[j];
00290             tmp[j] = a*uptr[rim1]*bet;
00291 
00292             bet=b+a*(dptr[ri]-tmp[j]*lptr[ri]);
00293             QL_ENSURE(bet != 0.0, "division by zero");
00294             bet=1.0/bet;
00295 
00296             retVal[ri] = (r[ri]-a*lptr[ri]*retVal[rim1])*bet;
00297             rim1 = ri;
00298         }
00299         // cannot be j>=0 with Size j
00300         for (Size j=layout->size()-2; j>0; --j)
00301             retVal[reverseIndex_[j]] -= tmp[j+1]*retVal[reverseIndex_[j+1]];
00302         retVal[reverseIndex_[0]] -= tmp[1]*retVal[reverseIndex_[1]];
00303 
00304         return retVal;
00305     }
