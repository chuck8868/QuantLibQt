00149     template <class T>
00150     Real InterpolatedHazardRateCurve<T>::hazardRateImpl(Time t) const {
00151         if (t <= this->times_.back())
00152             return this->interpolation_(t, true);
00153 
00154         // flat hazard rate extrapolation
00155         return this->data_.back();
00156     }

00158     template <class T>
00159     Probability
00160     InterpolatedHazardRateCurve<T>::survivalProbabilityImpl(Time t) const {
00161         if (t == 0.0)
00162             return 1.0;
00163 
00164         Real integral;
00165         if (t <= this->times_.back()) {
00166             integral = this->interpolation_.primitive(t, true);
00167         } else {
00168             // flat hazard rate extrapolation
00169             integral = this->interpolation_.primitive(this->times_.back(), true)
00170                      + this->data_.back()*(t - this->times_.back());
00171         }
00172         return std::exp(-integral);
00173     }

00250     template <class T>
00251     void InterpolatedHazardRateCurve<T>::initialize()
00252     {
00253         QL_REQUIRE(dates_.size() >= T::requiredPoints,
00254                    "not enough input dates given");
00255         QL_REQUIRE(this->data_.size() == dates_.size(),
00256                    "dates/data count mismatch");
00257 
00258         this->times_.resize(dates_.size());
00259         this->times_[0] = 0.0;
00260         for (Size i=1; i<dates_.size(); ++i) {
00261             QL_REQUIRE(dates_[i] > dates_[i-1],
00262                        "invalid date (" << dates_[i] << ", vs "
00263                        << dates_[i-1] << ")");
00264             this->times_[i] = dayCounter().yearFraction(dates_[0], dates_[i]);
00265             QL_REQUIRE(!close(this->times_[i], this->times_[i-1]),
00266                        "two dates correspond to the same time "
00267                        "under this curve's day count convention");
00268             QL_REQUIRE(this->data_[i] >= 0.0, "negative hazard rate");
00269         }
00270 
00271         this->interpolation_ =
00272             this->interpolator_.interpolate(this->times_.begin(),
00273                                             this->times_.end(),
00274                                             this->data_.begin());
00275         this->interpolation_.update();
00276     }
