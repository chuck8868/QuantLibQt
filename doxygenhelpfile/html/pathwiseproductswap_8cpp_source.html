00057                    "accruals.size() does not equal numberOfRates or 1");

00060                    "strikes.size() does not equal numberOfRates or 1" );

00066     bool MarketModelPathwiseSwap::nextTimeStep(
00067             const CurveState& currentState,
00068             std::vector<Size>& numberCashFlowsThisStep,
00069             std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated) 
00070     {
00071         Rate liborRate = currentState.forwardRate(currentIndex_);
00072         cashFlowsGenerated[0][0].timeIndex = currentIndex_+1;
00073 
00074         cashFlowsGenerated[0][0].amount[0] =
00075                      (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_]*multiplier_;
00076 
00077         numberCashFlowsThisStep[0] = 1;
00078         
00079         for (Size i=1; i <= numberRates_; ++i)
00080                 cashFlowsGenerated[0][0].amount[i] =0;
00081 
00082         cashFlowsGenerated[0][0].amount[currentIndex_+1] = accruals_[currentIndex_]*multiplier_;
00083             
00084         ++currentIndex_;
00085         return (currentIndex_ == strikes_.size());
