00092     template <class PathType> inline
00093     Real LongstaffSchwartzPathPricer<PathType>::operator()
00094         (const PathType& path) const {
00095         if (calibrationPhase_) {
00096             // store paths for the calibration
00097             paths_.push_back(path);
00098             // result doesn't matter
00099             return 0.0;
00100         }
00101 
00102         const Size len = EarlyExerciseTraits<PathType>::pathLength(path);
00103         Real price = (*pathPricer_)(path, len-1);
00104         for (Size i=len-2; i>0; --i) {
00105             price*=dF_[i];
00106 
00107             const Real exercise = (*pathPricer_)(path, i);
00108             if (exercise > 0.0) {
00109                 const StateType regValue = pathPricer_->state(path, i);
00110 
00111                 Real continuationValue = 0.0;
00112                 for (Size l=0; l<v_.size(); ++l) {
00113                     continuationValue += coeff_[i][l] * v_[l](regValue);
00114                 }
00115 
00116                 if (continuationValue < exercise) {
00117                     price = exercise;
00118                 }
00119             }
00120         }
00121 
00122         return price*dF_[0];
