00124     void CapFloorTermVolCurve::checkInputs() const
00125     {
00126         QL_REQUIRE(!optionTenors_.empty(), "empty option tenor vector");
00127         QL_REQUIRE(nOptionTenors_==vols_.size(),
00128                    "mismatch between number of option tenors (" <<
00129                    nOptionTenors_ << ") and number of volatilities (" <<
00130                    vols_.size() << ")");
00131         QL_REQUIRE(optionTenors_[0]>0*Days,
00132                    "negative first option tenor: " << optionTenors_[0]);
00133         for (Size i=1; i<nOptionTenors_; ++i)
00134             QL_REQUIRE(optionTenors_[i]>optionTenors_[i-1],
00135                        "non increasing option tenor: " << io::ordinal(i) <<
00136                        " is " << optionTenors_[i-1] << ", " <<
00137                        io::ordinal(i+1) << " is " << optionTenors_[i]);
