00026     Real genericLongstaffSchwartzRegression(
00027                 std::vector<std::vector<NodeData> >& simulationData,
00028                 std::vector<std::vector<Real> >& basisCoefficients) {
00029 
00030         Size steps = simulationData.size();
00031         basisCoefficients.resize(steps-1);
00032 
00033         for (Size i=steps-1; i!=0; --i) {
00034 
00035             std::vector<NodeData>& exerciseData = simulationData[i];
00036 
00037             // 1) find the covariance matrix of basis function values and
00038             //    deflated cash-flows
00039             Size N = exerciseData.front().values.size();
00040             std::vector<Real> temp(N+1);
00041             SequenceStatistics stats(N+1);
00042 
00043             Size j;
00044             for (j=0; j<exerciseData.size(); ++j) {
00045                 if (exerciseData[j].isValid) {
00046                     std::copy(exerciseData[j].values.begin(),
00047                               exerciseData[j].values.end(),
00048                               temp.begin());
00049                     temp.back() = exerciseData[j].cumulatedCashFlows
00050                                 - exerciseData[j].controlValue;
00051 
00052                     stats.add(temp);
00053                 }
00054             }
00055 
00056             std::vector<Real> means = stats.mean();
00057             Matrix covariance = stats.covariance();
00058 
00059             Matrix C(N,N);
00060             Array target(N);
00061             for (Size k=0; k<N; ++k) {
00062                 target[k] = covariance[k][N] + means[k]*means[N];
00063                 for (Size l=0; l<=k; ++l)
00064                     C[k][l] = C[l][k] = covariance[k][l] + means[k]*means[l];
00065             }
00066 
00067             // 2) solve for least squares regression
00068             Array alphas = SVD(C).solveFor(target);
00069             basisCoefficients[i-1].resize(N);
00070             std::copy(alphas.begin(), alphas.end(),
00071                       basisCoefficients[i-1].begin());
00072 
00073             // 3) use exercise strategy to divide paths into exercise and
00074             //    non-exercise domains
00075             for (j=0; j<exerciseData.size(); ++j) {
00076                 if (exerciseData[j].isValid) {
00077                     Real exerciseValue = exerciseData[j].exerciseValue;
00078                     Real continuationValue =
00079                         exerciseData[j].cumulatedCashFlows;
00080                     Real estimatedContinuationValue =
00081                         std::inner_product(
00082                                  exerciseData[j].values.begin(),
00083                                  exerciseData[j].values.end(),
00084                                  alphas.begin(),
00085                                  exerciseData[j].controlValue);
00086 
00087                     // for exercise paths, add deflated rebate to
00088                     // deflated cash-flows at previous time frame;
00089                     // for non-exercise paths, add deflated cash-flows to
00090                     // deflated cash-flows at previous time frame
00091                     Real value = estimatedContinuationValue <= exerciseValue ?
00092                                  exerciseValue :
00093                                  continuationValue;
00094 
00095                     simulationData[i-1][j].cumulatedCashFlows += value;
00096                 }
00097             }
00098         }
00099 
00100         // the value of the product can now be estimated by averaging
00101         // over all paths
00102         Statistics estimate;
00103         std::vector<NodeData>& estimatedData = simulationData[0];
00104         for (Size j=0; j<estimatedData.size(); ++j)
00105             estimate.add(estimatedData[j].cumulatedCashFlows);
00106 
00107         return estimate.mean();
00108     }
