00091     void SquareRootAndersen::DoOneSubStep(Real& vt, Real z, Size j)
00092     {
00093 
00094         Real eminuskT = eMinuskDt_[j];
00095         Real m = theta_+(vt-theta_)*eminuskT;
00096         Real s2= vt*epsilon_*epsilon_*eminuskT*(1-eminuskT)/k_
00097                 + theta_*epsilon_*epsilon_*(1- eminuskT)*(1- eminuskT)/(2*k_);
00098         Real s = std::sqrt(s2);
00099         Real psi = s*s/(m*m);
00100         if (psi<= PsiC_)
00101         {
00102             Real psiinv = 1.0/psi;
00103             Real b2 = 2.0*psiinv -1+std::sqrt(2*psiinv*(2*psiinv-1.0));
00104             Real b = std::sqrt(b2);
00105             Real a= m/(1+b2);
00106             vt= a*(b+z)*(b+z);
00107         }
00108         else
00109         {
00110             Real p = (psi-1.0)/(psi+1.0);
00111             Real beta = (1.0-p)/m;
00112             Real u = CumulativeNormalDistribution()(z);
00113 
00114             if (u < p)
00115             {
00116                 vt=0;
00117                 return;
00118             }
00119 
00120              vt = std::log((1.0-p)/(1.0-u))/beta;
00121         }
