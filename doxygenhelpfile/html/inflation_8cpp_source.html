00070 
00071         std::vector<boost::shared_ptr<BootstrapHelper<T> > > instruments;
00072         for (Size i=0; i<N; i++) {
00073             Date maturity = iiData[i].date;
00074             Handle<Quote> quote(boost::shared_ptr<Quote>(
00075                 new SimpleQuote(iiData[i].rate/100.0)));
00076             boost::shared_ptr<BootstrapHelper<T> > anInstrument(new U(
