00071     Real CPICouponPricer::optionletPrice(Option::Type optionType,
00072                                                   Real effStrike) const {
00073         Date fixingDate = coupon_->fixingDate();
00074         if (fixingDate <= Settings::instance().evaluationDate()) {
00075             // the amount is determined
00076             Real a, b;
00077             if (optionType==Option::Call) {
00078                 a = coupon_->indexFixing();
00079                 b = effStrike;
00080             } else {
00081                 a = effStrike;
00082                 b = coupon_->indexFixing();
00083             }
00084             return std::max(a - b, 0.0)* coupon_->accrualPeriod()*discount_;
00085         } else {
00086             // not yet determined, use Black/DD1/Bachelier/whatever from Impl
00087             QL_REQUIRE(!capletVolatility().empty(),
00088                        "missing optionlet volatility");
00089             Real stdDev =
00090             std::sqrt(capletVolatility()->totalVariance(fixingDate,
00091                                                         effStrike));
00092             Rate fixing = optionletPriceImp(optionType,
00093                                             effStrike,
00094                                             adjustedFixing(),
00095                                             stdDev);
00096             return fixing * coupon_->accrualPeriod() * discount_;
00097         }
00098     }
