00060

00065         BOOST_FAIL(d2 << " still a business day for original TARGET instance");
00066

00068     Calendar c3 = TARGET();
00069     if (c3.isHoliday(d1))
00070         BOOST_FAIL(d1 << " still a holiday for generic TARGET instance");
00071     if (c3.isBusinessDay(d2))
00072         BOOST_FAIL(d2 << " still a business day for generic TARGET instance");
00073

00075     if (c2.isBusinessDay(d1))
00076         BOOST_FAIL(d1 << " business day for New York");

00078         BOOST_FAIL(d2 << " holiday for New York");
00079

00081     c3.addHoliday(d1);

00083

00926 
00927     Calendar c = TARGET(); // any calendar would be OK
00928 
00929     Date eom, counter = Date::minDate();
00930     Date last = Date::maxDate() - 2*Months;
00931 
00932     while (counter<=last) {
00933         eom = c.endOfMonth(counter);
00934         // check that eom is eom
00935         if (!c.isEndOfMonth(eom))
00936             BOOST_FAIL("\n  "
00937                        << eom.weekday() << " " << eom
00938                        << " is not the last business day in "
00939                        << eom.month() << " " << eom.year()

00942         if (eom.month()!=counter.month())
00943             BOOST_FAIL("\n  "
00944                        << eom
00945                        << " is not in the same month as "
00946                        << counter);
00947         counter = counter + 1;
