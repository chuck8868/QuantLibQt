00149     Rate ZeroInflationTermStructure::zeroRate(const Date &d, const Period& instObsLag,
00150                                               bool forceLinearInterpolation,
00151                                               bool extrapolate) const {
00152 
00153         Period useLag = instObsLag;
00154         if (instObsLag == Period(-1,Days)) {
00155             useLag = observationLag();
00156         }
00157 
00158         Rate zeroRate;
00159         if (forceLinearInterpolation) {
00160             std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
00161             dd.second = dd.second + Period(1,Days);
00162             Real dp = dd.second - dd.first;
00163             Real dt = d - dd.first;
00164             // if we are interpolating we only check the exact point
00165             // this prevents falling off the end at curve maturity
00166             InflationTermStructure::checkRange(d, extrapolate);
00167             Time t1 = timeFromReference(dd.first);
00168             Time t2 = timeFromReference(dd.second);
00169             zeroRate = zeroRateImpl(t1) + zeroRateImpl(t2) * (dt/dp);
00170         } else {
00171             if (indexIsInterpolated()) {
00172                 InflationTermStructure::checkRange(d-useLag, extrapolate);
00173                 Time t = timeFromReference(d-useLag);
00174                 zeroRate = zeroRateImpl(t);
00175             } else {
00176                 std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
00177                 InflationTermStructure::checkRange(dd.first, extrapolate);
00178                 Time t = timeFromReference(dd.first);
00179                 zeroRate = zeroRateImpl(t);
00180             }
00181         }
00182 
00183         if (hasSeasonality()) {
00184             zeroRate = seasonality()->correctZeroRate(d-useLag, zeroRate, *this);
00185         }
00186         return zeroRate;
00187     }

00283     std::pair<Date,Date> inflationPeriod(const Date& d,
00284                                          Frequency frequency) {
00285 
00286         Month month = d.month();
00287         Year year = d.year();
00288 
00289         Month startMonth, endMonth;
00290         switch (frequency) {
00291           case Annual:
00292             startMonth = January;
00293             endMonth = December;
00294             break;
00295           case Semiannual:
00296             startMonth = Month(6*((month-1)/6) + 1);
00297             endMonth = Month(startMonth + 5);
00298             break;
00299           case Quarterly:
00300             startMonth = Month(3*((month-1)/3) + 1);
00301             endMonth = Month(startMonth + 2);
00302             break;
00303           case Monthly:
00304             startMonth = endMonth = month;
00305             break;
00306           default:
00307             QL_FAIL("Frequency not handled: " << frequency);
00308             break;
00309         };
00310 
00311         Date startDate = Date(1, startMonth, year);
00312         Date endDate = Date::endOfMonth(Date(1, endMonth, year));
00313 
00314         return std::make_pair(startDate,endDate);
00315     }

00318     Time inflationYearFraction(Frequency f, bool indexIsInterpolated,
00319                                const DayCounter &dayCounter,
00320                                const Date &d1, const Date &d2) {
00321 
00322         Time t=0;
00323         if (indexIsInterpolated) {
00324             // N.B. we do not use linear interpolation between flat
00325             // fixing forecasts for forecasts.  This avoids awkwardnesses
00326             // when bootstrapping the inflation curve.
00327             t = dayCounter.yearFraction(d1, d2);
00328         } else {
00329             // I.e. fixing is constant for the whole inflation period.
00330             // Use the value for half way along the period.
00331             // But the inflation time is the time between period starts
00332             std::pair<Date,Date> limD1 = inflationPeriod(d1, f);
00333             std::pair<Date,Date> limD2 = inflationPeriod(d2, f);
00334             t = dayCounter.yearFraction(limD1.first, limD2.first);
00335         }
00336 
00337         return t;
00338     }
