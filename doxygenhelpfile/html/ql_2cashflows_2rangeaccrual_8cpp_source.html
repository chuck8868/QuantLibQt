00194     std::vector<Real> RangeAccrualPricerByBgm::driftsOverPeriod(Real U,
00195                                                                 Real lambdaS,
00196                                                                 Real lambdaT,
00197                                                                 Real correlation) const{
00198         std::vector<Real> result;
00199 
00200         const Real p = (U-startTime_)/accrualFactor_;
00201         const Real q = (endTime_-U)/accrualFactor_;
00202         const Real L0T = initialValues_.back();
00203 
00204         const Real driftBeforeFixing =
00205                 p*accrualFactor_*L0T/(1.+L0T*accrualFactor_)*(p*lambdaT*lambdaT + q*lambdaS*lambdaT*correlation) +
00206                 q*lambdaS*lambdaS + p*lambdaS*lambdaT*correlation
00207                 -0.5*lambda(U,lambdaS,lambdaT)*lambda(U,lambdaS,lambdaT);
00208         const Real driftAfterFixing = (p*accrualFactor_*L0T/(1.+L0T*accrualFactor_)-0.5)*lambdaT*lambdaT;
00209 
00210         result.push_back(driftBeforeFixing);
00211         result.push_back(driftAfterFixing);
00212 
00213         return result;
00214     }

00358     Real RangeAccrualPricerByBgm::digitalPriceWithoutSmile(Real strike,
00359                                         Real initialValue,
00360                                         Real expiry,
00361                                         Real deflator) const {
00362 
00363         Real lambdaS = smilesOnExpiry_->volatility(strike);
00364         Real lambdaT = smilesOnPayment_->volatility(strike);
00365 
00366         std::vector<Real> lambdaU = lambdasOverPeriod(expiry, lambdaS, lambdaT);
00367         const Real variance =
00368             startTime_*lambdaU[0]*lambdaU[0]+(expiry-startTime_)*lambdaU[1]*lambdaU[1];

00370         Real lambdaSATM = smilesOnExpiry_->volatility(initialValue);
00371         Real lambdaTATM = smilesOnPayment_->volatility(initialValue);
00372         //drift of Lognormal process (of Libor) "a_U()" nel paper
00373         std::vector<Real> muU = driftsOverPeriod(expiry, lambdaSATM, lambdaTATM, correlation_);
00374         const Real adjustment = (startTime_*muU[0]+(expiry-startTime_)*muU[1]);
00375 
00376 
00377        Real d2 = (std::log(initialValue/strike) + adjustment - 0.5*variance)/std::sqrt(variance);
00378 
00379        CumulativeNormalDistribution phi;
00380        const Real result = deflator*phi(d2);
00381 
00382        QL_REQUIRE(result > 0.,
00383            "RangeAccrualPricerByBgm::digitalPriceWithoutSmile: result< 0. Result:"<<result);
00384        QL_REQUIRE(result/deflator <= 1.,
00385             "RangeAccrualPricerByBgm::digitalPriceWithoutSmile: result/deflator > 1. Ratio: "
00386             << result/deflator << " result: " << result<< " deflator: " << deflator);

00448                                         Real forward,
00449                                         Real expiry,
00450                                         Real deflator) const {
00451 
00452         const Real previousStrike = strike - eps_/2;
00453         const Real nextStrike = strike + eps_/2;
00454 
00455         const Real derSmileS = (smilesOnExpiry_->volatility(nextStrike)-
00456                                  smilesOnExpiry_->volatility(previousStrike))/eps_;
00457         const Real derSmileT = (smilesOnPayment_->volatility(nextStrike)-
00458                                  smilesOnPayment_->volatility(previousStrike))/eps_;
00459 
00460         Real lambdaS = smilesOnExpiry_->volatility(strike);
00461         Real lambdaT = smilesOnPayment_->volatility(strike);
00462         //Real lambdaU = lambda(expiry, lambdaS, lambdaT);
00463 
00464         Real derLambdaDerK = derLambdaDerLambdaS(expiry) * derSmileS +
00465                              derLambdaDerLambdaT(expiry) * derSmileT;
00466         //Real derDriftDerK = derDriftDerLambdaS(expiry, lambdaS, lambdaT, correlation_)*derSmileS +
00467         //                      derDriftDerLambdaT(expiry, lambdaS, lambdaT, correlation_)*derSmileT +
00468         //                      lambdaU * derLambdaDerK;
00469 
00470         Real lambdaSATM = smilesOnExpiry_->volatility(forward);
00471         Real lambdaTATM = smilesOnPayment_->volatility(forward);
00472         std::vector<Real> lambdasOverPeriodU = lambdasOverPeriod(expiry, lambdaS, lambdaT);
00473         //drift of Lognormal process (of Libor) "a_U()" nel paper
00474         std::vector<Real> muU = driftsOverPeriod(expiry, lambdaSATM, lambdaTATM, correlation_);
00475 
00476         const Real variance = std::max(startTime_, 0.)*lambdasOverPeriodU[0]*lambdasOverPeriodU[0] +
00477                        std::min(expiry-startTime_, expiry)*lambdasOverPeriodU[1]*lambdasOverPeriodU[1];
00478 
00479         const Real forwardAdjustment = std::exp(std::max(startTime_, 0.)*muU[0] +
00480                                          std::min(expiry-startTime_, expiry)*muU[1]);
00481         const Real forwardAdjusted = forward * forwardAdjustment;
00482 
00483         const Real d1 = (std::log(forwardAdjusted/strike)+0.5*variance)/std::sqrt(variance);
00484 
00485         const Real sqrtOfTimeToExpiry = (std::max(startTime_, 0.)*lambdasOverPeriodU[0] +
00486                                 std::min(expiry-startTime_, expiry)*lambdasOverPeriodU[1])*
00487                                 (1./std::sqrt(variance));
00488

00490         NormalDistribution psi;

00492                        // - forwardAdjusted*phi(d1)*expiry*derDriftDerK;

00498             << result/deflator << " result: " << result<< " deflator: " << deflator);

00652 
00653         // the following is not always correct
00654         Calendar calendar = schedule_.calendar();
00655

00660         for (Size i=0; i<n; ++i) {
00661             refStart = start = schedule_.date(i);

00664             if (i==0   && !schedule_.isRegular(i+1)) {
00665                 BusinessDayConvention bdc = schedule_.businessDayConvention();
