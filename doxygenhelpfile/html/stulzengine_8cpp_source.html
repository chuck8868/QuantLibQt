00033         // calculate the value of euro min basket call
00034         Real euroTwoAssetMinBasketCall(Real forward1, Real forward2,
00035                                        Real strike,
00036                                        DiscountFactor riskFreeDiscount,
00037                                        Real variance1, Real variance2,
00038                                        Real rho) {
00039 
00040             Real stdDev1 = std::sqrt(variance1);
00041             Real stdDev2 = std::sqrt(variance2);
00042 
00043             Real variance = variance1 + variance2 - 2*rho*stdDev1*stdDev2;
00044             Real stdDev = std::sqrt(variance);
00045 
00046             Real modRho1 = (rho * stdDev2 - stdDev1) / stdDev;
00047             Real modRho2 = (rho * stdDev1 - stdDev2) / stdDev;
00048 
00049             Real D1 = (std::log(forward1/forward2) + 0.5*variance) / stdDev;
00050 
00051             Real alfa, beta, gamma;
00052             if (strike != 0.0) {
00053                 BivariateCumulativeNormalDistribution bivCNorm =
00054                     BivariateCumulativeNormalDistribution(rho);
00055                 BivariateCumulativeNormalDistribution bivCNormMod2 =
00056                     BivariateCumulativeNormalDistribution(modRho2);
00057                 BivariateCumulativeNormalDistribution bivCNormMod1 =
00058                     BivariateCumulativeNormalDistribution(modRho1);
00059 
00060                 Real D1_1 =
00061                     (std::log(forward1/strike) + 0.5*variance1) / stdDev1;
00062                 Real D1_2 =
00063                     (std::log(forward2/strike) + 0.5*variance2) / stdDev2;
00064                 alfa = bivCNormMod1(D1_1, -D1);
00065                 beta = bivCNormMod2(D1_2, D1 - stdDev);
00066                 gamma = bivCNorm(D1_1 - stdDev1, D1_2 - stdDev2);
00067             } else {
00068                 CumulativeNormalDistribution cum;
00069                 alfa = cum(-D1);
00070                 beta = cum(D1 - stdDev);
00071                 gamma = 1.0;
00072             }
00073 
00074             return riskFreeDiscount *
00075                 (forward1*alfa + forward2*beta - strike*gamma);
00076 
00077         }
