00119     Real Joshi4::computeUpProb(Real k, Real dj) const {
00120         Real alpha = dj/(std::sqrt(8.0));
00121         Real alpha2 = alpha*alpha;
00122         Real alpha3 = alpha*alpha2;
00123         Real alpha5 = alpha3*alpha2;
00124         Real alpha7 = alpha5*alpha2;
00125         Real beta = -0.375*alpha-alpha3;
00126         Real gamma = (5.0/6.0)*alpha5 + (13.0/12.0)*alpha3
00127             +(25.0/128.0)*alpha;
00128         Real delta = -0.1025 *alpha- 0.9285 *alpha3
00129             -1.43 *alpha5 -0.5 *alpha7;
00130         Real p =0.5;
00131         Real rootk = std::sqrt(k);
00132         p+= alpha/rootk;
00133         p+= beta /(k*rootk);
00134         p+= gamma/(k*k*rootk);
00135         // delete next line to get results for j three tree
00136         p+= delta/(k*k*k*rootk);
00137         return p;
00138     }
