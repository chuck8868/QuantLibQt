00084     Real AnalyticPTDHestonEngine::Fj_Helper::operator()(Real phi) const {
00085 
00086         // avoid numeric overflow for phi->0. 
00087         // todo: use l'Hospital's rule use to get lim_{phi->0}
00088         phi = std::max(Real(std::numeric_limits<float>::epsilon()), phi);
00089         
00090         std::complex<Real> D = 0.0;
00091         std::complex<Real> C = 0.0;
00092 
00093         for (Size i=timeGrid_.size()-1; i > 0; --i) {
00094             const Time begin = timeGrid_[i-1];
00095             if (begin < term_) {
00096                 const Time end = std::min(term_, timeGrid_[i]);
00097                 const Time tau = end-begin;
00098                 const Time t   = 0.5*(end+begin);
00099                 
00100                 const Real rho = model_->rho(t);
00101                 const Real sigma = model_->sigma(t);
00102                 const Real kappa = model_->kappa(t);
00103                 const Real theta = model_->theta(t);
00104                 
00105                 const Real sigma2 = sigma*sigma;
00106                 const Real t0 = kappa - ((j_== 1)? rho*sigma : 0);
00107                 const Real rpsig = rho*sigma*phi;
00108 
00109                 const std::complex<Real> t1 = t0+std::complex<Real>(0, -rpsig);
00110                 const std::complex<Real> d  = std::sqrt(t1*t1 - sigma2*phi
00111                                  *std::complex<Real>(-phi, (j_== 1)? 1 : -1));
00112                 const std::complex<Real> g = (t1-d)/(t1+d);
00113                 const std::complex<Real> gt 
00114                                        = (t1-d - D*sigma2)/(t1+d - D*sigma2);
00115                 
00116                 D = (t1+d)/sigma2*(g-gt*std::exp(-d*tau))
00117                     /(1.0-gt*std::exp(-d*tau));
00118                 
00119                 const std::complex<Real> lng 
00120                     = std::log((1.0 - gt*std::exp(-d*tau))/(1.0 - gt));
00121                 
00122                 C =(kappa*theta)/sigma2*((t1-d)*tau-2.0*lng)
00123                     + std::complex<Real>(0.0, phi*(r_[i-1]-q_[i-1])*tau) + C;
00124             }
00125         }
00126         return std::exp(v0_*D+C+std::complex<Real>(0.0, phi*(x_ - sx_))).imag()
00127                 /phi; 
00128     }

00151     void AnalyticPTDHestonEngine::calculate() const {
00152         // this is an european option pricer
00153         QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
00154                 "not an European option");
00155 
00156         // plain vanilla
00157         boost::shared_ptr<PlainVanillaPayoff> payoff =
00158             boost::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
00159         QL_REQUIRE(payoff, "non-striked payoff given");
00160         
00161         const Real v0 = model_->v0();
00162         const Real spotPrice = model_->s0();
00163         QL_REQUIRE(spotPrice > 0.0, "negative or null underlying given");
00164         
00165         const Real strike = payoff->strike();
00166         const Real term 
00167             = model_->riskFreeRate()->dayCounter().yearFraction(
00168                                      model_->riskFreeRate()->referenceDate(), 
00169                                      arguments_.exercise->lastDate());
00170         const Real riskFreeDiscount = model_->riskFreeRate()->discount(
00171                                             arguments_.exercise->lastDate());
00172         const Real dividendDiscount = model_->dividendYield()->discount(
00173                                             arguments_.exercise->lastDate());
00174 
00175         //average values
00176         const TimeGrid& timeGrid = model_->timeGrid();
00177         const Size n = timeGrid.size()-1;
00178         Real kappaAvg = 0.0, thetaAvg = 0.0,  sigmaAvg=0.0, rhoAvg = 0.0;
00179 
00180         for (Size i=1; i <= n; ++i) {
00181             const Time t = 0.5*(timeGrid[i-1] + timeGrid[i]);
00182             kappaAvg += model_->kappa(t);
00183             thetaAvg += model_->theta(t);
00184             sigmaAvg += model_->sigma(t);
00185             rhoAvg   += model_->rho(t);
00186         }
00187         kappaAvg/=n; thetaAvg/=n; sigmaAvg/=n; rhoAvg/=n;
00188         
00189         const Real c_inf = std::min(10.0, std::max(0.0001,
00190                 std::sqrt(1.0-square<Real>()(rhoAvg))/sigmaAvg))
00191                 *(v0 + kappaAvg*thetaAvg*term);
00192 
00193         const Real p1 = integration_->calculate(c_inf,
00194                                 Fj_Helper(model_, this, term, strike, 1))/M_PI;
00195 
00196         const Real p2 = integration_->calculate(c_inf,
00197                                 Fj_Helper(model_, this, term, strike, 2))/M_PI;
00198 
00199         switch (payoff->optionType())
00200         {
00201           case Option::Call:
00202             results_.value = spotPrice*dividendDiscount*(p1+0.5)
00203                             - strike*riskFreeDiscount*(p2+0.5);
00204             break;
00205           case Option::Put:
00206             results_.value = spotPrice*dividendDiscount*(p1-0.5)
00207                             - strike*riskFreeDiscount*(p2-0.5);
00208             break;
00209           default:
00210             QL_FAIL("unknown option type");
00211         }
