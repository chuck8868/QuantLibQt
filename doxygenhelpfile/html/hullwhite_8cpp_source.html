00040     boost::shared_ptr<Lattice> HullWhite::tree(const TimeGrid& grid) const {
00041 
00042         TermStructureFittingParameter phi(termStructure());
00043         boost::shared_ptr<ShortRateDynamics> numericDynamics(
00044                                              new Dynamics(phi, a(), sigma()));
00045         boost::shared_ptr<TrinomialTree> trinomial(
00046                          new TrinomialTree(numericDynamics->process(), grid));
00047         boost::shared_ptr<ShortRateTree> numericTree(
00048                          new ShortRateTree(trinomial, numericDynamics, grid));
00049 
00050         typedef TermStructureFittingParameter::NumericalImpl NumericalImpl;
00051         boost::shared_ptr<NumericalImpl> impl =
00052             boost::dynamic_pointer_cast<NumericalImpl>(phi.implementation());
00053         impl->reset();
00054         for (Size i=0; i<(grid.size() - 1); i++) {
00055             Real discountBond = termStructure()->discount(grid[i+1]);
00056             const Array& statePrices = numericTree->statePrices(i);
00057             Size size = numericTree->size(i);
00058             Time dt = numericTree->timeGrid().dt(i);
00059             Real dx = trinomial->dx(i);
00060             Real x = trinomial->underlying(i,0);
00061             Real value = 0.0;
00062             for (Size j=0; j<size; j++) {
00063                 value += statePrices[j]*std::exp(-x*dt);
00064                 x += dx;
00065             }
00066             value = std::log(value/discountBond)/dt;
00067             impl->set(grid[i], value);
00068         }
00069         return numericTree;

00072     Real HullWhite::A(Time t, Time T) const {
00073         DiscountFactor discount1 = termStructure()->discount(t);
00074         DiscountFactor discount2 = termStructure()->discount(T);
00075         Rate forward = termStructure()->forwardRate(t, t,
00076                                                     Continuous, NoFrequency);
00077         Real temp = sigma()*B(t,T);
00078         Real value = B(t,T)*forward - 0.25*temp*temp*B(0.0,2.0*t);
00079         return std::exp(value)*discount2/discount1;

00087                                        Time maturity,
00088                                        Time bondMaturity) const {
00089 
00090         Real _a = a();
00091         Real v;
00092         if (_a < std::sqrt(QL_EPSILON)) {

00095             v = sigma()*B(maturity, bondMaturity)*

00104     Rate HullWhite::convexityBias(Real futuresPrice,
00105                                   Time t,
00106                                   Time T,
00107                                   Real sigma,
00108                                   Real a) {
00109         QL_REQUIRE(futuresPrice>=0.0,
00110             "negative futures price (" << futuresPrice << ") not allowed");
00111         QL_REQUIRE(t>=0.0,
00112             "negative t (" << t << ") not allowed");
00113         QL_REQUIRE(T>=t,
00114             "T (" << T << ") must not be less than t (" << t << ")");
00115         QL_REQUIRE(sigma>=0.0,
00116             "negative sigma (" << sigma << ") not allowed");
00117         QL_REQUIRE(a>=0.0,
00118             "negative a (" << a << ") not allowed");
00119 
00120         Time deltaT = (T-t);
00121         Real tempDeltaT = (1.-std::exp(-a*deltaT)) / a;
00122         Real halfSigmaSquare = sigma*sigma/2.0;
00123 
00124         // lambda adjusts for the fact that the underlying is an interest rate
00125         Real lambda = halfSigmaSquare * (1.-std::exp(-2.0*a*t)) / a *
00126             tempDeltaT * tempDeltaT;
00127 
00128         Real tempT = (1.0 - std::exp(-a*t)) / a;
00129 
00130         // phi is the MtM adjustment
00131         Real phi = halfSigmaSquare * tempDeltaT * tempT * tempT;
00132 
00133         // the adjustment
00134         Real z = lambda + phi;
00135 
00136         Rate futureRate = (100.0-futuresPrice)/100.0;
00137         return (1.0-std::exp(-z)) * (futureRate + 1.0/(T-t));
00138     }
