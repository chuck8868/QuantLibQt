00077     Real YoYInflationCouponPricer::optionletPrice(Option::Type optionType,
00078                                                   Real effStrike) const {
00079         Date fixingDate = coupon_->fixingDate();
00080         if (fixingDate <= Settings::instance().evaluationDate()) {
00081             // the amount is determined
00082             Real a, b;
00083             if (optionType==Option::Call) {
00084                 a = coupon_->indexFixing();
00085                 b = effStrike;
00086             } else {
00087                 a = effStrike;
00088                 b = coupon_->indexFixing();
00089             }
00090             return std::max(a - b, 0.0)* coupon_->accrualPeriod()*discount_;
00091         } else {
00092             // not yet determined, use Black/DD1/Bachelier/whatever from Impl
00093             QL_REQUIRE(!capletVolatility().empty(),
00094                        "missing optionlet volatility");
00095             Real stdDev =
00096             std::sqrt(capletVolatility()->totalVariance(fixingDate,
00097                                                         effStrike));
00098             Rate fixing = optionletPriceImp(optionType,
00099                                             effStrike,
00100                                             adjustedFixing(),
00101                                             stdDev);
00102             return fixing * coupon_->accrualPeriod() * discount_;
00103         }
00104     }

00117     void YoYInflationCouponPricer::initialize(const InflationCoupon& coupon) {
00118         coupon_ = dynamic_cast<const YoYInflationCoupon*>(&coupon);
00119         QL_REQUIRE(coupon_, "year-on-year inflation coupon needed");
00120         gearing_ = coupon_->gearing();
00121         spread_ = coupon_->spread();
00122         paymentDate_ = coupon_->date();
00123         rateCurve_ = boost::dynamic_pointer_cast<YoYInflationIndex>(coupon.index())
00124             ->yoyInflationTermStructure()
00125             ->nominalTermStructure();
00126 
00127         // past or future fixing is managed in YoYInflationIndex::fixing()
00128         // use yield curve from index (which sets discount)
00129 
00130         discount_ = 1.0;
00131         if (paymentDate_ > rateCurve_->referenceDate())
00132             discount_ = rateCurve_->discount(paymentDate_);
00133 
00134         spreadLegValue_ = spread_ * coupon_->accrualPeriod()* discount_;
