00051             return -yearFraction(d2,d1,d3,d4);

00067             Integer(0.5+12*Real(refPeriodEnd-refPeriodStart)/365);
00068 
00069         // for short periods...
00070         if (months == 0) {
00071             // ...take the reference period as 1 year from d1
00072             refPeriodStart = d1;
00073             refPeriodEnd = d1 + 1*Years;
00074             months = 12;
00075         }
00076

00078 
00079         if (d2 <= refPeriodEnd) {
00080             // here refPeriodEnd is a future (notional?) payment date
00081             if (d1 >= refPeriodStart) {
00082                 // here refPeriodStart is the last (maybe notional)
00083                 // payment date.
00084                 // refPeriodStart <= d1 <= d2 <= refPeriodEnd
00085                 // [maybe the equality should be enforced, since
00086                 // refPeriodStart < d1 <= d2 < refPeriodEnd
00087                 // could give wrong results] ???
00088                 return period*Real(dayCount(d1,d2)) /
00089                     dayCount(refPeriodStart,refPeriodEnd);
00090             } else {
00091                 // here refPeriodStart is the next (maybe notional)
00092                 // payment date and refPeriodEnd is the second next
00093                 // (maybe notional) payment date.

00095                 // AND d2 <= refPeriodEnd
00096                 // this case is long first coupon
00097 
00098                 // the last notional payment date

00101                     return yearFraction(d1, refPeriodStart, previousRef,
00102                                         refPeriodStart) +
00103                         yearFraction(refPeriodStart, d2, refPeriodStart,
00104                                      refPeriodEnd);
00105                 else
00106                     return yearFraction(d1,d2,previousRef,refPeriodStart);
00107             }
00108         } else {
00109             // here refPeriodEnd is the last (notional?) payment date
00110             // d1 < refPeriodEnd < d2 AND refPeriodStart < refPeriodEnd
00111             QL_REQUIRE(refPeriodStart<=d1,
00112                        "invalid dates: "
00113                        "d1 < refPeriodStart < refPeriodEnd < d2");

00117             Time sum = yearFraction(d1, refPeriodEnd,
