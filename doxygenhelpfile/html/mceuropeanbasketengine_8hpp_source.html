00174     template <class RNG, class S>
00175     inline TimeGrid MCEuropeanBasketEngine<RNG,S>::timeGrid() const {
00176 
00177         Time residualTime = processes_->time(
00178                                        this->arguments_.exercise->lastDate());
00179         if (timeSteps_ != Null<Size>()) {
00180             return TimeGrid(residualTime, timeSteps_);
00181         } else if (timeStepsPerYear_ != Null<Size>()) {
00182             Size steps = static_cast<Size>(timeStepsPerYear_*residualTime);
00183             return TimeGrid(residualTime, std::max<Size>(steps, 1));
00184         } else {
00185             QL_FAIL("time steps not specified");
