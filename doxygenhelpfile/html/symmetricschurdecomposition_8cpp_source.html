00027     : diagonal_(s.rows()), eigenVectors_(s.rows(), s.columns(), 0.0) {
00028 
00029         QL_REQUIRE(s.rows() > 0 && s.columns() > 0, "null matrix given");
00030         QL_REQUIRE(s.rows()==s.columns(), "input matrix must be square");

00032         Size size = s.rows();
00033         for (Size q=0; q<size; q++) {
00034             diagonal_[q] = s[q][q];
00035             eigenVectors_[q][q] = 1.0;
00036         }
00037         Matrix ss = s;

00039         std::vector<Real> tmpDiag(diagonal_.begin(), diagonal_.end());
00040         std::vector<Real> tmpAccumulate(size, 0.0);
00041         Real threshold, epsPrec = 1e-15;
00042         bool keeplooping = true;

00044         do {
00045             //main loop
00046             Real sum = 0;
00047             for (Size a=0; a<size-1; a++) {
00048                 for (Size b=a+1; b<size; b++) {
00049                     sum += std::fabs(ss[a][b]);
00050                 }
00051             }

00053             if (sum==0) {
00054                 keeplooping = false;
00055             } else {
00056                 /* To speed up computation a threshold is introduced to
00057                    make sure it is worthy to perform the Jacobi rotation
00058                 */
00059                 if (ite<5) threshold = 0.2*sum/(size*size);
00060                 else       threshold = 0.0;
00061 
00062                 Size j, k, l;
00063                 for (j=0; j<size-1; j++) {
00064                     for (k=j+1; k<size; k++) {
00065                         Real sine, rho, cosin, heig, tang, beta;
00066                         Real smll = std::fabs(ss[j][k]);
00067                         if(ite> 5 &&
00068                            smll<epsPrec*std::fabs(diagonal_[j]) &&
00069                            smll<epsPrec*std::fabs(diagonal_[k])) {
00070                                 ss[j][k] = 0;
00071                         } else if (std::fabs(ss[j][k])>threshold) {
00072                             heig = diagonal_[k]-diagonal_[j];
00073                             if (smll<epsPrec*std::fabs(heig)) {
00074                                 tang = ss[j][k]/heig;
00075                             } else {
00076                                 beta = 0.5*heig/ss[j][k];
00077                                 tang = 1.0/(std::fabs(beta)+
00078                                     std::sqrt(1+beta*beta));
00079                                 if (beta<0)
00080                                     tang = -tang;
00081                             }
00082                             cosin = 1/std::sqrt(1+tang*tang);
00083                             sine = tang*cosin;
00084                             rho = sine/(1+cosin);
00085                             heig = tang*ss[j][k];
00086                             tmpAccumulate[j] -= heig;
00087                             tmpAccumulate[k] += heig;
00088                             diagonal_[j] -= heig;
00089                             diagonal_[k] += heig;
00090                             ss[j][k] = 0.0;
00091                             for (l=0; l+1<=j; l++)

00093                             for (l=j+1; l<=k-1; l++)
00094                                 jacobiRotate_(ss, rho, sine, j, l, l, k);
00095                             for (l=k+1; l<size; l++)
00096                                 jacobiRotate_(ss, rho, sine, j, l, k, l);
00097                             for (l=0;   l<size; l++)
00098                                 jacobiRotate_(eigenVectors_,
00099                                                   rho, sine, l, j, l, k);
00100                         }
00101                     }
00102                 }
00103                 for (k=0; k<size; k++) {
00104                     tmpDiag[k] += tmpAccumulate[k];
00105                     diagonal_[k] = tmpDiag[k];
00106                     tmpAccumulate[k] = 0.0;
00107                 }
00108             }
00109         } while (++ite<=maxIterations && keeplooping);

00112                   "Too many iterations (" << maxIterations << ") reached");

00120             std::copy(eigenVectors_.column_begin(col),
00121                       eigenVectors_.column_end(col), eigenVector.begin());

00123                 diagonal_[col], eigenVector);
00124         }

00126             std::greater<std::pair<Real, std::vector<Real> > >());

00129             // check for round-off errors
00130             diagonal_[col] =
00131                 (std::fabs(temp[col].first/maxEv)<1e-16 ? 0.0 :
00132                                                           temp[col].first);

00134             if (temp[col].second[0]<0.0)
00135                 sign = -1.0;
00136             for (row=0; row<size; row++) {
00137                 eigenVectors_[row][col] = sign * temp[col].second[row];
00138             }
00139         }
