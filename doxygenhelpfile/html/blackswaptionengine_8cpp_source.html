00066     void BlackSwaptionEngine::calculate() const {
00067         static const Spread basisPoint = 1.0e-4;
00068 
00069         Date exerciseDate = arguments_.exercise->date(0);
00070 
00071         // the part of the swap preceding exerciseDate should be truncated
00072         // to avoid taking into account unwanted cashflows
00073         VanillaSwap swap = *arguments_.swap;
00074 
00075         Rate strike = swap.fixedRate();
00076 
00077         // using the discounting curve
00078         // swap.iborIndex() might be using a different forwarding curve
00079         swap.setPricingEngine(boost::shared_ptr<PricingEngine>(new
00080             DiscountingSwapEngine(discountCurve_, false)));
00081         Rate atmForward = swap.fairRate();
00082 
00083         // Volatilities are quoted for zero-spreaded swaps.
00084         // Therefore, any spread on the floating leg must be removed
00085         // with a corresponding correction on the fixed leg.
00086         if (swap.spread()!=0.0) {
00087             Spread correction = swap.spread() *
00088                 std::fabs(swap.floatingLegBPS()/swap.fixedLegBPS());
00089             strike -= correction;
00090             atmForward -= correction;
00091             results_.additionalResults["spreadCorrection"] = correction;
00092         } else {
00093             results_.additionalResults["spreadCorrection"] = 0.0;
00094         }
00095         results_.additionalResults["strike"] = strike;
00096         results_.additionalResults["atmForward"] = atmForward;
00097 
00098         // using the discounting curve
00099         swap.setPricingEngine(boost::shared_ptr<PricingEngine>(
00100                            new DiscountingSwapEngine(discountCurve_, false)));
00101         Real annuity;
00102         switch(arguments_.settlementType) {
00103           case Settlement::Physical: {
00104               annuity = std::fabs(swap.fixedLegBPS())/basisPoint;
00105               break;
00106           }
00107           case Settlement::Cash: {
00108               const Leg& fixedLeg = swap.fixedLeg();
00109               boost::shared_ptr<FixedRateCoupon> firstCoupon =
00110                   boost::dynamic_pointer_cast<FixedRateCoupon>(fixedLeg[0]);
00111               DayCounter dayCount = firstCoupon->dayCounter();
00112               Real fixedLegCashBPS =
00113                   CashFlows::bps(fixedLeg,
00114                                  InterestRate(atmForward, dayCount, Compounded, Annual),
00115                                  false, discountCurve_->referenceDate()) ;
00116               annuity = std::fabs(fixedLegCashBPS/basisPoint);
00117               break;
00118           }
00119           default:
00120             QL_FAIL("unknown settlement type");
00121         }
00122         results_.additionalResults["annuity"] = annuity;
00123 
00124         // the swap length calculation might be improved using the value date
00125         // of the exercise date
00126         Time swapLength =  vol_->swapLength(exerciseDate,
00127                                                    arguments_.floatingPayDates.back());
00128         results_.additionalResults["swapLength"] = swapLength;
00129 
00130         Real variance = vol_->blackVariance(exerciseDate,
00131                                                    swapLength,
00132                                                    strike);
00133         Real stdDev = std::sqrt(variance);
00134         results_.additionalResults["stdDev"] = stdDev;
00135         Option::Type w = (arguments_.type==VanillaSwap::Payer) ?
00136                                                 Option::Call : Option::Put;
00137         results_.value = blackFormula(w, strike, atmForward, stdDev, annuity);
00138 
00139         Time exerciseTime = vol_->timeFromReference(exerciseDate);
00140         results_.additionalResults["vega"] = std::sqrt(exerciseTime) *
00141             blackFormulaStdDevDerivative(strike, atmForward, stdDev, annuity);
00142     }
