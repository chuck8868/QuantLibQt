00036         template <class I1, class I2, class Interpolator>
00037         class LogInterpolationImpl
00038             : public Interpolation::templateImpl<I1,I2> {
00039           public:
00040             LogInterpolationImpl(const I1& xBegin, const I1& xEnd,
00041                                  const I2& yBegin,
00042                                  const Interpolator& factory = Interpolator())
00043             : Interpolation::templateImpl<I1,I2>(xBegin, xEnd, yBegin),
00044               logY_(xEnd-xBegin) {
00045                 interpolation_ = factory.interpolate(this->xBegin_,
00046                                                      this->xEnd_,
00047                                                      logY_.begin());
00048             }
00049             void update() {
00050                 for (Size i=0; i<logY_.size(); ++i) {
00051                     QL_REQUIRE(this->yBegin_[i]>0.0,
00052                                "invalid value (" << this->yBegin_[i]
00053                                << ") at index " << i);
00054                     logY_[i] = std::log(this->yBegin_[i]);
00055                 }
00056                 interpolation_.update();
00057             }
00058             Real value(Real x) const {
00059                 return std::exp(interpolation_(x, true));
00060             }
00061             Real primitive(Real) const {
00062                 QL_FAIL("LogInterpolation primitive not implemented");
00063             }
00064             Real derivative(Real x) const {
00065                 return value(x)*interpolation_.derivative(x, true);
00066             }
00067             Real secondDerivative(Real x) const {
00068                 return derivative(x)*interpolation_.derivative(x, true) +
00069                             value(x)*interpolation_.secondDerivative(x, true);
00070             }
00071           private:
00072             std::vector<Real> logY_;
00073             Interpolation interpolation_;
00074         };
