00063                const Real discountBondPrice,
00064                const boost::shared_ptr<ShortRateTree>& tree)
00065         : size_(tree->size(i)),
00066           dt_(tree->timeGrid().dt(i)),
00067           xMin_(xMin), dx_(dx),
00068           statePrices_(tree->statePrices(i)),
00069           discountBondPrice_(discountBondPrice){}
00070 
00071         Real operator()(const Real theta) const {
00072             Real value = discountBondPrice_;
00073             Real x = xMin_;
00074             for (Size j=0; j<size_; j++) {
00075                 Real discount = std::exp(- fInverse_(theta+x)*dt_);
00076                 value -= statePrices_[j]*discount;
00077                 x += dx_;
00078             }
00079 
00080             return value;
00081         }

00186         Real vMax = 50.0;

00188         for (Size i=0; i<(grid.size() - 1); i++) {
00189             Real discountBond = termStructure()->discount(grid[i+1]);
