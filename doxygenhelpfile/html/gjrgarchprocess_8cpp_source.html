00056     Disposable<Array> GJRGARCHProcess::drift(Time t, const Array& x) const {
00057         Array tmp(2);
00058         const Real N = CumulativeNormalDistribution()(lambda_);
00059         const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
00060         const Real q2 = 1.0 + lambda_*lambda_;
00061         const Real q3 = lambda_*n + N + lambda_*lambda_*N;
00062         const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
00063                          : (discretization_ == Reflection) ? - std::sqrt(-x[1])
00064                          : 0.0;
00065 
00066         tmp[0] = riskFreeRate_->forwardRate(t, t, Continuous)
00067                - dividendYield_->forwardRate(t, t, Continuous)
00068                - 0.5 * vol * vol;
00069 
00070         tmp[1] = daysPerYear_*daysPerYear_*omega_ + daysPerYear_*(beta_ 
00071                                              + alpha_*q2 + gamma_*q3 - 1.0) *
00072            ((discretization_==PartialTruncation) ? x[1] : vol*vol);
00073         return tmp;
00074     }

00076     Disposable<Matrix> GJRGARCHProcess::diffusion(Time, const Array& x) const {
00077         /* the correlation matrix is
00078            |  1   rho |
00079            | rho   1  |
00080            whose square root (which is used here) is
00081            |  1          0       |
00082            | rho   std::sqrt(1-rho^2) |
00083         */
00084         Matrix tmp(2,2);
00085         const Real N = CumulativeNormalDistribution()(lambda_);
00086         const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
00087         const Real sigma2 = 2.0 + 4.0*lambda_*lambda_;
00088         const Real q3 = lambda_*n + N + lambda_*lambda_*N;
00089         const Real Eml_e4 = lambda_*lambda_*lambda_*n + 5.0*lambda_*n 
00090             + 3.0*N + lambda_*lambda_*lambda_*lambda_*N 
00091             + 6.0*lambda_*lambda_*N;
00092         const Real sigma3 = Eml_e4 - q3*q3;
00093         const Real sigma12 = -2.0*lambda_;
00094         const Real sigma13 = -2.0*n - 2*lambda_*N;
00095         const Real sigma23 = 2.0*N + sigma12*sigma13;
00096         const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
00097                          : (discretization_ == Reflection) ? - std::sqrt(-x[1])
00098                          : 1e-8; // set vol to (almost) zero but still
00099                                  // expose some correlation information
00100         const Real rho1 = std::sqrt(daysPerYear_)*(alpha_*sigma12 
00101                                             + gamma_*sigma13) * vol * vol;
00102         const Real rho2 = vol*vol*std::sqrt(daysPerYear_)
00103             *std::sqrt(alpha_*alpha_*(sigma2 - sigma12*sigma12) 
00104                        + gamma_*gamma_*(sigma3 - sigma13*sigma13) 
00105                        + 2.0*alpha_*gamma_*(sigma23 - sigma12*sigma13)); 
00106 
00107             // tmp[0][0], tmp[0][1] are the coefficients of dW_1 and dW_2 
00108             // in asset return stochastic process
00109         tmp[0][0] = vol;  tmp[0][1] = 0.0;
00110         tmp[1][0] = rho1; tmp[1][1] = rho2;
00111         return tmp;
00112     }

00122     Disposable<Array> GJRGARCHProcess::evolve(Time t0, const Array& x0,
00123                                             Time dt, const Array& dw) const {
00124         Array retVal(2);
00125         Real vol, mu, nu;
00126 
00127         const Real sdt = std::sqrt(dt);
00128         const Real N = CumulativeNormalDistribution()(lambda_);
00129         const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
00130         const Real sigma2 = 2.0 + 4.0*lambda_*lambda_;
00131         const Real q2 = 1.0 + lambda_*lambda_;
00132         const Real q3 = lambda_*n + N + lambda_*lambda_*N;
00133         const Real Eml_e4 = lambda_*lambda_*lambda_*n + 5.0*lambda_*n 
00134             + 3.0*N + lambda_*lambda_*lambda_*lambda_*N 
00135             + 6.0*lambda_*lambda_*N;
00136         const Real sigma3 = Eml_e4 - q3*q3;
00137         const Real sigma12 = -2.0*lambda_;
00138         const Real sigma13 = -2.0*n - 2*lambda_*N;
00139         const Real sigma23 = 2.0*N + sigma12*sigma13;
00140         const Real rho1 = std::sqrt(daysPerYear_)*(alpha_*sigma12 + gamma_*sigma13);
00141         const Real rho2 = std::sqrt(daysPerYear_)
00142             *std::sqrt(alpha_*alpha_*(sigma2 - sigma12*sigma12) 
00143                        + gamma_*gamma_*(sigma3 - sigma13*sigma13) 
00144                        + 2.0*alpha_*gamma_*(sigma23 - sigma12*sigma13));
00145 
00146         switch (discretization_) {
00147           // For the definition of PartialTruncation, FullTruncation
00148           // and Reflection  see Lord, R., R. Koekkoek and D. van Dijk (2006),
00149           // "A Comparison of biased simulation schemes for
00150           //  stochastic volatility models",
00151           // Working Paper, Tinbergen Institute
00152           case PartialTruncation:
00153             vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00154             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00155                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00156                     - 0.5 * vol * vol;
00157             nu = daysPerYear_*daysPerYear_*omega_ 
00158                 + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * x0[1];
00159 
00160             retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
00161             retVal[1] = x0[1] + nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
00162             break;
00163           case FullTruncation:
00164             vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00165             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00166                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00167                     - 0.5 * vol * vol;
00168             nu = daysPerYear_*daysPerYear_*omega_ 
00169                 + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * vol *vol;
00170 
00171             retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
00172             retVal[1] = x0[1] + nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
00173             break;
00174           case Reflection:
00175             vol = std::sqrt(std::fabs(x0[1]));
00176             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00177                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00178                     - 0.5 * vol*vol;
00179             nu = daysPerYear_*daysPerYear_*omega_ 
00180                 + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * vol * vol;
00181 
00182             retVal[0] = x0[0]*std::exp(mu*dt+vol*dw[0]*sdt);
00183             retVal[1] = vol*vol
00184                         +nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
00185             break;
00186           default:
00187             QL_FAIL("unknown discretization schema");
