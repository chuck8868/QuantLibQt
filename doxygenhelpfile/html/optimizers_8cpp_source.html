00047     class OneDimensionalPolynomialDegreeN : public CostFunction {
00048       public:
00049         OneDimensionalPolynomialDegreeN(const Array& coefficients)
00050         : coefficients_(coefficients),
00051           polynomialDegree_(coefficients.size()-1),odd(true) {}
00052 
00053         Real value(const Array& x) const {
00054             QL_REQUIRE(x.size()==1,"independent variable must be 1 dimensional");
00055             Real y = 0;
00056             for (Size i=0; i<=polynomialDegree_; ++i)
00057                 y += coefficients_[i]*std::pow(x[0],static_cast<int>(i));
00058             return y;
00059         }
00060 
00061         Disposable<Array> values(const Array& x) const{
00062             QL_REQUIRE(x.size()==1,"independent variable must be 1 dimensional");
00063             Array y(1);
00064             y[0] = value(x);
00065             return y;
00066         }
00067 
00068       private:
00069         const Array coefficients_;
00070         const Size polynomialDegree_;
00071         mutable bool odd;
00072     };
