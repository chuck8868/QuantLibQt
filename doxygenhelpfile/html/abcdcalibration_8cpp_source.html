00067     void AbcdCalibration::compute() {
00068         if (vegaWeighted_) {
00069             Real weightsSum = 0.0;
00070             for (Size i=0; i<times_.size() ; i++) {
00071                 Real stdDev = std::sqrt(blackVols_[i]* blackVols_[i]* times_[i]);
00072                 // when strike==forward, the blackFormulaStdDevDerivative becomes
00073                 weights_[i] = CumulativeNormalDistribution().derivative(.5*stdDev);
00074                 weightsSum += weights_[i];
00075             }
00076             // weight normalization
00077             for (Size i=0; i<times_.size() ; i++) {
00078                 weights_[i] /= weightsSum;
00079             }
00080         }
00081 
00082         // there is nothing to optimize
00083         if (aIsFixed_ && bIsFixed_ && cIsFixed_ && dIsFixed_) {
00084             abcdEndCriteria_ = EndCriteria::None;
00085             //error_ = interpolationError();
00086             //maxError_ = interpolationMaxError();
00087             return;
00088         } else {
00089 
00090             AbcdError costFunction(this);
00091             transformation_ = boost::shared_ptr<ParametersTransformation>(new
00092                 AbcdParametersTransformation);
00093 
00094             Array guess(4);
00095             guess[0] = a_;
00096             guess[1] = b_;
00097             guess[2] = c_;
00098             guess[3] = d_;
00099 
00100             std::vector<bool> parameterAreFixed(4);
00101             parameterAreFixed[0] = aIsFixed_;
00102             parameterAreFixed[1] = bIsFixed_;
00103             parameterAreFixed[2] = cIsFixed_;
00104             parameterAreFixed[3] = dIsFixed_;
00105 
00106             Array inversedTransformatedGuess(transformation_->inverse(guess));
00107 
00108             ProjectedCostFunction projectedAbcdCostFunction(costFunction,
00109                             inversedTransformatedGuess, parameterAreFixed);
00110 
00111             Array projectedGuess
00112                 (projectedAbcdCostFunction.project(inversedTransformatedGuess));
00113 
00114             NoConstraint constraint;
00115             Problem problem(projectedAbcdCostFunction, constraint, projectedGuess);
00116             abcdEndCriteria_ = optMethod_->minimize(problem, *endCriteria_);
00117             Array projectedResult(problem.currentValue());
00118             Array transfResult(projectedAbcdCostFunction.include(projectedResult));
00119 
00120             Array result = transformation_->direct(transfResult);
00121             a_ = result[0];
00122             b_ = result[1];
00123             c_ = result[2];
00124             d_ = result[3];
00125 
00126             validateAbcdParameters(a_, b_, c_, d_);
00127         }
00128     }

00151     std::vector<Real> AbcdCalibration::k(const std::vector<Real>& t,
00152                                          const std::vector<Real>& blackVols) const {

00154                "mismatch between number of times (" << t.size() <<

00163     Real AbcdCalibration::error() const {
00164         Size n = times_.size();
00165         Real error, squaredError = 0.0;
00166         for (Size i=0; i<times_.size() ; i++) {
00167             error = (value(times_[i]) - blackVols_[i]);
00168             squaredError += error * error *(weights_[i]);
00169         }
00170         return std::sqrt(n*squaredError/(n-1));
00171     }

00173     Real AbcdCalibration::maxError() const {
