00031     LogNormalFwdRateEulerConstrained::LogNormalFwdRateEulerConstrained(
00032                         const boost::shared_ptr<MarketModel>& marketModel,
00033                         const BrownianGeneratorFactory& factory,
00034                         const std::vector<Size>& numeraires,
00035                         Size initialStep)
00036     : marketModel_(marketModel),
00037       numeraires_(numeraires),
00038       initialStep_(initialStep),
00039       numberOfRates_(marketModel->numberOfRates()),
00040       numberOfFactors_(marketModel_->numberOfFactors()),
00041       curveState_(marketModel->evolution().rateTimes()),
00042       forwards_(marketModel->initialRates()),
00043       displacements_(marketModel->displacements()),
00044       logForwards_(numberOfRates_), initialLogForwards_(numberOfRates_),
00045       drifts1_(numberOfRates_), initialDrifts_(numberOfRates_),
00046       brownians_(numberOfFactors_), correlatedBrownians_(numberOfRates_),
00047       alive_(marketModel->evolution().firstAliveRate())
00048     {
00049         checkCompatibility(marketModel->evolution(), numeraires);
00050 
00051         Size steps = marketModel->evolution().numberOfSteps();
00052 
00053         generator_ = factory.create(numberOfFactors_, steps-initialStep_);
00054         currentStep_ = initialStep_;
00055 
00056         calculators_.reserve(steps);
00057         variances_.reserve(steps);
00058         fixedDrifts_.reserve(steps);
00059         for (Size j=0; j<steps; ++j) {
00060             const Matrix& A = marketModel_->pseudoRoot(j);
00061             calculators_.push_back(LMMDriftCalculator(A,
00062                 displacements_,
00063                 marketModel->evolution().rateTaus(),
00064                 numeraires[j],
00065                 alive_[j]));
00066             std::vector<Real> fixed(numberOfRates_);
00067             std::vector<Real> variances(numberOfRates_);
00068             for (Size k=0; k<numberOfRates_; ++k) {
00069                 Real variance =
00070                     std::inner_product(A.row_begin(k), A.row_end(k),
00071                     A.row_begin(k), 0.0);
00072                 variances[k] = variance;
00073                 fixed[k] = -0.5*variance;
00074             }
00075             variances_.push_back(variances);
00076             fixedDrifts_.push_back(fixed);
00077         }
00078 
00079         setForwards(marketModel_->initialRates());
00080     }

00160     Real LogNormalFwdRateEulerConstrained::advanceStep()
00161     {
00162         // we're going from T1 to T2
00163 
00164         // a) compute drifts D1 at T1;
00165         if (currentStep_ > initialStep_) {
00166             calculators_[currentStep_].compute(forwards_, drifts1_);
00167         } else {
00168             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00169                       drifts1_.begin());
00170         }
00171 
00172         // b) evolve forwards up to T2 using D1;
00173         Real weight = generator_->nextStep(brownians_);
00174         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00175         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00176 
00177         Size alive = alive_[currentStep_];
00178         for (Size i=alive; i<numberOfRates_; i++) {
00179             logForwards_[i] += drifts1_[i] + fixedDrift[i];
00180             logForwards_[i] +=
00181                 std::inner_product(A.row_begin(i), A.row_end(i),
00182                 brownians_.begin(), 0.0);
00183         }
00184 
00185         // check constraint active
00186         if (isConstraintActive_[currentStep_]) {
00187             Size index = startIndexOfSwapRate_[currentStep_];
00188 
00189             // compute error
00190             Real requiredShift =  rateConstraints_[currentStep_] - logForwards_[index];
00191 
00192             Real multiplier = requiredShift/variances_[currentStep_][index];
00193 
00194             // now shift each rate by multiplier * weighting of index rate
00195             // across the step
00196             for (Size i=alive; i<numberOfRates_; i++) {
00197                 // we only need a small part of cov matrix
00198                 logForwards_[i] += multiplier*covariances_[currentStep_][i];
00199             }
00200 
00201             // now we need to calculate the LR shift, we have moved Gaussian Z_k by multiplier * a_{index,k}
00202             // divide original density by density of shifted normal
00203             Real weightsEffect = 1.0;
00204 
00205             CumulativeNormalDistribution phi;
00206             for (Size k=0; k < numberOfFactors_; k++) {
00207                 Real shift = multiplier * A[index][k];
00208                 Real originalDensity = phi.derivative(brownians_[k]+shift);
00209                 // the density of the draw after changes in original measure
00210                 Real newDensity = phi.derivative(brownians_[k]);
00211                 // the density of the draw after changes in new measure, shifts cancel
00212                 weightsEffect*= originalDensity/newDensity;
00213             }
00214 
00215             weight *= weightsEffect;
00216         }
00217 
00218         for (Size i=alive; i<numberOfRates_; i++)
00219             forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
00220 
00221         // c) update curve state
00222         curveState_.setOnForwardRates(forwards_);
00223 
00224         ++currentStep_;
00225 
00226         return weight;
00227     }
