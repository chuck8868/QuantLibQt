00045     YoYInflationCapFloor::YoYInflationCapFloor(YoYInflationCapFloor::Type type,
00046                        const Leg& yoyLeg,
00047                        const std::vector<Rate>& capRates,
00048                        const std::vector<Rate>& floorRates)
00049     : type_(type), yoyLeg_(yoyLeg),
00050     capRates_(capRates), floorRates_(floorRates) {
00051         if (type_ == Cap || type_ == Collar) {
00052             QL_REQUIRE(!capRates_.empty(), "no cap rates given");
00053             capRates_.reserve(yoyLeg_.size());
00054             while (capRates_.size() < yoyLeg_.size())
00055                 capRates_.push_back(capRates_.back());
00056         }
00057         if (type_ == Floor || type_ == Collar) {
00058             QL_REQUIRE(!floorRates_.empty(), "no floor rates given");
00059             floorRates_.reserve(yoyLeg_.size());
00060             while (floorRates_.size() < yoyLeg_.size())
00061                 floorRates_.push_back(floorRates_.back());
00062         }
00063         Leg::const_iterator i;
00064         for (i = yoyLeg_.begin(); i != yoyLeg_.end(); ++i)
00065             registerWith(*i);
00066 
00067         registerWith(Settings::instance().evaluationDate());
00068     }

00134     void YoYInflationCapFloor::setupArguments(PricingEngine::arguments* args) const {
00135         YoYInflationCapFloor::arguments* arguments =
00136         dynamic_cast<YoYInflationCapFloor::arguments*>(args);
00137         QL_REQUIRE(arguments != 0, "wrong argument type");
00138 
00139         Size n = yoyLeg_.size();
00140 
00141         arguments->startDates.resize(n);
00142         arguments->fixingDates.resize(n);
00143         arguments->payDates.resize(n);
00144         arguments->accrualTimes.resize(n);
00145         arguments->nominals.resize(n);
00146         arguments->gearings.resize(n);
00147         arguments->capRates.resize(n);
00148         arguments->floorRates.resize(n);
00149         arguments->spreads.resize(n);
00150 
00151         arguments->type = type_;
00152 
00153         for (Size i=0; i<n; ++i) {
00154             shared_ptr<YoYInflationCoupon> coupon =
00155             boost::dynamic_pointer_cast<YoYInflationCoupon>(
00156                                                             yoyLeg_[i]);
00157             QL_REQUIRE(coupon, "non-YoYInflationCoupon given");
00158             arguments->startDates[i] = coupon->accrualStartDate();
00159             arguments->fixingDates[i] = coupon->fixingDate();
00160             arguments->payDates[i] = coupon->date();
00161 
00162             // this is passed explicitly for precision
00163             arguments->accrualTimes[i] = coupon->accrualPeriod();
00164 
00165             arguments->nominals[i] = coupon->nominal();
00166             Spread spread = coupon->spread();
00167             Real gearing = coupon->gearing();
00168             arguments->gearings[i] = gearing;
00169             arguments->spreads[i] = spread;
00170 
00171             if (type_ == Cap || type_ == Collar)
00172                 arguments->capRates[i] = (capRates_[i]-spread)/gearing;
00173             else
00174                 arguments->capRates[i] = Null<Rate>();
00175 
00176             if (type_ == Floor || type_ == Collar)
00177                 arguments->floorRates[i] = (floorRates_[i]-spread)/gearing;
00178             else
00179                 arguments->floorRates[i] = Null<Rate>();
00180         }
00181     }
