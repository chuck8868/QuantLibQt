00324     std::vector<Size> makeMeasure(const MarketModelMultiProduct& product,
00325                                   MeasureType measureType) {
00326         std::vector<Size> result;
00327         EvolutionDescription evolution(product.evolution());
00328         switch (measureType) {
00329           case ProductSuggested:
00330             result = product.suggestedNumeraires();
00331             break;
00332           case Terminal:
00333             result = terminalMeasure(evolution);
00334             if (!isInTerminalMeasure(evolution, result)) {
00335                 Array a(result.size());
00336                 std::copy(result.begin(), result.end(), a.begin());
00337                 BOOST_ERROR("\nfailure in verifying Terminal measure:\n" << a);
00338             }
00339             break;
00340           case MoneyMarket:
00341             result = moneyMarketMeasure(evolution);
00342             if (!isInMoneyMarketMeasure(evolution, result)) {
00343                 Array a(result.size());
00344                 std::copy(result.begin(), result.end(), a.begin());
00345                 BOOST_ERROR("\nfailure in verifying MoneyMarket measure:\n" << a);
00346             }
00347             break;
00348           case MoneyMarketPlus:
00349             result = moneyMarketPlusMeasure(evolution, measureOffset_);
00350             if (!isInMoneyMarketPlusMeasure(evolution, result, measureOffset_)) {
00351                 Array a(result.size());
00352                 std::copy(result.begin(), result.end(), a.begin());
00353                 BOOST_ERROR("\nfailure in verifying MoneyMarketPlus(" <<
00354                             measureOffset_ << ") measure:\n" << a);
00355             }
00356             break;
00357           default:
00358             QL_FAIL("unknown measure type");
00359         }
00360         checkCompatibility(evolution, result);
00361         if (printReport_) {
00362             Array num(result.size());
00363             std::copy(result.begin(), result.end(), num.begin());
00364             BOOST_MESSAGE("    " << measureTypeToString(measureType) << ": " << num);
00365         }
00366         return result;
00367     }

00411

00415     // Size numberOfSteps = evolution.numberOfSteps();

00438     bool lowestRoot = true;
00439     bool useFullApprox = false;

00444         BOOST_MESSAGE("lowestRoot:         " << lowestRoot);
