00165     Real AmericanPayoffAtHit::delta() const {
00166         Real tempDelta = - spot_ * stdDev_;
00167         Real DalphaDs = DalphaDd1_/tempDelta;
00168         Real DbetaDs  = DbetaDd2_/tempDelta;
00169 
00170         Real DforwardDs, DXDs;
00171         if (inTheMoney_) {
00172             DforwardDs = 0.0;
00173             DXDs       = 0.0;
00174         } else {
00175             DforwardDs = -muPlusLambda_  * forward_ / spot_;
00176             DXDs       = -muMinusLambda_ * X_       / spot_;
00177         }
00178 
00179         return K_ * (
00180               DalphaDs * forward_ + alpha_ * DforwardDs
00181             + DbetaDs  * X_       + beta_  * DXDs
00182             );
