00102     inline void LazyObject::update() {
00103         // forwards notifications only the first time
00104         if (calculated_) {
00105             // set to false early
00106             // 1) to prevent infinite recursion
00107             // 2) otherways non-lazy observers would be served obsolete
00108             //    data because of calculated_ being still true
00109             calculated_ = false;
00110             // observers don't expect notifications from frozen objects
00111             if (!frozen_)
00112                 notifyObservers();
00113                 // exiting notifyObservers() calculated_ could be
00114                 // already true because of non-lazy observers
00115         }
00116     }

00118     inline void LazyObject::recalculate() {
00119         bool wasFrozen = frozen_;
00120         calculated_ = frozen_ = false;
00121         try {
00122             calculate();
00123         } catch (...) {
00124             frozen_ = wasFrozen;
00125             notifyObservers();
00126             throw;
00127         }
00128         frozen_ = wasFrozen;
00129         notifyObservers();
00130     }

00137         // send notifications, just in case we lost any,
00138         // but only once, i.e. if it was frozen

00145     inline void LazyObject::calculate() const {
