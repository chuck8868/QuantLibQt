00036     LineSearchBasedMethod::minimize(Problem& P,
00037                                     const EndCriteria& endCriteria) {
00038         // Initializations
00039         Real ftol = endCriteria.functionEpsilon();
00040         Size maxStationaryStateIterations_

00044         Array x_ = P.currentValue();              // store the starting point
00045         Size iterationNumber_ = 0;
00046         // dimension line search
00047         lineSearch_->searchDirection() = Array(x_.size());
00048         bool done = false;
00049 
00050         // function and squared norm of gradient values;
00051         Real fnew, fold, gold2;
00052         Real fdiff;
00053         // classical initial value for line-search step
00054         Real t = 1.0;
00055         // Set gradient g at the size of the optimization problem
00056         // search direction
00057         Size sz = lineSearch_->searchDirection().size();
00058         Array prevGradient(sz), d(sz), sddiff(sz), direction(sz);
00059         // Initialize cost function, gradient prevGradient and search direction
00060         P.setFunctionValue(P.valueAndGradient(prevGradient, x_));
00061         P.setGradientNormValue(DotProduct(prevGradient, prevGradient));
00062         lineSearch_->searchDirection() = -prevGradient;
00063 
00064         bool first_time = true;
00065         // Loop over iterations
00066         do {
00067             // Linesearch
00068             if (!first_time)
00069                 prevGradient = lineSearch_->lastGradient();
00070             t = (*lineSearch_)(P, ecType, endCriteria, t);
00071             // don't throw: it can fail just because maxIterations exceeded
00072             //QL_REQUIRE(lineSearch_->succeed(), "line-search failed!");
00073             if (lineSearch_->succeed())
00074             {
00075                 // Updates
00076

00078                 x_ = lineSearch_->lastX();

00080                 fold = P.functionValue();
00081                 P.setFunctionValue(lineSearch_->lastFunctionValue());
00082                 // New gradient and search direction vectors

00084                 // orthogonalization coef
00085                 gold2 = P.gradientNormValue();

00088                 // conjugate gradient search direction

00090 
00091                 sddiff = direction - lineSearch_->searchDirection();

00093                 // Now compute accuracy and check end criteria
00094                 // Numerical Recipes exit strategy on fx (see NR in C++, p.423)
00095                 fnew = P.functionValue();

00102                     endCriteria.checkMaxIterations(iterationNumber_, ecType);
00103                     return ecType;
00104                 }
00105                 P.setCurrentValue(x_);      // update problem current value
00106                 ++iterationNumber_;         // Increase iteration number
00107                 first_time = false;
00108             } else {
00109                 done = true;
00110             }
00111         } while (!done);
