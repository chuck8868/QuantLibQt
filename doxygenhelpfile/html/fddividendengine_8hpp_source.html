00053             const Dividend *dividend =
00054                 dynamic_cast<const Dividend *>(this->events_[i].get());
00055             if (dividend) {
00056                 return dividend->amount();

00058                 return 0.0;

00062             Real dividend = getDividendAmount(i);
00063             Real discount =
00064                 this->process_->riskFreeRate()->discount(
00065                                                    this->events_[i]->date()) /
00066                 this->process_->dividendYield()->discount(
00067                                                     this->events_[i]->date());

00180             this->getDiscountedDividend(step) / this->center_ + 1.0;
00181         this->sMin_ *= scaleFactor;
