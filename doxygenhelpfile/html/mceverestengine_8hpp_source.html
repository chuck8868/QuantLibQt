00171     template <class RNG, class S>
00172     inline TimeGrid MCEverestEngine<RNG,S>::timeGrid() const {
00173         Time residualTime = processes_->time(
00174                                        this->arguments_.exercise->lastDate());
00175         if (timeSteps_ != Null<Size>()) {
00176             return TimeGrid(residualTime, timeSteps_);
00177         } else if (timeStepsPerYear_ != Null<Size>()) {
00178             Size steps = static_cast<Size>(timeStepsPerYear_*residualTime);
00179             return TimeGrid(residualTime, std::max<Size>(steps, 1));
00180         } else {
00181             QL_FAIL("time steps not specified");
00182         }
