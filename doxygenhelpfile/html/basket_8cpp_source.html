00062         for(Size i=0; i<notionals_.size(); i++)
00063             registerWith(rrModels_[i]);

00066         // At this point Issuers in the pool might or might not have
00067         //   probability term structures for the defultKeys(eventType+
00068         //   currency+seniority) entering in this basket. This is not
00069         //   neccessarily a problem.
00070         for (Size i = 0; i < notionals_.size(); i++) {
00071             basketNotional_ += notionals_[i];
00072             attachmentAmount_ += notionals_[i] * attachmentRatio_;
00073             detachmentAmount_ += notionals_[i] * detachmentRatio_;

00150             boost::shared_ptr<DefaultEvent> credEvent =

00153                                                        defaultKeys_[i]);
00154             // to do: adjust for settlement notional accrued convexity, see doc
00155             if (credEvent) {

00273     Real Basket::scenarioTrancheLoss(Date endDate) const {
00274         Real A = attachmentAmount_;
00275         Real D = detachmentAmount_;
00276         Date today = Settings::instance().evaluationDate();
00277         Real t2 = ActualActual().yearFraction(today, endDate);
00278         Real L = 0.0;
00279         for (Size i = 0; i < scenarioLoss_.size(); i++) {
00280             if (scenarioLoss_[i].time <= t2)
00281                 L += scenarioLoss_[i].amount;
00282             else break;
00283         }
00284         return std::min(L, D) - std::min(L, A);

00287     vector<Loss> Basket::scenarioIncrementalTrancheLosses(Date startDate,
00288                                                           Date endDate) const {
00289         vector<Loss> losses;
00290         Real A = attachmentAmount_;
00291         Real D = detachmentAmount_;
00292         Date today = Settings::instance().evaluationDate();
00293         Real tmin = ActualActual().yearFraction(today, startDate);
00294         Real tmax = ActualActual().yearFraction(today, endDate);
00295         Real TL1 = 0.0;
00296         Real L = 0.0;
00297         for (Size i = 0; i < scenarioLoss_.size(); i++) {
00298             Real t = scenarioLoss_[i].time;
00299             if (t > tmax && endDate != Date::maxDate()) break;
00300             if (t < tmin && startDate != Date::minDate()) continue;
00301             L += scenarioLoss_[i].amount;
00302             Real TL2 = std::min(L, D) - std::min(L, A);
00303             Real increment = TL2 - TL1;
00304             TL1 = TL2;
00305             losses.push_back(Loss(t, increment));
00306         }
00307         return losses;
00308     }
