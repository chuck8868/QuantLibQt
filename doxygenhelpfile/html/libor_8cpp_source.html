00086     Date Libor::valueDate(const Date& fixingDate) const {
00087 
00088         QL_REQUIRE(isValidFixingDate(fixingDate),
00089                    "Fixing date " << fixingDate << " is not valid");
00090 
00091         // http://www.bba.org.uk/bba/jsp/polopoly.jsp?d=225&a=1412 :
00092         // For all currencies other than EUR and GBP the period between
00093         // Fixing Date and Value Date will be two London business days
00094         // after the Fixing Date, or if that day is not both a London
00095         // business day and a business day in the principal financial centre
00096         // of the currency concerned, the next following day which is a
00097         // business day in both centres shall be the Value Date.
00098         Date d = fixingCalendar().advance(fixingDate, fixingDays_, Days);
00099         return jointCalendar_.adjust(d);
00100     }

00102     Date Libor::maturityDate(const Date& valueDate) const {
00103         // Where a deposit is made on the final business day of a
00104         // particular calendar month, the maturity of the deposit shall
00105         // be on the final business day of the month in which it matures
00106         // (not the corresponding date in the month of maturity). Or in
00107         // other words, in line with market convention, BBA LIBOR rates
00108         // are dealt on an end-end basis. For instance a one month
00109         // deposit for value 28th February would mature on 31st March,
00110         // not the 28th of March.
00111         return jointCalendar_.advance(valueDate, tenor_, convention_,
00112                                                          endOfMonth());
00113     }
