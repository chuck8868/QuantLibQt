00039                                              const Date& referenceDate,
00040                                              const DayCounter& dayCounter)
00041     : DiscretizedOption(boost::shared_ptr<DiscretizedAsset>(),
00042                         args.exercise->type(),
00043                         std::vector<Time>()),
00044       arguments_(args) {
00045 
00046         exerciseTimes_.resize(arguments_.exercise->dates().size());
00047         for (Size i=0; i<exerciseTimes_.size(); ++i)
00048             exerciseTimes_[i] =
00049                 dayCounter.yearFraction(referenceDate,
00050                                         arguments_.exercise->date(i));
00051 
00052         // Date adjustments can get time vectors out of synch.
00053         // Here, we try and collapse similar dates which could cause
00054         // a mispricing.
00055         for (Size i=0; i<arguments_.exercise->dates().size(); i++) {
00056             Date exerciseDate = arguments_.exercise->date(i);
00057             for (Size j=0; j<arguments_.fixedPayDates.size(); j++) {
00058                 if (withinNextWeek(exerciseDate,
00059                                    arguments_.fixedPayDates[j])
00060                     // coupons in the future are dealt with below
00061                     && arguments_.fixedResetDates[j] < referenceDate)
00062                     arguments_.fixedPayDates[j] = exerciseDate;
00063             }
00064             for (Size j=0; j<arguments_.fixedResetDates.size(); j++) {
00065                 if (withinPreviousWeek(exerciseDate,
00066                                        arguments_.fixedResetDates[j]))
00067                     arguments_.fixedResetDates[j] = exerciseDate;
00068             }
00069             for (Size j=0; j<arguments_.floatingResetDates.size(); j++) {
00070                 if (withinPreviousWeek(exerciseDate,
00071                                        arguments_.floatingResetDates[j]))
00072                     arguments_.floatingResetDates[j] = exerciseDate;
00073             }
00074         }
00075 
00076         Time lastFixedPayment =
00077             dayCounter.yearFraction(referenceDate,
00078                                     arguments_.fixedPayDates.back());
00079         Time lastFloatingPayment =
00080             dayCounter.yearFraction(referenceDate,
00081                                     arguments_.floatingPayDates.back());
00082         lastPayment_ = std::max(lastFixedPayment,lastFloatingPayment);
00083 
00084         underlying_ = boost::shared_ptr<DiscretizedAsset>(
00085                                             new DiscretizedSwap(arguments_,
00086                                                                 referenceDate,
00087                                                                 dayCounter));
