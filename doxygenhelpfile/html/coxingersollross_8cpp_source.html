00064     Real CoxIngersollRoss::A(Time t, Time T) const {
00065         Real sigma2 = sigma()*sigma();
00066         Real h = std::sqrt(k()*k() + 2.0*sigma2);
00067         Real numerator = 2.0*h*std::exp(0.5*(k()+h)*(T-t));
00068         Real denominator = 2.0*h + (k()+h)*(std::exp((T-t)*h) - 1.0);
00069         Real value = std::log(numerator/denominator)*
00070             2.0*k()*theta()/sigma2;
00071         return std::exp(value);
00072     }

00074     Real CoxIngersollRoss::B(Time t, Time T) const {
00075         Real h = std::sqrt(k()*k() + 2.0*sigma()*sigma());
00076         Real temp = std::exp((T-t)*h) - 1.0;
00077         Real numerator = 2.0*temp;
00078         Real denominator = 2.0*h + (k()+h)*temp;
00079         Real value = numerator/denominator;
00080         return value;

00084                                               Real strike,

00086 
00087         QL_REQUIRE(strike>0.0, "strike must be positive");
00088         DiscountFactor discountT = discountBond(0.0, t, x0());

00090 
00091         if (t < QL_EPSILON) {
00092             switch(type) {
00093               case Option::Call:
00094                 return std::max<Real>(discountS - strike, 0.0);
00095               case Option::Put:
00096                 return std::max<Real>(strike - discountS, 0.0);
00097               default: QL_FAIL("unsupported option type");
00098             }
00099         }
00100 
00101         Real sigma2 = sigma()*sigma();
00102         Real h = std::sqrt(k()*k() + 2.0*sigma2);
00103         Real b = B(t,s);
00104 
00105         Real rho = 2.0*h/(sigma2*(std::exp(h*t) - 1.0));
00106         Real psi = (k() + h)/sigma2;
00107 
00108         Real df = 4.0*k()*theta()/sigma2;
00109         Real ncps = 2.0*rho*rho*x0()*std::exp(h*t)/(rho+psi+b);
00110         Real ncpt = 2.0*rho*rho*x0()*std::exp(h*t)/(rho+psi);
00111 
00112         NonCentralChiSquareDistribution chis(df, ncps);
00113         NonCentralChiSquareDistribution chit(df, ncpt);
00114 
00115         Real z = std::log(A(t,s)/strike)/b;
00116         Real call = discountS*chis(2.0*z*(rho+psi+b)) -
00117             strike*discountT*chit(2.0*z*(rho+psi));
00118 
00119         if (type == Option::Call)
00120             return call;
00121         else
00122             return call - discountS + strike*discountT;
00123     }
