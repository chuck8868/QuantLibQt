00030                            Real mid,
00031                            Real high,
00032                            Real tolerance,
00033                            Size maxIt,
00034                            const F& objectiveFunction) {

00036             Real x = W*low+(1-W)*high;
00037             if (mid > low && mid < high)
00038                 x = mid;
00039 
00040             Real midValue = objectiveFunction(x);
00041 
00042             Size iterations = 0;
00043             while (high-low > tolerance && iterations < maxIt) {
00044                 if (x - low > high -x) { // left interval is bigger
00045                     Real tentativeNewMid = W*low+(1-W)*x;
00046                     Real tentativeNewMidValue =
00047                         objectiveFunction(tentativeNewMid);

00049                     if (tentativeNewMidValue < midValue) { // go left
00050                         high =x;

00052                         midValue = tentativeNewMidValue;
00053                     } else { // go right
00054                         low = tentativeNewMid;
00055                     }
00056                 } else {

00149     {
00150    //     Real x1 = alpha_ - std::sqrt(s_*s_-x2*x2);
00151 
00152         Real x2sq = s_*s_ - (x1-alpha_)*(x1-alpha_);

00165     bool SphereCylinderOptimizer::findByProjection(Real& y1,
00166                                                    Real& y2,
00167                                                    Real& y3) const {
00168         Real z1moved = z1_-alpha_;
00169         Real distance = std::sqrt( z1moved*z1moved + z2_*z2_);
00170         Real scale = s_/distance;
00171         Real y1moved = z1moved*scale;
00172         y1 = alpha_+ y1moved;
00173         y2 = scale*z2_;
00174         Real residual = r_*r_ - y1*y1 -y2*y2;
00175         if (residual >=0.0) {
00176             y3 = std::sqrt(residual);
00177             return true;
00178         }
00179         // we are outside the sphere
00180         if (!isIntersectionNonEmpty()) {
00181             y3=0.0;
00182             return false;
00183         }
00184 
00185        // intersection is non-empty but projection point is outside sphere
00186        // so take rightmost point
00187        y3 = 0.0;
00188        y1 = topValue_;
00189        y2 = std::sqrt(r_*r_ -y1*y1);
00190 
00191        return true;
