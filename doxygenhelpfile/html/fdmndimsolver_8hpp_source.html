00075     template <Size N> inline
00076     FdmNdimSolver<N>::FdmNdimSolver(
00077                         const FdmSolverDesc& solverDesc,
00078                         const FdmSchemeDesc& schemeDesc,
00079                         const boost::shared_ptr<FdmLinearOpComposite>& op)
00080     : solverDesc_(solverDesc),
00081       schemeDesc_(schemeDesc),
00082       op_(op),
00083       thetaCondition_(new FdmSnapshotCondition(
00084         0.99*std::min(1.0/365.0,
00085                 solverDesc.condition->stoppingTimes().empty()
00086                 ? solverDesc.maturity :
00087                   solverDesc.condition->stoppingTimes().front()))),
00088       conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
00089                                                         solverDesc.condition)),
00090       x_            (solverDesc.mesher->layout()->dim().size()),
00091       initialValues_(solverDesc.mesher->layout()->size()),
00092       extrapolation_(std::vector<bool>(N, false)) {
00093 
00094         const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
00095         const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
00096 
00097         QL_REQUIRE(layout->dim().size() == N, "solver dim " << N
00098                     << "does not fit to layout dim " << layout->size());
00099 
00100         for (Size i=0; i < N; ++i) {
00101             x_[i].reserve(layout->dim()[i]);
00102         }
00103 
00104         const FdmLinearOpIterator endIter = layout->end();
00105         for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
00106              ++iter) {
00107 
00108             initialValues_[iter.index()] = solverDesc_.calculator
00109                                 ->avgInnerValue(iter, solverDesc.maturity);
00110 
00111             const std::vector<Size>& c = iter.coordinates();
00112             for (Size i=0; i < N; ++i) {
00113                 if (!(std::accumulate(c.begin(), c.end(), 0)-c[i])) {
00114                     x_[i].push_back(mesher->location(iter, i));
00115                 }
00116             }
00117         }
00118 
00119         f_ = boost::shared_ptr<data_table>(new data_table(x_));
00120     }

00125         Array rhs(initialValues_.size());

00130                            solverDesc_.timeSteps, solverDesc_.dampingSteps);

00132         const boost::shared_ptr<FdmLinearOpLayout> layout

00135         const FdmLinearOpIterator endIter = layout->end();

00147     Real FdmNdimSolver<N>::thetaAt(const std::vector<Real>& x) const {
00148         QL_REQUIRE(conditions_->stoppingTimes().front() > 0.0,
00149                    "stopping time at zero-> can't calculate theta");
00150         calculate();
00151         const Array& rhs = thetaCondition_->getValues();
00152         const boost::shared_ptr<FdmLinearOpLayout> layout
00153                                             = solverDesc_.mesher->layout();
00154 
00155         data_table f(x_);
00156 
00157         const FdmLinearOpIterator endIter = layout->end();
00158         for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
00159              ++iter) {
00160             setValue(f, iter.coordinates(), rhs[iter.index()]);
00161         }
00162 
00163         return (MultiCubicSpline<N>(x_, f)(x)
00164                         - interpolateAt(x)) / thetaCondition_->getTime();
