00155 
00156         QL_REQUIRE(!optionTenors_.empty(), "empty option tenor vector");
00157         QL_REQUIRE(nOptionTenors_==vols_.rows(),

00163         for (Size i=1; i<nOptionTenors_; ++i)
00164             QL_REQUIRE(optionTenors_[i]>optionTenors_[i-1],
00165                        "non increasing option tenor: " << io::ordinal(i) <<

00167                        io::ordinal(i+1) << " is " << optionTenors_[i]);
00168 
00169         QL_REQUIRE(nStrikes_==vols_.columns(),
00170                    "mismatch between strikes(" << strikes_.size() <<
00171                    ") and vol columns (" << vols_.columns() << ")");
00172         for (Size j=1; j<nStrikes_; ++j)
00173             QL_REQUIRE(strikes_[j-1]<strikes_[j],
00174                        "non increasing strikes: " << io::ordinal(j) <<
00175                        " is " << io::rate(strikes_[j-1]) << ", " <<
00176                        io::ordinal(j+1) << " is " << io::rate(strikes_[j]));
