00028     Real CumulativeNormalDistribution::operator()(Real z) const {
00029         //QL_REQUIRE(!(z >= average_ && 2.0*average_-z > average_),
00030         //           "not a real number. ");
00031         z = (z - average_) / sigma_;
00032 
00033         Real result = 0.5 * ( 1.0 + errorFunction_( z*M_SQRT_2 ) );
00034         if (result<=1e-8) { //todo: investigate the threshold level
00035             // Asymptotic expansion for very negative z following (26.2.12)
00036             // on page 408 in M. Abramowitz and A. Stegun,
00037             // Pocketbook of Mathematical Functions, ISBN 3-87144818-4.
00038             Real sum=1.0, zsqr=z*z, i=1.0, g=1.0, x, y,
00039                  a=QL_MAX_REAL, lasta;
00040             do {
00041                 lasta=a;
00042                 x = (4.0*i-3.0)/zsqr;
00043                 y = x*((4.0*i-1)/zsqr);
00044                 a = g*(x-y);
00045                 sum -= a;
00046                 g *= y;
00047                 ++i;
00048                 a = std::fabs(a);
00049             } while (lasta>a && a>=std::fabs(sum*QL_EPSILON));
00050             result = -gaussian_(z)/z*sum;
00051         }
00052         return result;
00053     }

00089     Real InverseCumulativeNormal::tail_value(Real x) {
00090         if (x <= 0.0 || x >= 1.0) {
00091             // try to recover if due to numerical error
00092             if (close_enough(x, 1.0)) {
00093                 return QL_MAX_REAL; // largest value available
00094             } else if (std::fabs(x) < QL_EPSILON) {
00095                 return QL_MIN_REAL; // largest negative value available
00096             } else {
00097                 QL_FAIL("InverseCumulativeNormal(" << x
00098                         << ") undefined: must be 0 < x < 1");
00099             }
00100         }
00101 
00102         Real z;
00103         if (x < x_low_) {
00104             // Rational approximation for the lower region 0<x<u_low
00105             z = std::sqrt(-2.0*std::log(x));
00106             z = (((((c1_*z+c2_)*z+c3_)*z+c4_)*z+c5_)*z+c6_) /
00107                 ((((d1_*z+d2_)*z+d3_)*z+d4_)*z+1.0);
00108         } else {
00109             // Rational approximation for the upper region u_high<x<1
00110             z = std::sqrt(-2.0*std::log(1.0-x));
00111             z = -(((((c1_*z+c2_)*z+c3_)*z+c4_)*z+c5_)*z+c6_) /
00112                 ((((d1_*z+d2_)*z+d3_)*z+d4_)*z+1.0);
00113         }
00114 
00115         return z;
00116     }

00138     Real MoroInverseCumulativeNormal::operator()(Real x) const {
00139         QL_REQUIRE(x > 0.0 && x < 1.0,
00140                    "MoroInverseCumulativeNormal(" << x
00141                    << ") undefined: must be 0<x<1");
00142 
00143         Real result;
00144         Real temp=x-0.5;
00145 
00146         if (std::fabs(temp) < 0.42) {
00147             // Beasley and Springer, 1977
00148             result=temp*temp;
00149             result=temp*
00150                 (((a3_*result+a2_)*result+a1_)*result+a0_) /
00151                 ((((b3_*result+b2_)*result+b1_)*result+b0_)*result+1.0);
00152         } else {
00153             // improved approximation for the tail (Moro 1995)
00154             if (x<0.5)
00155                 result = x;
00156             else
00157                 result=1.0-x;
00158             result = std::log(-std::log(result));
00159             result = c0_+result*(c1_+result*(c2_+result*(c3_+result*
00160                                    (c4_+result*(c5_+result*(c6_+result*
00161                                                        (c7_+result*c8_)))))));
00162             if (x<0.5)
00163                 result=-result;
00164         }
00165 
00166         return average_ + result*sigma_;
00167     }
