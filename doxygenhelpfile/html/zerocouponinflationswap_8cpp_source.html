00028     /* Generally inflation indices are available with a lag of 1month
00029        and then observed with a lag of 2-3 months depending whether
00030        they use an interpolated fixing or not.  Here, we make the
00031        swap use the interpolation of the index to avoid incompatibilities.
00032     */
00033     ZeroCouponInflationSwap::ZeroCouponInflationSwap(
00034         Type type,
00035         Real nominal,
00036         const Date& startDate,  // start date of contract (only)
00037         const Date& maturity,   // this is pre-adjustment!
00038         const Calendar& fixCalendar,
00039         BusinessDayConvention fixConvention,
00040         const DayCounter& dayCounter,
00041         Rate fixedRate,
00042         const boost::shared_ptr<ZeroInflationIndex> &infIndex,
00043         const Period& observationLag,
00044         bool adjustInfObsDates,
00045         Calendar infCalendar,
00046         BusinessDayConvention infConvention)
00047     : Swap(2), type_(type), nominal_(nominal),
00048       startDate_(startDate), maturityDate_(maturity),
00049       fixCalendar_(fixCalendar), fixConvention_(fixConvention),
00050       fixedRate_(fixedRate), infIndex_(infIndex),
00051       observationLag_(observationLag), adjustInfObsDates_(adjustInfObsDates),
00052       infCalendar_(infCalendar), infConvention_(infConvention),
00053       dayCounter_(dayCounter) {
00054         // first check compatibility of index and swap definitions
00055         if (infIndex_->interpolated()) {
00056             Period pShift(infIndex_->frequency());
00057             QL_REQUIRE(observationLag_ - pShift > infIndex_->availabilityLag(),
00058                        "inconsistency between swap observation of index " << observationLag_ <<
00059                        " index availability " << infIndex_->availabilityLag() <<
00060                        " interpolated index period " << pShift <<
00061                        " and index availability " << infIndex_->availabilityLag() <<
00062                        " need (obsLag-index period) > availLag");
00063         } else {
00064             QL_REQUIRE(infIndex_->availabilityLag() < observationLag_,
00065                        "index tries to observe inflation fixings that do not yet exist: "
00066                        << " availability lag " << infIndex_->availabilityLag()
00067                        << " versus obs lag = " << observationLag_);
00068         }
00069 
00070         if (infCalendar_==Calendar()) infCalendar_ = fixCalendar_;
00071         if (infConvention_==BusinessDayConvention()) infConvention_ = fixConvention_;
00072 
00073         if (adjustInfObsDates_) {
00074             baseDate_ = infCalendar_.adjust(startDate - observationLag_, infConvention_);
00075             obsDate_ = infCalendar_.adjust(maturity - observationLag_, infConvention_);
00076         } else {
00077             baseDate_ = startDate - observationLag_;
00078             obsDate_ = maturity - observationLag_;
00079         }
00080 
00081         Date infPayDate = infCalendar_.adjust(maturity, infConvention_);
00082         Date fixedPayDate = fixCalendar_.adjust(maturity, fixConvention_);
00083 
00084         // At this point the index may not be able to forecast
00085         // i.e. do not want to force the existence of an inflation
00086         // term structure before allowing users to create instruments.
00087         Real T = inflationYearFraction(infIndex_->frequency(), infIndex_->interpolated(),
00088                                        dayCounter_, baseDate_, obsDate_);
00089         // N.B. the -1.0 is because swaps only exchange growth, not notionals as well
00090         Real fixedAmount = nominal * ( std::pow(1.0 + fixedRate, T) - 1.0 );
00091 
00092         legs_[0].push_back(boost::shared_ptr<CashFlow>(
00093             new SimpleCashFlow(fixedAmount, fixedPayDate)));
00094         bool growthOnly = true;
00095         legs_[1].push_back(boost::shared_ptr<CashFlow>(
00096             new IndexedCashFlow(nominal,infIndex,baseDate_,obsDate_,infPayDate,growthOnly)));
00097 
00098         for (Size j=0; j<2; ++j) {
00099             for (Leg::iterator i = legs_[j].begin(); i!= legs_[j].end(); ++i)
00100                 registerWith(*i);
00101         }
00102 
00103         switch (type_) {
00104             case Payer:
00105                 payer_[0] = +1.0;
00106                 payer_[1] = -1.0;
00107                 break;
00108             case Receiver:
00109                 payer_[0] = -1.0;
00110                 payer_[1] = +1.0;
00111                 break;
00112             default:
00113                 QL_FAIL("Unknown zero-inflation-swap type");
00114         }
00115 
00116     }

00140 
00141         boost::shared_ptr<IndexedCashFlow> icf =
00142         boost::dynamic_pointer_cast<IndexedCashFlow>(legs_[1].at(0));

00145         // +1 because the IndexedCashFlow has growthOnly=true

00147         Real T = inflationYearFraction(infIndex_->frequency(),
00148                                        infIndex_->interpolated(),
00149                                        dayCounter_, baseDate_, obsDate_);

00151         return std::pow(growth,1.0/T) - 1.0;
