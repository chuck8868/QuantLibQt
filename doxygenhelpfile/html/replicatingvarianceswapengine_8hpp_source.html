00094     inline void ReplicatingVarianceSwapEngine::computeOptionWeights(
00095                                     const std::vector<Real>& availStrikes,
00096                                     const Option::Type type,
00097                                     weights_type& optionWeights) const {
00098         if (availStrikes.empty())
00099             return;
00100 
00101         std::vector<Real> strikes = availStrikes;
00102 
00103         // add end-strike for piecewise approximation
00104         switch (type) {
00105           case Option::Call:
00106             std::sort(strikes.begin(), strikes.end());
00107             strikes.push_back(strikes.back() + dk_);
00108             break;
00109           case Option::Put:
00110             std::sort(strikes.begin(), strikes.end(), std::greater<Real>());
00111             strikes.push_back(std::max(strikes.back() - dk_, 0.0));
00112             break;
00113           default:
00114             QL_FAIL("invalid option type");
00115         }
00116 
00117         // remove duplicate strikes
00118         std::vector<Real>::iterator last =
00119             std::unique(strikes.begin(), strikes.end());
00120         strikes.erase(last, strikes.end());
00121 
00122         // compute weights
00123         Real f = strikes.front();
00124         Real slope, prevSlope = 0.0;
00125 
00126 
00127 
00128 
00129         for (std::vector<Real>::const_iterator k=strikes.begin();
00130              // added end-strike discarded
00131              k<strikes.end()-1;
00132              k++) {
00133             slope = std::fabs((computeLogPayoff(*(k+1), f) -
00134                                computeLogPayoff(*k, f))/
00135                               (*(k+1) - *k));
00136             boost::shared_ptr<StrikedTypePayoff> payoff(
00137                                             new PlainVanillaPayoff(type, *k));
00138             if ( k == strikes.begin() )
00139                 optionWeights.push_back(std::make_pair(payoff,slope));
00140             else
00141                 optionWeights.push_back(
00142                                    std::make_pair(payoff, slope - prevSlope));
00143             prevSlope = slope;
00144         }
00145     }

00148     inline Real ReplicatingVarianceSwapEngine::computeLogPayoff(
00149                          const Real strike,
00150                          const Real callPutStrikeBoundary) const {
00151         Real f = callPutStrikeBoundary;
00152         return (2.0/residualTime()) * (((strike - f)/f) - std::log(strike/f));

00157     Real ReplicatingVarianceSwapEngine::computeReplicatingPortfolio(
00158                                     const weights_type& optionWeights) const {
00159 
00160         boost::shared_ptr<Exercise> exercise(
00161                                new EuropeanExercise(arguments_.maturityDate));
00162         boost::shared_ptr<PricingEngine> optionEngine(
00163                                         new AnalyticEuropeanEngine(process_));
00164         Real optionsValue = 0.0;
00165 
00166         for (weights_type::const_iterator i = optionWeights.begin();
00167              i < optionWeights.end(); ++i) {
00168             boost::shared_ptr<StrikedTypePayoff> payoff = i->first;
00169             EuropeanOption option(payoff, exercise);
00170             option.setPricingEngine(optionEngine);
00171             Real weight = i->second;
00172             optionsValue += option.NPV() * weight;
00173         }
00174 
00175         Real f = optionWeights.front().first->strike();
00176         return 2.0 * riskFreeRate() -
00177             2.0/residualTime() *
00178             (((underlying()/riskFreeDiscount() - f)/f) +
00179              std::log(f/underlying())) +
00180             optionsValue/riskFreeDiscount();
00181     }

00185     inline void ReplicatingVarianceSwapEngine::calculate() const {
00186         weights_type optionWeigths;
00187         computeOptionWeights(callStrikes_, Option::Call, optionWeigths);
00188         computeOptionWeights(putStrikes_, Option::Put, optionWeigths);
00189 
00190         results_.variance = computeReplicatingPortfolio(optionWeigths);
00191 
00192         DiscountFactor riskFreeDiscount =
00193             process_->riskFreeRate()->discount(arguments_.maturityDate);
00194         Real multiplier;
00195         switch (arguments_.position) {
00196           case Position::Long:
00197             multiplier = 1.0;
00198             break;
00199           case Position::Short:
00200             multiplier = -1.0;
00201             break;
00202           default:
00203             QL_FAIL("Unknown position");
00204         }
00205         results_.value = multiplier * riskFreeDiscount * arguments_.notional *
00206             (results_.variance - arguments_.strike);
00207 
00208         results_.additionalResults["optionWeights"] = optionWeigths;
00209     }
