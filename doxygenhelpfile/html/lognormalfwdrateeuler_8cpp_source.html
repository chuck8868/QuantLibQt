00029     LogNormalFwdRateEuler::LogNormalFwdRateEuler(
00030                            const boost::shared_ptr<MarketModel>& marketModel,
00031                            const BrownianGeneratorFactory& factory,
00032                            const std::vector<Size>& numeraires,
00033                            Size initialStep)
00034     : marketModel_(marketModel),
00035       numeraires_(numeraires),
00036       initialStep_(initialStep),
00037       numberOfRates_(marketModel->numberOfRates()),
00038       numberOfFactors_(marketModel_->numberOfFactors()),
00039       curveState_(marketModel->evolution().rateTimes()),
00040       forwards_(marketModel->initialRates()),
00041       displacements_(marketModel->displacements()),
00042       logForwards_(numberOfRates_), initialLogForwards_(numberOfRates_),
00043       drifts1_(numberOfRates_), initialDrifts_(numberOfRates_),
00044       brownians_(numberOfFactors_), correlatedBrownians_(numberOfRates_),
00045       alive_(marketModel->evolution().firstAliveRate())
00046     {
00047         checkCompatibility(marketModel->evolution(), numeraires);
00048 
00049         Size steps = marketModel->evolution().numberOfSteps();
00050 
00051         generator_ = factory.create(numberOfFactors_, steps-initialStep_);
00052 
00053         currentStep_ = initialStep_;
00054 
00055         calculators_.reserve(steps);
00056         fixedDrifts_.reserve(steps);
00057         for (Size j=0; j<steps; ++j) {
00058             const Matrix& A = marketModel_->pseudoRoot(j);
00059             calculators_.push_back(LMMDriftCalculator(A,
00060                                                    displacements_,
00061                                                    marketModel->evolution().rateTaus(),
00062                                                    numeraires[j],
00063                                                    alive_[j]));
00064             std::vector<Real> fixed(numberOfRates_);
00065             for (Size k=0; k<numberOfRates_; ++k) {
00066                 Real variance =
00067                     std::inner_product(A.row_begin(k), A.row_end(k),
00068                                        A.row_begin(k), 0.0);
00069                 fixed[k] = -0.5*variance;
00070             }
00071             fixedDrifts_.push_back(fixed);
00072         }
00073 
00074         setForwards(marketModel_->initialRates());
00075     }

00102     Real LogNormalFwdRateEuler::advanceStep()
00103     {
00104         // we're going from T1 to T2
00105 
00106         // a) compute drifts D1 at T1;
00107         if (currentStep_ > initialStep_) {
00108             calculators_[currentStep_].compute(forwards_, drifts1_);
00109         } else {
00110             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00111                       drifts1_.begin());
00112         }
00113 
00114         // b) evolve forwards up to T2 using D1;
00115         Real weight = generator_->nextStep(brownians_);
00116         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00117         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00118 
00119         Size alive = alive_[currentStep_];
00120         for (Size i=alive; i<numberOfRates_; i++) {
00121             logForwards_[i] += drifts1_[i] + fixedDrift[i];
00122             logForwards_[i] +=
00123                 std::inner_product(A.row_begin(i), A.row_end(i),
00124                                    brownians_.begin(), 0.0);
00125             forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
00126         }
00127 
00128         // same as PC evolver with two steps dropped
00129 
00130         // c) update curve state
00131         curveState_.setOnForwardRates(forwards_);
00132 
00133         ++currentStep_;
00134 
00135         return weight;
00136     }
