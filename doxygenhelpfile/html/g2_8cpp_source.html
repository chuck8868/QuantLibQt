00057     Real G2::sigmaP(Time t, Time s) const {
00058         Real temp = 1.0 - std::exp(-(a()+b())*t);
00059         Real temp1 = 1.0 - std::exp(-a()*(s-t));
00060         Real temp2 = 1.0 - std::exp(-b()*(s-t));
00061         Real a3 = a()*a()*a();
00062         Real b3 = b()*b()*b();
00063         Real sigma2 = sigma()*sigma();
00064         Real eta2 = eta()*eta();
00065         Real value =
00066             0.5*sigma2*temp1*temp1*(1.0 - std::exp(-2.0*a()*t))/a3 +
00067             0.5*eta2*temp2*temp2*(1.0 - std::exp(-2.0*b()*t))/b3 +
00068             2.0*rho()*sigma()*eta()/(a()*b()*(a()+b()))*
00069             temp1*temp2*temp;
00070         return std::sqrt(value);
00071     }

00087     Real G2::V(Time t) const {
00088         Real expat = std::exp(-a()*t);
00089         Real expbt = std::exp(-b()*t);
00090         Real cx = sigma()/a();
00091         Real cy = eta()/b();
00092         Real valuex = cx*cx*(t + (2.0*expat-0.5*expat*expat-1.5)/a());
00093         Real valuey = cy*cy*(t + (2.0*expbt-0.5*expbt*expbt-1.5)/b());
00094         Real value = 2.0*rho()*cx*cy* (t + (expat - 1.0)/a()
00095                                          + (expbt - 1.0)/b()
00096                                          - (expat*expbt-1.0)/(a()+b()));
00097         return valuex + valuey + value;
00098     }
