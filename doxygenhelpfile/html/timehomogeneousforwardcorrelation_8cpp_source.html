00050     std::vector<Matrix> TimeHomogeneousForwardCorrelation::evolvedMatrices(
00051                                     const Matrix& fwdCorrelation) {
00052         Size numberOfRates = fwdCorrelation.rows();
00053         std::vector<Matrix> correlations(numberOfRates, Matrix(numberOfRates,
00054                                                                numberOfRates,
00055                                                                0.0));
00056         for (Size k=0; k<correlations.size(); ++k) {
00057             // proper diagonal values
00058             for (Size i=k; i<numberOfRates; ++i)
00059                 correlations[k][i][i] = 1.0;
00060             // copy only time homogeneous values
00061             for (Size i=k; i<numberOfRates; ++i) {
00062                 for (Size j=k; j<i; ++j) {
00063                     correlations[k][i][j] = correlations[k][j][i] =
00064                         fwdCorrelation[i-k][j-k];
00065                 }
00066             }
00067         }
00068         return correlations;
00069     }
