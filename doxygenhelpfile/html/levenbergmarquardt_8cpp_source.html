00113     void LevenbergMarquardt::fcn(int, int n, double* x, double* fvec, int*) {
00114         Array xt(n);
00115         std::copy(x, x+n, xt.begin());
00116         // constraint handling needs some improvement in the future:

00122             std::copy(initCostValues_.begin(), initCostValues_.end(), fvec);
