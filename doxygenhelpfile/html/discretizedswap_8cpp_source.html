00085     void DiscretizedSwap::preAdjustValuesImpl() {
00086         // floating payments
00087         for (Size i=0; i<floatingResetTimes_.size(); i++) {
00088             Time t = floatingResetTimes_[i];
00089             if (t >= 0.0 && isOnTime(t)) {
00090                 DiscretizedDiscountBond bond;
00091                 bond.initialize(method(), floatingPayTimes_[i]);
00092                 bond.rollback(time_);
00093 
00094                 Real nominal = arguments_.nominal;
00095                 Time T = arguments_.floatingAccrualTimes[i];
00096                 Spread spread = arguments_.floatingSpreads[i];
00097                 Real accruedSpread = nominal*T*spread;
00098                 for (Size j=0; j<values_.size(); j++) {
00099                     Real coupon = nominal * (1.0 - bond.values()[j])
00100                                 + accruedSpread * bond.values()[j];
00101                     if (arguments_.type == VanillaSwap::Payer)
00102                         values_[j] += coupon;
00103                     else
00104                         values_[j] -= coupon;
00105                 }
00106             }
00107         }
00108         // fixed payments
00109         for (Size i=0; i<fixedResetTimes_.size(); i++) {
00110             Time t = fixedResetTimes_[i];
00111             if (t >= 0.0 && isOnTime(t)) {
00112                 DiscretizedDiscountBond bond;
00113                 bond.initialize(method(), fixedPayTimes_[i]);
00114                 bond.rollback(time_);
00115 
00116                 Real fixedCoupon = arguments_.fixedCoupons[i];
00117                 for (Size j=0; j<values_.size(); j++) {
00118                     Real coupon = fixedCoupon*bond.values()[j];
00119                     if (arguments_.type == VanillaSwap::Payer)
00120                         values_[j] -= coupon;
00121                     else
00122                         values_[j] += coupon;
00123                 }
00124             }
00125         }

00128     void DiscretizedSwap::postAdjustValuesImpl() {
00129         // fixed coupons whose reset time is in the past won't be managed
00130         // in preAdjustValues()
00131         for (Size i=0; i<fixedPayTimes_.size(); i++) {
00132             Time t = fixedPayTimes_[i];
00133             Time reset = fixedResetTimes_[i];
00134             if (t >= 0.0 && isOnTime(t) && reset < 0.0) {
00135                 Real fixedCoupon = arguments_.fixedCoupons[i];
00136                 if (arguments_.type==VanillaSwap::Payer)
00137                     values_ -= fixedCoupon;
00138                 else
00139                     values_ += fixedCoupon;
00140             }
00141         }
00142         // the same applies to floating payments whose rate is already fixed
00143         for (Size i=0; i<floatingPayTimes_.size(); i++) {
00144             Time t = floatingPayTimes_[i];
00145             Time reset = floatingResetTimes_[i];
00146             if (t >= 0.0 && isOnTime(t) && reset < 0.0) {
00147                 Real currentFloatingCoupon = arguments_.floatingCoupons[i];
00148                 QL_REQUIRE(currentFloatingCoupon != Null<Real>(),
00149                            "current floating coupon not given");
00150                 if (arguments_.type == VanillaSwap::Payer)
00151                     values_ += currentFloatingCoupon;
00152                 else
00153                     values_ -= currentFloatingCoupon;
00154             }
00155         }
