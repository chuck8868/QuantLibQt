00045     void InterpolatingCPICapFloorEngine::calculate() const
00046     {
00047         Real npv = 0.0;
00048 
00049         // what is the difference between the observationLag of the surface
00050         // and the observationLag of the cap/floor?
00051         // \TODO next line will fail if units are different
00052         Period lagDiff = arguments_.observationLag - priceSurf_->observationLag();
00053         // next line will fail if units are different if Period() is not well written
00054         QL_REQUIRE(lagDiff >= Period(0, Months), "InterpolatingCPICapFloorEngine: "
00055                    "lag difference must be non-negative: " << lagDiff);
00056 
00057         // we now need an effective maturity to use in the price surface because this uses
00058         // maturity of calibration instruments as its time axis, N.B. this must also
00059         // use the roll because the surface does
00060         Date effectiveMaturity = arguments_.payDate - lagDiff;
00061 
00062 
00063         // what interpolation do we use? Index / flat / linear
00064         if (arguments_.observationInterpolation == CPI::AsIndex) {
00065             // same as index means we can just use the price surface
00066             // since this uses the index
00067             if (arguments_.type == Option::Call) {
00068                 npv = priceSurf_->capPrice(effectiveMaturity, arguments_.strike);
00069             } else {
00070                 npv = priceSurf_->floorPrice(effectiveMaturity, arguments_.strike);
00071             }
00072 
00073 
00074         } else {
00075             std::pair<Date,Date> dd = inflationPeriod(effectiveMaturity, arguments_.infIndex->frequency());
00076             Real priceStart = 0.0;
00077 
00078             if (arguments_.type == Option::Call) {
00079                 priceStart = priceSurf_->capPrice(dd.first, arguments_.strike);
00080             } else {
00081                 priceStart = priceSurf_->floorPrice(dd.first, arguments_.strike);
00082             }
00083 
00084             // if we use a flat index vs the interpolated one ...
00085             if (arguments_.observationInterpolation == CPI::Flat) {
00086                 // then use the price for the first day in the period because the value cannot change after then
00087                 npv = priceStart;
00088 
00089             } else {
00090                 // linear interpolation will be very close
00091                 Real priceEnd = 0.0;
00092                 if (arguments_.type == Option::Call) {
00093                     priceEnd = priceSurf_->capPrice((dd.second+Period(1,Days)), arguments_.strike);
00094                 } else {
00095                     priceEnd = priceSurf_->floorPrice((dd.second+Period(1,Days)), arguments_.strike);
00096                 }
00097 
00098                 npv = priceStart + (priceEnd - priceStart) * (effectiveMaturity - dd.first)
00099                 / ( (dd.second+Period(1,Days)) - dd.first); // can't get to next period'
00100             }
00101 
00102         }
00103         results_.value = npv;
00104     }
