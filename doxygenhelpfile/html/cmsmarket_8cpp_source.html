00113     void CmsMarket::performCalculations() const {
00114         for (Size j=0; j<nSwapIndexes_; ++j) {
00115           Real mktPrevPart = 0.0, mdlPrevPart = 0.0;
00116           for (Size i=0; i<nExercise_; ++i) {
00117 
00118             // **** market
00119 
00120             mktBidSpreads_[i][j] = bidAskSpreads_[i][j*2]->value();
00121             mktAskSpreads_[i][j] = bidAskSpreads_[i][j*2+1]->value();
00122             mktSpreads_[i][j] = (mktBidSpreads_[i][j]+mktAskSpreads_[i][j])/2;
00123 
00124             const Leg& spotFloatLeg = spotSwaps_[i][j]->leg(1);
00125             spotFloatLegNPV_[i][j] = CashFlows::npv(spotFloatLeg,
00126                                                     **discTS_,
00127                                                     false, discTS_->referenceDate());
00128             spotFloatLegBPS_[i][j] = CashFlows::bps(spotFloatLeg,
00129                                                     **discTS_,
00130                                                     false, discTS_->referenceDate());
00131 
00132             // imply the spot CMS leg NPV from the spot ibor floating leg NPV
00133             mktSpotCmsLegNPV_[i][j] = -(spotFloatLegNPV_[i][j] +
00134                                 spotFloatLegBPS_[i][j]*mktSpreads_[i][j]/1e-4);
00135             // fwd CMS legs can be computed as differences between spot legs
00136             mktFwdCmsLegNPV_[i][j] = mktSpotCmsLegNPV_[i][j] - mktPrevPart;
00137             mktPrevPart = mktSpotCmsLegNPV_[i][j];
00138 
00139             // **** model
00140 
00141             // calculate the forward swap (the time consuming part)
00142             mdlFwdCmsLegNPV_[i][j] = fwdSwaps_[i][j]->legNPV(0);
00143             errFwdCmsLegNPV_[i][j] = mdlFwdCmsLegNPV_[i][j] -
00144                                                 mktFwdCmsLegNPV_[i][j];
00145 
00146             // spot CMS legs can be computed as incremental sum of forward legs
00147             mdlSpotCmsLegNPV_[i][j] = mdlPrevPart + mdlFwdCmsLegNPV_[i][j];
00148             mdlPrevPart = mdlSpotCmsLegNPV_[i][j];
00149             errSpotCmsLegNPV_[i][j] = mdlSpotCmsLegNPV_[i][j] -
00150                                                 mktSpotCmsLegNPV_[i][j];
00151 
00152             // equilibriums spread over ibor leg
00153             Real npv = spotFloatLegNPV_[i][j] + mdlSpotCmsLegNPV_[i][j];
00154             mdlSpreads_[i][j] = - npv/spotFloatLegBPS_[i][j]*1e-4;
00155             errSpreads_[i][j] = mdlSpreads_[i][j] - mktSpreads_[i][j];
00156           }
00157         }
00158     }

00205     Real CmsMarket::weightedMean(const Matrix& var,
00206                                  const Matrix& w) {
00207         Real mean = 0.0;
00208         for (Size i=0; i<nExercise_; ++i) {
00209             for (Size j=0; j<nSwapIndexes_; ++j) {
00210                 mean += w[i][j]*var[i][j]*var[i][j];
00211             }
00212         }
00213         mean = std::sqrt(mean/(nExercise_*nSwapIndexes_));
00214         return mean;
00215     }
