00117             if (x < x_low_ || x_high_ < x) {
00118                 z = tail_value(x);
00119             } else {
00120                 z = x - 0.5;
00121                 Real r = z*z;
00122                 z = (((((a1_*r+a2_)*r+a3_)*r+a4_)*r+a5_)*r+a6_)*z /
00123                     (((((b1_*r+b2_)*r+b3_)*r+b4_)*r+b5_)*r+1.0);
00124             }
00125 
00126             // The relative error of the approximation has absolute value less
00127             // than 1.15e-9.  One iteration of Halley's rational method (third

00134             z -= r/(1+0.5*z*r);
00135             #endif

00137             return z;

00244     inline Real NormalDistribution::operator()(Real x) const {
00245         Real deltax = x-average_;
00246         Real exponent = -(deltax*deltax)/denominator_;
00247         // debian alpha had some strange problem in the very-low range
00248         return exponent <= -690.0 ? 0.0 :  // exp(x) < 1.0e-300 anyway
00249             normalizationFactor_*std::exp(exponent);
