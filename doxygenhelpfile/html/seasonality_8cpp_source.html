00064     bool MultiplicativePriceSeasonality::isConsistent(const InflationTermStructure& iTS) const
00065     {
00066         // If multi-year is the specification consistent with the term structure start date?
00067         // We do NOT test daily seasonality because this will, in general, never be consistent
00068         // given weekends, holidays, leap years, etc.
00069         if(this->frequency() == Daily) return true;
00070         if(Size(this->frequency()) == seasonalityFactors().size()) return true;
00071 
00072         // how many years do you need to test?
00073         Size nTest = seasonalityFactors().size() / this->frequency();
00074         // ... relative to the start of the inflation curve
00075         std::pair<Date,Date> lim = inflationPeriod(iTS.baseDate(), iTS.frequency());
00076         Date curveBaseDate = lim.second;
00077         Real factorBase = this->seasonalityFactor(curveBaseDate);
00078 
00079         Real eps = 0.00001;
00080         for (Size i = 1; i < nTest; i++) {
00081             Real factorAt = this->seasonalityFactor(curveBaseDate+Period(i,Years));
00082             QL_REQUIRE(fabs(factorAt-factorBase)<eps,"seasonality is inconsistent with inflation term structure, factors "
00083                        << factorBase << " and later factor " << factorAt << ", " << i << " years later from inflation curve "
00084                        <<" with base date at " << curveBaseDate);
00085         }
00086 
00087         return true;
00088     }

00140     Real MultiplicativePriceSeasonality::seasonalityFactor(const Date &to) const {
00141 
00142         Date from = seasonalityBaseDate();
00143         Frequency factorFrequency = frequency();
00144         Size nFactors = seasonalityFactors().size();
00145         Period factorPeriod(factorFrequency);
00146         Size which = 0;
00147         if (from==to) {
00148             which = 0;
00149         } else {
00150             // days, weeks, months, years are the only time unit possibilities
00151             Integer diffDays = std::abs(to - from);  // in days
00152             Integer dir = 1;
00153             if(from > to)dir = -1;
00154             Integer diff;
00155             if (factorPeriod.units() == Days) {
00156                 diff = dir*diffDays;
00157             } else if (factorPeriod.units() == Weeks) {
00158                 diff = dir * (diffDays / 7);
00159             } else if (factorPeriod.units() == Months) {
00160                 std::pair<Date,Date> lim = inflationPeriod(to, factorFrequency);
00161                 diff = diffDays / (31*factorPeriod.length());
00162                 Date go = from + dir*diff*factorPeriod;
00163                 while ( !(lim.first <= go && go <= lim.second) ) {
00164                     go += dir*factorPeriod;
00165                     diff++;
00166                 }
00167                 diff=dir*diff;
00168             } else if (factorPeriod.units() == Years) {
00169                 QL_FAIL("seasonality period time unit is not allowed to be : " << factorPeriod.units());
00170             } else {
00171                 QL_FAIL("Unknown time unit: " << factorPeriod.units());
00172             }
00173             // now adjust to the available number of factors, direction dependent
00174 
00175             if (dir==1) {
00176                 which = diff % nFactors;
00177             } else {
00178                 which = (nFactors - (-diff % nFactors)) % nFactors;
00179             }
00180         }
00181 
00182         return seasonalityFactors()[which];
00183     }

00186     Rate MultiplicativePriceSeasonality::seasonalityCorrection(Rate rate,
00187                                                                const Date& atDate,
00188                                                                const DayCounter& dc,
00189                                                                const Date& curveBaseDate,
00190                                                                const bool isZeroRate) const {
00191         // need _two_ corrections in order to get: seasonality = factor[atDate-seasonalityBase] / factor[reference-seasonalityBase]
00192         // i.e. for ZERO inflation rates you have the true fixing at the curve base so this factor must be normalized to one
00193         //      for YoY inflation rates your reference point is the year before
00194 
00195         Real factorAt = this->seasonalityFactor(atDate);
00196 
00197         //Getting seasonality correction for either ZC or YoY
00198         Rate f;
00199         if (isZeroRate) {
00200             Rate factorBase = this->seasonalityFactor(curveBaseDate);
00201             Real seasonalityAt = factorAt / factorBase;
00202             Time timeFromCurveBase = dc.yearFraction(curveBaseDate, atDate);
00203             f = pow(seasonalityAt, 1/timeFromCurveBase);
00204         }
00205         else {
00206             Rate factor1Ybefore = this->seasonalityFactor(atDate - Period(1,Years));
00207             f = factorAt / factor1Ybefore;
00208         }
00209 
00210         return (rate + 1)*f - 1;
00211     }
