00074     UnitOfMeasureConversion UnitOfMeasureConversionManager::lookup(
00075                                    const CommodityType& commodityType,
00076                                    const UnitOfMeasure& source,
00077                                    const UnitOfMeasure& target,
00078                                    UnitOfMeasureConversion::Type type) const {
00079         if (type == UnitOfMeasureConversion::Direct) {
00080             return directLookup(commodityType,source,target);
00081         } else if (!source.triangulationUnitOfMeasure().empty()) {
00082             const UnitOfMeasure& link = source.triangulationUnitOfMeasure();
00083             if (link == target)
00084                 return directLookup(commodityType,source,link);
00085             else
00086                 return UnitOfMeasureConversion::chain(
00087                                       directLookup(commodityType,source,link),
00088                                       lookup(commodityType,link,target));
00089         } else if (!target.triangulationUnitOfMeasure().empty()) {
00090             const UnitOfMeasure& link = target.triangulationUnitOfMeasure();
00091             if (source == link)
00092                 return directLookup(commodityType,link,target);
00093             else
00094                 return UnitOfMeasureConversion::chain(
00095                                      lookup(commodityType,source,link),
00096                                      directLookup(commodityType,link,target));
00097         } else {
00098             return smartLookup(commodityType,source,target);
00099         }

00144              i != data_.end(); ++i) {
00145             if (matches(*i, commodityType, source, target)) {
00146                 return *i;

00167     UnitOfMeasureConversion UnitOfMeasureConversionManager::smartLookup(
00168                               const CommodityType& commodityType,
00169                               const UnitOfMeasure& source,
00170                               const UnitOfMeasure& target,
00171                               list<string> forbidden) const {
00172 
00173         try {
00174             return directLookup(commodityType,source,target);
00175         } catch (Error&) {
00176             ; // no direct conversion available; turn to smart lookup.
00177         }
00178 
00179         // The source unit is forbidden to subsequent lookups in order
00180         // to avoid cycles.
00181         forbidden.push_back(source.code());
00182 
00183         for (list<UnitOfMeasureConversion>::const_iterator i = data_.begin();
00184              i != data_.end(); ++i) {
00185             // we look for conversion data which involve our source unit...
00186             if (matches(*i, commodityType, source)) {
00187                 const UnitOfMeasure& other =
00188                     source == i->source() ? i->target() : i->source();
00189                 if (find(forbidden.begin(),forbidden.end(),
00190                          other.code()) == forbidden.end()) {
00191                     // if we can get to the target from here...
00192                     try {
00193                         UnitOfMeasureConversion tail =
00194                             smartLookup(commodityType,other,target);
00195                         // ..we're done.
00196                         return UnitOfMeasureConversion::chain(*i,tail);
00197                     } catch (Error&) {
00198                         // otherwise, we just discard this conversion.
00199                         ;
00200                     }
00201                 }
00202             }
00203         }
00204 
00205         // if the loop completed, we have no way to return the
00206         // requested conversion.
00207         QL_FAIL("no conversion available for "
00208                 << commodityType.code() << " from "
00209                 << source.code() << " to " << target.code());
00210     }
