00103     inline Rate DriftTermStructure::zeroYieldImpl(Time t) const {
00104         // warning: here it is assumed that
00105         //          a) all TS have the same daycount.
00106         //          b) all TS have the same referenceDate
00107         //          It should be QL_REQUIREd
00108         return riskFreeTS_->zeroRate(t, Continuous, NoFrequency, true)
00109              - dividendTS_->zeroRate(t, Continuous, NoFrequency, true)
