00079             for(Size i=0; i<v.size(); ++i) {

00086         // build order N+1 tuples from order N tuples
00087         VV next_order_tuples(const VV& v) {
00088             const Size order = std::accumulate(v[0].begin(), v[0].end(), 0u);
00089             const Size dim = v[0].size();
00090 
00091             check_tuples(v, dim, order);
00092 
00093             // the set of unique tuples
00094             std::set<std::vector<Size> > tuples;
00095             std::vector<Size> x;
00096             for(Size i=0; i<dim; ++i) {
00097                 // increase i-th value in every tuple by 1
00098                 for(Size j=0; j<v.size(); ++j) {
00099                     x = v[j];
00100                     x[i] += 1;
00101                     tuples.insert(x);
00102                 }
00103             }
00104 
00105             VV ret(tuples.begin(), tuples.end());

00112     VF_R LsmBasisSystem::pathBasisSystem(Size order, PolynomType polyType) {
00113         VF_R ret(order+1);
00114         for (Size i=0; i<=order; ++i) {
00115             switch (polyType) {
00116               case Monomial:
00117                 ret[i] = MonomialFct(i);
00118                 break;
00119               case Laguerre:
00120                 ret[i] = boost::bind(ptr_w, GaussLaguerrePolynomial(), i, _1);
00121                 break;
00122               case Hermite:
00123                 ret[i] = boost::bind(ptr_w, GaussHermitePolynomial(), i, _1);
00124                 break;
00125               case Hyperbolic:
00126                 ret[i] = boost::bind(ptr_w, GaussHyperbolicPolynomial(), i, _1);
00127                 break;
00128               case Legendre:
00129                 ret[i] = boost::bind(ptr_w, GaussLegendrePolynomial(), i, _1);
00130                 break;
00131               case Chebyshev:
00132                 ret[i] = boost::bind(ptr_w, GaussChebyshevPolynomial(), i, _1);
00133                 break;
00134               case Chebyshev2nd:
00135                 ret[i] = boost::bind(ptr_w,GaussChebyshev2ndPolynomial(),i, _1);

00137               default:
00138                 QL_FAIL("unknown regression type");
00139             }
00140         }

00144     VF_A LsmBasisSystem::multiPathBasisSystem(Size dim, Size order,
00145                                               PolynomType polyType) {
00146         QL_REQUIRE(dim>0, "zero dimension");
00147         // get single factor basis
00148         VF_R pathBasis = pathBasisSystem(order, polyType);
00149         VF_A ret;
00150         // 0-th order term
00151         VF_R term(dim, pathBasis[0]);
00152         ret.push_back(MultiDimFct(term));
00153         // start with all 0 tuple
00154         VV tuples(1, std::vector<Size>(dim));
00155         // add multi-factor terms
00156         for(Size i=1; i<=order; ++i) {
00157             tuples = next_order_tuples(tuples);
00158             // now we have all tuples of order i
00159             // for each tuple add the corresponding term
00160             for(Size j=0; j<tuples.size(); ++j) {
00161                 for(Size k=0; k<dim; ++k)
00162                     term[k] = pathBasis[tuples[j][k]];
00163                 ret.push_back(MultiDimFct(term));
00164             }
00165         }
00166         return ret;
