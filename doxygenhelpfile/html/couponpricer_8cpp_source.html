00040     void BlackIborCouponPricer::initialize(const FloatingRateCoupon& coupon) {
00041 
00042         gearing_ = coupon.gearing();
00043         spread_ = coupon.spread();
00044         accrualPeriod_ = coupon.accrualPeriod();
00045 
00046         index_ = dynamic_pointer_cast<IborIndex>(coupon.index());
00047         if (!index_) {
00048             // check if the coupon was right
00049             const IborCoupon* c = dynamic_cast<const IborCoupon*>(&coupon);
00050             QL_REQUIRE(c, "IborCoupon required");
00051             // coupon was right, index is not
00052             QL_FAIL("IborIndex required");
00053         }
00054         Handle<YieldTermStructure> rateCurve =
00055                                             index_->forwardingTermStructure();
00056 
00057         Date paymentDate = coupon.date();
00058         if (paymentDate > rateCurve->referenceDate())
00059             discount_ = rateCurve->discount(paymentDate);
00060         else
00061             discount_ = 1.0;
00062 
00063         spreadLegValue_ = spread_ * accrualPeriod_ * discount_;
00064 
00065         coupon_ = &coupon;
00066     }

00068     Real BlackIborCouponPricer::optionletPrice(Option::Type optionType,
00069                                                Real effStrike) const {
00070         Date fixingDate = coupon_->fixingDate();
00071         if (fixingDate <= Settings::instance().evaluationDate()) {
00072             // the amount is determined
00073             Real a, b;
00074             if (optionType==Option::Call) {
00075                 a = coupon_->indexFixing();
00076                 b = effStrike;
00077             } else {
00078                 a = effStrike;
00079                 b = coupon_->indexFixing();
00080             }
00081             return std::max(a - b, 0.0)* accrualPeriod_*discount_;
00082         } else {
00083             // not yet determined, use Black model
00084             QL_REQUIRE(!capletVolatility().empty(),
00085                        "missing optionlet volatility");
00086             Real stdDev =
00087                 std::sqrt(capletVolatility()->blackVariance(fixingDate,
00088                                                             effStrike));
00089             Rate fixing = blackFormula(optionType,
00090                                        effStrike,
00091                                        adjustedFixing(),
00092                                        stdDev);
00093             return fixing * accrualPeriod_ * discount_;
00094         }
00095     }

00097     Rate BlackIborCouponPricer::adjustedFixing(Rate fixing) const {
00098 
00099         if (fixing == Null<Rate>())
00100             fixing = coupon_->indexFixing();
00101 
00102         if (!coupon_->isInArrears())
00103             return fixing;
00104 
00105         QL_REQUIRE(!capletVolatility().empty(),
00106                    "missing optionlet volatility");
00107         Date d1 = coupon_->fixingDate();
00108         Date referenceDate = capletVolatility()->referenceDate();
00109         if (d1 <= referenceDate)
00110             return fixing;
00111 
00112         // see Hull, 4th ed., page 550
00113         Date d2 = index_->maturityDate(d1);
00114         Time tau = index_->dayCounter().yearFraction(d1, d2);
00115         Real variance = capletVolatility()->blackVariance(d1, fixing);
00116         Spread adjustement = fixing*fixing*variance*tau/(1.0+fixing*tau);
00117         return fixing + adjustement;
00118     }
