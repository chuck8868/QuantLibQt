00084     void NthToDefault::performCalculations() const {
00085         QL_REQUIRE(!yieldTS_.empty(), "no yield term structure set");
00086 
00087         for (Size i = 0; i < probabilities_.size(); i++) {
00088             QL_REQUIRE(!probabilities_[i].empty(),
00089                        "no default term structure set");
00090             if (i > 0) {
00091                 QL_REQUIRE(probabilities_[i]->referenceDate() ==
00092                            probabilities_[i-1]->referenceDate(),
00093                            "basket reference dates do not match");
00094             }
00095         }
00096 
00097         errorEstimate_ = Null<Real>();
00098 
00099         NPV_ = 0.0;
00100         Real premiumValue = 0.0;
00101         Real accrualValue = 0.0;
00102         Real claimValue = 0.0;
00103         Date d, d0;
00104         for (Size i = 0; i < premiumLeg_.size(); i++) {
00105             boost::shared_ptr<FixedRateCoupon> coupon =
00106                 boost::dynamic_pointer_cast<FixedRateCoupon>(premiumLeg_[i]);
00107             Date d = premiumLeg_[i]->date();
00108             if (d > yieldTS_->referenceDate()) {
00109                 premiumValue += premiumLeg_[i]->amount()
00110                     * yieldTS_->discount(d)
00111                     * (1.0 - defaultProbability(d));
00112 
00113                 if (coupon->accrualStartDate() >= yieldTS_->referenceDate())
00114                     d = coupon->accrualStartDate();
00115                 else
00116                     d = yieldTS_->referenceDate();
00117 
00118                 // do steps of specified size
00119                 d0 = d;
00120                 Period stepSize = integrationStepSize_;
00121                 Probability defProb0 = defaultProbability(d0);
00122                 do {
00123                     DiscountFactor disc = yieldTS_->discount(d);
00124                     Probability defProb1 = defaultProbability(d);
00125                     Probability dcfdd = defProb1 - defProb0;
00126 
00127                     defProb0 = defProb1;
00128 
00129                     if (settlePremiumAccrual_)
00130                         accrualValue += coupon->accruedAmount(d)*disc*dcfdd;
00131 
00132                     claimValue -= claim_->amount(d, nominal_, recoveryRate_)
00133                                   * disc * dcfdd;
00134 
00135                     d0 = d;
00136                     d = d0 + stepSize;
00137                     // reduce step size ?
00138                     if (stepSize != 1*Days && d > coupon->accrualEndDate()) {
00139                         stepSize = 1*Days;
00140                         //d = NullCalendar().advance(d0,1,stepSize,Unadjusted);
00141                         d = d0 + stepSize;
00142                     }
00143                 }
00144                 while (d <= coupon->accrualEndDate());
00145             }
00146         }
00147 
00148         if (side_ == Protection::Buyer) {
00149             premiumValue *= -1;
00150             accrualValue *= -1;
00151             claimValue *= -1;
00152         }
00153 
00154         NPV_ = premiumValue + accrualValue + claimValue;
00155 
00156         fairPremium_ = -premiumRate_*claimValue/(premiumValue+accrualValue);
00157     }

00159     Probability NthToDefault::defaultProbability(const Date& d) const {
00160         if (d <= probabilities_.front()->referenceDate())
00161             return 0.0;
00162 
00163         std::vector<Real> defProb(probabilities_.size());
00164         for (Size j = 0; j < probabilities_.size(); j++)
00165             defProb[j] = probabilities_[j]->defaultProbability(d);
00166 
00167         ProbabilityOfAtLeastNEvents op(n_);
00168 
00169         return copula_->integral(op, defProb);
00170     }
