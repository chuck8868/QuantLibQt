00078     void CTSMMCapletCalibration::performChecks(
00079                     const EvolutionDescription& evolution,
00080                     const PiecewiseConstantCorrelation&  corr,
00081                     const std::vector<boost::shared_ptr<
00082                                 PiecewiseConstantVariance> >&
00083                                             displacedSwapVariances,
00084                     const std::vector<Volatility>& mktCapletVols,
00085                     const CurveState& cs)
00086     {
00087         const std::vector<Time>& evolutionTimes = evolution.evolutionTimes();
00088         QL_REQUIRE(evolutionTimes==corr.times(),
00089                    "evolutionTimes "
00090                    << io::sequence(evolutionTimes)
00091                    << " not equal to correlation times "
00092                    << io::sequence(corr.times()));
00093 
00094         const std::vector<Time>& rateTimes = evolution.rateTimes();
00095         QL_REQUIRE(rateTimes==cs.rateTimes(),
00096                    "mismatch between EvolutionDescription and "
00097                    "CurveState rate times");
00098 
00099         Size numberOfRates = evolution.numberOfRates();
00100         QL_REQUIRE(numberOfRates==displacedSwapVariances.size(),
00101                    "mismatch between EvolutionDescription number of rates ("
00102                    << numberOfRates << ") and displacedSwapVariances size ("
00103                    << displacedSwapVariances.size() << ")");
00104         QL_REQUIRE(numberOfRates==corr.numberOfRates(),
00105                    "mismatch between EvolutionDescription number of rates ("
00106                    << numberOfRates << ") and corr number of rates ("
00107                    << corr.numberOfRates() << ")");
00108         QL_REQUIRE(numberOfRates==mktCapletVols.size(),
00109                    "mismatch between EvolutionDescription number of rates ("
00110                    << numberOfRates << ") and mktCapletVols size ("
00111                    << mktCapletVols.size() << ")");
00112         QL_REQUIRE(numberOfRates==cs.numberOfRates(),
00113                    "mismatch between EvolutionDescription number of rates ("
00114                    << numberOfRates << ") and CurveState  number of rates ("
00115                    << cs.numberOfRates() << ")");
00116 
00117         std::vector<Time> temp(rateTimes.begin(), rateTimes.end()-1);
00118         QL_REQUIRE(temp==evolutionTimes,
00119                    "mismatch between evolutionTimes and rateTimes");
00120 
00121         Volatility lastSwaptionVol =
00122             displacedSwapVariances.back()->totalVolatility(numberOfRates-1);
00123         QL_REQUIRE(close(lastSwaptionVol, mktCapletVols[numberOfRates-1]),
00124                    "last caplet vol (" << std::setprecision(16) <<
00125                    mktCapletVols[numberOfRates-1] <<
00126                    ") must be equal to last swaption vol (" <<
00127                    lastSwaptionVol << "); discrepancy is " <<
00128                    lastSwaptionVol-mktCapletVols[numberOfRates-1]);
00129     }

00133                                            Natural maxIterations,

00135 
00136                                            Natural innerSolvingMaxIterations,
00137                                            Real innerSolvingTolerance) {

00145         // initialize working variables
00146         usedCapletVols_ = mktCapletVols_;
00147 
00148         for (Size i=0; i<numberOfRates_; ++i)
00149             mktSwaptionVols_[i]=displacedSwapVariances_[i]->totalVolatility(i);
00150 
00151         std::vector<Spread> displacements(numberOfRates_,
00152                                           displacement_);
00153         const std::vector<Time>& rateTimes = evolution_.rateTimes();
00154         Natural iterations = 0;
00155 
00156         // calibration loop
00157         do {
00158             failures_ = calibrationImpl_(numberOfFactors,
00159                                          innerSolvingMaxIterations,
00160                                          innerSolvingTolerance);
00161 
00162             boost::shared_ptr<MarketModel> ctsmm(new
00163                 PseudoRootFacade(swapCovariancePseudoRoots_,
00164                                  rateTimes,
00165                                  cs_->coterminalSwapRates(),
00166                                  displacements));
00167             const Matrix& swaptionTotCovariance =
00168                 ctsmm->totalCovariance(numberOfRates_-1);
00169 
00170             CotSwapToFwdAdapter flmm(ctsmm);
00171             const Matrix& capletTotCovariance =
00172                 flmm.totalCovariance(numberOfRates_-1);
00173 
00174             // check fit
00175             capletRmsError_ = swaptionRmsError_ = 0.0;
00176             capletMaxError_ = swaptionMaxError_ = -1.0;
00177 
00178             for (Size i=0; i<numberOfRates_; ++i) {
00179                 mdlSwaptionVols_[i] = std::sqrt(swaptionTotCovariance[i][i]/rateTimes[i]);
00180                 Real swaptionError = std::fabs(mktSwaptionVols_[i]-mdlSwaptionVols_[i]);
00181                 swaptionRmsError_ += swaptionError*swaptionError;
00182                 swaptionMaxError_ = std::max(swaptionMaxError_, swaptionError);
00183 
00184                 mdlCapletVols_[i] = std::sqrt(capletTotCovariance[i][i]/rateTimes[i]);
00185                 Real capletError = std::fabs(mktCapletVols_[i]-mdlCapletVols_[i]);
00186                 capletRmsError_ += capletError*capletError;
00187                 capletMaxError_ = std::max(capletMaxError_, capletError);
00188 
00189                 if (i < numberOfRates_-1)
00190                     usedCapletVols_[i] *= mktCapletVols_[i]/mdlCapletVols_[i];
00191             }
00192             swaptionRmsError_ = std::sqrt(swaptionRmsError_/numberOfRates_);
00193             capletRmsError_ = std::sqrt(capletRmsError_/numberOfRates_);
00194             ++iterations;
00195         } while (iterations<maxIterations &&
00196                  capletRmsError_>capletVolTolerance);
00197 
00198          boost::shared_ptr<MarketModel> ctsmm(new
00199                 PseudoRootFacade(swapCovariancePseudoRoots_,
00200                                  rateTimes,
00201                                  cs_->coterminalSwapRates(),
00202                                  displacements));
00203 
00204          timeDependentCalibratedSwaptionVols_.clear();
00205          for (Size i=0; i<numberOfRates_; ++i)
00206              timeDependentCalibratedSwaptionVols_.push_back(
00207                 ctsmm->timeDependentVolatility(i));
00208 
00209         // calculate deformationSize_ ??
00210         calibrated_ = true;
