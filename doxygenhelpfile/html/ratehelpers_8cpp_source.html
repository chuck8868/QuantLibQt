00164     Real FuturesRateHelper::impliedQuote() const {
00165         QL_REQUIRE(termStructure_ != 0, "term structure not set");
00166         Rate forwardRate = (termStructure_->discount(earliestDate_) /
00167             termStructure_->discount(latestDate_)-1.0)/yearFraction_;
00168         Rate convAdj = convAdj_.empty() ? 0.0 : convAdj_->value();
00169         QL_ENSURE(convAdj >= 0.0,
00170                   "Negative (" << convAdj <<
00171                   ") futures convexity adjustment");
00172         Rate futureRate = forwardRate + convAdj;
00173         return 100.0 * (1.0 - futureRate);
00174     }

00558     void SwapRateHelper::initializeDates() {
00559 
00560         // 1. do not pass the spread here, as it might be a Quote
00561         //    i.e. it can dinamically change
00562         // 2. input discount curve Handle might be empty now but it could
00563         //    be assigned a curve later; use a RelinkableHandle here
00564         swap_ = MakeVanillaSwap(tenor_, iborIndex_, 0.0, fwdStart_)
00565             .withDiscountingTermStructure(discountRelinkableHandle_)
00566             .withFixedLegDayCount(fixedDayCount_)
00567             .withFixedLegTenor(Period(fixedFrequency_))
00568             .withFixedLegConvention(fixedConvention_)
00569             .withFixedLegTerminationDateConvention(fixedConvention_)
00570             .withFixedLegCalendar(calendar_)
00571             .withFloatingLegCalendar(calendar_);
00572 
00573         earliestDate_ = swap_->startDate();
00574 
00575         // Usually...
00576         latestDate_ = swap_->maturityDate();
00577         // ...but due to adjustments, the last floating coupon might
00578         // need a later date for fixing
00579         #ifdef QL_USE_INDEXED_COUPON
00580         shared_ptr<FloatingRateCoupon> lastFloating =
00581             boost::dynamic_pointer_cast<FloatingRateCoupon>(
00582                                                  swap_->floatingLeg().back());
00583         Date fixingValueDate =
