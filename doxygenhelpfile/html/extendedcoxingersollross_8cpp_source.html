00053         Real ps = termStructure()->discount(s);

00060     Real ExtendedCoxIngersollRoss::discountBondOption(Option::Type type,
00061                                                       Real strike,
00062                                                       Time t, Time s) const {
00063 
00064         QL_REQUIRE(strike>0.0, "strike must be positive");
00065 
00066         DiscountFactor discountT = termStructure()->discount(t);
00067         DiscountFactor discountS = termStructure()->discount(s);
00068         if (t < QL_EPSILON) {
00069             switch(type) {
00070               case Option::Call:
00071                 return std::max<Real>(discountS - strike, 0.0);
00072               case Option::Put:
00073                 return std::max<Real>(strike - discountS, 0.0);
00074               default: QL_FAIL("unsupported option type");
00075             }
00076         }
00077 
00078         Real sigma2 = sigma()*sigma();
00079         Real h = std::sqrt(k()*k() + 2.0*sigma2);
00080         Real r0 = termStructure()->forwardRate(0.0, 0.0,
00081                                                Continuous, NoFrequency);
00082         Real b = B(t,s);
00083 
00084         Real rho = 2.0*h/(sigma2*(std::exp(h*t) - 1.0));
00085         Real psi = (k() + h)/sigma2;
00086 
00087         Real df = 4.0*k()*theta()/sigma2;
00088         Real ncps = 2.0*rho*rho*(r0-phi_(0.0))*std::exp(h*t)/(rho+psi+b);
00089         Real ncpt = 2.0*rho*rho*(r0-phi_(0.0))*std::exp(h*t)/(rho+psi);
00090 
00091         NonCentralChiSquareDistribution chis(df, ncps);
00092         NonCentralChiSquareDistribution chit(df, ncpt);
00093 
00094         Real z = std::log(CoxIngersollRoss::A(t,s)/strike)/b;
00095         Real call = discountS*chis(2.0*z*(rho+psi+b)) -
00096             strike*discountT*chit(2.0*z*(rho+psi));
00097         if (type == Option::Call)
00098             return call;
00099         else
00100             return call - discountS + strike*discountT;
00101     }
