00104                 payoff, riskFreeDiscount, dividendDiscount, variance,
00105                 tolerance);

00142             Real V_E_h = part1 + part2 + part3;

00144             Real b = (1-h) * alpha * lambda_prime / (2*(2*lambda + beta - 1));
00145             Real c = - ((1 - h) * alpha / (2 * lambda + beta - 1)) *
00146                 (V_E_h / (hA) + 1 / h + lambda_prime / (2*lambda + beta - 1));

00148             Real chi = temp_spot_ratio * (b * temp_spot_ratio + c);

00150             if (phi*(Sk-spot) > 0) {
00151                 results_.value = black.value() +
00152                     hA * std::pow((spot/Sk), lambda) / (1 - chi);
00153             } else {
00154                 results_.value = phi * (spot - payoff->strike());
00155             }

00164                 (phi * (Sk - payoff->strike()) - black_Sk) * std::pow((spot/Sk), lambda);
00165 
00166             results_.gamma = phi * dividendDiscount * normalDist (phi*d1_Sk) /
00167                                         (spot * std::sqrt(variance))

00169                               + 2 * chi_prime * chi_prime / ((1 - chi) * (1 - chi) * (1 - chi))
00170                               + chi_double_prime / ((1 - chi) * (1 - chi))

00172                             * (phi * (Sk - payoff->strike()) - black_Sk)
