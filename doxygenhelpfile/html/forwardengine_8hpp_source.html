00075     template <class Engine>
00076     void ForwardVanillaEngine<Engine>::setup() const {
00077 
00078         boost::shared_ptr<StrikedTypePayoff> argumentsPayoff =
00079             boost::dynamic_pointer_cast<StrikedTypePayoff>(
00080                 this->arguments_.payoff);
00081         QL_REQUIRE(argumentsPayoff, "wrong payoff given");
00082 
00083         boost::shared_ptr<StrikedTypePayoff> payoff(
00084                    new PlainVanillaPayoff(argumentsPayoff->optionType(),
00085                                           this->arguments_.moneyness *
00086                                           process_->x0()));
00087 
00088         // maybe the forward value is "better", in some fashion
00089         // the right level is needed in order to interpolate
00090         // the vol
00091         Handle<Quote> spot = process_->stateVariable();
00092         QL_REQUIRE(spot->value() >= 0.0, "negative or null underlting given");
00093         Handle<YieldTermStructure> dividendYield(
00094             boost::shared_ptr<YieldTermStructure>(
00095                new ImpliedTermStructure(process_->dividendYield(),
00096                                         this->arguments_.resetDate)));
00097         Handle<YieldTermStructure> riskFreeRate(
00098             boost::shared_ptr<YieldTermStructure>(
00099                new ImpliedTermStructure(process_->riskFreeRate(),
00100                                         this->arguments_.resetDate)));
00101         // The following approach is ok if the vol is at most
00102         // time dependant. It is plain wrong if it is asset dependant.
00103         // In the latter case the right solution would be stochastic
00104         // volatility or at least local volatility (which unfortunately
00105         // implies an unrealistic time-decreasing smile)
00106         Handle<BlackVolTermStructure> blackVolatility(
00107             boost::shared_ptr<BlackVolTermStructure>(
00108                 new ImpliedVolTermStructure(process_->blackVolatility(),
00109                                             this->arguments_.resetDate)));
00110 
00111         boost::shared_ptr<GeneralizedBlackScholesProcess> fwdProcess(
00112                        new GeneralizedBlackScholesProcess(spot, dividendYield,
00113                                                           riskFreeRate,
00114                                                           blackVolatility));
00115 
00116         originalEngine_ = boost::shared_ptr<Engine>(new Engine(fwdProcess));
00117 
00118         originalArguments_ =
00119             dynamic_cast<VanillaOption::arguments*>(
00120                                              originalEngine_->getArguments());
00121         QL_REQUIRE(originalArguments_, "wrong engine type");
00122         originalResults_ =
00123             dynamic_cast<const VanillaOption::results*>(
00124                                                originalEngine_->getResults());
00125         QL_REQUIRE(originalResults_, "wrong engine type");
00126 
00127         originalArguments_->payoff = payoff;
00128         originalArguments_->exercise = this->arguments_.exercise;
00129 
00130         originalArguments_->validate();
00131     }

00140     template <class Engine>
00141     void ForwardVanillaEngine<Engine>::getOriginalResults() const {
00142 
00143         DayCounter rfdc = process_->riskFreeRate()->dayCounter();
00144         DayCounter divdc = process_->dividendYield()->dayCounter();
00145         Time resetTime = rfdc.yearFraction(
00146                                      process_->riskFreeRate()->referenceDate(),
00147                                      this->arguments_.resetDate);
00148         DiscountFactor discQ = process_->dividendYield()->discount(
00149                                                   this->arguments_.resetDate);
00150 
00151         this->results_.value = discQ * originalResults_->value;
00152         // I need the strike derivative here ...
00153         this->results_.delta = discQ * (originalResults_->delta +
00154             this->arguments_.moneyness * originalResults_->strikeSensitivity);
00155         this->results_.gamma = 0.0;
00156         this->results_.theta = process_->dividendYield()->
00157             zeroRate(this->arguments_.resetDate, divdc, Continuous, NoFrequency)
00158             * this->results_.value;
00159         this->results_.vega  = discQ * originalResults_->vega;
00160         this->results_.rho   = discQ *  originalResults_->rho;
00161         this->results_.dividendRho = - resetTime * this->results_.value
00162             + discQ * originalResults_->dividendRho;
00163     }
