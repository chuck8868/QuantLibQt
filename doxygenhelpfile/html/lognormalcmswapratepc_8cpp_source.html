00109     Real LogNormalCmSwapRatePc::advanceStep()
00110     {
00111         // we're going from T1 to T2
00112 
00113         // a) compute drifts D1 at T1;
00114         if (currentStep_ > initialStep_)
00115             calculators_[currentStep_].compute(curveState_, drifts1_);
00116         else
00117             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00118                       drifts1_.begin());
00119 
00120         // b) evolve forwards up to T2 using D1;
00121         Real weight = generator_->nextStep(brownians_);
00122         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00123         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00124 
00125         Size i, alive = alive_[currentStep_];
00126         for (i=alive; i<numberOfRates_; ++i) {
00127             logSwapRates_[i] += drifts1_[i] + fixedDrift[i];
00128             logSwapRates_[i] +=
00129                 std::inner_product(A.row_begin(i), A.row_end(i),
00130                                    brownians_.begin(), 0.0);
00131             swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
00132         }
00133 
00134         // intermediate curve state update
00135         curveState_.setOnCMSwapRates(swapRates_);
00136 
00137         // c) recompute drifts D2 using the predicted forwards;
00138         calculators_[currentStep_].compute(curveState_, drifts2_);
00139 
00140         // d) correct forwards using both drifts
00141         for (i=alive; i<numberOfRates_; ++i) {
00142             logSwapRates_[i] += (drifts2_[i]-drifts1_[i])/2.0;
00143             swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
00144         }
00145 
00146         // e) update curve state
00147         //curveState_.setOnCMSwapRates(swapRates_, alive);
00148         curveState_.setOnCMSwapRates(swapRates_);
00149 
00150         ++currentStep_;
00151 
00152         return weight;
00153     }
