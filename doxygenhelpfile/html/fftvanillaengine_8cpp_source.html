00060     std::complex<Real> FFTVanillaEngine::complexFourierTransform(std::complex<Real> u) const
00061     {
00062         std::complex<Real> i1(0, 1);
00063 
00064         Real s = process_->x0();
00065 
00066         std::complex<Real> phi = std::exp(i1 * u * (log(s) - (var_ * t_) / 2.0) 
00067             - (var_ * u * u * t_) / 2.0); 
00068         phi = phi * std::pow(dividendDiscount_/ riskFreeDiscount_, i1 * u);
