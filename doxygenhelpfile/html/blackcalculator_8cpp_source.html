00067     void BlackCalculator::initialize(const shared_ptr<StrikedTypePayoff>& p) {
00068         QL_REQUIRE(strike_>=0.0,
00069                    "strike (" << strike_ << ") must be non-negative");
00070         QL_REQUIRE(forward_>0.0,
00071                    "forward (" << forward_ << ") must be positive");
00072         //QL_REQUIRE(displacement_>=0.0,
00073         //           "displacement (" << displacement_ << ") must be non-negative");
00074         QL_REQUIRE(stdDev_>=0.0,
00075                    "stdDev (" << stdDev_ << ") must be non-negative");
00076         QL_REQUIRE(discount_>0.0,
00077                    "discount (" << discount_ << ") must be positive");
00078 
00079         if (stdDev_>=QL_EPSILON) {
00080             if (close(strike_, 0.0)) {
00081                 d1_ = QL_MAX_REAL;
00082                 d2_ = QL_MAX_REAL;
00083                 cum_d1_ = 1.0;
00084                 cum_d2_ = 1.0;
00085                 n_d1_ = 0.0;
00086                 n_d2_ = 0.0;
00087             } else {
00088                 d1_ = std::log(forward_/strike_)/stdDev_ + 0.5*stdDev_;
00089                 d2_ = d1_-stdDev_;
00090                 CumulativeNormalDistribution f;
00091                 cum_d1_ = f(d1_);
00092                 cum_d2_ = f(d2_);
00093                 n_d1_ = f.derivative(d1_);
00094                 n_d2_ = f.derivative(d2_);
00095             }
00096         } else {
00097             if (close(forward_, strike_)) {
00098                 d1_ = 0;
00099                 d2_ = 0;
00100                 cum_d1_ = 0.5;
00101                 cum_d2_ = 0.5;
00102                 n_d1_ = M_SQRT_2 * M_1_SQRTPI;
00103                 n_d2_ = M_SQRT_2 * M_1_SQRTPI;
00104             } else if (forward_>strike_) {
00105                 d1_ = QL_MAX_REAL;
00106                 d2_ = QL_MAX_REAL;
00107                 cum_d1_ = 1.0;
00108                 cum_d2_ = 1.0;
00109                 n_d1_ = 0.0;
00110                 n_d2_ = 0.0;
00111             } else {
00112                 d1_ = QL_MIN_REAL;
00113                 d2_ = QL_MIN_REAL;
00114                 cum_d1_ = 0.0;
00115                 cum_d2_ = 0.0;
00116                 n_d1_ = 0.0;
00117                 n_d2_ = 0.0;
00118             }
00119         }
00120 
00121         x_ = strike_;
00122         DxDstrike_ = 1.0;
00123 
00124         // the following one will probably disappear as soon as
00125         // super-share will be properly handled
00126         DxDs_ = 0.0;
00127 
00128         // this part is always executed.
00129         // in case of plain-vanilla payoffs, it is also the only part
00130         // which is executed.
00131         switch (p->optionType()) {
00132           case Option::Call:
00133             alpha_     =  cum_d1_;//  N(d1)
00134             DalphaDd1_ =    n_d1_;//  n(d1)
00135             beta_      = -cum_d2_;// -N(d2)
00136             DbetaDd2_  = -  n_d2_;// -n(d2)
00137             break;
00138           case Option::Put:
00139             alpha_     = -1.0+cum_d1_;// -N(-d1)
00140             DalphaDd1_ =        n_d1_;//  n( d1)
00141             beta_      =  1.0-cum_d2_;//  N(-d2)
00142             DbetaDd2_  =     -  n_d2_;// -n( d2)
00143             break;
00144           default:
00145             QL_FAIL("invalid option type");
00146         }
00147 
00148         // now dispatch on type.
00149 
00150         Calculator calc(*this);
00151         p->accept(calc);
00152     }

00204     Real BlackCalculator::delta(Real spot) const {
00205 
00206         QL_REQUIRE(spot > 0.0, "positive spot value required: " <<
00207                    spot << " not allowed");
00208 
00209         Real DforwardDs = forward_ / spot;
00210 
00211         Real temp = stdDev_*spot;
00212         Real DalphaDs = DalphaDd1_/temp;
00213         Real DbetaDs  = DbetaDd2_/temp;
00214         Real temp2 = DalphaDs * forward_ + alpha_ * DforwardDs
00215                       +DbetaDs  * x_       + beta_  * DxDs_;
00216 
00217         return discount_ * temp2;

00232         Real val = value();
00233         Real del = delta(spot);
00234         if (val>QL_EPSILON)
00235             return del/val*spot;
00236         else if (std::fabs(del)<QL_EPSILON)
00237             return 0.0;
00238         else if (del>0.0)
00239             return QL_MAX_REAL;

00257     Real BlackCalculator::gamma(Real spot) const {
00258 
00259         QL_REQUIRE(spot > 0.0, "positive spot value required: " <<
00260                    spot << " not allowed");
00261 
00262         Real DforwardDs = forward_ / spot;
00263 
00264         Real temp = stdDev_*spot;
00265         Real DalphaDs = DalphaDd1_/temp;
00266         Real DbetaDs  = DbetaDd2_/temp;
00267 
00268         Real D2alphaDs2 = - DalphaDs/spot*(1+d1_/stdDev_);
00269         Real D2betaDs2  = - DbetaDs /spot*(1+d2_/stdDev_);
00270 
00271         Real temp2 = D2alphaDs2 * forward_ + 2.0 * DalphaDs * DforwardDs
00272                       +D2betaDs2  * x_       + 2.0 * DbetaDs  * DxDs_;
00273 
00274         return  discount_ * temp2;
00275     }

00303     Real BlackCalculator::vega(Time maturity) const {
00304         QL_REQUIRE(maturity>=0.0,
00305                    "negative maturity not allowed");
00306 
00307         Real temp = std::log(strike_/forward_)/variance_;
00308         // actually DalphaDsigma / SQRT(T)
00309         Real DalphaDsigma = DalphaDd1_*(temp+0.5);
00310         Real DbetaDsigma  = DbetaDd2_ *(temp-0.5);
00311 
00312         Real temp2 = DalphaDsigma * forward_ + DbetaDsigma * x_;
00313 
00314         return discount_ * std::sqrt(maturity) * temp2;
