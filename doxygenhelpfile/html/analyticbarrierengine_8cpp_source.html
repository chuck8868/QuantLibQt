00035     void AnalyticBarrierEngine::calculate() const {
00036 
00037         boost::shared_ptr<PlainVanillaPayoff> payoff =
00038             boost::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
00039         QL_REQUIRE(payoff, "non-plain payoff given");
00040         QL_REQUIRE(payoff->strike()>0.0,
00041                    "strike must be positive");
00042 
00043         Real strike = payoff->strike();
00044         Real spot = process_->x0();
00045         QL_REQUIRE(spot >= 0.0, "negative or null underlying given");
00046         QL_REQUIRE(!triggered(spot), "barrier touched");
00047 
00048         Barrier::Type barrierType = arguments_.barrierType;
00049 
00050         switch (payoff->optionType()) {
00051           case Option::Call:
00052             switch (barrierType) {
00053               case Barrier::DownIn:
00054                 if (strike >= barrier())
00055                     results_.value = C(1,1) + E(1);
00056                 else
00057                     results_.value = A(1) - B(1) + D(1,1) + E(1);
00058                 break;
00059               case Barrier::UpIn:
00060                 if (strike >= barrier())
00061                     results_.value = A(1) + E(-1);
00062                 else
00063                     results_.value = B(1) - C(-1,1) + D(-1,1) + E(-1);
00064                 break;
00065               case Barrier::DownOut:
00066                 if (strike >= barrier())
00067                     results_.value = A(1) - C(1,1) + F(1);
00068                 else
00069                     results_.value = B(1) - D(1,1) + F(1);
00070                 break;
00071               case Barrier::UpOut:
00072                 if (strike >= barrier())
00073                     results_.value = F(-1);
00074                 else
00075                     results_.value = A(1) - B(1) + C(-1,1) - D(-1,1) + F(-1);
00076                 break;
00077             }
00078             break;
00079           case Option::Put:
00080             switch (barrierType) {
00081               case Barrier::DownIn:
00082                 if (strike >= barrier())
00083                     results_.value = B(-1) - C(1,-1) + D(1,-1) + E(1);
00084                 else
00085                     results_.value = A(-1) + E(1);
00086                 break;
00087               case Barrier::UpIn:
00088                 if (strike >= barrier())
00089                     results_.value = A(-1) - B(-1) + D(-1,-1) + E(-1);
00090                 else
00091                     results_.value = C(-1,-1) + E(-1);
00092                 break;
00093               case Barrier::DownOut:
00094                 if (strike >= barrier())
00095                     results_.value = A(-1) - B(-1) + C(1,-1) - D(1,-1) + F(1);
00096                 else
00097                     results_.value = F(1);
00098                 break;
00099               case Barrier::UpOut:
00100                 if (strike >= barrier())
00101                     results_.value = B(-1) - D(-1,-1) + F(-1);
00102                 else
00103                     results_.value = A(-1) - C(-1,-1) + F(-1);
00104                 break;
00105             }
00106             break;
00107           default:
00108             QL_FAIL("unknown type");
00109         }
00110     }

00171     Real AnalyticBarrierEngine::A(Real phi) const {
00172         Real x1 =
00173             std::log(underlying()/strike())/stdDeviation() + muSigma();
00174         Real N1 = f_(phi*x1);
00175         Real N2 = f_(phi*(x1-stdDeviation()));
00176         return phi*(underlying() * dividendDiscount() * N1

00211     Real AnalyticBarrierEngine::E(Real eta) const {
00212         if (rebate() > 0) {
00213             Real powHS0 = std::pow(barrier()/underlying(), 2 * mu());
00214             Real x2 =
00215                 std::log(underlying()/barrier())/stdDeviation() + muSigma();
00216             Real y2 =
00217                 std::log(barrier()/underlying())/stdDeviation() + muSigma();
00218             Real N1 = f_(eta*(x2 - stdDeviation()));
00219             Real N2 = f_(eta*(y2 - stdDeviation()));
00220             return rebate() * riskFreeDiscount() * (N1 - powHS0 * N2);
00221         } else {
00222             return 0.0;
00223         }
00224     }

00226     Real AnalyticBarrierEngine::F(Real eta) const {
00227         if (rebate() > 0) {
00228             Rate m = mu();
00229             Volatility vol = volatility();
00230             Real lambda = std::sqrt(m*m + 2.0*riskFreeRate()/(vol * vol));
00231             Real HS = barrier()/underlying();
00232             Real powHSplus = std::pow(HS, m + lambda);
00233             Real powHSminus = std::pow(HS, m - lambda);
00234 
00235             Real sigmaSqrtT = stdDeviation();
00236             Real z = std::log(barrier()/underlying())/sigmaSqrtT
00237                 + lambda * sigmaSqrtT;
00238 
00239             Real N1 = f_(eta * z);
00240             Real N2 = f_(eta * (z - 2.0 * lambda * sigmaSqrtT));
00241             return rebate() * (powHSplus * N1 + powHSminus * N2);
00242         } else {
00243             return 0.0;
00244         }
