00152     template <class S>
00153     Real GenericRiskStatistics<S>::regret(Real target) const {
00154         // average over the range below the target
00155         std::pair<Real,Size> result =
00156             this->expectationValue(compose(square<Real>(),
00157                                            std::bind2nd(std::minus<Real>(),
00158                                                         target)),
00159                                    std::bind2nd(std::less<Real>(),  target));
00160         Real x = result.first;
00161         Size N = result.second;
00162         QL_REQUIRE(N > 1,
00163                    "samples under target <= 1, unsufficient");
00164         return (N/(N-1.0))*x;
00165     }
