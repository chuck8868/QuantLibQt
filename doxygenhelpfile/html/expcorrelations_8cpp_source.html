00031     Disposable<Matrix> exponentialCorrelations(
00032                                         const std::vector<Time>& rateTimes,
00033                                         Real longTermCorr,
00034                                         Real beta,
00035                                         Real gamma,
00036                                         Time time) {
00037         // preliminary checks
00038         checkIncreasingTimes(rateTimes);
00039         QL_REQUIRE(longTermCorr<=1.0 && longTermCorr>=0.0,
00040                    "Long term correlation (" << longTermCorr <<
00041                    ") outside [0;1] interval");
00042         QL_REQUIRE(beta>=0.0,
00043                    "beta (" << beta <<
00044                    ") must be greater than zero");
00045         QL_REQUIRE(gamma<=1.0 && gamma>=0.0,
00046                    "gamma (" << gamma <<
00047                    ") outside [0;1] interval");
00048 
00049         // Calculate correlation matrix
00050         Size nbRows = rateTimes.size()-1;
00051         Matrix correlations(nbRows, nbRows, 0.0);
00052         for (Size i=0; i<nbRows; ++i) {
00053             // correlation is defined only between
00054             // (alive) stochastic rates...
00055             if (time<=rateTimes[i]) {
00056                 correlations[i][i] = 1.0;
00057                 for (Size j=0; j<i; ++j) {
00058                     if (time<=rateTimes[j]) {
00059                         correlations[i][j] = correlations[j][i] =
00060                             longTermCorr + (1.0-longTermCorr) *
00061                             std::exp(-beta*std::fabs(
00062                                 std::pow(rateTimes[i]-time, gamma) -
00063                                 std::pow(rateTimes[j]-time, gamma)
00064                                 )
00065                             );
00066                     }
00067                 }
00068             }
00069         }
00070         return correlations;
00071     }
