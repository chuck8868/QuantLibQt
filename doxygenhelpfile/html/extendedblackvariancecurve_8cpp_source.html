00058     void ExtendedBlackVarianceCurve::setVariances() {
00059         variances_[0] = 0.0;
00060         for (Size j=1; j<=volatilities_.size(); j++) {
00061             Volatility sigma = volatilities_[j-1]->value();
00062             variances_[j] = times_[j] * sigma * sigma;
00063             QL_REQUIRE(variances_[j]>=variances_[j-1]
00064                        || !forceMonotoneVariance_,
00065                        "variance must be non-decreasing");
00066         }
