00147 
00148         QL_REQUIRE(!bondHelpers_.empty(), "no bondHelpers given");
00149 
00150         maxDate_ = Date::minDate();
00151         Date refDate = referenceDate();
00152 
00153         // double check bond quotes still valid and/or instruments not expired
00154         for (Size i=0; i<bondHelpers_.size(); ++i) {
00155             shared_ptr<Bond> bond = bondHelpers_[i]->bond();
00156             QL_REQUIRE(bondHelpers_[i]->quote()->isValid(),

00161                        io::ordinal(i+1) << " bond settlemente date (" <<
00162                        bondSettlement << ") before curve reference date (" <<

00164             QL_REQUIRE(BondFunctions::isTradable(*bond, bondSettlement),
00165                        io::ordinal(i+1) << " bond non tradable at " <<
00166                        bondSettlement << " settlement date (maturity"
00167                        " being " << bond->maturityDate() << ")");
00168             maxDate_ = std::max(maxDate_, bondHelpers_[i]->latestDate());
00169             bondHelpers_[i]->setTermStructure(
00170                                   const_cast<FittedBondDiscountCurve*>(this));

00267                                                        const Array& x) const {

00270         const DayCounter& dc = fittingMethod_->curve_->dayCounter();
00271 
00272         Real squaredError = 0.0;
00273         Size n = fittingMethod_->curve_->bondHelpers_.size();
00274         for (Size i=0; i<n; ++i) {
00275 
00276             shared_ptr<Bond> bond =
00277                             fittingMethod_->curve_->bondHelpers_[i]->bond();
00278             Date bondSettlement = bond->settlementDate();
00279 
00280             // CleanPrice_i = sum( cf_k * d(t_k) ) - accruedAmount
00281             Real modelPrice = - bond->accruedAmount(bondSettlement);
00282             const Leg& cf = bond->cashflows();
00283             for (Size k=firstCashFlow_[i]; k<cf.size(); ++k) {
00284                 Time tenor = dc.yearFraction(refDate, cf[k]->date());
00285                 modelPrice += cf[k]->amount() *
00286                                     fittingMethod_->discountFunction(x, tenor);
00287             }
00288 
00289             // adjust price (NPV) for forward settlement
00290             if (bondSettlement != refDate ) {
00291                 Time tenor = dc.yearFraction(refDate, bondSettlement);
00292                 modelPrice /= fittingMethod_->discountFunction(x, tenor);
00293             }
00294             Real marketPrice =
00295                 fittingMethod_->curve_->bondHelpers_[i]->quote()->value();
00296             Real error = modelPrice - marketPrice;
00297             Real weightedError = fittingMethod_->weights_[i] * error;
00298             squaredError += weightedError * weightedError;
00299         }
00300         return squaredError;
