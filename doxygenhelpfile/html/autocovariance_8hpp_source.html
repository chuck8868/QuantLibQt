00036         // Outputs double FT for a given input:
00037         // input -> FFT -> norm -> FFT -> out
00038         template <typename ForwardIterator>
00039         std::vector<std::complex<Real> > double_ft(ForwardIterator begin,
00040                                                    ForwardIterator end) {
00041             std::size_t nData = std::distance(begin, end);
00042             std::size_t order = FastFourierTransform::min_order(nData) + 1;
00043             FastFourierTransform fft(order);
00044             std::vector<std::complex<Real> > ft(fft.output_size());
00045             fft.transform(begin, end, ft.begin());
00046             Array tmp(ft.size(), 0.0);
00047             std::transform(ft.begin(), ft.end(), tmp.begin(), std::norm<Real>);
00048             std::fill(ft.begin(), ft.end(), std::complex<Real>());
00049             fft.transform(tmp.begin(), tmp.end(), ft.begin());
00050             return ft;
00051         }

00054         // Calculates and subtracts mean from the input data; returns mean
00055         template <typename InputIterator, typename OutputIterator>
00056         Real remove_mean(InputIterator begin, InputIterator end,
00057                          OutputIterator out) {
00058             Real mean(0.0);
00059             std::size_t n = 1;
00060             for (InputIterator it = begin; it != end; ++it, ++n)
00061                 mean = (mean*Real(n-1) + *it)/n;
00062             std::transform(begin, end, out,
00063                            std::bind2nd(std::minus<Real>(), mean));
00064             return mean;
00065         }

00141     template <typename ForwardIterator, typename OutputIterator>
00142     void autocorrelations(ForwardIterator begin, ForwardIterator end,
00143                           OutputIterator out, std::size_t maxLag) {
00144         using namespace detail;
00145         std::size_t nData = std::distance(begin, end);
00146         QL_REQUIRE(maxLag < nData,
00147                    "number of correlations must be less than data size");
00148         const std::vector<std::complex<Real> >& ft = double_ft(begin, end);
00149         std::size_t nFrq = ft.size();
00150         Real variance = ft[0].real() / (nFrq * nData);
00151         *out++ = variance * nData / Real(nData-1);
00152         for (std::size_t k = 1; k <= maxLag; ++k)
00153             *out++ = ft[k].real() / (variance * nFrq * (nData - k));
