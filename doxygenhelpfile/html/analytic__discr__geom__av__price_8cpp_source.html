00038     void AnalyticDiscreteGeometricAveragePriceAsianEngine::calculate() const {
00039 
00040         /* this engine cannot really check for the averageType==Geometric
00041            since it can be used as control variate for the Arithmetic version
00042         QL_REQUIRE(arguments_.averageType == Average::Geometric,
00043                    "not a geometric average option");
00044         */
00045 
00046         QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
00047                    "not an European Option");
00048 
00049         Real runningLog;
00050         Size pastFixings;
00051         if (arguments_.averageType == Average::Geometric) {
00052             QL_REQUIRE(arguments_.runningAccumulator>0.0,
00053                        "positive running product required: "
00054                        << arguments_.runningAccumulator << " not allowed");
00055             runningLog =
00056                 std::log(arguments_.runningAccumulator);
00057             pastFixings = arguments_.pastFixings;
00058         } else {  // it is being used as control variate
00059             runningLog = 1.0;
00060             pastFixings = 0;
00061         }
00062 
00063         boost::shared_ptr<PlainVanillaPayoff> payoff =
00064             boost::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
00065         QL_REQUIRE(payoff, "non-plain payoff given");
00066 
00067         Date referenceDate = process_->riskFreeRate()->referenceDate();
00068         DayCounter rfdc  = process_->riskFreeRate()->dayCounter();
00069         DayCounter divdc = process_->dividendYield()->dayCounter();
00070         DayCounter voldc = process_->blackVolatility()->dayCounter();
00071         std::vector<Time> fixingTimes;
00072         Size i;
00073         for (i=0; i<arguments_.fixingDates.size(); i++) {
00074             if (arguments_.fixingDates[i]>=referenceDate) {
00075                 Time t = voldc.yearFraction(referenceDate,
00076                     arguments_.fixingDates[i]);
00077                 fixingTimes.push_back(t);
00078             }
00079         }
00080 
00081         Size remainingFixings = fixingTimes.size();
00082         Size numberOfFixings = pastFixings + remainingFixings;
00083         Real N = static_cast<Real>(numberOfFixings);
00084 
00085         Real pastWeight   = pastFixings/N;
00086         Real futureWeight = 1.0-pastWeight;
00087 
00088         Time timeSum = std::accumulate(fixingTimes.begin(),
00089                                        fixingTimes.end(), 0.0);
00090 
00091         Volatility vola = process_->blackVolatility()->blackVol(
00092                                               arguments_.exercise->lastDate(),
00093                                               payoff->strike());
00094         Real temp = 0.0;
00095         for (i=pastFixings+1; i<numberOfFixings; i++)
00096             temp += fixingTimes[i-pastFixings-1]*(N-i);
00097         Real variance = vola*vola /N/N * (timeSum+ 2.0*temp);
00098         Real dsigG_dsig = std::sqrt((timeSum + 2.0*temp))/N;
00099         Real sigG = vola * dsigG_dsig;
00100         Real dmuG_dsig = -(vola * timeSum)/N;
00101 
00102         Date exDate = arguments_.exercise->lastDate();
00103         Rate dividendRate = process_->dividendYield()->
00104             zeroRate(exDate, divdc, Continuous, NoFrequency);
00105         Rate riskFreeRate = process_->riskFreeRate()->
00106             zeroRate(exDate, rfdc, Continuous, NoFrequency);
00107         Rate nu = riskFreeRate - dividendRate - 0.5*vola*vola;
00108 
00109         Real s = process_->stateVariable()->value();
00110         QL_REQUIRE(s > 0.0, "positive underlying value required");
00111 
00112         Size M = (pastFixings == 0 ? 1 : pastFixings);
00113         Real muG = pastWeight * runningLog/M +
00114             futureWeight * std::log(s) + nu*timeSum/N;
00115         Real forwardPrice = std::exp(muG + variance / 2.0);
00116 
00117         DiscountFactor riskFreeDiscount = process_->riskFreeRate()->discount(
00118                                              arguments_.exercise->lastDate());
00119 
00120         BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),
00121                               riskFreeDiscount);
00122 
00123         results_.value = black.value();
00124         results_.delta = futureWeight*black.delta(forwardPrice)*forwardPrice/s;
00125         results_.gamma = forwardPrice*futureWeight/(s*s)
00126                 *(  black.gamma(forwardPrice)*futureWeight*forwardPrice
00127                   - pastWeight*black.delta(forwardPrice) );
00128 
00129         Real Nx_1, nx_1;
00130         CumulativeNormalDistribution CND;
00131         NormalDistribution ND;
00132         if (sigG > QL_EPSILON) {
00133             Real x_1  = (muG-std::log(payoff->strike())+variance)/sigG;
00134             Nx_1 = CND(x_1);
00135             nx_1 = ND(x_1);
00136         } else {
00137             Nx_1 = (muG > std::log(payoff->strike()) ? 1.0 : 0.0);
00138             nx_1 = 0.0;
00139         }
00140         results_.vega = forwardPrice * riskFreeDiscount *
00141                    ( (dmuG_dsig + sigG * dsigG_dsig)*Nx_1 + nx_1*dsigG_dsig );
00142 
00143         if (payoff->optionType() == Option::Put)
00144             results_.vega -= riskFreeDiscount * forwardPrice *
00145                                               (dmuG_dsig + sigG * dsigG_dsig);
00146 
00147         Time tRho = rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
00148                                       arguments_.exercise->lastDate());
00149         results_.rho = black.rho(tRho)*timeSum/(N*tRho)
00150                       - (tRho-timeSum/N)*results_.value;
00151 
00152         Time tDiv = divdc.yearFraction(
00153                            process_->dividendYield()->referenceDate(),
00154                            arguments_.exercise->lastDate());
00155 
00156         results_.dividendRho = black.dividendRho(tDiv)*timeSum/(N*tDiv);
00157 
00158         results_.strikeSensitivity = black.strikeSensitivity();
00159 
00160         results_.theta = blackScholesTheta(process_,
00161                                            results_.value,
00162                                            results_.delta,
00163                                            results_.gamma);
00164     }
