00043     Date
00044     YoYOptionletVolatilitySurface::baseDate() const {
00045 
00046         // Depends on interpolation, or not, of observed index
00047         // and observation lag with which it was built.
00048         // We want this to work even if the index does not
00049         // have a yoy term structure.
00050         if (indexIsInterpolated()) {
00051             return referenceDate() - observationLag();
00052         } else {
00053             return inflationPeriod(referenceDate() - observationLag(),
00054                                    frequency()).first;
00055         }

00087     Volatility
00088     YoYOptionletVolatilitySurface::volatility(const Date& maturityDate,
00089                                               Rate strike,
00090                                               const Period &obsLag,
00091                                               bool extrapolate) const {
00092 
00093         Period useLag = obsLag;
00094         if (obsLag==Period(-1,Days)) {
00095             useLag = observationLag();
00096         }
00097 
00098         if (indexIsInterpolated()) {
00099             YoYOptionletVolatilitySurface::checkRange(maturityDate-useLag, strike, extrapolate);
00100             Time t = timeFromReference(maturityDate-useLag);
00101             return volatilityImpl(t,strike);
00102         } else {
00103             std::pair<Date,Date> dd = inflationPeriod(maturityDate-useLag, frequency());
00104             YoYOptionletVolatilitySurface::checkRange(dd.first, strike, extrapolate);
00105             Time t = timeFromReference(dd.first);
00106             return volatilityImpl(t,strike);
00107         }

00123     YoYOptionletVolatilitySurface::timeFromBase(const Date &maturityDate,
00124                                                 const Period& obsLag) const {
00125 
00126         Period useLag = obsLag;
00127         if (obsLag==Period(-1,Days)) {
00128             useLag = observationLag();
00129         }
00130 
00131         Date useDate;
00132         if (indexIsInterpolated()) {
00133             useDate = maturityDate - useLag;

00135             useDate = inflationPeriod(maturityDate - useLag,
