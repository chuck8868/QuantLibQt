00049                 Size n = dt.size(),
00050                      i = 0;
00051 
00052                 Real compoundFactor = 1.0;
00053 
00054                 // already fixed part
00055                 Date today = Settings::instance().evaluationDate();
00056                 while (fixingDates[i]<today && i<n) {
00057                     // rate must have been fixed
00058                     Rate pastFixing = IndexManager::instance().getHistory(
00059                                                 index->name())[fixingDates[i]];
00060                     QL_REQUIRE(pastFixing != Null<Real>(),

00064                     ++i;
00065                 }
00066 
00067                 // today is a border case
00068                 if (fixingDates[i] == today && i<n) {
00069                     // might have been fixed
00070                     try {
00071                         Rate pastFixing = IndexManager::instance().getHistory(
00072                                                 index->name())[fixingDates[i]];
00073                         if (pastFixing != Null<Real>()) {
00074                             compoundFactor *= (1.0 + pastFixing*dt[i]);
00075                             ++i;
00076                         } else {
00077                             ;   // fall through and forecast
00078                         }
00079                     } catch (Error&) {
00080                         ;       // fall through and forecast
00081                     }
00082                 }
00083 
00084                 // forward part using telescopic property in order
00085                 // to avoid the evaluation of multiple forward fixings
00086                 if (i<n) {
00087                     Handle<YieldTermStructure> curve =
00088                         index->forwardingTermStructure();
00089                     QL_REQUIRE(!curve.empty(),
00090                                "null term structure set to this instance of "<<
00091                                index->name());
00092 
00093                     const vector<Date>& dates = coupon_->valueDates();
00094                     DiscountFactor startDiscount = curve->discount(dates[i]);
00095                     DiscountFactor endDiscount = curve->discount(dates[n]);
00096 
00097                     compoundFactor *= startDiscount/endDiscount;
00098                 }
00099 
00100                 Rate rate = (compoundFactor - 1.0) / coupon_->accrualPeriod();
00101                 return coupon_->gearing() * rate + coupon_->spread();
00102             }
