00031         class Integrand : std::unary_function<Real,Real> {
00032         public:
00033             Integrand(const boost::shared_ptr<StrikedTypePayoff>& payoff,
00034                 Real s0, Real t, Real riskFreeDiscount, Real dividendDiscount,
00035                 Real sigma, Real nu, Real theta)
00036                 : payoff_(payoff), s0_(s0), t_(t), riskFreeDiscount_(riskFreeDiscount),
00037                     dividendDiscount_(dividendDiscount),
00038                     sigma_(sigma), nu_(nu), theta_(theta) {
00039                 omega_ = std::log(1.0 - theta_ * nu_ - (sigma_ * sigma_ * nu_) / 2.0) / nu_;
00040                 // We can precompute the denominator of the gamma pdf (does not depend on x)
00041                 // shape = t_/nu_, scale = nu_
00042                 GammaFunction gf;
00043                 gammaDenom_ = std::exp(gf.logValue(t_ / nu_)) * std::pow(nu_, t_ / nu_);
00044             }
00045 
00046             Real operator()(Real x) const {
00047                 // Compute adjusted black scholes price
00048                 Real s0_adj = s0_ * std::exp(theta_ * x + omega_ * t_ + (sigma_ * sigma_ * x) / 2.0);
00049                 Real vol_adj = sigma_ * sqrt(x / t_);
00050                 vol_adj *= sqrt(t_);
00051 
00052                 BlackScholesCalculator bs(payoff_, s0_adj, dividendDiscount_, vol_adj, riskFreeDiscount_);
00053                 Real bsprice = bs.value();
00054 
00055                 // Multiply by gamma distribution
00056                 Real gamp = (std::pow(x, t_ / nu_ - 1.0) * std::exp(-x / nu_)) / gammaDenom_;
00057                 Real result = bsprice * gamp;
00058                 return result;
00059             }
00060 
00061         private:
00062             boost::shared_ptr<StrikedTypePayoff> payoff_;
00063             Real s0_;
00064             Real t_;
00065             Real riskFreeDiscount_;
00066             Real dividendDiscount_;
00067             Rate sigma_;
00068             Real nu_;
00069             Real theta_;
00070             Real omega_;
00071             Real gammaDenom_;
00072         };
00073     }
