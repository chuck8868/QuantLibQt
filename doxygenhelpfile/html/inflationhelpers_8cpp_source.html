00032     ZeroCouponInflationSwapHelper::ZeroCouponInflationSwapHelper(
00033         const Handle<Quote>& quote,
00034         const Period& swapObsLag,   // <= index availability lag
00035         const Date& maturity,
00036         const Calendar& calendar,   // index may have null calendar as valid on every day
00037         BusinessDayConvention paymentConvention,
00038         const DayCounter& dayCounter,
00039         const boost::shared_ptr<ZeroInflationIndex>& zii)
00040     : BootstrapHelper<ZeroInflationTermStructure>(quote),
00041     swapObsLag_(swapObsLag), maturity_(maturity), calendar_(calendar),
00042     paymentConvention_(paymentConvention), dayCounter_(dayCounter),
00043     zii_(zii) {
00044 
00045         if (zii_->interpolated()) {
00046             // if interpolated then simple
00047             earliestDate_ = maturity_ - swapObsLag_;
00048             latestDate_ = maturity_ - swapObsLag_;
00049         } else {
00050             // but if NOT interpolated then the value is valid
00051             // for every day in an inflation period so you actually
00052             // get an extended validity, however for curve building
00053             // just put the first date because using that convention
00054             // for the base date throughout
00055             std::pair<Date,Date> limStart = inflationPeriod(maturity_ - swapObsLag_,
00056                                                             zii_->frequency());
00057             earliestDate_ = limStart.first;
00058             latestDate_ = limStart.first;
00059         }
00060 
00061         // check that the observation lag of the swap
00062         // is compatible with the availability lag of the index AND
00063         // it's interpolation (assuming the start day is spot)
00064         if (zii_->interpolated()) {
00065             Period pShift(zii_->frequency());
00066             QL_REQUIRE(swapObsLag_ - pShift > zii_->availabilityLag(),
00067                        "inconsistency between swap observation of index "
00068                        << swapObsLag_ <<
00069                        " index availability " << zii_->availabilityLag() <<
00070                        " index period " << pShift <<
00071                        " and index availability " << zii_->availabilityLag() <<
00072                        " need (obsLag-index period) > availLag");
00073         }
00074 
00075         registerWith(Settings::instance().evaluationDate());
00076     }
