00076        classes which rely on the difference between the previous day's
00077        close price and today's open price. */
00078     template <class T>
00079     class GarmanKlassOpenClose : public T {
00080     protected:
00081         Real f_;
00082         Real a_;
00083     public:
00084         GarmanKlassOpenClose(Real y, Real marketOpenFraction,
00085                              Real a) :
00086         T(y), f_(marketOpenFraction), a_(a) {};
00087         TimeSeries<Volatility>
00088         calculate(const TimeSeries<IntervalPrice> &quoteSeries) {
00089             TimeSeries<Volatility> retval;
00090             TimeSeries<IntervalPrice>::const_iterator prev, next, cur, start;
00091             start = quoteSeries.begin();
00092             ++start;
00093             for (cur = start; cur != quoteSeries.end(); ++cur) {
00094                 prev = cur; --prev;
00095                 Real c0 = std::log(prev->second.close());
00096                 Real o1 = std::log(cur->second.open());
00097                 Real sigma2 =
00098                     a_ * (o1 - c0) * (o1 - c0) / f_ +
00099                     (1-a_) * T::calculatePoint(cur->second) /
00100                     (1-f_);
00101 
00102                 retval[cur->first] = std::sqrt(sigma2/T::yearFraction_);
00103             }
00104             return retval;
00105         }
00106     };
