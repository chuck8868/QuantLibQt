00050     void FDVanillaEngine::setGridLimits(Real center, Time t) const {
00051         QL_REQUIRE(center > 0.0, "negative or null underlying given");
00052         center_ = center;
00053         Size newGridPoints = safeGridPoints(gridPoints_, t);
00054         if (newGridPoints > intrinsicValues_.size()) {
00055             intrinsicValues_ = SampledCurve(newGridPoints);
00056         }
00057 
00058         Real volSqrtTime = std::sqrt(process_->blackVolatility()
00059                                      ->blackVariance(t, center_));
00060 
00061         // the prefactor fine tunes performance at small volatilities
00062         Real prefactor = 1.0 + 0.02/volSqrtTime;
00063         Real minMaxFactor = std::exp(4.0 * prefactor * volSqrtTime);
00064         sMin_ = center_/minMaxFactor;  // underlying grid min value
00065         sMax_ = center_*minMaxFactor;  // underlying grid max value
00066     }

00068     void FDVanillaEngine::ensureStrikeInGrid() const {
00069         // ensure strike is included in the grid
00070         boost::shared_ptr<StrikedTypePayoff> striked_payoff =
00071             boost::dynamic_pointer_cast<StrikedTypePayoff>(payoff_);
00072         if (!striked_payoff)
00073             return;
00074         Real requiredGridValue = striked_payoff->strike();
00075 
00076         if(sMin_ > requiredGridValue/safetyZoneFactor_){
00077             sMin_ = requiredGridValue/safetyZoneFactor_;
00078             // enforce central placement of the underlying
00079             sMax_ = center_/(sMin_/center_);
00080         }
00081         if(sMax_ < requiredGridValue*safetyZoneFactor_){
00082             sMax_ = requiredGridValue*safetyZoneFactor_;
00083             // enforce central placement of the underlying
00084             sMin_ = center_/(sMax_/center_);
00085         }
00086     }

00116     // safety check to be sure we have enough grid points.
00117     Size FDVanillaEngine::safeGridPoints(Size gridPoints,
00118                                          Time residualTime) const {
00119         static const Size minGridPoints = 10;
00120         static const Size minGridPointsPerYear = 2;
00121         return std::max(gridPoints,
00122                         residualTime > 1.0 ?
00123                             static_cast<Size>((minGridPoints +

00125                                                 minGridPointsPerYear))
