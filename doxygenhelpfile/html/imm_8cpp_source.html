00169         Date refDate = (date == Date() ?
00170                         Date(Settings::instance().evaluationDate()) :
00171                         date);
00172         Year y = refDate.year();

00174 
00175         Size offset = mainCycle ? 3 : 1;
00176         Size skipMonths = offset-(m%offset);
00177         if (skipMonths != offset || refDate.dayOfMonth() > 21) {
00178             skipMonths += Size(m);
00179             if (skipMonths<=12) {
00180                 m = QuantLib::Month(skipMonths);

00188         if (result<=refDate)
00189             result = nextDate(Date(22, m, y), mainCycle);
