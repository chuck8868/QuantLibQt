00086     void MersenneTwisterUniformRng::seedInitialization(unsigned long seed) {
00087         /* initializes mt with a seed */
00088         unsigned long s = (seed != 0 ? seed : SeedGenerator::instance().get());
00089         mt[0]= s & 0xffffffffUL;
00090         for (mti=1; mti<N; mti++) {
00091             mt[mti] =
00092                 (1812433253UL * (mt[mti-1] ^ (mt[mti-1] >> 30)) + mti);
00093             /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
00094             /* In the previous versions, MSBs of the seed affect   */
00095             /* only MSBs of the array mt[].                        */
00096             /* 2002/01/09 modified by Makoto Matsumoto             */
00097             mt[mti] &= 0xffffffffUL;
00098             /* for >32 bit machines */
00099         }
00100     }

00102     MersenneTwisterUniformRng::MersenneTwisterUniformRng(
00103                                       const std::vector<unsigned long>& seeds) {
00104         seedInitialization(19650218UL);
00105         Size i=1, j=0, k = (N>seeds.size() ? N : seeds.size());
00106         for (; k; k--) {
00107             mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 30)) * 1664525UL))
00108                 + seeds[j] + j; /* non linear */
00109             mt[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
00110             i++; j++;
00111             if (i>=N) { mt[0] = mt[N-1]; i=1; }
00112             if (j>=seeds.size()) j=0;
00113         }
00114         for (k=N-1; k; k--) {
00115             mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 30)) * 1566083941UL))
00116                 - i; /* non linear */
00117             mt[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
00118             i++;
00119             if (i>=N) { mt[0] = mt[N-1]; i=1; }
00120         }
00121 
00122         mt[0] = UPPER_MASK; /*MSB is 1; assuring non-zero initial array*/
00123     }

00125     void MersenneTwisterUniformRng::twist() const {
00126         static const unsigned long mag01[2]={0x0UL, MATRIX_A};
00127         /* mag01[x] = x * MATRIX_A  for x=0,1 */
00128         Size kk;
00129         unsigned long y;
00130 
00131         for (kk=0;kk<N-M;kk++) {
00132             y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
00133             mt[kk] = mt[kk+M] ^ (y >> 1) ^ mag01[y & 0x1UL];
00134         }
00135         for (;kk<N-1;kk++) {
00136             y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
00137             mt[kk] = mt[kk+(M-N)] ^ (y >> 1) ^ mag01[y & 0x1UL];
00138         }
00139         y = (mt[N-1]&UPPER_MASK)|(mt[0]&LOWER_MASK);
00140         mt[N-1] = mt[M-1] ^ (y >> 1) ^ mag01[y & 0x1UL];
00141 
00142         mti = 0;
00143     }
