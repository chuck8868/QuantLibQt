00054         Real HullWhiteCapFloorPricer::operator()(const Path& path) const {
00055 
00056             bool isCap = (args_.type == CapFloor::Cap);
00057             Real npv = 0.0;
00058             Time Tb = forwardMeasureTime_;
00059 
00060             Size pastFixings = 0;
00061             for (Size i = 0; i<fixingTimes_.size(); i++) {
00062                 Time tau = args_.accrualTimes[i];
00063                 Time start = startTimes_[i],
00064                      end = endTimes_[i],
00065                      fixing = fixingTimes_[i];
00066                 if (end <= 0.0) {
00067                     // the fixing is in the past...
00068                     pastFixings++;
00069                     // ...and the caplet is expired; nothing more to do.
00070                 } else {
00071                     Rate ri_1, ri_2, currentLibor;
00072                     if (fixing <= 0.0) {
00073                         // current caplet. The fixing is in the past...
00074                         pastFixings++;
00075                         // ...so it is determined.
00076                         currentLibor = args_.forwards[i];
00077                         // However, the short rate at caplet expiry is not.
00078                         ri_2 = path[i-pastFixings+2];
00079                     } else {
00080                         // future caplet. Everything is to be forecast.
00081                         // The number of past fixings is used as an offset
00082                         // to index into the path.
00083                         ri_1 = path[i-pastFixings+1];
00084                         ri_2 = path[i-pastFixings+2];
00085 
00086                         DiscountFactor d1 =
00087                             model_->discountBond(fixing, start, ri_1);
00088                         DiscountFactor d2 =
00089                             model_->discountBond(fixing, end, ri_1);
00090                         currentLibor = (d1/d2-1)/tau;
00091                     }
00092 
00093                     Real accrualFactor =
00094                         1.0/model_->discountBond(end, Tb, ri_2);
00095 
00096                     Rate strike = isCap?
00097                         args_.capRates[i] :
00098                         args_.floorRates[i];
00099                     Real payoff = isCap?
00100                         std::max(currentLibor - strike, 0.0) :
00101                     std::max(strike - currentLibor, 0.0);
00102 
00103                     npv += payoff * tau * args_.gearings[i] *
00104                            args_.nominals[i] * accrualFactor;
00105                 }
00106             }
00107             npv *= endDiscount_;
00108             return  npv;
