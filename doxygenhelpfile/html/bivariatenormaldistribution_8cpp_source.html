00056     Real BivariateCumulativeNormalDistributionDr78::operator()(Real a,
00057                                                                Real b) const {
00058 
00059         CumulativeNormalDistribution cumNormalDist;
00060         Real CumNormDistA = cumNormalDist(a);
00061         Real CumNormDistB = cumNormalDist(b);
00062         Real MaxCumNormDistAB = std::max(CumNormDistA, CumNormDistB);
00063         Real MinCumNormDistAB = std::min(CumNormDistA, CumNormDistB);
00064 
00065         if (1.0-MaxCumNormDistAB<1e-15)
00066             return MinCumNormDistAB;
00067 
00068         if (MinCumNormDistAB<1e-15)
00069             return MinCumNormDistAB;
00070 
00071         Real a1 = a / std::sqrt(2.0 * (1.0 - rho_*rho_));
00072         Real b1 = b / std::sqrt(2.0 * (1.0 - rho_*rho_));
00073 
00074         Real result=-1.0;
00075 
00076         if (a<=0.0 && b<=0 && rho_<=0) {
00077             Real sum=0.0;
00078             for (Size i=0; i<5; i++) {
00079                 for (Size j=0;j<5; j++) {
00080                     sum += x_[i]*x_[j]*
00081                         std::exp(a1*(2.0*y_[i]-a1)+b1*(2.0*y_[j]-b1)
00082                                  +2.0*rho_*(y_[i]-a1)*(y_[j]-b1));
00083                 }
00084             }
00085             result = std::sqrt(1.0 - rho_*rho_)/M_PI*sum;
00086         } else if (a<=0 && b>=0 && rho_>=0) {
00087             BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(-rho_);
00088             result= CumNormDistA - bivCumNormalDist(a, -b);
00089         } else if (a>=0.0 && b<=0.0 && rho_>=0.0) {
00090             BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(-rho_);
00091             result= CumNormDistB - bivCumNormalDist(-a, b);
00092         } else if (a>=0.0 && b>=0.0 && rho_<=0.0) {
00093             result= CumNormDistA + CumNormDistB -1.0 + (*this)(-a, -b);
00094         } else if (a*b*rho_>0.0) {
00095             Real rho1 = (rho_*a-b)*(a>0.0 ? 1.0: -1.0)/
00096                 std::sqrt(a*a-2.0*rho_*a*b+b*b);
00097             BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(rho1);
00098 
00099             Real rho2 = (rho_*b-a)*(b>0.0 ? 1.0: -1.0)/
00100                 std::sqrt(a*a-2.0*rho_*a*b+b*b);
00101             BivariateCumulativeNormalDistributionDr78 CBND2(rho2);
00102 
00103             Real delta = (1.0-(a>0.0 ? 1.0: -1.0)*(b>0.0 ? 1.0: -1.0))/4.0;
00104 
00105             result= bivCumNormalDist(a, 0.0) + CBND2(b, 0.0) - delta;
00106         } else {
00107             QL_FAIL("case not handled");
00108         }
00109 
00110         return result;
00111     }

00118         class eqn3 { /* Relates to eqn3 Genz 2004 */
00119           public:
00120             eqn3(Real h, Real k, Real asr) {
00121                 hk_ = h * k;
00122                 hs_  = (h * h + k * k) / 2;
00123                 asr_ = asr;
00124             }
00125             Real operator()(Real x) const {
00126                 Real sn = std::sin(asr_ * (-x + 1) * 0.5);
00127                 return std::exp((sn * hk_ - hs_) / (1.0 - sn * sn));
00128             }
00129           private:
00130             Real hk_, asr_, hs_;
00131         };

00133         class eqn6 { /* Relates to eqn6 Genz 2004 */
00134           public:
00135             eqn6(Real a, Real c, Real d, Real bs, Real hk)
00136             : a_(a), c_(c), d_(d), bs_(bs), hk_(hk) {}
00137             Real operator()(Real x) const {
00138                 Real xs = a_ * (-x + 1);
00139                 xs = std::fabs(xs*xs);
00140                 Real rs = std::sqrt(1 - xs);
00141                 Real asr = -(bs_ / xs + hk_) / 2;
00142                 if (asr > -100.0) {
00143                     return (a_ * std::exp(asr) *
00144                             (std::exp(-hk_ * (1 - rs) / (2 * (1 + rs))) / rs -
00145                              (1 + c_ * xs * (1 + d_ * xs))));
00146                 } else {
00147                     return 0.0;
00148                 }
00149             }
00150           private:
00151             Real a_, c_, d_, bs_, hk_;
00152         };

00167     Real BivariateCumulativeNormalDistributionWe04DP::operator()(
00168                                                        Real x, Real y) const {
00169 
00170         /* The implementation is described at section 2.4 "Hybrid
00171            Numerical Integration Algorithms" of "Numerical Computation
00172            of Rectangular Bivariate an Trivariate Normal and t
00173            Probabilities", Genz (2004), Statistics and Computing 14,
00174            151-160. (available at
00175            www.sci.wsu.edu/math/faculty/henz/homepage)
00176 
00177            The Gauss-Legendre quadrature have been extracted to
00178            TabulatedGaussLegendre (x,w zero-based)
00179 
00180            Tthe functions ot be integrated numerically have been moved
00181            to classes eqn3 and eqn6
00182 
00183            Change some magic numbers to M_PI */
00184 
00185         TabulatedGaussLegendre gaussLegendreQuad(20);
00186         if (std::fabs(correlation_) < 0.3) {
00187             gaussLegendreQuad.order(6);
00188         } else if (std::fabs(correlation_) < 0.75) {
00189             gaussLegendreQuad.order(12);
00190         }
00191 
00192         Real h = -x;
00193         Real k = -y;
00194         Real hk = h * k;
00195         Real BVN = 0.0;
00196 
00197         if (std::fabs(correlation_) < 0.925)
00198         {
00199             if (std::fabs(correlation_) > 0)
00200             {
00201                 Real asr = std::asin(correlation_);
00202                 eqn3 f(h,k,asr);
00203                 BVN = gaussLegendreQuad(f);
00204                 BVN *= asr * (0.25 / M_PI);
00205             }
00206             BVN += cumnorm_(-h) * cumnorm_(-k);
00207         }
00208         else
00209         {
00210             if (correlation_ < 0)
00211             {
00212                 k *= -1;
00213                 hk *= -1;
00214             }
00215             if (std::fabs(correlation_) < 1)
00216             {
00217                 Real Ass = (1 - correlation_) * (1 + correlation_);
00218                 Real a = std::sqrt(Ass);
00219                 Real bs = (h-k)*(h-k);
00220                 Real c = (4 - hk) / 8;
00221                 Real d = (12 - hk) / 16;
00222                 Real asr = -(bs / Ass + hk) / 2;
00223                 if (asr > -100)
00224                 {
00225                     BVN = a * std::exp(asr) *
00226                         (1 - c * (bs - Ass) * (1 - d * bs / 5) / 3 +
00227                          c * d * Ass * Ass / 5);
00228                 }
00229                 if (-hk < 100)
00230                 {
00231                     Real B = std::sqrt(bs);
00232                     BVN -= std::exp(-hk / 2) * 2.506628274631 *
00233                         cumnorm_(-B / a) * B *
00234                         (1 - c * bs * (1 - d * bs / 5) / 3);
00235                 }
00236                 a /= 2;
00237                 eqn6 f(a,c,d,bs,hk);
00238                 BVN += gaussLegendreQuad(f);
00239                 BVN /= (-2.0 * M_PI);
00240             }
00241 
00242             if (correlation_ > 0) {
00243                 BVN += cumnorm_(-std::max(h, k));
00244             } else {
00245                 BVN *= -1;
00246                 if (k > h) {
00247                     BVN += cumnorm_(k) - cumnorm_(h);
00248                 }
00249             }
00250         }
00251         return BVN;
00252     }
