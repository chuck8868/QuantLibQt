00031             Real del = 1.0/a_;
00032             Real sum = del;
00033             for (Size n=1; n<=100; n++) {
00034                 ap += 1.0;
00035                 del *= x/ap;
00036                 sum += del;

00038                     return sum*std::exp(-x + a_*std::log(x) - gln);
