00099     inline Real PoissonDistribution::operator()(BigNatural k) const {
00100         if (mu_==0.0) {
00101             if (k==0) return 1.0;
00102             else      return 0.0;

00104         Real logFactorial = Factorial::ln(k);
00105         return std::exp(k*std::log(mu_) - logFactorial - mu_);
