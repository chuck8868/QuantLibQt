00028     OptionletStripper::OptionletStripper(
00029             const boost::shared_ptr<CapFloorTermVolSurface>& termVolSurface,
00030             const boost::shared_ptr<IborIndex>& iborIndex)
00031     : termVolSurface_(termVolSurface),
00032       iborIndex_(iborIndex),
00033       nStrikes_(termVolSurface->strikes().size()) {
00034         
00035         registerWith(termVolSurface);
00036         registerWith(iborIndex_);
00037         registerWith(Settings::instance().evaluationDate());
00038 
00039         Period indexTenor = iborIndex_->tenor();
00040         Period maxCapFloorTenor = termVolSurface->optionTenors().back();
00041 
00042         // optionlet tenors and capFloor lengths
00043         optionletTenors_.push_back(indexTenor);
00044         capFloorLengths_.push_back(optionletTenors_.back()+indexTenor);
00045         QL_REQUIRE(maxCapFloorTenor>=capFloorLengths_.back(),
00046                    "too short (" << maxCapFloorTenor <<
00047                    ") capfloor term vol termVolSurface");
00048         Period nextCapFloorLength = capFloorLengths_.back()+indexTenor;
00049         while (nextCapFloorLength<=maxCapFloorTenor) {
00050             optionletTenors_.push_back(capFloorLengths_.back());
00051             capFloorLengths_.push_back(nextCapFloorLength);
00052             nextCapFloorLength += indexTenor;
00053         }
00054         nOptionletTenors_ = optionletTenors_.size();
00055         
00056         optionletVolatilities_ =
00057             vector<vector<Volatility> >(nOptionletTenors_, 
00058                                         vector<Volatility>(nStrikes_));
00059         optionletStrikes_ = vector<vector<Rate> >(nOptionletTenors_,
00060                                                   termVolSurface->strikes());
00061         optionletDates_ = vector<Date>(nOptionletTenors_);
00062         optionletTimes_ = vector<Time>(nOptionletTenors_);
00063         atmOptionletRate_ = vector<Rate>(nOptionletTenors_);
00064         optionletPaymentDates_ = vector<Date>(nOptionletTenors_);
00065         optionletAccrualPeriods_ = vector<Time>(nOptionletTenors_);
00066     }
