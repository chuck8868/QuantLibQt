00045     Swap::Swap(const std::vector<Leg>& legs,
00046                const std::vector<bool>& payer)

00055             if (payer[j]) payer_[j]=-1.0;
00056             for (Leg::iterator i = legs_[j].begin(); i!= legs_[j].end(); ++i)
00057                 registerWith(*i);
00058         }
00059     }

00143     Date Swap::startDate() const {
00144         QL_REQUIRE(!legs_.empty(), "no legs given");
00145         Date d = CashFlows::startDate(legs_[0]);
00146         for (Size j=1; j<legs_.size(); ++j)
00147             d = std::min(d, CashFlows::startDate(legs_[j]));
00148         return d;
00149     }
