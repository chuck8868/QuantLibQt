00027     SyntheticCDO::SyntheticCDO (const boost::shared_ptr<Basket> basket,
00028                                 Protection::Side side,
00029                                 const Schedule& schedule,
00030                                 Rate upfrontRate,
00031                                 Rate runningRate,
00032                                 const DayCounter& dayCounter,
00033                                 BusinessDayConvention paymentConvention,
00034                                 const Handle<YieldTermStructure>& yieldTS)
00035         : basket_(basket),
00036           side_(side),
00037           schedule_(schedule),
00038           upfrontRate_(upfrontRate),
00039           runningRate_(runningRate),
00040           dayCounter_(dayCounter),
00041           paymentConvention_(paymentConvention),
00042           yieldTS_(yieldTS) {
00043         QL_REQUIRE (basket->names().size() > 0, "basket is empty");
00044 
00045         registerWith (yieldTS_);
00046         const boost::shared_ptr<Pool> pool = basket->pool();
00047 
00048         Date today = Settings::instance().evaluationDate();
00049         // register with probabilities if the corresponding issuer is
00050         // alive under this name contractual conditions
00051         for (Size i = 0; i < basket->names().size(); i++) {
00052             if(!pool->get(basket->names()[i]).
00053                 defaultedBetween(schedule.dates()[0],
00054                                  today,
00055                                  basket->defaultKeys()[i]))
00056             registerWith(pool->get(basket->names()[i]).
00057                 defaultProbability(basket->defaultKeys()[i]));
00058             // To do: the basket could do this last line on its own without so much
00059             //    travelling, update its interface? some RR models depend on the
00060             //    probabilities and they will be registered with them. Strictly
00061             //    speaking the basket does not need directly to be registered
00062             //    with the probs.
00063             /*
00064                 we used to be registering with Issuers which are not observables. However
00065                 this might be what we want: Issuer registration might give us registration
00066                 with probabilities and with creditEvents at the same time.
00067             */
00068         }
00069         // register with recoveries:
00070         registerWith(basket_);
00071     }
