00058     void RiskyAssetSwap::performCalculations() const {
00059         // order of calls is essential
00060         floatAnnuity_   = floatAnnuity();
00061         fixedAnnuity_   = fixedAnnuity();
00062         parCoupon_      = parCoupon();
00063 
00064         if (coupon_ == Null<Rate>())  coupon_ = parCoupon_;
00065 
00066         recoveryValue_  = recoveryValue();
00067         riskyBondPrice_ = riskyBondPrice();
00068 
00069         NPV_ = riskyBondPrice_
00070             - coupon_ * fixedAnnuity_
00071             + yieldTS_->discount (fixedSchedule_.dates().front())
00072             - yieldTS_->discount (fixedSchedule_.dates().back())
00073             + spread_ * floatAnnuity_;
00074 
00075         NPV_ *= nominal_;
00076 
00077         if (fixedPayer_ == false)
00078             NPV_ *= -1;
00079     }

00111     Real RiskyAssetSwap::recoveryValue() const {
00112         Real recoveryValue = 0;
00113         // simple Euler integral to evaluate the recovery value
00114         for (Size i = 1; i < fixedSchedule_.size(); i++) {
00115             TimeUnit stepSize = Days;
00116             Date d;
00117             if (fixedSchedule_[i-1] >= defaultTS_->referenceDate())
00118                 d = fixedSchedule_[i-1];
00119             else
00120                 d = defaultTS_->referenceDate();
00121             Date d0 = d;
00122             do {
00123                 double disc = yieldTS_->discount (d);
00124                 double dd   = defaultTS_->defaultDensity (d, true);
00125                 double dcf  = defaultTS_->dayCounter().yearFraction (d0, d);
00126 
00127                 recoveryValue  += disc * dd * dcf;
00128 
00129                 d0 = d;
00130 
00131                 d = NullCalendar().advance (d0, 1, stepSize, Unadjusted);
00132             }
00133             while (d < fixedSchedule_[i]);
00134         }
00135         recoveryValue *= recoveryRate_;
00136 
00137         return recoveryValue;
00138     }
