00068     void DefaultProbabilityTermStructure::setJumps() {
00069         if (jumpDates_.empty() && !jumps_.empty()) { // turn of year dates
00070             jumpDates_.resize(nJumps_);
00071             jumpTimes_.resize(nJumps_);
00072             Year y = referenceDate().year();
00073             for (Size i=0; i<nJumps_; ++i)
00074                 jumpDates_[i] = Date(31, December, y+i);
00075         } else { // fixed dats
00076             QL_REQUIRE(jumpDates_.size()==nJumps_,
00077                        "mismatch between number of jumps (" << nJumps_ <<
00078                        ") and jump dates (" << jumpDates_.size() << ")");
00079         }
00080         for (Size i=0; i<nJumps_; ++i)
00081             jumpTimes_[i] = timeFromReference(jumpDates_[i]);
00082         latestReference_ = referenceDate();
00083     }

00085     Probability DefaultProbabilityTermStructure::survivalProbability(
00086                                                      Time t,
00087                                                      bool extrapolate) const {
00088         checkRange(t, extrapolate);
00089 
00090         if (!jumps_.empty()) {
00091             Probability jumpEffect = 1.0;
00092             for (Size i=0; i<nJumps_ && jumpTimes_[i]<t; ++i) {
00093                 QL_REQUIRE(jumps_[i]->isValid(),
00094                            "invalid " << io::ordinal(i+1) << " jump quote");
00095                 DiscountFactor thisJump = jumps_[i]->value();
00096                 QL_REQUIRE(thisJump > 0.0 && thisJump <= 1.0,
00097                            "invalid " << io::ordinal(i+1) << " jump value: " <<
00098                            thisJump);
00099                 jumpEffect *= thisJump;
00100             }
00101             return jumpEffect * survivalProbabilityImpl(t);
00102         }
00103 
00104         return survivalProbabilityImpl(t);
00105     }
