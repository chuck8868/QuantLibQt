00052     Real BlackCallableFixedRateBondEngine::spotIncome() const {
00054         Date settlement = arguments_.settlementDate;
00055         Leg cf = arguments_.cashflows;
00056         Date optionMaturity = arguments_.putCallSchedule[0]->date();
00057 
00058         /* the following assumes
00059            1. cashflows are in ascending order !
00060            2. income = coupons paid between settlementDate() and put/call date
00061         */
00062         Real income = 0.0;
00063         for (Size i = 0; i < cf.size() - 1; ++i) {
00064             if (!cf[i]->hasOccurred(settlement, false)) {
00065                 if (cf[i]->hasOccurred(optionMaturity, false)) {
00066                     income += cf[i]->amount() *
00067                               discountCurve_->discount(cf[i]->date());
00068                 } else {
00069                     break;
00070                 }
00071             }
00072         }
00073         return income/discountCurve_->discount(settlement);
00074     }

00113         dayCounter = volatility_->dayCounter();

00158                                                  volatility_->referenceDate(),
00159                                                  exerciseDate);
00160         Real embeddedOptionValue =
00161             blackFormula(type,
00162                          cashStrike,
00163                          fwdCashPrice,
00164                          priceVol*std::sqrt(exerciseTime));

00166         if (type == Option::Call) {
00167             results_.value = npv - embeddedOptionValue;
