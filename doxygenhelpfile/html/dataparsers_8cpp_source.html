00073     Period PeriodParser::parseOnePeriod(const std::string& str) {
00074         QL_REQUIRE(str.length()>1, "single period require a string of at "

00076 
00077         Size iPos = str.find_first_of("DdWwMmYy");

00079                    str.substr(str.length()-1, str.length()) << "' unit");
00080         TimeUnit units = Days;
00081         char abbr = std::toupper(str[iPos]);
00082         if      (abbr == 'D') units = Days;
00083         else if (abbr == 'W') units = Weeks;
00084         else if (abbr == 'M') units = Months;
00085         else if (abbr == 'Y') units = Years;
00086 
00087         Size nPos = str.find_first_of("-+0123456789");
00088         QL_REQUIRE(nPos<iPos, "no numbers of " << units << " provided");
00089         Integer n;
00090         try {
00091             n = io::to_integer(str.substr(nPos,iPos));
00092                 //boost::lexical_cast<Integer>(str.substr(nPos,iPos));
00093         } catch (std::exception& e) {
00094             QL_FAIL("unable to parse the number of units of " << units <<
00095                     " in '" << str << "'. Error:" << e.what());
00096         }
00097 
00098         return Period(n, units);
