00078     BigInteger Business252::Impl::dayCount(const Date& d1,
00079                                            const Date& d2) const {
00080         if (sameMonth(d1,d2) || d1 >= d2) {
00081             // we treat the case of d1 > d2 here, since we'd need a
00082             // second cache to get it right (our cached figures are
00083             // for first included, last excluded and might have to be
00084             // changed going the other way.)
00085             return calendar_.businessDaysBetween(d1, d2);
00086         } else if (sameYear(d1,d2)) {
00087             Cache& cache = monthlyFigures_[calendar_.name()];
00088             BigInteger total = 0;
00089             Date d;
00090             // first, we get to the beginning of next month.
00091             d = Date(1,d1.month(),d1.year()) + 1*Months;
00092             total += calendar_.businessDaysBetween(d1, d);
00093             // then, we add any whole months (whose figures might be
00094             // cached already) in the middle of our period.
00095             while (!sameMonth(d,d2)) {
00096                 total += businessDays(cache, calendar_,
00097                                       d.month(), d.year());
00098                 d += 1*Months;
00099             }
00100             // finally, we get to the end of the period.
00101             total += calendar_.businessDaysBetween(d, d2);
00102             return total;
00103         } else {
00104             Cache& cache = monthlyFigures_[calendar_.name()];
00105             OuterCache& outerCache = yearlyFigures_[calendar_.name()];
00106             BigInteger total = 0;
00107             Date d;
00108             // first, we get to the beginning of next year.
00109             // The first bit gets us to the end of this month...
00110             d = Date(1,d1.month(),d1.year()) + 1*Months;
00111             total += calendar_.businessDaysBetween(d1, d);
00112             // ...then we add any remaining months, possibly cached
00113             for (Integer m = Integer(d1.month())+1; m <= 12; ++m) {
00114                 total += businessDays(cache, calendar_,
00115                                       Month(m), d.year());
00116             }
00117             // then, we add any whole year in the middle of our period.
00118             d = Date(1,January,d1.year()+1);
00119             while (!sameYear(d,d2)) {
00120                 total += businessDays(outerCache, cache,
00121                                       calendar_, d.year());

00123             }
00124             // finally, we get to the end of the period.
00125             // First, we add whole months...
00126             for (Integer m = 1; m<Integer(d2.month()); ++m) {
00127                 total += businessDays(cache, calendar_,
00128                                       Month(m), d2.year());
00129             }
00130             // ...then the last bit.
00131             d = Date(1,d2.month(),d2.year());
00132             total += calendar_.businessDaysBetween(d, d2);
