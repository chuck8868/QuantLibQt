00098     HybridHestonHullWhiteProcess::diffusion(Time t, const Array& x) const {
00099         Matrix retVal(3,3);
00100 
00101         Array xt(2); xt[0] = x[0]; xt[1] = x[1];
00102         Matrix m = hestonProcess_->diffusion(t, xt);
00103         retVal[0][0] = m[0][0]; retVal[0][1] = 0.0;     retVal[0][2] = 0.0;
00104         retVal[1][0] = m[1][0]; retVal[1][1] = m[1][1]; retVal[1][2] = 0.0;
00105         
00106         const Real sigma = hullWhiteProcess_->sigma();
00107         retVal[2][0] = corrEquityShortRate_ * sigma;
00108         retVal[2][1] = - retVal[2][0]*retVal[1][0] / retVal[1][1];
00109         retVal[2][2] = std::sqrt( sigma*sigma - retVal[2][1]*retVal[2][1] 
00110                                               - retVal[2][0]*retVal[2][0] );
00111         
00112         return retVal;

00116     HybridHestonHullWhiteProcess::evolve(Time t0, const Array& x0,
00117                                          Time dt, const Array& dw) const {
00118 
00119         const Rate r         = x0[2];
00120         const Real a         = hullWhiteProcess_->a();
00121         const Real sigma     = hullWhiteProcess_->sigma();
00122         const Real rho       = corrEquityShortRate_;
00123         const Real xi        = hestonProcess_->rho();
00124         const Volatility eta = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00125         const Time s = t0;
00126         const Time t = t0 + dt;
00127         const Time T = T_;
00128         const Rate dy
00129             = hestonProcess_->dividendYield()->forwardRate(s, t, Continuous,
00130                                                            NoFrequency);
00131 
00132         const Real df
00133             = std::log(  hestonProcess_->riskFreeRate()->discount(t)
00134                        / hestonProcess_->riskFreeRate()->discount(s));
00135 
00136         const Real eaT=std::exp(-a*T);
00137         const Real eat=std::exp(-a*t);
00138         const Real eas=std::exp(-a*s);
00139         const Real iat=1.0/eat;
00140         const Real ias=1.0/eas;
00141 
00142         const Real m1 = -(dy+0.5*eta*eta)*dt - df;
00143 
00144         const Real m2 = -rho*sigma*eta/a*(dt-1/a*eaT*(iat-ias));
00145 
00146         const Real m3 = (r - hullWhiteProcess_->alpha(s))
00147                        *hullWhiteProcess_->B(s,t);
00148 
00149         const Real m4 = sigma*sigma/(2*a*a)
00150             *(dt + 2/a*(eat-eas) - 1/(2*a)*(eat*eat-eas*eas));
00151 
00152         const Real m5 = -sigma*sigma/(a*a)
00153             *(dt - 1/a*(1-eat*ias) - 1/(2*a)*eaT*(iat-2*ias+eat*ias*ias));
00154 
00155         const Real mu = m1 + m2 + m3 + m4 + m5;
00156 
00157         Array retVal(3);
00158         
00159         const Real eta2 = hestonProcess_->sigma() * eta;
00160         const Real nu
00161             = hestonProcess_->kappa()*(hestonProcess_->theta() - eta*eta);
00162 
00163         retVal[1] = x0[1] + nu*dt + eta2*std::sqrt(dt)
00164                                           *(xi*dw[0]+std::sqrt(1-xi*xi)*dw[1]);
00165 
00166         if (discretization_ == BSMHullWhite) {
00167             const Real v1 = eta*eta*dt 
00168                 + sigma*sigma/(a*a)*(dt - 2/a*(1 - eat*ias) 
00169                                         + 1/(2*a)*(1 - eat*eat*ias*ias))
00170                 + 2*sigma*eta/a*rho*(dt - 1/a*(1 - eat*ias));
00171             const Real v2 = hullWhiteProcess_->variance(t0, r, dt);
00172             const Real v12 = (1-eat*ias)*(sigma*eta/a*rho + sigma*sigma/(a*a))
00173                             - sigma*sigma/(2*a*a)*(1 - eat*eat*ias*ias);
00174     
00175             QL_REQUIRE(v1 > 0.0 && v2 > 0.0, "zero or negative variance given");
00176             
00177             // terminal rho must be between -maxRho and +maxRho
00178             const Real rhoT 
00179                 = std::min(maxRho_, std::max(-maxRho_, v12/std::sqrt(v1*v2)));
00180             QL_REQUIRE(    rhoT <= 1.0 && rhoT >= -1.0
00181                        && 1-rhoT*rhoT/(1-xi*xi) >= 0.0, 
00182                        "invalid terminal correlation");
00183             
00184             const Real dw_0 =  dw[0];
00185             const Real dw_2 =  rhoT*dw[0]- rhoT*xi/std::sqrt(1-xi*xi)*dw[1] 
00186                              + std::sqrt(1 - rhoT*rhoT/(1-xi*xi))*dw[2];        
00187     
00188             retVal[2] = hullWhiteProcess_->evolve(t0, r, dt, dw_2);
00189     
00190             const Real vol = std::sqrt(v1)*dw_0;
00191             retVal[0] = x0[0]*std::exp(mu + vol);
00192         }
00193         else if (discretization_ == Euler) {
00194             const Real dw_2 =  rho*dw[0]- rho*xi/std::sqrt(1-xi*xi)*dw[1] 
00195                              + std::sqrt(1 - rho*rho/(1-xi*xi))*dw[2];        
00196     
00197             retVal[2] = hullWhiteProcess_->evolve(t0, r, dt, dw_2);
00198     
00199             const Real vol = eta*std::sqrt(dt)*dw[0];
00200             retVal[0] = x0[0]*std::exp(mu + vol);            
00201         }
00202         else
00203             QL_FAIL("unknown discretization scheme");
00204 
00205         return retVal;
00206     }
