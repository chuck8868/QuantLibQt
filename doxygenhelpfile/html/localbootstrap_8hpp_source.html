00135         validCurve_ = false;
00136         Size nInsts = ts_->instruments_.size();
00137 
00138         // ensure rate helpers are sorted
00139         std::sort(ts_->instruments_.begin(), ts_->instruments_.end(),
00140                   detail::BootstrapHelperSorter());
00141 
00142         // check that there is no instruments with the same maturity
00143         for (Size i=1; i<nInsts; ++i) {
00144             Date m1 = ts_->instruments_[i-1]->latestDate(),
00145                  m2 = ts_->instruments_[i]->latestDate();
00146             QL_REQUIRE(m1 != m2,
00147                        "two instruments have the same maturity ("<< m1 <<")");
00148         }
00149 
00150         // check that there is no instruments with invalid quote
00151         for (Size i=0; i<nInsts; ++i)
00152             QL_REQUIRE(ts_->instruments_[i]->quote()->isValid(),
00153                        io::ordinal(i+1) << " instrument (maturity: " <<
00154                        ts_->instruments_[i]->latestDate() <<
00155                        ") has an invalid quote");

00158         for (Size i=0; i<nInsts; ++i) {
00159             // don't try this at home!
00160             // This call creates instruments, and removes "const".
00161             // There is a significant interaction with observability.
00162             ts_->instruments_[i]->setTermStructure(const_cast<Curve*>(ts_));
00163         }
00164         // set initial guess only if the current curve cannot be used as guess
00165         if (validCurve_)
00166             QL_ENSURE(ts_->data_.size() == nInsts+1,
00167                       "dimension mismatch: expected " << nInsts+1 <<
00168                       ", actual " << ts_->data_.size());
00169         else {
00170             ts_->data_ = std::vector<Rate>(nInsts+1);
00171             ts_->data_[0] = Traits::initialValue(ts_);
00172         }
00173 
00174         // calculate dates and times
00175         ts_->dates_ = std::vector<Date>(nInsts+1);
00176         ts_->times_ = std::vector<Time>(nInsts+1);
00177         ts_->dates_[0] = Traits::initialDate(ts_);

00179         for (Size i=0; i<nInsts; ++i) {
00180             ts_->dates_[i+1] = ts_->instruments_[i]->latestDate();
00181             ts_->times_[i+1] = ts_->timeFromReference(ts_->dates_[i+1]);
00182             if (!validCurve_)
00183                 ts_->data_[i+1] = ts_->data_[i];
00184         }
00185 
00186         LevenbergMarquardt solver(ts_->accuracy_,
00187                                   ts_->accuracy_,
00188                                   ts_->accuracy_);
00189         EndCriteria endCriteria(100, 10, 0.00, ts_->accuracy_, 0.00);
00190         PositiveConstraint posConstraint;
00191         NoConstraint noConstraint;
00192         Constraint& solverConstraint = forcePositive_ ?
00193             static_cast<Constraint&>(posConstraint) :
00194             static_cast<Constraint&>(noConstraint);
00195 
00196         // now start the bootstrapping.
00197         Size iInst = localisation_-1;
00198 
00199         Size dataAdjust = Curve::interpolator_type::dataSizeAdjustment;
00200 
00201         do {
00202             Size initialDataPt = iInst+1-localisation_+dataAdjust;
00203             Array startArray(localisation_+1-dataAdjust);
00204             for (Size j = 0; j < startArray.size()-1; ++j)

00208             // time... but the local interpolator can make an
00209             // approximation for the final localisation period.
00210             // e.g. if the localisation is 2, then the first section
00211             // of the curve will be solved using the first 2
00212             // instruments... with the local interpolator making
00213             // suitable boundary conditions.

00216                                               ts_->times_.begin(),
00217                                               ts_->times_.begin()+(iInst + 2),

00219                                               localisation_,
00220                                               ts_->interpolation_,
00221                                               nInsts+1);
00222 
00223             if (iInst >= localisation_) {

00225                     Traits::guess(iInst, ts_, false, 0); // ?
00226             } else {
00227                 startArray[localisation_-dataAdjust] = ts_->data_[0];
00228             }
00229 
00230             PenaltyFunction<Curve> currentCost(
00231                         ts_,
00232                         initialDataPt,
00233                         ts_->instruments_.begin() + (iInst - localisation_+1),
00234                         ts_->instruments_.begin() + (iInst+1));
00235 
00236             Problem toSolve(currentCost, solverConstraint, startArray);
00237 
00238             EndCriteria::Type endType = solver.minimize(toSolve, endCriteria);
00239 
00240             // check the end criteria
