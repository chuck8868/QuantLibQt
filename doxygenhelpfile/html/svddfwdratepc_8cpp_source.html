00037       volProcess_(volProcess),
00038       firstVolatilityFactor_(firstVolatilityFactor),
00039       volatilityFactorStep_(volatilityFactorStep),
00040       volFactorsPerStep_(volProcess->variatesPerStep()),
00041       numeraires_(numeraires),
00042       initialStep_(initialStep),
00043       isVolVariate_(false,volProcess->variatesPerStep()+marketModel_->numberOfFactors()),
00044       numberOfRates_(marketModel->numberOfRates()),
00045       numberOfFactors_(marketModel_->numberOfFactors()),
00046       curveState_(marketModel->evolution().rateTimes()),
00047       forwards_(marketModel->initialRates()),
00048       displacements_(marketModel->displacements()),
00049       logForwards_(numberOfRates_), initialLogForwards_(numberOfRates_),
00050       drifts1_(numberOfRates_), drifts2_(numberOfRates_),
00051       initialDrifts_(numberOfRates_), allBrownians_(volProcess->variatesPerStep()+marketModel_->numberOfFactors()), 
00052       brownians_(numberOfFactors_),

00057         QL_REQUIRE(initialStep ==0, "initial step zero only supported currently. ");
00058         checkCompatibility(marketModel->evolution(), numeraires);
00059 
00060         Size steps = marketModel->evolution().numberOfSteps();
00061 
00062         generator_ = factory.create(numberOfFactors_+volFactorsPerStep_, steps-initialStep_);
00063 
00064         currentStep_ = initialStep_;
00065 
00066         calculators_.reserve(steps);
00067         fixedDrifts_.reserve(steps);
00068         for (Size j=0; j<steps; ++j) 
00069         {
00070             const Matrix& A = marketModel_->pseudoRoot(j);
00071             calculators_.push_back(
00072                 LMMDriftCalculator(A,
00073                                    displacements_,
00074                                    marketModel->evolution().rateTaus(),
00075                                    numeraires[j],
00076                                    alive_[j]));
00077             std::vector<Real> fixed(numberOfRates_);
00078             for (Size k=0; k<numberOfRates_; ++k) 
00079             {
00080                 Real variance =
00081                     std::inner_product(A.row_begin(k), A.row_end(k),
00082                                        A.row_begin(k), 0.0);
00083                 fixed[k] = -0.5*variance;
00084             }
00085             fixedDrifts_.push_back(fixed);
00086         }
00087 
00088         setForwards(marketModel_->initialRates());
00089 
00090         Size variatesPerStep = numberOfFactors_+volFactorsPerStep_;
00091 
00092         firstVolatilityFactor_ = std::min(firstVolatilityFactor_,variatesPerStep - volFactorsPerStep_);
00093 
00094         Size volIncrement = (variatesPerStep - firstVolatilityFactor_)/volFactorsPerStep_;
00095         
00096         for (Size i=0; i < volFactorsPerStep_; ++i)
00097             isVolVariate_[firstVolatilityFactor_+i*volIncrement] = true;

00128     Real SVDDFwdRatePc::advanceStep()
00129     {
00130         // we're going from T1 to T2
00131 
00132         // a) compute drifts D1 at T1;
00133         if (currentStep_ > initialStep_) 
00134         {
00135             calculators_[currentStep_].compute(forwards_, drifts1_);
00136         } 
00137         else 
00138         {
00139             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00140                       drifts1_.begin());
00141         }
00142 
00143         // b) evolve forwards up to T2 using D1;
00144         Real weight = generator_->nextStep(allBrownians_);
00145 
00146         // divide Brownians between vol process and forward process
00147 
00148         for (Size i=0, j=0, k=0; i < allBrownians_.size(); ++i)
00149             if ( isVolVariate_[i])
00150             {
00151                 volBrownians_[j] = allBrownians_[i];
00152                 ++j;
00153             }
00154             else
00155             {
00156                 brownians_[k] = allBrownians_[i];
00157                 ++k;
00158             }
00159 
00160 
00161         // get sd for step
00162 
00163         Real weight2 = volProcess_->nextstep(volBrownians_);
00164         Real sdMultiplier = volProcess_->stepSd();
00165         Real varianceMultiplier = sdMultiplier*sdMultiplier;
00166 
00167         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00168         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00169 
00170         Size alive = alive_[currentStep_];
00171         for (Size i=alive; i<numberOfRates_; ++i) {
00172             logForwards_[i] += varianceMultiplier*(drifts1_[i] + fixedDrift[i]);
00173             logForwards_[i] += sdMultiplier*
00174                 std::inner_product(A.row_begin(i), A.row_end(i),
00175                                    brownians_.begin(), 0.0);
00176             forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
00177         }
00178 
00179         // c) recompute drifts D2 using the predicted forwards;
00180         calculators_[currentStep_].compute(forwards_, drifts2_);
00181 
00182         // d) correct forwards using both drifts
00183         for (Size i=alive; i<numberOfRates_; ++i) {
00184             logForwards_[i] += varianceMultiplier*(drifts2_[i]-drifts1_[i])/2.0;
00185             forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
00186         }
00187 
00188         // e) update curve state
00189         curveState_.setOnForwardRates(forwards_);
00190 
00191         ++currentStep_;
00192 
00193         return weight*weight2;
00194     }
