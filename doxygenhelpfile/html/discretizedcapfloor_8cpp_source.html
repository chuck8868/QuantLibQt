00053     void DiscretizedCapFloor::preAdjustValuesImpl() {
00054         for (Size i=0; i<startTimes_.size(); i++) {

00057                 Time tenor = arguments_.accrualTimes[i];
00058                 DiscretizedDiscountBond bond;
00059                 bond.initialize(method(), end);
00060                 bond.rollback(time_);
00061 
00062                 CapFloor::Type type = arguments_.type;
00063                 Real gearing = arguments_.gearings[i];
00064                 Real nominal = arguments_.nominals[i];
00065 
00066                 if ( (type == CapFloor::Cap) ||
00067                      (type == CapFloor::Collar)) {
00068                     Real accrual = 1.0 + arguments_.capRates[i]*tenor;
00069                     Real strike = 1.0/accrual;
00070                     for (Size j=0; j<values_.size(); j++)
00071                         values_[j] += nominal*accrual*gearing*
00072                             std::max<Real>(strike - bond.values()[j], 0.0);
00073                 }
00074 
00075                 if ( (type == CapFloor::Floor) ||
00076                      (type == CapFloor::Collar)) {
00077                     Real accrual = 1.0 + arguments_.floorRates[i]*tenor;
00078                     Real strike = 1.0/accrual;
00079                     Real mult = (type == CapFloor::Floor)?1.0:-1.0;
00080                     for (Size j=0; j<values_.size(); j++)
00081                         values_[j] += nominal*accrual*mult*gearing*
00082                             std::max<Real>(bond.values()[j] - strike, 0.0);

00084             }
