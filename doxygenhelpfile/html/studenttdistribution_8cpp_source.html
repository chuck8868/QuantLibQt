00026     Real StudentDistribution::operator()(Real x) const {
00027         static GammaFunction G;
00028         Real g1 = std::exp (G.logValue(0.5 * (n_ + 1)));
00029         Real g2 = std::exp (G.logValue(0.5 * n_));
00030 
00031         Real power = std::pow (1. + x*x / n_, 0.5 * (n_ + 1));
00032 
00033         return g1 / (g2 * power * std::sqrt (M_PI * n_));
