00173     void YearOnYearInflationSwap::fetchResults(const PricingEngine::results* r) const {
00174         static const Spread basisPoint = 1.0e-4;
00175 
00176         // copy from VanillaSwap
00177         // works because similarly simple instrument
00178         // that we always expect to be priced with a swap engine
00179 
00180         Swap::fetchResults(r);
00181 
00182         const YearOnYearInflationSwap::results* results =
00183         dynamic_cast<const YearOnYearInflationSwap::results*>(r);
00184         if (results) { // might be a swap engine, so no error is thrown
00185             fairRate_ = results->fairRate;
00186             fairSpread_ = results->fairSpread;
00187         } else {
00188             fairRate_ = Null<Rate>();
00189             fairSpread_ = Null<Spread>();
00190         }
00191 
00192         if (fairRate_ == Null<Rate>()) {
00193             // calculate it from other results
00194             if (legBPS_[0] != Null<Real>())
00195                 fairRate_ = fixedRate_ - NPV_/(legBPS_[0]/basisPoint);
00196         }
00197         if (fairSpread_ == Null<Spread>()) {
00198             // ditto
00199             if (legBPS_[1] != Null<Real>())
00200                 fairSpread_ = spread_ - NPV_/(legBPS_[1]/basisPoint);
00201         }
00202
