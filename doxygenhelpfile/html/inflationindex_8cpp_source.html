00081                                     bool /*forecastTodaysFixing*/) const {
00082         if (!needsForecast(aFixingDate)) {
00083             const TimeSeries<Real>& ts = timeSeries();
00084             Real pastFixing = ts[aFixingDate];
00085             QL_REQUIRE(pastFixing != Null<Real>(),
00086                        "Missing " << name() << " fixing for " << aFixingDate);
00087             Real theFixing = pastFixing;
00088             if (interpolated_) {
00089                 // fixings stored flat & for every day
00090                 std::pair<Date,Date> lim =
00091                     inflationPeriod(aFixingDate, frequency_);
00092                 if (aFixingDate == lim.first) {
00093                     // we don't actually need the next fixing
00094                     theFixing = pastFixing;
00095                 } else {
00096                     Date fixingDate2 = aFixingDate + Period(frequency_);
00097                     Real pastFixing2 = ts[fixingDate2];
00098                     QL_REQUIRE(pastFixing2 != Null<Real>(),
00099                                "Missing " << name() << " fixing for " << fixingDate2);
00100                     // now linearly interpolate
00101                     Real daysInPeriod = lim.second+1 - lim.first;
00102                     theFixing = pastFixing
00103                         + (pastFixing2-pastFixing)*(aFixingDate-lim.first)/daysInPeriod;
00104                 }
00105             }
00106             return theFixing;
00107         } else {
00108             return forecastFixing(aFixingDate);

00113     bool ZeroInflationIndex::needsForecast(const Date& fixingDate) const {
00114 
00115         // Stored fixings are always non-interpolated.
00116         // If an interpolated fixing is required then
00117         // the availability lag + one inflation period
00118         // must have passed to use historical fixings
00119         // (because you need the next one to interpolate).
00120         // The interpolation is calculated (linearly) on demand.
00121 
00122         Date today = Settings::instance().evaluationDate();
00123         Date todayMinusLag = today - availabilityLag_;
00124 
00125         Date historicalFixingKnown =
00126             inflationPeriod(todayMinusLag, frequency_).first-1;
00127         Date latestNeededDate = fixingDate;
00128 
00129         if (interpolated_) { // might need the next one too
00130             std::pair<Date,Date> p = inflationPeriod(fixingDate, frequency_);
00131             if (fixingDate > p.first)
00132                 latestNeededDate += Period(frequency_);
00133         }
00134 
00135         if (latestNeededDate <= historicalFixingKnown) {
00136             // the fixing date is well before the availability lag, so
00137             // we know that fixings were provided.
00138             return false;
00139         } else {
00140             // we're not sure, but the fixing might be there so we
00141             // check.  Todo: check which fixings are not possible, to
00142             // avoid using fixings in the future
00143             Real f = timeSeries()[latestNeededDate];
00144             return (f == Null<Real>());
00145         }
00146     }

00149     Rate ZeroInflationIndex::forecastFixing(const Date& fixingDate) const {
00150         // the term structure is relative to the fixing value at the base date.
00151         Date baseDate = zeroInflation_->baseDate();
00152         QL_REQUIRE(!needsForecast(baseDate),
00153                    name() << " index fixing at base date is not available");
00154         Real baseFixing = fixing(baseDate);
00155         Date effectiveFixingDate;
00156         if (interpolated()) {
00157             effectiveFixingDate = fixingDate;
00158         } else {
00159             // start of period is the convention
00160             // so it's easier to do linear interpolation on fixings
00161             effectiveFixingDate = inflationPeriod(fixingDate, frequency()).first;
00162         }
00163 
00164         // no observation lag because it is the fixing for the date
00165         // but if index is not interpolated then that fixing is constant
00166         // for each period, hence the t uses the effectiveFixingDate
00167         // However, it's slightly safe to get the zeroRate with the
00168         // fixingDate to avoid potential problems at the edges of periods
00169         Time t = zeroInflation_->dayCounter().yearFraction(baseDate, effectiveFixingDate);
00170         bool forceLinearInterpolation = false;
00171         Rate zero = zeroInflation_->zeroRate(fixingDate, Period(0,Days), forceLinearInterpolation);
00172         // Annual compounding is the convention for zero inflation rates (or quotes)
00173         return baseFixing * std::pow(1.0 + zero, t);
00174     }
