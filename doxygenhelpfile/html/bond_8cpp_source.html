00113     Real Bond::notional(Date d) const {
00114         if (d == Date())
00115             d = settlementDate();
00116 
00117         if (d > notionalSchedule_.back()) {
00118             // after maturity
00119             return 0.0;
00120         }
00121 
00122         // After the check above, d is between the schedule
00123         // boundaries.  We search starting from the second notional
00124         // date, since the first is null.  After the call to
00125         // lower_bound, *i is the earliest date which is greater or
00126         // equal than d.  Its index is greater or equal to 1.
00127         std::vector<Date>::const_iterator i =
00128             std::lower_bound(notionalSchedule_.begin()+1,
00129                              notionalSchedule_.end(), d);
00130         Size index = std::distance(notionalSchedule_.begin(), i);
00131 
00132         if (d < notionalSchedule_[index]) {
00133             // no doubt about what to return
00134             return notionals_[index-1];
00135         } else {
00136             // d is equal to a redemption date.
00137             // As per bond conventions, the payment has occurred;
00138             // the bond already changed notional.
00139             return notionals_[index];
00140         }
00141     }

00300     void Bond::addRedemptionsToCashflows(const std::vector<Real>& redemptions) {
00301         // First, we gather the notional information from the cashflows
00302         calculateNotionalsFromCashflows();
00303         // Then, we create the redemptions based on the notional
00304         // information and we add them to the cashflows vector after
00305         // the coupons.
00306         redemptions_.clear();
00307         for (Size i=1; i<notionalSchedule_.size(); ++i) {
00308             Real R = i < redemptions.size() ? redemptions[i] :
00309                      !redemptions.empty()   ? redemptions.back() :
00310                                               100.0;
00311             Real amount = (R/100.0)*(notionals_[i-1]-notionals_[i]);
00312             boost::shared_ptr<CashFlow> payment;
00313             if (i < notionalSchedule_.size()-1)
00314                 payment.reset(new AmortizingPayment(amount,
00315                                                     notionalSchedule_[i]));
00316             else
00317                 payment.reset(new Redemption(amount, notionalSchedule_[i]));
00318             cashflows_.push_back(payment);
00319             redemptions_.push_back(payment);
00320         }
00321         // stable_sort now moves the redemptions to the right places
00322         // while ensuring that they follow coupons with the same date.
00323         std::stable_sort(cashflows_.begin(), cashflows_.end(),
00324                          earlier_than<boost::shared_ptr<CashFlow> >());
00325     }

00353     void Bond::calculateNotionalsFromCashflows() {
00354         notionalSchedule_.clear();
00355         notionals_.clear();
00356 
00357         Date lastPaymentDate = Date();
00358         notionalSchedule_.push_back(Date());
00359         for (Size i=0; i<cashflows_.size(); ++i) {
00360             boost::shared_ptr<Coupon> coupon =
00361                 boost::dynamic_pointer_cast<Coupon>(cashflows_[i]);
00362             if (!coupon)
00363                 continue;
00364 
00365             Real notional = coupon->nominal();
00366             // we add the notional only if it is the first one...
00367             if (notionals_.empty()) {
00368                 notionals_.push_back(coupon->nominal());
00369                 lastPaymentDate = coupon->date();
00370             } else if (!close(notional, notionals_.back())) {
00371                 // ...or if it has changed.
00372                 QL_REQUIRE(notional < notionals_.back(),
00373                            "increasing coupon notionals");
00374                 notionals_.push_back(coupon->nominal());
00375                 // in this case, we also add the last valid date for
00376                 // the previous one...
00377                 notionalSchedule_.push_back(lastPaymentDate);
00378                 // ...and store the candidate for this one.
00379                 lastPaymentDate = coupon->date();
00380             } else {
00381                 // otherwise, we just extend the valid range of dates
00382                 // for the current notional.
00383                 lastPaymentDate = coupon->date();
00384             }
00385         }
00386         QL_REQUIRE(!notionals_.empty(), "no coupons provided");
00387         notionals_.push_back(0.0);
00388         notionalSchedule_.push_back(lastPaymentDate);
00389     }
