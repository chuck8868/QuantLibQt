00045     Real blackFormula(Option::Type optionType,
00046                       Real strike,
00047                       Real forward,
00048                       Real stdDev,
00049                       Real discount,
00050                       Real displacement)
00051     {
00052         checkParameters(strike, forward, displacement);
00053         QL_REQUIRE(stdDev>=0.0,
00054                    "stdDev (" << stdDev << ") must be non-negative");
00055         QL_REQUIRE(discount>0.0,
00056                    "discount (" << discount << ") must be positive");
00057 
00058         if (stdDev==0.0)
00059             return std::max((forward-strike)*optionType, Real(0.0))*discount;
00060 
00061         forward = forward + displacement;
00062         strike = strike + displacement;
00063 
00064         // since displacement is non-negative strike==0 iff displacement==0
00065         // so returning forward*discount is OK
00066         if (strike==0.0)
00067             return (optionType==Option::Call ? forward*discount : 0.0);
00068 
00069         Real d1 = std::log(forward/strike)/stdDev + 0.5*stdDev;
00070         Real d2 = d1 - stdDev;
00071         CumulativeNormalDistribution phi;
00072         Real nd1 = phi(optionType*d1);
00073         Real nd2 = phi(optionType*d2);
00074         Real result = discount * optionType * (forward*nd1 - strike*nd2);
00075         QL_ENSURE(result>=0.0,
00076                   "negative value (" << result << ") for " <<
00077                   stdDev << " stdDev, " <<
00078                   optionType << " option, " <<
00079                   strike << " strike , " <<
00080                   forward << " forward");
00081         return result;
00082     }

00093     Real blackFormulaImpliedStdDevApproximation(Option::Type optionType,
00094                                                 Real strike,
00095                                                 Real forward,
00096                                                 Real blackPrice,
00097                                                 Real discount,
00098                                                 Real displacement)
00099     {
00100         checkParameters(strike, forward, displacement);
00101         QL_REQUIRE(blackPrice>=0.0,
00102                    "blackPrice (" << blackPrice << ") must be non-negative");
00103         QL_REQUIRE(discount>0.0,
00104                    "discount (" << discount << ") must be positive");
00105 
00106         Real stdDev;
00107         forward = forward + displacement;
00108         strike = strike + displacement;
00109         if (strike==forward)
00110             // Brenner-Subrahmanyan (1988) and Feinstein (1988) ATM approx.
00111             stdDev = blackPrice/discount*std::sqrt(2.0 * M_PI)/forward;
00112         else {
00113             // Corrado and Miller extended moneyness approximation
00114             Real moneynessDelta = optionType*(forward-strike);
00115             Real moneynessDelta_2 = moneynessDelta/2.0;
00116             Real temp = blackPrice/discount - moneynessDelta_2;
00117             Real moneynessDelta_PI = moneynessDelta*moneynessDelta/M_PI;
00118             Real temp2 = temp*temp-moneynessDelta_PI;
00119             if (temp2<0.0) // approximation breaks down, 2 alternatives:
00120                 // 1. zero it
00121                 temp2=0.0;
00122                 // 2. Manaster-Koehler (1982) efficient Newton-Raphson seed
00123                 //return std::fabs(std::log(forward/strike))*std::sqrt(2.0);
00124             temp2 = std::sqrt(temp2);
00125             temp += temp2;
00126             temp *= std::sqrt(2.0 * M_PI);
00127             stdDev = temp/(forward+strike);
00128         }
00129         QL_ENSURE(stdDev>=0.0,
00130                   "stdDev (" << stdDev << ") must be non-negative");
00131         return stdDev;
00132     }

00145     class BlackImpliedStdDevHelper {
00146       public:
00147         BlackImpliedStdDevHelper(Option::Type optionType,
00148                                  Real strike,
00149                                  Real forward,
00150                                  Real undiscountedBlackPrice,
00151                                  Real displacement = 0.0)
00152         : halfOptionType_(0.5*optionType), signedStrike_(optionType*(strike+displacement)),
00153           signedForward_(optionType*(forward+displacement)),
00154           undiscountedBlackPrice_(undiscountedBlackPrice)
00155         {
00156             checkParameters(strike, forward, displacement);
00157             QL_REQUIRE(undiscountedBlackPrice>=0.0,
00158                        "undiscounted Black price (" <<
00159                        undiscountedBlackPrice << ") must be non-negative");
00160             signedMoneyness_ = optionType*std::log((forward+displacement)/(strike+displacement));
00161         }
00162         Real operator()(Real stdDev) const {
00163             #if defined(QL_EXTRA_SAFETY_CHECKS)
00164             QL_REQUIRE(stdDev>=0.0,
00165                        "stdDev (" << stdDev << ") must be non-negative");
00166             #endif
00167             if (stdDev==0.0)
00168                 return std::max(signedForward_-signedStrike_, Real(0.0))
00169                                                    - undiscountedBlackPrice_;
00170             Real temp = halfOptionType_*stdDev;
00171             Real d = signedMoneyness_/stdDev;
00172             Real signedD1 = d + temp;
00173             Real signedD2 = d - temp;
00174             Real result = signedForward_ * N_(signedD1)
00175                 - signedStrike_ * N_(signedD2);
00176             // numerical inaccuracies can yield a negative answer
00177             return std::max(Real(0.0), result) - undiscountedBlackPrice_;
00178         }
00179         Real derivative(Real stdDev) const {
00180             #if defined(QL_EXTRA_SAFETY_CHECKS)
00181             QL_REQUIRE(stdDev>=0.0,
00182                        "stdDev (" << stdDev << ") must be non-negative");
00183             #endif
00184             Real signedD1 = signedMoneyness_/stdDev + halfOptionType_*stdDev;
00185             return signedForward_*N_.derivative(signedD1);
00186         }
00187       private:
00188         Real halfOptionType_;
00189         Real signedStrike_, signedForward_;
00190         Real undiscountedBlackPrice_, signedMoneyness_;
00191         CumulativeNormalDistribution N_;
00192     };

00195     Real blackFormulaImpliedStdDev(Option::Type optionType,
00196                                    Real strike,
00197                                    Real forward,
00198                                    Real blackPrice,
00199                                    Real discount,
00200                                    Real displacement,
00201                                    Real guess,
00202                                    Real accuracy,
00203                                    Natural maxIterations)
00204     {
00205         checkParameters(strike, forward, displacement);
00206 
00207         QL_REQUIRE(discount>0.0,
00208                    "discount (" << discount << ") must be positive");
00209 
00210         QL_REQUIRE(blackPrice>=0.0,
00211                    "option price (" << blackPrice << ") must be non-negative");
00212         // check the price of the "other" option implied by put-call paity
00213         Real otherOptionPrice = blackPrice - optionType*(forward-strike)*discount;
00214         QL_REQUIRE(otherOptionPrice>=0.0,
00215                    "negative " << Option::Type(-1*optionType) <<
00216                    " price (" << otherOptionPrice <<
00217                    ") implied by put-call parity. No solution exists for " <<
00218                    optionType << " strike " << strike <<
00219                    ", forward " << forward <<
00220                    ", price " << blackPrice <<
00221                    ", deflator " << discount);
00222 
00223         // solve for the out-of-the-money option which has
00224         // greater vega/price ratio, i.e.
00225         // it is numerically more robust for implied vol calculations
00226         if (optionType==Option::Put && strike>forward) {
00227             optionType = Option::Call;
00228             blackPrice = otherOptionPrice;
00229         }
00230         if (optionType==Option::Call && strike<forward) {
00231             optionType = Option::Put;
00232             blackPrice = otherOptionPrice;
00233         }
00234 
00235         strike = strike + displacement;
00236         forward = forward + displacement;
00237 
00238         if (guess==Null<Real>())
00239             guess = blackFormulaImpliedStdDevApproximation(
00240                 optionType, strike, forward, blackPrice, discount, displacement);
00241         else
00242             QL_REQUIRE(guess>=0.0,
00243                        "stdDev guess (" << guess << ") must be non-negative");
00244         BlackImpliedStdDevHelper f(optionType, strike, forward,
00245                                    blackPrice/discount);
00246         NewtonSafe solver;
00247         solver.setMaxEvaluations(maxIterations);
00248         Real minSdtDev = 0.0, maxStdDev = 24.0; // 24 = 300% * sqrt(60)
00249         Real stdDev = solver.solve(f, accuracy, guess, minSdtDev, maxStdDev);
00250         QL_ENSURE(stdDev>=0.0,
00251                   "stdDev (" << stdDev << ") must be non-negative");
00252         return stdDev;
00253     }

00271                                         Real stdDev,
00272                                         Real displacement) {

00349     Real bachelierBlackFormula(Option::Type optionType,
00350                                Real strike,
00351                                Real forward,
00352                                Real stdDev,
00353                                Real discount)
00354     {
00355         QL_REQUIRE(stdDev>=0.0,
00356                    "stdDev (" << stdDev << ") must be non-negative");
00357         QL_REQUIRE(discount>0.0,
00358                    "discount (" << discount << ") must be positive");
00359         Real d = (forward-strike)*optionType, h = d/stdDev;
00360         if (stdDev==0.0)
00361             return discount*std::max(d, 0.0);
00362         CumulativeNormalDistribution phi;
00363         Real result = discount*(stdDev*phi.derivative(h) + d*phi(h));
00364         QL_ENSURE(result>=0.0,
00365                   "negative value (" << result << ") for " <<
00366                   stdDev << " stdDev, " <<
00367                   optionType << " option, " <<
00368                   strike << " strike , " <<
00369                   forward << " forward");
