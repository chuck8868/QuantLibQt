00098         dy = ((std::fabs(y) > 0.001) ? y*0.0001 : 0.000001);
00099         strikep=strike*std::exp(dy);
00100         strikem=strike/std::exp(dy);
00101         w  = blackTS_->blackVariance(t, strike,  true);
00102         wp = blackTS_->blackVariance(t, strikep, true);
00103         wm = blackTS_->blackVariance(t, strikem, true);
00104         dwdy = (wp-wm)/(2.0*dy);
00105         d2wdy2 = (wp-2.0*w+wm)/(dy*dy);
00106 
00107         // time derivative
00108         Real dt, wpt, wmt, dwdt;
00109         if (t==0.0) {
00110             dt = 0.0001;
00111             DiscountFactor drpt = riskFreeTS_->discount(t+dt, true);
00112             DiscountFactor dqpt = dividendTS_->discount(t+dt, true);           
00113             Real strikept = strike*dr*dqpt/(drpt*dq);

00117                       "decreasing variance at strike " << strike
00118                       << " between time " << t << " and time " << t+dt);

00120         } else {

00122             DiscountFactor drpt = riskFreeTS_->discount(t+dt, true);
00123             DiscountFactor drmt = riskFreeTS_->discount(t-dt, true);

00135                       << " between time " << t << " and time " << t+dt);

00137                       "decreasing variance at strike " << strike
00138                       << " between time " << t-dt << " and time " << t);
