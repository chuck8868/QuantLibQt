00383                 Real price = floorPrice_(t,fStrikes_[j]);

00385                 if (minSwapRate > tmpMinSwapRateIntersection)
00386                     tmpMinSwapRateIntersection = minSwapRate;
00387             }
00388             for (Size j=0; j<cStrikes_.size(); ++j) {
00389                 Real price = capPrice_(t,cStrikes_[j]);
00390                 Real maxSwapRate = cStrikes_[j] + price / (sumDiscount * 10000);
00391                 if (maxSwapRate < tmpMaxSwapRateIntersection)
00392                     tmpMaxSwapRateIntersection = maxSwapRate;
00393             }
