00218         Real alpha = dj/(std::sqrt(8.0));
00219         Real alpha2 = alpha*alpha;
00220         Real alpha3 = alpha*alpha2;

00222         Real alpha7 = alpha5*alpha2;
00223         Real beta = -0.375*alpha-alpha3;

00227             -1.43 *alpha5 -0.5 *alpha7;
00228         Real p =0.5;
00229         Real rootk= std::sqrt(k);
00230         p+= alpha/rootk;
00231         p+= beta /(k*rootk);
00232         p+= gamma/(k*k*rootk);
00233         // delete next line to get results for j three tree
00234         p+= delta/(k*k*k*rootk);
