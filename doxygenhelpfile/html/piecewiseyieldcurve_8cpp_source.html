00846                        const I& interpolator = I()) {
00847 
00848         PiecewiseYieldCurve<T,I> curve(vars.settlement, vars.instruments,
00849                                        Actual360(),

00851                                        interpolator);
00852         // necessary to trigger bootstrap
00853         curve.recalculate();
00854 
00855         typedef typename T::template curve<I>::type base_curve;
00856 
00857         base_curve copiedCurve = curve;
00858 
00859         // the two curves should be the same.
00860         Time t = 2.718;
00861         Rate r1 = curve.zeroRate(t, Continuous);
00862         Rate r2 = copiedCurve.zeroRate(t, Continuous);
00863         if (!close(r1, r2)) {
00864             BOOST_ERROR("failed to link original and copied curve");
00865         }
00866 
00867         for (Size i=0; i<vars.rates.size(); ++i) {
00868             vars.rates[i]->setValue(vars.rates[i]->value() + 0.001);
00869         }
00870 
00871         // now the original curve should have changed; the copied
00872         // curve should not.
00873         Rate r3 = curve.zeroRate(t, Continuous);
00874         Rate r4 = copiedCurve.zeroRate(t, Continuous);
00875         if (close(r1, r3)) {
00876             BOOST_ERROR("failed to modify original curve");
00877         }
00878         if (!close(r2,r4)) {
00879             BOOST_ERROR(
00880                     "failed to break link between original and copied curve");
00881         }
