00031     GaussianQuadrature::GaussianQuadrature(
00032                                 Size n,
00033                                 const GaussianOrthogonalPolynomial& orthPoly)
00034     : x_(n), w_(n) {
00035 
00036         // set-up matrix to compute the roots and the weights
00037         Array e(n-1);
00038 
00039         Size i;
00040         for (i=1; i < n; ++i) {
00041             x_[i] = orthPoly.alpha(i);
00042             e[i-1] = std::sqrt(orthPoly.beta(i));
00043         }
00044         x_[0] = orthPoly.alpha(0);
00045 
00046         TqrEigenDecomposition tqr(
00047                                x_, e,
00048                                TqrEigenDecomposition::OnlyFirstRowEigenVector,
00049                                TqrEigenDecomposition::Overrelaxation);
00050 
00051         x_ = tqr.eigenvalues();
00052         const Matrix& ev = tqr.eigenvectors();
00053 
00054         Real mu_0 = orthPoly.mu_0();
00055         for (i=0; i<n; ++i) {
00056             w_[i] = mu_0*ev[0][i]*ev[0][i] / orthPoly.w(x_[i]);
00057         }
00058     }

00061     void TabulatedGaussLegendre::order(Size order) {
00062         switch(order) {
00063           case(6):
00064             order_=order; x_=x6; w_=w6; n_=n6;
00065             break;
00066           case(7):
00067             order_=order; x_=x7; w_=w7; n_=n7;
00068             break;
00069           case(12):
00070             order_=order; x_=x12; w_=w12; n_=n12;
00071             break;
00072           case(20):
00073             order_=order; x_=x20; w_=w20; n_=n20;
00074             break;
00075           default:
00076             QL_FAIL("order " << order << " not supported");
00077         }
00078     }
