00285     std::vector<Size> makeMeasure(const MarketModelMultiProduct& product,
00286                                   MeasureType measureType) {
00287         std::vector<Size> result;
00288         EvolutionDescription evolution(product.evolution());
00289         switch (measureType) {
00290           case ProductSuggested:
00291             result = product.suggestedNumeraires();
00292             break;
00293           case Terminal:
00294             result = terminalMeasure(evolution);
00295             if (!isInTerminalMeasure(evolution, result)) {
00296                 Array a(result.size());
00297                 std::copy(result.begin(), result.end(), a.begin());
00298                 BOOST_ERROR("failure in verifying Terminal measure:\n" << a);
00299             }
00300             break;
00301           case MoneyMarket:
00302             result = moneyMarketMeasure(evolution);
00303             if (!isInMoneyMarketMeasure(evolution, result)) {
00304                 Array a(result.size());
00305                 std::copy(result.begin(), result.end(), a.begin());
00306                 BOOST_ERROR("failure in verifying MoneyMarket measure:\n" << a);
00307             }
00308             break;
00309           case MoneyMarketPlus:
00310             result = moneyMarketPlusMeasure(evolution, measureOffset_);
00311             if (!isInMoneyMarketPlusMeasure(evolution, result,
00312                                             measureOffset_)) {
00313                 Array a(result.size());
00314                 std::copy(result.begin(), result.end(), a.begin());
00315                 BOOST_ERROR("failure in verifying MoneyMarketPlus(" <<
00316                             measureOffset_ << ") measure:\n" << a);
00317             }
00318             break;
00319           default:
00320             QL_FAIL("unknown measure type");
00321         }
00322         checkCompatibility(evolution, result);
00323         if (printReport_) {
00324             Array num(result.size());
00325             std::copy(result.begin(), result.end(), num.begin());
00326             BOOST_MESSAGE("    " << measureTypeToString(measureType) << ": "
00327                           << num);
00328         }
00329         return result;
00330     }

00366     void checkCMSAndSwaptions(
00367               const SequenceStatisticsInc& stats,
00368               const Rate fixedRate,
00369               const std::vector<boost::shared_ptr<StrikedTypePayoff> >&
00370                                                               displacedPayoff,
00371               const boost::shared_ptr<MarketModel>, // marketModel,
00372               const std::string& config) {
00373         std::vector<Real> results = stats.mean();
00374         std::vector<Real> errors = stats.errorEstimate();
00375         std::vector<Real> discrepancies(todaysForwards.size());
00376 
00377         Size N = todaysForwards.size();
00378         // check Swaps
00379         Real maxError = QL_MIN_REAL;
00380         LMMCurveState curveState_lmm(rateTimes);
00381         curveState_lmm.setOnForwardRates(todaysForwards);
00382 
00383         std::vector<Real> expectedNPVs(todaysCMSwapRates.size());
00384         Real errorThreshold = 0.5;
00385         for (Size i=0; i<N; ++i) {
00386             Real expectedNPV = curveState_lmm.cmSwapAnnuity(i, i, spanningForwards)
00387                 * (todaysCMSwapRates[i]-fixedRate) * todaysDiscounts[i];
00388             expectedNPVs[i] = expectedNPV;
00389             discrepancies[i] = (results[i]-expectedNPVs[i])/errors[i];
00390             maxError = std::max(std::fabs(discrepancies[i]), maxError);
00391         }
00392 
00393         if (maxError > errorThreshold) {
00394             BOOST_MESSAGE(config);
00395             for (Size i=0; i<N; ++i) {
00396                 BOOST_MESSAGE(io::ordinal(i+1) << " CMS NPV: "
00397                               << io::rate(results[i])
00398                               << " +- " << io::rate(errors[i])
00399                               << "; expected: " << io::rate(expectedNPVs[i])
00400                               << "; discrepancy/error = "
00401                               << discrepancies[N-1-i]
00402                               << " standard errors");
00403             }
00404             BOOST_ERROR("test failed");
00405         }
00406 
00407         // check Swaptions
00408         maxError = 0;
00409 
00410         std::vector<Rate> expectedSwaptions(N);
00411         std::vector<Real> stdDevSwaptions(N);
00412         for (Size i=0; i<N; ++i) {
00413             Real expectedSwaption =
00414                 BlackCalculator(displacedPayoff[i],
00415                                 todaysCMSwapRates[i]+displacement,
00416                                 volatilities[i]*std::sqrt(rateTimes[i]),
00417                                 curveState_lmm.cmSwapAnnuity(i,i, spanningForwards)
00418                                 * todaysDiscounts[i]).value();
00419             expectedSwaptions[i] = expectedSwaption;
00420             discrepancies[i] = (results[N+i]-expectedSwaptions[i])/errors[N+i];
00421             maxError = std::max(std::fabs(discrepancies[i]), maxError);
00422         }
00423         errorThreshold = 2.0;
00424 
00425         if (maxError > errorThreshold) {
00426             BOOST_MESSAGE(config);
00427             for (Size i=1; i<=N; ++i) {
00428                 BOOST_MESSAGE(io::ordinal(i) << " Swaption: "
00429                               << io::rate(results[2*N-i])
00430                               << " +- " << io::rate(errors[2*N-i])
00431                               << "; expected: " << io::rate(expectedSwaptions[N-i])
00432                               << "; discrepancy/error = "
00433                               << io::percent(discrepancies[N-i])
00434                               << " standard errors");
00435             }
00436             BOOST_ERROR("test failed");
00437         }
