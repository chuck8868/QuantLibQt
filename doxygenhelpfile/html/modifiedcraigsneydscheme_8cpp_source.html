00036         QL_REQUIRE(t-dt_ > -1e-8, "a step towards negative time given");
00037         map_->setTime(std::max(0.0, t-dt_), t);
00038         bcSet_.setTime(std::max(0.0, t-dt_));
00039 
00040         bcSet_.applyBeforeApplying(*map_);
00041         Array y = a + dt_*map_->apply(a);
00042         bcSet_.applyAfterApplying(y);
00043 
00044         Array y0 = y;
00045 
00046         for (Size i=0; i < map_->size(); ++i) {
00047             Array rhs = y - theta_*dt_*map_->apply_direction(i, a);
00048             y = map_->solve_splitting(i, rhs, -theta_*dt_);
00049         }
00050 
00051         bcSet_.applyBeforeApplying(*map_);

00053                   +(0.5-mu_)*dt_*map_->apply(y-a);;
00054         bcSet_.applyAfterApplying(yt);
00055 
00056         for (Size i=0; i < map_->size(); ++i) {
00057             Array rhs = yt - theta_*dt_*map_->apply_direction(i, a);
00058             yt = map_->solve_splitting(i, rhs, -theta_*dt_);
00059         }
00060         bcSet_.applyAfterSolving(yt);

00062         a = yt;
