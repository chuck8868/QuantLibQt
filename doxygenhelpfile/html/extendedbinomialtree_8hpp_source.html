00075         : public ExtendedBinomialTree<T> {
00076       public:
00077         ExtendedEqualProbabilitiesBinomialTree(
00078                         const boost::shared_ptr<StochasticProcess1D>& process,
00079                         Time end,
00080                         Size steps)
00081         : ExtendedBinomialTree<T>(process, end, steps) {}
00082         virtual ~ExtendedEqualProbabilitiesBinomialTree() {}
00083 
00084         Real underlying(Size i, Size index) const {
00085             Time stepTime = i*this->dt_;
00086             BigInteger j = 2*BigInteger(index) - BigInteger(i);
00087             // exploiting the forward value tree centering
00088             return this->x0_*std::exp(i*this->driftStep(stepTime) + j*this->upStep(stepTime));
00089         }

00092       protected:
