00038                 const std::vector<Date>& fixingDates = coupon_->fixingDates();
00039                 const boost::shared_ptr<InterestRateIndex>& index =
00040                     coupon_->index();
00041 
00042                 Natural cutoffDays = 0; // to be verified
00043                 Date startDate = coupon_->accrualStartDate() - cutoffDays,
00044                      endDate = coupon_->accrualEndDate() - cutoffDays,
00045                      d1 = startDate,
00046                      d2 = startDate;
00047

00050                             "first fixing date valid after period start");
00051                 QL_REQUIRE (index->valueDate(fixingDates.back()) >= endDate,
00052                             "last fixing date valid before period end");
00053 
00054                 Rate avgBMA = 0.0;
00055                 Integer days = 0;
00056                 for (Size i=0; i<fixingDates.size() - 1; ++i) {
00057                     Date valueDate = index->valueDate(fixingDates[i]);
00058                     Date nextValueDate = index->valueDate(fixingDates[i+1]);
00059 
00060                     if (fixingDates[i] >= endDate || valueDate >= endDate)
00061                         break;
00062                     if (fixingDates[i+1] < startDate
00063                         || nextValueDate <= startDate)
00064                         continue;
00065 
00066                     d2 = std::min(nextValueDate, endDate);
00067 
00068                     avgBMA += index->fixing(fixingDates[i]) * (d2 - d1);
00069 
00070                     days += d2 - d1;
00071                     d1 = d2;
00072                 }
00073                 avgBMA /= (endDate - startDate);
00074 
00075                 QL_ENSURE(days == endDate - startDate,
00076                           "averaging days " << days << " differ from "
00077                           "interest days " << (endDate - startDate));
00078 
00079                 return coupon_->gearing()*avgBMA + coupon_->spread();
