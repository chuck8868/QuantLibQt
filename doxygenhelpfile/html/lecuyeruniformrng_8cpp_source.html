00042     LecuyerUniformRng::LecuyerUniformRng(long seed)
00043     : buffer(LecuyerUniformRng::bufferSize) {
00044         // Need to prevent seed=0, so use seed=0 to have a "random" seed
00045         temp2 = temp1 = (seed != 0 ? seed : SeedGenerator::instance().get());
00046         // Load the shuffle table (after 8 warm-ups)
00047         for (int j=bufferSize+7; j>=0; j--) {
00048             long k = temp1/q1;
00049             temp1 = a1*(temp1-k*q1)-k*r1;
00050             if (temp1 < 0)
00051                 temp1 += m1;
00052             if (j < bufferSize)
00053                 buffer[j] = temp1;
00054         }
00055         y = buffer[0];
00056     }

00058     LecuyerUniformRng::sample_type LecuyerUniformRng::next() const {
00059         long k = temp1/q1;
00060         // Compute temp1=(a1*temp1) % m1
00061         // without overflows (Schrage's method)
00062         temp1 = a1*(temp1-k*q1)-k*r1;
00063         if (temp1 < 0)
00064             temp1 += m1;
00065         k = temp2/q2;
00066         // Compute temp2=(a2*temp2) % m2
00067         // without overflows (Schrage's method)
00068         temp2 = a2*(temp2-k*q2)-k*r2;
00069         if (temp2 < 0)
00070             temp2 += m2;
00071         // Will be in the range 0..bufferSize-1
00072         int j = y/bufferNormalizer;
00073         // Here temp1 is shuffled, temp1 and temp2 are
00074         // combined to generate output
00075         y = buffer[j]-temp2;
00076         buffer[j] = temp1;
00077         if (y < 1)
00078             y += m1-1;
00079         double result = y/double(m1);
00080         // users don't expect endpoint values
00081         if (result > maxRandom)
00082             result = (double) maxRandom;
00083         return sample_type(result,1.0);
