00132     bool DefaultEvent::matchesDefaultKey(
00133         const DefaultProbKey& contractKey) const {
00134         if(bondsCurrency_ != contractKey.currency()) return false;
00135         // a contract with NoSeniority matches all events
00136         if((bondsSeniority_ != contractKey.seniority())
00137             && (contractKey.seniority() != NoSeniority))
00138             return false;
00139         // loop on all event types in the contract and chek if we match any,
00140         //   calls derived types
00141         for(Size i=0; i<contractKey.size(); i++) {
00142             if(this->matchesEventType(contractKey.eventTypes()[i])) return true;
00143         }
00144         return false;
00145     }

00157     bool FailureToPayEvent::matchesEventType(
00158         const boost::shared_ptr<DefaultType>& contractEvType) const {
00159         boost::shared_ptr<FailureToPay> eveType =
00160             boost::dynamic_pointer_cast<FailureToPay>(contractEvType);
00161         // this chekcs the atomic types, no need to call parents method
00162         if(!eveType) return false;
00163         if(defaultedAmount_ < eveType->amountRequired()) return false;
00164         Date today = Settings::instance().evaluationDate();
00165         if(!this->hasOccurred(today - eveType->gracePeriod(), true))
00166             return false;
00167         return true;
