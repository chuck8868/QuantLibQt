00036     void KnuthUniformRng::ranf_start(long seed) {
00037         int t,s,j;
00038         std::vector<double> u(KK+KK-1),ul(KK+KK-1);
00039         double ulp=(1.0/(1L<<30))/(1L<<22);                // 2 to the -52
00040         double ss=2.0*ulp*((seed&0x3fffffff)+2);
00041 
00042         for (j=0;j<KK;j++) {
00043             u[j]=ss; ul[j]=0.0;                    // bootstrap the buffer
00044             ss+=ss; if (ss>=1.0) ss-=1.0-2*ulp; // cyclic shift of 51 bits
00045         }
00046         for (;j<KK+KK-1;j++) u[j]=ul[j]=0.0;
00047         u[1]+=ulp;ul[1]=ulp;            // make u[1] (and only u[1]) "odd"
00048         s=seed&0x3fffffff;
00049         t=TT-1; while (t) {
00050             for (j=KK-1;j>0;--j) ul[j+j]=ul[j],u[j+j]=u[j];    // "square"
00051             for (j=KK+KK-2;j>KK-LL;j-=2)
00052                 ul[KK+KK-1-j]=0.0,u[KK+KK-1-j]=u[j]-ul[j];
00053             for (j=KK+KK-2;j>=KK;--j) if(ul[j]) {
00054                 ul[j-(KK-LL)]=ulp-ul[j-(KK-LL)],
00055                     u[j-(KK-LL)]=mod_sum(u[j-(KK-LL)],u[j]);
00056                 ul[j-KK]=ulp-ul[j-KK],u[j-KK]=mod_sum(u[j-KK],u[j]);
00057             }
00058             if (is_odd(s)) {                            // "multiply by z"
00059                 for (j=KK;j>0;--j)  ul[j]=ul[j-1],u[j]=u[j-1];
00060                 ul[0]=ul[KK],u[0]=u[KK];    // shift the buffer cyclically
00061                 if (ul[KK]) ul[LL]=ulp-ul[LL],u[LL]=mod_sum(u[LL],u[KK]);
00062             }
00063             if (s) s>>=1; else t--;
00064         }
00065         for (j=0;j<LL;j++) ran_u[j+KK-LL]=u[j];
00066         for (;j<KK;j++) ran_u[j-LL]=u[j];
00067     }

00069     void KnuthUniformRng::ranf_array(std::vector<double>& aa,
00070                                      int n) const {
00071         int i,j;
00072         for (j=0;j<KK;j++) aa[j]=ran_u[j];
00073         for (;j<n;j++) aa[j]=mod_sum(aa[j-KK],aa[j-LL]);
00074         for (i=0;i<LL;i++,j++) ran_u[i]=mod_sum(aa[j-KK],aa[j-LL]);
00075         for (;i<KK;i++,j++) ran_u[i]=mod_sum(aa[j-KK],ran_u[i-LL]);
