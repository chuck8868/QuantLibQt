00162         const std::vector<Date> & fixings = this->arguments_.fixingDates;
00163         const Size numberOfFixings = fixings.size();
00164 
00165         std::vector<Time> fixingTimes(numberOfFixings);

00167             fixingTimes[i] =

00195 
00196         QL_REQUIRE(fixings.size() == numberOfTimes, "Invalid dates/times");
00197

00200         std::vector<Handle<YieldTermStructure> > forwardTermStructures(numberOfTimes);
00201 
00202         const Handle<YieldTermStructure> & riskFreeRate = process->riskFreeRate();
00203 
00204         for (Size i = 0; i < numberOfTimes; ++i) {
00205             timePositions[i] = theTimeGrid.index(times[i]);
00206             discountFactors[i] = riskFreeRate->discount(times[i]);
00207             forwardTermStructures[i] = Handle<YieldTermStructure>(
00208                         new ImpliedTermStructure(riskFreeRate, fixings[i]));
