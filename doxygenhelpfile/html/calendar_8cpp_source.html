00045     Date Calendar::adjust(const Date& d,
00046                           BusinessDayConvention c) const {
00047         QL_REQUIRE(d != Date(), "null date");
00048 
00049         if (c == Unadjusted)
00050             return d;
00051 
00052         Date d1 = d;
00053         if (c == Following || c == ModifiedFollowing) {
00054             while (isHoliday(d1))
00055                 d1++;
00056             if (c == ModifiedFollowing) {
00057                 if (d1.month() != d.month()) {
00058                     return adjust(d,Preceding);
00059                 }
00060             }
00061         } else if (c == Preceding || c == ModifiedPreceding) {
00062             while (isHoliday(d1))
00063                 d1--;
00064             if (c == ModifiedPreceding && d1.month() != d.month()) {
00065                 return adjust(d,Following);
00066             }
00067         } else {
00068             QL_FAIL("unknown business-day convention");
00069         }
00070         return d1;
00071     }

00073     Date Calendar::advance(const Date& d,
00074                            Integer n, TimeUnit unit,
00075                            BusinessDayConvention c,
00076                            bool endOfMonth) const {
00077         QL_REQUIRE(d!=Date(), "null date");
00078         if (n == 0) {
00079             return adjust(d,c);
00080         } else if (unit == Days) {
00081             Date d1 = d;
00082             if (n > 0) {
00083                 while (n > 0) {
00084                     d1++;
00085                     while (isHoliday(d1))
00086                         d1++;
00087                     n--;
00088                 }
00089             } else {
00090                 while (n < 0) {
00091                     d1--;
00092                     while(isHoliday(d1))
00093                         d1--;
00094                     n++;
00095                 }
00096             }
00097             return d1;
00098         } else if (unit == Weeks) {
00099             Date d1 = d + n*unit;
00100             return adjust(d1,c);
00101         } else {
00102             Date d1 = d + n*unit;
00103 
00104             // we are sure the unit is Months or Years
00105             if (endOfMonth && isEndOfMonth(d))
00106                 return Calendar::endOfMonth(d1);
00107 
00108             return adjust(d1, c);
00109         }
00110     }

00119     BigInteger Calendar::businessDaysBetween(const Date& from,
00120                                              const Date& to,
00121                                              bool includeFirst,
00122                                              bool includeLast) const {
00123         BigInteger wd = 0;
00124         if (from != to) {
00125             if (from < to) {
00126                 // the last one is treated separately to avoid
00127                 // incrementing Date::maxDate()
00128                 for (Date d = from; d < to; ++d) {
00129                     if (isBusinessDay(d))
00130                         ++wd;
00131                 }
00132                 if (isBusinessDay(to))
00133                     ++wd;
00134             } else if (from > to) {
00135                 for (Date d = to; d < from; ++d) {
00136                     if (isBusinessDay(d))
00137                         ++wd;
00138                 }
00139                 if (isBusinessDay(from))
00140                     ++wd;
00141             }
00142 
00143             if (isBusinessDay(from) && !includeFirst)
00144                 wd--;
00145             if (isBusinessDay(to) && !includeLast)
00146                 wd--;
00147 
00148             if (from > to)
00149                 wd = -wd;
00150         }
00151 
00152         return wd;
00153     }
