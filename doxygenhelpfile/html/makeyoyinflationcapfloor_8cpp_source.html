00064         .withPaymentAdjustment(roll_)
00065         .withPaymentDayCounter(dayCounter_)
00066         .withNotionals(nominal_)
00067         ;
00068 
00069         if (firstCapletExcluded_)
00070             leg.erase(leg.begin());

00072         // only leaves the last coupon
00073         if (asOptionlet_ && leg.size() > 1) {
00074             Leg::iterator end = leg.end();  // Sun Studio needs an lvalue
00075             leg.erase(leg.begin(), --end);

00078         std::vector<Rate> strikeVector(1, strike_);
