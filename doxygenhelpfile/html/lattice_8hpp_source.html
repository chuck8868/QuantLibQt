00097     template <class Impl>
00098     void TreeLattice<Impl>::computeStatePrices(Size until) const {
00099         for (Size i=statePricesLimit_; i<until; i++) {
00100             statePrices_.push_back(Array(this->impl().size(i+1), 0.0));
00101             for (Size j=0; j<this->impl().size(i); j++) {
00102                 DiscountFactor disc = this->impl().discount(i,j);
00103                 Real statePrice = statePrices_[i][j];
00104                 for (Size l=0; l<n_; l++) {
00105                     statePrices_[i+1][this->impl().descendant(i,j,l)] +=
00106                         statePrice*disc*this->impl().probability(i,j,l);
00107                 }
00108             }
00109         }
00110         statePricesLimit_ = until;
00111     }

00141                                             Time to) const {
00142 
00143         Time from = asset.time();
00144 
00145         if (close(from,to))

00155         for (Integer i=iFrom-1; i>=iTo; --i) {
