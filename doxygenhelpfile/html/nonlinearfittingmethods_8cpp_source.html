00039     DiscountFactor ExponentialSplinesFitting::discountFunction(const Array& x,
00040                                                                Time t) const {
00041         DiscountFactor d = 0.0;
00042         Size N = size();
00043         Real kappa = x[N-1];
00044         Real coeff = 0;
00045 
00046         if (!constrainAtZero_) {
00047             for (Size i=0; i<N-1; ++i) {
00048                 d += x[i]* std::exp(-kappa * (i+1) * t);
00049             }
00050         } else {
00051             //  notation:
00052             //  d(t) = coeff* exp(-kappa*1*t) + x[0]* exp(-kappa*2*t) +
00053             //  x[1]* exp(-kappa*3*t) + ..+ x[7]* exp(-kappa*9*t)
00054             for (Size i=0; i<N-1; i++) {
00055                 d += x[i]* std::exp(-kappa * (i+2) * t);
00056                 coeff += x[i];
00057             }
00058             coeff = 1.0- coeff;
00059             d += coeff * std::exp(-kappa * t);
00060         }
00061         return d;
00062     }

00079     DiscountFactor NelsonSiegelFitting::discountFunction(const Array& x,
00080                                                          Time t) const {
00081         Real kappa = x[size()-1];
00082         Real zeroRate = x[0] + (x[1] + x[2])*
00083                         (1.0 - std::exp(-kappa*t))/
00084                         ((kappa+QL_EPSILON)*(t+QL_EPSILON)) -
00085                         (x[2])*std::exp(-kappa*t);
00086         DiscountFactor d = std::exp(-zeroRate * t) ;
00087         return d;

00158     DiscountFactor CubicBSplinesFitting::discountFunction(const Array& x,
00159                                                           Time t) const {
00160         DiscountFactor d = 0.0;
00161 
00162         if (!constrainAtZero_) {
00163             for (Size i=0; i<size_; ++i) {
00164                 d += x[i] * splines_(i,t);
00165             }
00166         } else {
00167             const Real T = 0.0;
00168             Real sum = 0.0;
00169             for (Size i=0; i<size_; ++i) {
00170                 if (i < N_) {
00171                     d += x[i] * splines_(i,t);
00172                     sum += x[i] * splines_(i,T);
00173                 } else {
00174                     d += x[i] * splines_(i+1,t);
00175                     sum += x[i] * splines_(i+1,T);
00176                 }
00177             }
00178             Real coeff = 1.0 - sum;
00179             coeff /= splines_(N_,T);
00180             d += coeff * splines_(N_,t);
00181         }
00182 
00183         return d;
