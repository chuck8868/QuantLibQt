00048     Disposable<Matrix> G2Process::diffusion(Time, const Array&) const {
00049         /* the correlation matrix is
00050            |  1   rho |
00051            | rho   1  |
00052            whose square root (which is used here) is
00053            |  1          0       |

00072     Disposable<Matrix> G2Process::stdDeviation(Time t0, const Array& x0,
00073                                                Time dt) const {
00074         /* the correlation matrix is
00075            |  1   rho |
00076            | rho   1  |
00077            whose square root (which is used here) is
00078            |  1          0       |
00079            | rho   sqrt(1-rho^2) |
00080         */
00081         Matrix tmp(2,2);
00082         Real sigma1 = xProcess_->stdDeviation(t0, x0[0], dt);
00083         Real sigma2 = yProcess_->stdDeviation(t0, x0[1], dt);
00084         Real expa = std::exp(-a_*dt), expb = std::exp(-b_*dt);
00085         Real H = (rho_*sigma_*eta_)/(a_+b_)*(1-expa*expb);
00086         Real den =
00087             (0.5*sigma_*eta_)*std::sqrt((1-expa*expa)*(1-expb*expb)/(a_*b_));
00088         Real newRho = H/den;
00089         tmp[0][0] = sigma1;
00090         tmp[0][1] = 0.0;
00091         tmp[1][0] = newRho*sigma2;
00092         tmp[1][1] = std::sqrt(1.0-newRho*newRho)*sigma2;
00093         return tmp;

00213     Real G2ForwardProcess::Mx_T(Real s, Real t, Real T) const {
00214         Real M;
00215         M = ( (sigma_*sigma_)/(a_*a_) + (rho_*sigma_*eta_)/(a_*b_) )
00216           * (1-std::exp(-a_*(t-s)));

00218               (std::exp(-a_*(T-t))-std::exp(-a_*(T+t-2*s)));
00219         M += -(rho_*sigma_*eta_)/(b_*(a_+b_))
