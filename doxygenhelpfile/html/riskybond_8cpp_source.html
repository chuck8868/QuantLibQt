00041     void RiskyBond::performCalculations() const {
00042         NPV_ = 0;
00043         Date today = Settings::instance().evaluationDate();
00044         std::vector<boost::shared_ptr<CashFlow> > cf = cashflows();
00045         Date d1 = effectiveDate();
00046         for (Size i = 0; i < cf.size(); i++) {
00047             Date d2 = cf[i]->date();
00048             if (d2 > today) {
00049                 d1 = max(today , d1);
00050                 Date defaultDate = d1 + (d2-d1)/2;
00051 
00052                 Real coupon = cf[i]->amount()
00053                     * defaultTS_->survivalProbability(d2);
00054                 Real recovery = notional(defaultDate) * recoveryRate_
00055                     * (defaultTS_->survivalProbability(d1)
00056                        -defaultTS_->survivalProbability(d2));
00057                 NPV_ += coupon * yieldTS()->discount(d2);
00058                 NPV_ += recovery * yieldTS()->discount(defaultDate);
00059             }
00060             d1 = d2;
