00172         const {

00174         Time t = process_->time(lastExerciseDate);
00175         if (this->timeSteps_ != Null<Size>()) {
00176             return TimeGrid(t, this->timeSteps_);

00178             Size steps = static_cast<Size>(this->timeStepsPerYear_*t);
00179             return TimeGrid(t, std::max<Size>(steps, 1));

00181             QL_FAIL("time steps not specified");
