00138     bool isInMoneyMarketPlusMeasure(const EvolutionDescription& evolution,
00139                                     const std::vector<Size>& numeraires,
00140                                     Size offset) {
00141         bool res = true;
00142         const std::vector<Time>& rateTimes = evolution.rateTimes();
00143         Size maxNumeraire=rateTimes.size()-1;
00144         QL_REQUIRE(offset<=maxNumeraire,
00145                    "offset (" << offset <<
00146                    ") is greater than the max allowed value for numeraire ("
00147                    << maxNumeraire << ")");
00148         const std::vector<Time>& evolutionTimes = evolution.evolutionTimes();
00149         for (Size i=0, j=0; i<evolutionTimes.size(); ++i) {
00150             while (rateTimes[j] < evolutionTimes[i])
00151                 j++;
00152             res = (numeraires[i] == std::min(j+offset, maxNumeraire)) && res;
00153         }
00154         return res;
00155     }
