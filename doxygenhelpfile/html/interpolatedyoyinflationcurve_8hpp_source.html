00103     template <class Interpolator>
00104     InterpolatedYoYInflationCurve<Interpolator>::
00105     InterpolatedYoYInflationCurve(const Date& referenceDate,
00106                                   const Calendar& calendar,
00107                                   const DayCounter& dayCounter,
00108                                   const Period& lag,
00109                                   Frequency frequency,
00110                                   bool indexIsInterpolated,
00111                                   const Handle<YieldTermStructure>& yTS,
00112                                   const std::vector<Date>& dates,
00113                                   const std::vector<Rate>& rates,
00114                                   const Interpolator& interpolator)
00115     : YoYInflationTermStructure(referenceDate, calendar, dayCounter, rates[0],
00116                                 lag, frequency, indexIsInterpolated,  yTS),
00117       InterpolatedCurve<Interpolator>(std::vector<Time>(), rates, interpolator),
00118       dates_(dates) {
00119 
00120         QL_REQUIRE(dates_.size()>1, "too few dates: " << dates_.size());
00121 
00122         // check that the data starts from the beginning,
00123         // i.e. referenceDate - lag, at least must be in the relevant
00124         // period
00125         std::pair<Date,Date> lim =
00126             inflationPeriod(yTS->referenceDate() - this->observationLag(), frequency);
00127         QL_REQUIRE(lim.first <= dates_[0] && dates_[0] <= lim.second,
00128                    "first data date is not in base period, date: " << dates_[0]
00129                    << " not within [" << lim.first << "," << lim.second << "]");
00130 
00131         QL_REQUIRE(this->data_.size() == dates_.size(),
00132                    "indices/dates count mismatch: "
00133                    << this->data_.size() << " vs " << dates_.size());
00134 
00135         this->times_.resize(dates_.size());
00136         this->times_[0] = timeFromReference(dates_[0]);
00137 
00138         for (Size i = 1; i < dates_.size(); i++) {
00139             QL_REQUIRE(dates_[i] > dates_[i-1],
00140                        "dates not sorted");
00141             // YoY inflation data may be positive or negative
00142             // but must be greater than -1
00143             QL_REQUIRE(this->data_[i] > -1.0,
00144                        "year-on-year inflation data < -100 %");
00145 
00146             // this can be negative
00147             this->times_[i] = timeFromReference(dates_[i]);
00148 
00149             QL_REQUIRE(!close(this->times_[i],this->times_[i-1]),
00150                        "two dates correspond to the same time "
00151                        "under this curve's day count convention");
00152         }
00153 
00154         this->interpolation_ =
00155             this->interpolator_.interpolate(this->times_.begin(),
00156                                             this->times_.end(),
00157                                             this->data_.begin());
00158         this->interpolation_.update();
00159     }
