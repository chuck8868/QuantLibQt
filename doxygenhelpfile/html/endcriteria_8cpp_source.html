00063     bool EndCriteria::checkStationaryPoint(const Real xOld,
00064                                            const Real xNew,
00065                                            Size& statStateIterations,
00066                                            EndCriteria::Type& ecType) const {
00067         if (std::fabs(xNew-xOld) >= rootEpsilon_) {
00068             statStateIterations = 0;
00069             return false;
00070         }
00071         ++statStateIterations;
00072         if (statStateIterations <= maxStationaryStateIterations_)
00073             return false;
00074         ecType = StationaryPoint;
00075         return true;
00076     }
