00090             Notice that while we want to compute:
00091             \f[
00092             EL(t) = \sum_{l_k}l_k P(l;t) =
00093               \sum_{l_k}l_k \int P(l_k;t|\omega) d\omega q(\omega)

00101             it easier this way.

00105                 integral_(boost::bind(

00227                 pIndepDistrib.begin();

00229                 // update prob if this name does not default
00230                 std::map<Real, Probability>::iterator matchIt
00231                     = pDistTemp.find(distIt->first);
00232                 if(matchIt != pDistTemp.end()) {
00233                     matchIt->second += distIt->second * (1.-pDef);
00234                 }else{
00235                     pDistTemp.insert(std::make_pair(distIt->first,
00236                         distIt->second * (1.-pDef)));
00237                 }
00238                 // and if it does
00239                 matchIt = pDistTemp.find(distIt->first + wk_[iName]);
00240                 if(matchIt != pDistTemp.end()) {
00241                     matchIt->second += distIt->second * pDef;
00242                 }else{
00243                     pDistTemp.insert(std::make_pair(
00244                         distIt->first+wk_[iName], distIt->second * pDef));
00245                 }
00246                 distIt++;
00247             }
00248             // copy back
00249             pIndepDistrib = pDistTemp;

00253         //   factor.

00257              unroll below to take profit of the fact that once we go over

00274         Real relativeMin = this->results_.xMin / loss_unit_;

00276         Size relativeMinIdx = std::floor(relativeMin);

00278             distIt = pIndepDistrib.lower_bound(relativeMinIdx),
00279             itTop  = pIndepDistrib.lower_bound(relativeMax);
