00052     ExchangeRate ExchangeRateManager::lookup(const Currency& source,
00053                                              const Currency& target,
00054                                              Date date,
00055                                              ExchangeRate::Type type) const {
00056 
00057         if (source == target)
00058             return ExchangeRate(source,target,1.0);
00059 
00060         if (date == Date())
00061             date = Settings::instance().evaluationDate();
00062 
00063         if (type == ExchangeRate::Direct) {
00064             return directLookup(source,target,date);
00065         } else if (!source.triangulationCurrency().empty()) {
00066             const Currency& link = source.triangulationCurrency();
00067             if (link == target)
00068                 return directLookup(source,link,date);
00069             else
00070                 return ExchangeRate::chain(directLookup(source,link,date),
00071                                            lookup(link,target,date));
00072         } else if (!target.triangulationCurrency().empty()) {
00073             const Currency& link = target.triangulationCurrency();
00074             if (source == link)
00075                 return directLookup(link,target,date);
00076             else
00077                 return ExchangeRate::chain(lookup(source,link,date),
00078                                            directLookup(link,target,date));
00079         } else {
00080             return smartLookup(source,target,date);
00081         }
00082     }

00148     ExchangeRate ExchangeRateManager::smartLookup(
00149                                          const Currency& source,
00150                                          const Currency& target,
00151                                          const Date& date,
00152                                          std::list<Integer> forbidden) const {
00153         // direct exchange rates are preferred.
00154         if (const ExchangeRate* direct = fetch(source,target,date))
00155             return *direct;
00156 
00157         // if none is found, turn to smart lookup. The source currency
00158         // is forbidden to subsequent lookups in order to avoid cycles.
00159         forbidden.push_back(source.numericCode());
00160         std::map<Key, std::list<Entry> >::const_iterator i;
00161         for (i = data_.begin(); i != data_.end(); ++i) {
00162             // we look for exchange-rate data which involve our source
00163             // currency...
00164             if (hashes(i->first, source) && !(i->second.empty())) {
00165                 // ...whose other currency is not forbidden...
00166                 const Entry& e = i->second.front();
00167                 const Currency& other =
00168                     source == e.rate.source() ?
00169                         e.rate.target() : e.rate.source();
00170                 if (std::find(forbidden.begin(),forbidden.end(),
00171                               other.numericCode()) == forbidden.end()) {
00172                     // ...and which carries information for the requested date.
00173                     if (const ExchangeRate* head = fetch(source,other,date)) {
00174                         // if we can get to the target from here...
00175                         try {
00176                             ExchangeRate tail = smartLookup(other,target,date,
00177                                                             forbidden);
00178                             // ..we're done.
00179                             return ExchangeRate::chain(*head,tail);
00180                         } catch (Error&) {
00181                             // otherwise, we just discard this rate.
00182                             ;
00183                         }
00184                     }
00185                 }
00186             }
00187         }
00188         // if the loop completed, we have no way to return the requested rate.
00189         QL_FAIL("no conversion available from "
00190                 << source.code() << " to " << target.code()
00191                 << " for " << date);
00192     }
