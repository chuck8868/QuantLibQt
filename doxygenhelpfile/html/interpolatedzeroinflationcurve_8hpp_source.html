00099     template <class Interpolator>
00100     InterpolatedZeroInflationCurve<Interpolator>::
00101     InterpolatedZeroInflationCurve(const Date& referenceDate,
00102                                    const Calendar& calendar,
00103                                    const DayCounter& dayCounter,
00104                                    const Period& lag,
00105                                    Frequency frequency,
00106                                    bool indexIsInterpolated,
00107                                    const Handle<YieldTermStructure>& yTS,
00108                                    const std::vector<Date>& dates,
00109                                    const std::vector<Rate>& rates,
00110                                    const Interpolator& interpolator)
00111     : ZeroInflationTermStructure(referenceDate, calendar, dayCounter, rates[0],
00112                                  lag, frequency, indexIsInterpolated, yTS),
00113       InterpolatedCurve<Interpolator>(std::vector<Time>(), rates, interpolator),
00114       dates_(dates) {
00115 
00116           QL_REQUIRE(dates_.size() > 1, "too few dates: " << dates_.size());
00117 
00118           // check that the data starts from the beginning,
00119           // i.e. referenceDate - lag, at least must be in the relevant
00120           // period
00121           std::pair<Date,Date> lim =
00122                 inflationPeriod(yTS->referenceDate() - this->observationLag(), frequency);
00123           QL_REQUIRE(lim.first <= dates_[0] && dates_[0] <= lim.second,
00124                    "first data date is not in base period, date: " << dates_[0]
00125                    << " not within [" << lim.first << "," << lim.second << "]");
00126 
00127           // by convention, if the index is not interpolated we pull all the dates
00128           // back to the start of their inflationPeriods
00129           // otherwise the time calculations will be inconsistent
00130           if (!indexIsInterpolated_) {
00131               for (Size i = 0; i < dates_.size(); i++) {
00132                   dates_[i] = inflationPeriod(dates_[i], frequency).first;
00133               }
00134           }
00135 
00136 
00137 
00138           QL_REQUIRE(this->data_.size() == dates_.size(),
00139                    "indices/dates count mismatch: "
00140                    << this->data_.size() << " vs " << dates_.size());
00141 
00142           this->times_.resize(dates_.size());
00143           this->times_[0] = timeFromReference(dates_[0]);
00144           for (Size i = 1; i < dates_.size(); i++) {
00145               QL_REQUIRE(dates_[i] > dates_[i-1],
00146                        "dates not sorted");
00147 
00148               // but must be greater than -1
00149               QL_REQUIRE(this->data_[i] > -1.0, "zero inflation data < -100 %");
00150 
00151               // this can be negative
00152               this->times_[i] = timeFromReference(dates_[i]);
00153               QL_REQUIRE(!close(this->times_[i],this->times_[i-1]),
00154                        "two dates correspond to the same time "
00155                        "under this curve's day count convention");
00156           }
00157 
00158           this->interpolation_ =
00159                 this->interpolator_.interpolate(this->times_.begin(),
00160                                                 this->times_.end(),
00161                                                 this->data_.begin());
00162           this->interpolation_.update();
00163     }
