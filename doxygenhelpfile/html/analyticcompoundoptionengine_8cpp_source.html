00033                    "Daughter strike must be positive");
00034 
00035         QL_REQUIRE(strikeMother()>0.0,
00036                    "Mother strike must be positive");
00037 
00038         QL_REQUIRE(spot() >= 0.0, "negative or null underlying given");
00039 
00040         /* Solver Setup ***************************************************/
00041         Date helpDate(process_->riskFreeRate()->referenceDate());
00042         Date helpMaturity=helpDate+(maturityDaughter()-maturityMother())*Days;
00043         Real vol =process_->blackVolatility()->blackVol(helpMaturity,

00045 
00046         Time helpTimeToMat=process_->time(helpMaturity);

00048 
00049         DiscountFactor dividendDiscount =
00050             process_->dividendYield()->discount(helpMaturity);
00051 
00052         DiscountFactor riskFreeDiscount =
00053             process_->riskFreeRate()->discount(helpMaturity);
00054 
00055

00057                 new ImpliedSpotHelper(dividendDiscount, riskFreeDiscount,
00058                                       vol, payoffDaughter(), strikeMother()));

00061         solver.setMaxEvaluations(1000);
00062         Real accuracy = 1.0e-6;
00063 
00064         Real X=0.0;
00065         Real sSolved=0.0;
00066         sSolved=solver.solve(*f, accuracy, strikeDaughter(), 1.0e-6, strikeDaughter()*1000.0);
00067         X=transformX(sSolved); // transform stock to return as in Wystup's book
00068         /* Solver Setup Finished*****************************************/
00069 
00070         Real phi=typeDaughter(); // -1 or 1
00071         Real w=typeMother(); // -1 or 1
00072 
00073         Real rho=std::sqrt(residualTimeMother()/residualTimeDaughter());
00074         BivariateCumulativeNormalDistributionDr78 N2(w*rho) ;
00075 
00076         DiscountFactor ddD=dividendDiscountDaughter();
00077         DiscountFactor rdD=riskFreeDiscountDaughter();
00078         //DiscountFactor ddM=dividendDiscountMother();
00079         DiscountFactor rdM=riskFreeDiscountMother();
00080 
00081         Real XmSM=X-stdDeviationMother();
00082         Real S=spot();
00083         Real dP=dPlus();
00084         Real dPT12=dPlusTau12(sSolved);
00085         Real vD=volatilityDaughter();
00086 
00087         Real dM=dMinus();
00088         Real strD=strikeDaughter();
00089         Real strM=strikeMother();
00090         Real rTM=residualTimeMother();
00091         Real rTD=residualTimeDaughter();
00092 
00093         Real rD=riskFreeRateDaughter();
00094         Real dD=dividendRateDaughter();
00095 
00096         Real tempRes=0.0;
00097         Real tempDelta=0.0;
00098         Real tempGamma=0.0;
00099         Real tempVega=0.0;
00100         Real tempTheta=0.0;
00101         Real N2XmSM=N2(-phi*w*XmSM,phi*dP);
00102         Real N2X=N2(-phi*w*X,phi*dM);
00103         Real NeX=N_(-phi*w*e(X));
00104         Real NX=N_(-phi*w*X);
00105         Real NT12=N_(phi*dPT12);
00106         Real ndP=n_(dP);
00107         Real nXm=n_(XmSM);
00108         Real invMTime=1/std::sqrt(rTM);
00109         Real invDTime=1/std::sqrt(rTD);
00110 
00111         tempRes=phi*w*S*ddD*N2XmSM-phi*w*strD*rdD*N2X-w*strM*rdM*NX;
00112         tempDelta=phi*w*ddD*N2XmSM;
00113         tempGamma=(ddD/(vD*S))*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
00114         tempVega=ddD*S*((1/invMTime)*nXm*NT12+w*(1/invDTime)*ndP*NeX);
00115 
00116         tempTheta+=phi*w*dD*S*ddD*N2XmSM-phi*w*rD*strD*rdD*N2X-w*rD*strM*rdM*NX;
00117         tempTheta-=0.5*vD*S*ddD*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
00118 
00119         results_.value=tempRes;
00120         results_.delta=tempDelta;
00121         results_.gamma=tempGamma;
00122         results_.vega=tempVega;
00123         results_.theta=tempTheta;
00124     }

00262     Real AnalyticCompoundOptionEngine::transformX(Real X) const{
00263 
00264         Real sd=stdDeviationMother();
00265         Real resX=riskFreeDiscountMother()*X/(spot()*dividendDiscountMother());
00266         resX=resX*std::exp(0.5*sd*sd);
