00192         boost::shared_ptr<YieldTermStructure> qTS = flatRate(today,qRate,dc);

00194         boost::shared_ptr<YieldTermStructure> rTS = flatRate(today,rRate,dc);
00195 
00196         for (Size i=0; i<LENGTH(types); i++) {
00197           for (Size j=0; j<LENGTH(strikes); j++) {
00198             for (Size k=0; k<LENGTH(lengths); k++) {
00199               Date exDate = today + lengths[k]*360;
00200               boost::shared_ptr<Exercise> exercise(
00201                                                 new EuropeanExercise(exDate));
00202               boost::shared_ptr<StrikedTypePayoff> payoff(new
00203                                     PlainVanillaPayoff(types[i], strikes[j]));
00204               // reference option
00205               boost::shared_ptr<VanillaOption> refOption =
00206                   makeOption(payoff, exercise, spot, qTS, rTS, volTS,
00207                              Analytic, Null<Size>());

00210                   makeOption(payoff, exercise, spot, qTS, rTS, volTS,

00212 
00213               for (Size l=0; l<LENGTH(underlyings); l++) {
00214                 for (Size m=0; m<LENGTH(qRates); m++) {
00215                   for (Size n=0; n<LENGTH(rRates); n++) {
00216                     for (Size p=0; p<LENGTH(vols); p++) {
00217                       Real u = underlyings[l];
00218                       Rate q = qRates[m],
00219                            r = rRates[n];
00220                       Volatility v = vols[p];
00221                       spot->setValue(u);
00222                       qRate->setValue(q);
00223                       rRate->setValue(r);

00234                            expected["delta"] = refOption->delta();

00236                            expected["theta"] = refOption->theta();

00238                            calculated["gamma"] = option->gamma();
00239                            calculated["theta"] = option->theta();
00240                       }
