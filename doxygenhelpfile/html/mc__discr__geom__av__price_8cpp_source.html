00034     Real GeometricAPOPathPricer::operator()(const Path& path) const {
00035         Size n = path.length() - 1;
00036         QL_REQUIRE(n>0, "the path cannot be empty");
00037 
00038         Real averagePrice;
00039         Real product = runningProduct_;
00040         Size fixings = n+pastFixings_;
00041         if (path.timeGrid().mandatoryTimes()[0]==0.0) {
00042             fixings += 1;
00043             product *= path.front();
00044         }
00045         // care must be taken not to overflow product
00046         Real maxValue = QL_MAX_REAL;
00047         averagePrice = 1.0;
00048         for (Size i=1; i<n+1; i++) {
00049             Real price = path[i];
00050             if (product < maxValue/price) {
00051                 product *= price;
00052             } else {
00053                 averagePrice *= std::pow(product, 1.0/fixings);
00054                 product = price;
00055             }
00056         }
00057         averagePrice *= std::pow(product, 1.0/fixings);
00058         return discount_ * payoff_(averagePrice);
00059     }
