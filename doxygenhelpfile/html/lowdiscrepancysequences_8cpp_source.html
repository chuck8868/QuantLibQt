00822     template <class T>
00823     void testGeneratorDiscrepancy(const T& generatorFactory,
00824                                   #ifndef PRINT_ONLY
00825                                   const Real * const discrepancy[8],
00826                                   const std::string&,
00827                                   const std::string&
00828                                   #else
00829                                   const Real * const [8],
00830                                   const std::string& fileName,
00831                                   const std::string& arrayName
00832                                   #endif
00833                                   ) {
00834 
00835         QL_TEST_START_TIMING
00836 
00837         #ifndef PRINT_ONLY
00838         Real tolerance = 1.0e-2;
00839         #endif
00840 
00841         std::vector<Real> point;
00842         Size dim;
00843         BigNatural seed = 123456;
00844         Real discr;
00845         // more than 1 discrepancy measures take long time
00846         Size sampleLoops = std::max<Size>(1, discrepancyMeasuresNumber);
00847 
00848         #ifdef PRINT_ONLY
00849         std::ofstream outStream(fileName.c_str());
00850         #endif
00851         for (Integer i = 0; i<8; i++) {
00852             #ifdef PRINT_ONLY
00853             outStream << std::endl;
00854             #endif
00855 
00856             dim = dimensionality[i];
00857             DiscrepancyStatistics stat(dim);
00858 
00859             typename T::generator_type rsg = generatorFactory.make(dim, seed);
00860 
00861             Size j, k=0, jMin=10;
00862             stat.reset();
00863             #ifdef PRINT_ONLY
00864             outStream << "const Real dim" << dim
00865                       << arrayName << "[] = {" ;
00866             #endif
00867             for (j=jMin; j<jMin+sampleLoops; j++) {
00868                 Size points = Size(std::pow(2.0, Integer(j)))-1;
00869                 for (; k<points; k++) {
00870                     point = rsg.nextSequence().value;
00871                     stat.add(point);
00872                 }
00873 
00874                 discr = stat.discrepancy();
00875 
00876                 #ifdef PRINT_ONLY
00877                 if (j!=jMin)
00878                     outStream << ", ";
00879                 outStream << QL_FIXED << std::setprecision(2) << discr;
00880                 #else
00881                 if (std::fabs(discr-discrepancy[i][j-jMin])>tolerance*discr) {
00882                     BOOST_ERROR(generatorFactory.name()
00883                                 << "discrepancy dimension " << dimensionality[i]
00884                                 << " at " << points << " samples is "
00885                                 << discr << " instead of "
00886                                 << discrepancy[i][j-jMin]);
00887                 }
00888                 #endif
00889             }
00890             #ifdef PRINT_ONLY
00891             outStream << "};" << std::endl;
00892             #endif
00893         }
00894         #ifdef PRINT_ONLY
00895         outStream.close();
00896         #endif
00897     }
