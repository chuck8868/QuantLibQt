00057     Disposable<Array> HestonProcess::drift(Time t, const Array& x) const {
00058         Array tmp(2);
00059         const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
00060                          : (discretization_ == Reflection) ? - std::sqrt(-x[1])
00061                          : 0.0;
00062 
00063         tmp[0] = riskFreeRate_->forwardRate(t, t, Continuous)
00064                - dividendYield_->forwardRate(t, t, Continuous)
00065                - 0.5 * vol * vol;
00066 
00067         tmp[1] = kappa_*
00068            (theta_-((discretization_==PartialTruncation) ? x[1] : vol*vol));
00069         return tmp;
00070     }

00072     Disposable<Matrix> HestonProcess::diffusion(Time, const Array& x) const {
00073         /* the correlation matrix is
00074            |  1   rho |
00075            | rho   1  |
00076            whose square root (which is used here) is
00077            |  1          0       |
00078            | rho   sqrt(1-rho^2) |
00079         */
00080         Matrix tmp(2,2);
00081         const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
00082                          : (discretization_ == Reflection) ? -std::sqrt(-x[1])
00083                          : 1e-8; // set vol to (almost) zero but still
00084                                  // expose some correlation information
00085         const Real sigma2 = sigma_ * vol;
00086         const Real sqrhov = std::sqrt(1.0 - rho_*rho_);
00087 
00088         tmp[0][0] = vol;          tmp[0][1] = 0.0;
00089         tmp[1][0] = rho_*sigma2;  tmp[1][1] = sqrhov*sigma2;
00090         return tmp;
00091     }

00093     Disposable<Array> HestonProcess::apply(const Array& x0,
00094                                            const Array& dx) const {
00095         Array tmp(2);
00096         tmp[0] = x0[0] * std::exp(dx[0]);
00097         tmp[1] = x0[1] + dx[1];

00101     Disposable<Array> HestonProcess::evolve(Time t0, const Array& x0,
00102                                             Time dt, const Array& dw) const {
00103         Array retVal(2);
00104         Real ncp, df, p, dy;
00105         Real vol, vol2, mu, nu;
00106 
00107         const Real sdt = std::sqrt(dt);
00108         const Real sqrhov = std::sqrt(1.0 - rho_*rho_);
00109 
00110         switch (discretization_) {
00111           // For the definition of PartialTruncation, FullTruncation
00112           // and Reflection  see Lord, R., R. Koekkoek and D. van Dijk (2006),
00113           // "A Comparison of biased simulation schemes for
00114           //  stochastic volatility models",
00115           // Working Paper, Tinbergen Institute
00116           case PartialTruncation:
00117             vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00118             vol2 = sigma_ * vol;
00119             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00120                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00121                     - 0.5 * vol * vol;
00122             nu = kappa_*(theta_ - x0[1]);
00123 
00124             retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
00125             retVal[1] = x0[1] + nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
00126             break;
00127           case FullTruncation:
00128             vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00129             vol2 = sigma_ * vol;
00130             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00131                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00132                     - 0.5 * vol * vol;
00133             nu = kappa_*(theta_ - vol*vol);
00134 
00135             retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
00136             retVal[1] = x0[1] + nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
00137             break;
00138           case Reflection:
00139             vol = std::sqrt(std::fabs(x0[1]));
00140             vol2 = sigma_ * vol;
00141             mu =    riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00142                   - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00143                     - 0.5 * vol*vol;
00144             nu = kappa_*(theta_ - vol*vol);
00145 
00146             retVal[0] = x0[0]*std::exp(mu*dt+vol*dw[0]*sdt);
00147             retVal[1] = vol*vol
00148                         +nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
00149             break;
00150           case NonCentralChiSquareVariance:
00151             // use Alan Lewis trick to decorrelate the equity and the variance
00152             // process by using y(t)=x(t)-\frac{rho}{sigma}\nu(t)
00153             // and Ito's Lemma. Then use exact sampling for the variance
00154             // process. For further details please read the wilmott thread
00155             // "QuantLib code is very high quality"
00156             vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
00157             mu =   riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00158                  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
00159                    - 0.5 * vol*vol;
00160 
00161             df  = 4*theta_*kappa_/(sigma_*sigma_);
00162             ncp = 4*kappa_*std::exp(-kappa_*dt)
00163                 /(sigma_*sigma_*(1-std::exp(-kappa_*dt)))*x0[1];
00164 
00165             p = CumulativeNormalDistribution()(dw[1]);
00166             if (p<0.0)
00167                 p = 0.0;
00168             else if (p >= 1.0)
00169                 p = 1.0-QL_EPSILON;
00170 
00171             retVal[1] = sigma_*sigma_*(1-std::exp(-kappa_*dt))/(4*kappa_)
00172                 *InverseNonCentralChiSquareDistribution(df, ncp, 100)(p);
00173 
00174             dy = (mu - rho_/sigma_*kappa_
00175                           *(theta_-vol*vol)) * dt + vol*sqrhov*dw[0]*sdt;
00176 
00177             retVal[0] = x0[0]*std::exp(dy + rho_/sigma_*(retVal[1]-x0[1]));
00178             break;
00179           case QuadraticExponential:
00180           case QuadraticExponentialMartingale:
00181           {
00182             // for details of the quadratic exponential discretization scheme
00183             // see Leif Andersen,
00184             // Efficient Simulation of the Heston Stochastic Volatility Model
00185             const Real ex = std::exp(-kappa_*dt);
00186 
00187             const Real m  =  theta_+(x0[1]-theta_)*ex;
00188             const Real s2 =  x0[1]*sigma_*sigma_*ex/kappa_*(1-ex)
00189                            + theta_*sigma_*sigma_/(2*kappa_)*(1-ex)*(1-ex);
00190             const Real psi = s2/(m*m);
00191 
00192             const Real g1 =  0.5;
00193             const Real g2 =  0.5;
00194                   Real k0 = -rho_*kappa_*theta_*dt/sigma_;
00195             const Real k1 =  g1*dt*(kappa_*rho_/sigma_-0.5)-rho_/sigma_;
00196             const Real k2 =  g2*dt*(kappa_*rho_/sigma_-0.5)+rho_/sigma_;
00197             const Real k3 =  g1*dt*(1-rho_*rho_);
00198             const Real k4 =  g2*dt*(1-rho_*rho_);
00199             const Real A  =  k2+0.5*k4;
00200 
00201             if (psi < 1.5) {
00202                 const Real b2 = 2/psi-1+std::sqrt(2/psi*(2/psi-1));
00203                 const Real b  = std::sqrt(b2);
00204                 const Real a  = m/(1+b2);
00205 
00206                 if (discretization_ == QuadraticExponentialMartingale) {
00207                     // martingale correction
00208                     QL_REQUIRE(A < 1/(2*a), "illegal value");
00209                     k0 = -A*b2*a/(1-2*A*a)+0.5*std::log(1-2*A*a)
00210                          -(k1+0.5*k3)*x0[1];
00211                 }
00212                 retVal[1] = a*(b+dw[1])*(b+dw[1]);
00213             }
00214             else {
00215                 const Real p = (psi-1)/(psi+1);
00216                 const Real beta = (1-p)/m;
00217 
00218                 const Real u = CumulativeNormalDistribution()(dw[1]);
00219 
00220                 if (discretization_ == QuadraticExponentialMartingale) {
00221                     // martingale correction
00222                     QL_REQUIRE(A < beta, "illegal value");
00223                     k0 = -std::log(p+beta*(1-p)/(beta-A))-(k1+0.5*k3)*x0[1];
00224                 }
00225                 retVal[1] = ((u <= p) ? 0.0 : std::log((1-p)/(1-u))/beta);
00226             }
00227 
00228             mu =   riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
00229                  - dividendYield_->forwardRate(t0, t0+dt, Continuous);
00230 
00231             retVal[0] = x0[0]*std::exp(mu*dt + k0 + k1*x0[1] + k2*retVal[1]
00232                                        +std::sqrt(k3*x0[1]+k4*retVal[1])*dw[0]);
00233           }
00234           break;
00235           default:
00236             QL_FAIL("unknown discretization schema");
00237         }
00238 
00239         return retVal;
00240     }
