00064     Volatility BlackVolTermStructure::blackForwardVol(Time time1,
00065                                                       Time time2,
00066                                                       Real strike,
00067                                                       bool extrapolate) const {
00068         QL_REQUIRE(time1 <= time2,
00069                    time1 << " later than " << time2);
00070         checkRange(time2, extrapolate);
00071         checkStrike(strike, extrapolate);
00072         if (time2==time1) {
00073             if (time1==0.0) {
00074                 Time epsilon = 1.0e-5;
00075                 Real var = blackVarianceImpl(epsilon, strike);
00076                 return std::sqrt(var/epsilon);
00077             } else {
00078                 Time epsilon = std::min<Time>(1.0e-5, time1);
00079                 Real var1 = blackVarianceImpl(time1-epsilon, strike);
00080                 Real var2 = blackVarianceImpl(time1+epsilon, strike);
00081                 QL_ENSURE(var2>=var1,
00082                           "variances must be non-decreasing");
00083                 return std::sqrt((var2-var1)/(2*epsilon));
00084             }
00085         } else {
00086             Real var1 = blackVarianceImpl(time1, strike);
00087             Real var2 = blackVarianceImpl(time2, strike);
00088             QL_ENSURE(var2 >= var1,
00089                       "variances must be non-decreasing");
00090             return std::sqrt((var2-var1)/(time2-time1));
00091         }
00092     }
