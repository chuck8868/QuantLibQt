00160     Real AveragingRatePricer::swapletPrice() const {
00161         // past or future fixing is managed in InterestRateIndex::fixing()
00162 
00163         Size nCount = initialValues_.size();
00164         double dAvgRate = 0.0, dTotalCvg = 0.0, dTotalPayment = 0.0;
00165         for (Size i=0; i<nCount; i++) {
00166             dTotalPayment += initialValues_[i] * observationCvg_[i];
00167             dTotalCvg += observationCvg_[i];
00168         }
00169 
00170         dAvgRate =  dTotalPayment/dTotalCvg;
00171 
00172         Real swapletPrice = dAvgRate*coupon_->accrualPeriod()*discount_;
00173         return gearing_ * swapletPrice + spreadLegValue_;
00174     }
