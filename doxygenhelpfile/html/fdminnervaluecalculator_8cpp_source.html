00074     Real FdmLogInnerValue::avgInnerValueCalc(
00075                                     const FdmLinearOpIterator& iter, Time t) {
00076         const Size dim = mesher_->layout()->dim()[direction_];
00077         const Size coord = iter.coordinates()[direction_];
00078         const Real loc = mesher_->location(iter,direction_);
00079         Real a = loc;
00080         Real b = loc;
00081         if (coord > 0) {
00082             a -= mesher_->dminus(iter, direction_)/2.0;
00083         }
00084         if (coord < dim-1) {
00085             b += mesher_->dplus(iter, direction_)/2.0;
00086         }
00087         boost::function1<Real, Real> f = compose(
00088             std::bind1st(std::mem_fun(&Payoff::operator()), payoff_.get()),
00089                          std::ptr_fun<Real,Real>(std::exp));
00090         
00091         Real retVal;
00092         try {
00093             const Real acc 
00094                 = ((f(a) != 0.0 || f(b) != 0.0) ? (f(a)+f(b))*5e-5 : 1e-4);
00095             retVal = SimpsonIntegral(acc, 8)(f, a, b)/(b-a);
00096         }
00097         catch (Error&) {
00098             // use default value
00099             retVal = innerValue(iter, t);
00100         }
00101                     
00102         return retVal;
00103     }
