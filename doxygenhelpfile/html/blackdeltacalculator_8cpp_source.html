00054     Real BlackDeltaCalculator::deltaFromStrike(Real strike) const {
00055 
00056         QL_REQUIRE(strike >=0.0,
00057                    "positive strike value required: " <<
00058                    strike << " not allowed");
00059 
00060         Real res=0.0;
00061 
00062         switch(dt_){
00063           case DeltaVolQuote::Spot:
00064             res=phi_*fDiscount_*cumD1(strike);
00065             break;
00066 
00067           case DeltaVolQuote::Fwd:
00068             res=phi_*cumD1(strike);
00069             break;
00070 
00071           case DeltaVolQuote::PaSpot:
00072             res=phi_*fDiscount_*cumD2(strike)*strike/forward_;
00073             break;
00074 
00075           case DeltaVolQuote::PaFwd:
00076             res=phi_*cumD2(strike)*strike/forward_;
00077             break;
00078 
00079           default:
00080             QL_FAIL("invalid delta type");
00081         }
00082         return res;
00083     }

00089     Real BlackDeltaCalculator::strikeFromDelta(Real delta,
00090                                                DeltaVolQuote::DeltaType dt)
00091                                                                         const{
00092         Real res=0.0;
00093         Real arg=0.0;
00094         InverseCumulativeNormal f;
00095 
00096         QL_REQUIRE(delta*phi_>=0.0, "Option type and delta are incoherent.");
00097 
00098         switch (dt) {
00099           case DeltaVolQuote::Spot:
00100             QL_REQUIRE(std::fabs(delta)<=fDiscount_,
00101                        "Spot delta out of range.");
00102 
00103             arg=-phi_*f(phi_*delta/fDiscount_)*stdDev_+0.5*stdDev_*stdDev_;
00104             res=forward_*exp(arg);
00105             break;
00106 
00107           case DeltaVolQuote::Fwd:
00108             QL_REQUIRE(std::fabs(delta)<=1.0,
00109                        "Forward delta out of range.");
00110 
00111             arg=-phi_*f(phi_*delta)*stdDev_+0.5*stdDev_*stdDev_;
00112             res=forward_*exp(arg);
00113             break;
00114 
00115           case DeltaVolQuote::PaSpot:
00116           case DeltaVolQuote::PaFwd: {
00117               // This has to be solved numerically. One of the
00118               // problems is that the premium adjusted call delta is
00119               // not monotonic in strike, such that two solutions
00120               // might occur. The one right to the max of the delta is
00121               // considered to be the correct strike.  Some proper
00122               // interval bounds for the strike need to be chosen, the
00123               // numerics can otherwise be very unreliable and
00124               // unstable.  I've chosen Brent over Newton, since the
00125               // interval can be specified explicitly and we can not
00126               // run into the area on the left of the maximum.  The
00127               // put delta doesn't have this property and can be
00128               // solved without any problems, but also numerically.
00129 
00130               BlackDeltaPremiumAdjustedSolverClass f(
00131                        ot_, dt , spot_,dDiscount_, fDiscount_, stdDev_,delta);
00132 
00133               Brent solver;
00134               solver.setMaxEvaluations(1000);
00135               Real accuracy = 1.0e-10;
00136 
00137               Real rightLimit=0.0;
00138               Real leftLimit=0.0;
00139 
00140               // Strike of not premium adjusted is always to the right of premium adjusted
00141               if (dt==DeltaVolQuote::PaSpot) {
00142                   rightLimit=strikeFromDelta(delta, DeltaVolQuote::Spot);
00143               } else {
00144                   rightLimit=strikeFromDelta(delta, DeltaVolQuote::Fwd);
00145               }
00146 
00147               if (phi_<0) { // if put
00148                   res=solver.solve(f, accuracy, rightLimit, 0.0, spot_*100.0);
00149                   break;
00150               } else {
00151 
00152                   // find out the left limit which is the strike
00153                   // corresponding to the value where premium adjusted
00154                   // deltas have their maximum.
00155 
00156                   BlackDeltaPremiumAdjustedMaxStrikeClass g(
00157                                ot_,dt, spot_,dDiscount_, fDiscount_, stdDev_);
00158 
00159                   leftLimit=solver.solve(g, accuracy, rightLimit*0.5,
00160                                          0.0, rightLimit);
00161 
00162                   Real guess=leftLimit+(rightLimit-leftLimit)*0.5;
00163 
00164                   res=solver.solve(f, accuracy, guess, leftLimit, rightLimit);
00165               } // end if phi<0 else
00166 
00167               break;
00168           }
00169 
00170           default:
00171             QL_FAIL("invalid delta type");
00172         }
00173 
00174         return res;
00175     }

00177     Real BlackDeltaCalculator::atmStrike(DeltaVolQuote::AtmType atmT) const {
00178 
00179         Real res=0.0;
00180 
00181         switch(atmT) {
00182           case DeltaVolQuote::AtmDeltaNeutral:
00183             if(dt_==DeltaVolQuote::Spot || dt_==DeltaVolQuote::Fwd){
00184                 res=fExpPos_;
00185             } else {
00186                 res=fExpNeg_;
00187             }
00188             break;
00189 
00190           case DeltaVolQuote::AtmFwd:
00191             res=forward_;
00192             break;
00193 
00194           case DeltaVolQuote::AtmGammaMax: case DeltaVolQuote::AtmVegaMax:
00195             res=fExpPos_;
00196             break;
00197 
00198           case DeltaVolQuote::AtmPutCall50:
00199             QL_REQUIRE(dt_==DeltaVolQuote::Fwd,
00200                        "|PutDelta|=CallDelta=0.50 only possible for forward delta.");
00201             res=fExpPos_;
00202             break;
00203 
00204           default:
00205             QL_FAIL("invalid atm type");
00206         }
00207 
00208         return res;

00212     Real BlackDeltaCalculator::cumD1(Real strike) const {
00213 
00214         Real d1_=0.0;
00215         Real cum_d1_pos_ = 1.0; // N(d1)
00216         Real cum_d1_neg_ = 0.0; // N(-d1)
00217 
00218         CumulativeNormalDistribution f;
00219 
00220         if (stdDev_>=QL_EPSILON) {
00221             if(strike>0) {
00222                 d1_ = std::log(forward_/strike)/stdDev_ + 0.5*stdDev_;
00223                 return f(phi_*d1_);
00224             }
00225         } else {
00226             if (forward_<strike) {
00227                 cum_d1_pos_ = 0.0;
00228                 cum_d1_neg_ = 1.0;
00229             } else if(forward_==strike){
00230                 d1_ = 0.5*stdDev_;
00231                 return f(phi_*d1_);
00232             }
00233         }
00234 
00235         if (phi_>0) { // if Call
00236             return cum_d1_pos_;
00237         } else {
00238             return cum_d1_neg_;
00239         }
00240     }
