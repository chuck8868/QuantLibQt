00047     void DiscretizedVanillaOption::postAdjustValuesImpl() {
00048 
00049         Time now = time();
00050         switch (arguments_.exercise->type()) {
00051           case Exercise::American:
00052             if (now <= stoppingTimes_[1] &&
00053                 now >= stoppingTimes_[0])
00054                 applySpecificCondition();
00055             break;
00056           case Exercise::European:
00057             if (isOnTime(stoppingTimes_[0]))
00058                 applySpecificCondition();
00059             break;
00060           case Exercise::Bermudan:
00061             for (Size i=0; i<stoppingTimes_.size(); i++) {
