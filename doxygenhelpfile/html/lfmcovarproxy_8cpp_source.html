00063                                                       const Array& x) const {
00064 
00065         Array  volatility  = volaModel_->volatility(t, x);
00066         Matrix correlation = corrModel_->correlation(t, x);
00067 
00068         Matrix tmp(size_, size_);
00069         for (Size i=0; i<size_; ++i) {
00070             for (Size j=0; j<size_; ++j) {
00071                 tmp[i][j] = volatility[i]*correlation[i][j]*volatility[j];
