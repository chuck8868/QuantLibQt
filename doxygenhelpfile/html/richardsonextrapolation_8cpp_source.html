00055     Real RichardsonExtrapolation::operator()(Real t) const {
00056 
00057         QL_REQUIRE(t > 1, "scaling factor must be greater than 1");
00058         QL_REQUIRE(n_ != Null<Real>(), "order of convergence must be known");
00059 
00060         const Real tk = std::pow(t,n_);

00062         return (tk*f_(delta_h_/t)-fdelta_h_)/(tk-1.0);

00066     const {

00068
