00107     Real LogNormalCotSwapRatePc::advanceStep()
00108     {
00109          //we're going from T1 to T2
00110 
00111         //a) compute drifts D1 at T1;
00112         if (currentStep_ > initialStep_)
00113             calculators_[currentStep_].compute(curveState_, drifts1_);
00114         else
00115             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00116                       drifts1_.begin());
00117 
00118         //b) evolve forwards up to T2 using D1;
00119         Real weight = generator_->nextStep(brownians_);
00120         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00121         const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
00122 
00123         Size i, alive = alive_[currentStep_];
00124         for (i=alive; i<numberOfRates_; ++i) {
00125             logSwapRates_[i] += drifts1_[i] + fixedDrift[i];
00126             logSwapRates_[i] +=
00127                 std::inner_product(A.row_begin(i), A.row_end(i),
00128                                    brownians_.begin(), 0.0);
00129             swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
00130         }
00131 
00132         // intermediate curve state update
00133         curveState_.setOnCoterminalSwapRates(swapRates_);
00134 
00135         //c) recompute drifts D2 using the predicted forwards;
00136         calculators_[currentStep_].compute(curveState_, drifts2_);
00137 
00138         //d) correct forwards using both drifts
00139         for (i=alive; i<numberOfRates_; ++i) {
00140             logSwapRates_[i] += (drifts2_[i]-drifts1_[i])/2.0;
00141             swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
00142         }
00143 
00144         //e) update curve state
00145         curveState_.setOnCoterminalSwapRates(swapRates_);
00146 
00147         ++currentStep_;
00148 
00149         return weight;
00150     }
