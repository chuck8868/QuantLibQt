00026     Disposable<std::vector<Real> >
00027     factorReduction(Matrix mtrx,
00028                     Size maxIters) {
00029         static Real tolerance = 1.e-6;
00030 
00031         QL_REQUIRE(mtrx.rows() == mtrx.columns(),
00032                    "Input matrix is not square");
00033 
00034         const Size n = mtrx.columns();
00035 
00036         #if defined(QL_EXTRA_SAFETY_CHECKS)
00037         // check symmetry
00038         for (Size iRow=0; iRow<mtrx.rows(); iRow++)
00039             for (Size iCol=0; iCol<iRow; iCol++)
00040                 QL_REQUIRE(mtrx[iRow][iCol] == mtrx[iCol][iRow],
00041                            "input matrix is not symmetric");
00042         QL_REQUIRE(*std::max_element(mtrx.begin(), mtrx.end()) <=1 &&
00043             *std::min_element(mtrx.begin(), mtrx.end()) >=-1,
00044                     "input matrix data is not correlation data");
00045         #endif
00046 
00047         // Initial guess value
00048         std::vector<Real> previousCorrels(n, 0.);
00049         for(Size iCol=0; iCol<n; iCol++) {
00050             for(Size iRow=0; iRow<n; iRow++)
00051                 previousCorrels[iCol] +=
00052                     mtrx[iRow][iCol] * mtrx[iRow][iCol];
00053             previousCorrels[iCol] =
00054                 std::sqrt((previousCorrels[iCol]-1.)/(n-1.));
00055         }
00056 
00057         // iterative solution
00058         Size iteration = 0;
00059         Real distance;
00060         do {
00061             // patch Matrix diagonal
00062             for(Size iCol=0; iCol<n; iCol++)
00063                 mtrx[iCol][iCol] =
00064                     previousCorrels[iCol];
00065             // compute eigenvector decomposition
00066             SymmetricSchurDecomposition ssDec(mtrx);
00067             //const Matrix& eigenVect = ssDec.eigenvectors();
00068             const Array&  eigenVals = ssDec.eigenvalues();
00069             Array::const_iterator itMaxEval =
00070                 std::max_element(eigenVals.begin(), eigenVals.end());
00071             // We do not need the max value, only the position of the
00072             //   corresponding eigenvector
00073             Size iMax = std::distance(eigenVals.begin(), itMaxEval);
00074             std::vector<Real> newCorrels, distances;
00075             for(Size iCol=0; iCol<n; iCol++) {
00076                 Real thisCorrel = mtrx[iMax][iCol];
00077                 newCorrels.push_back(thisCorrel);
00078                 // strictly is:
00079                 // abs(\sqrt{\rho}- \sqrt{\rho_{old}})/\sqrt{\rho_{old}}
00080                 distances.push_back(
00081                     std::abs(thisCorrel - previousCorrels[iCol])/
00082                         previousCorrels[iCol]);
00083             }
00084             previousCorrels = newCorrels;
00085             distance = *std::max_element(distances.begin(), distances.end());
00086         }while(distance > tolerance && ++iteration <= maxIters );
00087 
00088         // test it did not go up to the max iteration and the matrix can
00089         //   be reduced to one factor.
00090         QL_REQUIRE(iteration < maxIters,
00091                    "convergence not reached after " <<
00092                    iteration << " iterations");
00093 
00094         #if defined(QL_EXTRA_SAFETY_CHECKS)
00095         QL_REQUIRE(*std::max_element(previousCorrels.begin(),
00096                                      previousCorrels.end()) <=1 &&
00097                    *std::min_element(previousCorrels.begin(),
00098                                      previousCorrels.end()) >=-1,
00099                 "matrix can not be decomposed to a single factor dependence");
00100         #endif
00101 
00102         return previousCorrels;
00103     }
