00036     Real Vasicek::A(Time t, Time T) const {
00037         Real _a = a();
00038         if (_a < std::sqrt(QL_EPSILON)) {
00039             return 0.0;
00040         } else {
00041             Real sigma2 = sigma()*sigma();
00042             Real bt = B(t, T);
00043             return std::exp((b() + lambda()*sigma()/_a
00044                              - 0.5*sigma2/(_a*_a))*(bt - (T - t))
00045                              - 0.25*sigma2*bt*bt/_a);
00046         }
00047     }

00049     Real Vasicek::B(Time t, Time T) const {
00050         Real _a = a();
00051         if (_a < std::sqrt(QL_EPSILON))
00052             return (T - t);
00053         else
00054             return (1.0 - std::exp(-_a*(T - t)))/_a;
00055     }

00057     Real Vasicek::discountBondOption(Option::Type type,
00058                                      Real strike, Time maturity,
00059                                      Time bondMaturity) const {
00060 
00061         Real v;
00062         Real _a = a();
00063         if (std::fabs(maturity) < QL_EPSILON) {
00064             v = 0.0;
00065         } else if (_a < std::sqrt(QL_EPSILON)) {
00066             v = sigma()*B(maturity, bondMaturity)* std::sqrt(maturity);
00067         } else {
00068             v = sigma()*B(maturity, bondMaturity)*
00069                 std::sqrt(0.5*(1.0 - std::exp(-2.0*_a*maturity))/_a);
00070         }
00071         Real f = discountBond(0.0, bondMaturity, r0_);
00072         Real k = discountBond(0.0, maturity, r0_)*strike;
00073 
00074         return blackFormula(type, k, f, v);
00075     }
