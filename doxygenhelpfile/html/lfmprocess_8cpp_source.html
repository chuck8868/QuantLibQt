00077     Disposable<Array> LiborForwardModelProcess::drift(Time t,
00078                                                       const Array& x) const {
00079         Array f(size_, 0.0);
00080         Matrix covariance(lfmParam_->covariance(t, x));
00081 
00082         const Size m = nextIndexReset(t);
00083 
00084         for (Size k=m; k<size_; ++k) {
00085             m1[k] = accrualPeriod_[k]*x[k]/(1+accrualPeriod_[k]*x[k]);
00086             f[k]  = std::inner_product(m1.begin()+m, m1.begin()+k+1,
00087                                        covariance.column_begin(k)+m,0.0)
00088                     - 0.5*covariance[k][k];
00089         }
00090 
00091         return f;
00092     }

00115     Disposable<Array> LiborForwardModelProcess::evolve(
00116                                              Time t0, const Array& x0,
00117                                              Time dt, const Array& dw) const {
00118         /* predictor-corrector step to reduce discretization errors.
00119 
00120            Short - but slow - solution would be
00121 
00122            Array rnd_0     = stdDeviation(t0, x0, dt)*dw;
00123            Array drift_0   = discretization_->drift(*this, t0, x0, dt);
00124 
00125            return apply(x0, ( drift_0 + discretization_
00126                 ->drift(*this,t0,apply(x0, drift_0 + rnd_0),dt) )*0.5 + rnd_0);
00127 
00128            The following implementation does the same but is faster.
00129         */
00130 
00131         const Size m   = nextIndexReset(t0);
00132         const Real sdt = std::sqrt(dt);
00133 
00134         Array f(x0);
00135         Matrix diff       = lfmParam_->diffusion(t0, x0);
00136         Matrix covariance = lfmParam_->covariance(t0, x0);
00137 
00138         for (Size k=m; k<size_; ++k) {
00139             const Real y = accrualPeriod_[k]*x0[k];
00140             m1[k] = y/(1+y);
00141             const Real d = (
00142                 std::inner_product(m1.begin()+m, m1.begin()+k+1,
00143                                    covariance.column_begin(k)+m,0.0)
00144                 -0.5*covariance[k][k]) * dt;
00145 
00146             const Real r = std::inner_product(
00147                 diff.row_begin(k), diff.row_end(k), dw.begin(), 0.0)*sdt;
00148 
00149             const Real x = y*std::exp(d + r);
00150             m2[k] = x/(1+x);
00151             f[k] = x0[k] * std::exp(0.5*(d+
00152                  (std::inner_product(m2.begin()+m, m2.begin()+k+1,
00153                                      covariance.column_begin(k)+m,0.0)
00154                   -0.5*covariance[k][k])*dt)+ r);
00155         }
00156 
00157         return f;
00158     }
