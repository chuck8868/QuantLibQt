00070     boost::array<Real, 4> SabrVolSurface::sabrGuesses(const Date& d) const {
00071 
00072         // the guesses for sabr parameters are assumed to be piecewise constant
00073         if (d<=optionDates_[0]) return sabrGuesses_[0];
00074         Size i=0;
00075         while( d<optionDates_[i] && i<optionDates_.size()-1)
00076             ++i;
00077         return sabrGuesses_[i];
