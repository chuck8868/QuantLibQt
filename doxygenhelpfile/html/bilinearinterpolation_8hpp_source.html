00047             Real value(Real x, Real y) const {
00048                 Size i = this->locateX(x), j = this->locateY(y);
00049 
00050                 Real z1 = this->zData_[j][i];
00051                 Real z2 = this->zData_[j][i+1];
00052                 Real z3 = this->zData_[j+1][i];
00053                 Real z4 = this->zData_[j+1][i+1];
00054 
00055                 Real t=(x-this->xBegin_[i])/
00056                     (this->xBegin_[i+1]-this->xBegin_[i]);
00057                 Real u=(y-this->yBegin_[j])/
00058                     (this->yBegin_[j+1]-this->yBegin_[j]);
00059 
00060                 return (1.0-t)*(1.0-u)*z1 + t*(1.0-u)*z2
00061                      + (1.0-t)*u*z3 + t*u*z4;
00062             }
00063         };
