00030     FwdToCotSwapAdapter::FwdToCotSwapAdapter(
00031                                const boost::shared_ptr<MarketModel>& fwdModel)
00032     : fwdModel_(fwdModel),
00033       numberOfFactors_(fwdModel->numberOfFactors()),
00034       numberOfRates_(fwdModel->numberOfRates()),
00035       numberOfSteps_(fwdModel->numberOfSteps()),
00036       pseudoRoots_(numberOfSteps_, Matrix(numberOfRates_, numberOfFactors_))
00037     {
00038 
00039         const std::vector<Spread>& displacements =
00040             fwdModel_->displacements();
00041         for (Size i = 1; i<displacements.size(); ++i) {
00042             QL_REQUIRE(displacements[i]==displacements[0],
00043                        io::ordinal(i+1) << " displacement (" <<
00044                        displacements[i] << ") not equal to the previous ones"
00045                        " (" << displacements[0] << ")");
00046         }
00047 
00048         const std::vector<Time>& rateTimes =
00049             fwdModel_->evolution().rateTimes();
00050         // we must ensure we step through all rateTimes
00051         const std::vector<Time>& evolutionTimes =
00052             fwdModel_->evolution().evolutionTimes();
00053         for (Size i = 0;
00054              i<rateTimes.size() && rateTimes[i]<=evolutionTimes.back(); ++i) {
00055             QL_REQUIRE(std::find(evolutionTimes.begin(), evolutionTimes.end(),
00056                                  rateTimes[i])!=evolutionTimes.end(),
00057                                  "skipping " << io::ordinal(i+1) << " rate time");
00058         }
00059 
00060         LMMCurveState cs(rateTimes);
00061         const std::vector<Rate>& initialFwdRates =
00062             fwdModel_->initialRates();
00063         cs.setOnForwardRates(initialFwdRates);
00064         initialRates_ = cs.coterminalSwapRates();
00065 
00066         Matrix zedMatrix = SwapForwardMappings::coterminalSwapZedMatrix(
00067             cs, displacements[0]);
00068 
00069 
00070         const std::vector<Size>& alive =
00071             fwdModel_->evolution().firstAliveRate();
00072         for (Size k = 0; k<numberOfSteps_; ++k) {
00073             pseudoRoots_[k]=zedMatrix*fwdModel_->pseudoRoot(k);
00074             for (Size i=0; i<alive[k]; ++i)
00075                 std::fill(pseudoRoots_[k].row_begin(i),
00076                           pseudoRoots_[k].row_end(i),
00077                           0.0);
00078         }
00079     }
