00151     template <class T>
00152     Rate InterpolatedZeroCurve<T>::zeroYieldImpl(Time t) const {
00153         if (t <= this->times_.back())
00154             return this->interpolation_(t, true);
00155 
00156         // flat fwd extrapolation
00157         Time tMax = this->times_.back();
00158         Rate zMax = this->data_.back();
00159         Rate instFwdMax = zMax + tMax * this->interpolation_.derivative(tMax);
00160         return (zMax * tMax + instFwdMax * (t-tMax)) / t;
00161     }

00238     template <class T>
00239     void InterpolatedZeroCurve<T>::initialize()
00240     {
00241         QL_REQUIRE(dates_.size() >= T::requiredPoints,
00242                    "not enough input dates given");
00243         QL_REQUIRE(this->data_.size() == dates_.size(),
00244                    "dates/data count mismatch");
00245 
00246         this->times_.resize(dates_.size());
00247         this->times_[0] = 0.0;
00248         for (Size i=1; i<dates_.size(); ++i) {
00249             QL_REQUIRE(dates_[i] > dates_[i-1],
00250                        "invalid date (" << dates_[i] << ", vs "
00251                        << dates_[i-1] << ")");
00252             this->times_[i] = dayCounter().yearFraction(dates_[0], dates_[i]);
00253             QL_REQUIRE(!close(this->times_[i],this->times_[i-1]),
00254                        "two dates correspond to the same time "
00255                        "under this curve's day count convention");
00256             #if !defined(QL_NEGATIVE_RATES)
00257             QL_REQUIRE(this->data_[i] > 0.0, "non-positive yield");
00258             // positive yields are not enough to ensure non-negative fwd rates
00259             // so here's a stronger requirement
00260             QL_REQUIRE(this->data_[i]   * this->times_[i] -
00261                        this->data_[i-1] * this->times_[i-1] >= 0.0,
00262                        "negative forward rate implied by the zero yield " <<
00263                        io::rate(this->data_[i]) << " at " << dates_[i] <<
00264                        " (t=" << this->times_[i] << ") after the zero yield " <<
00265                        io::rate(this->data_[i-1]) << " at " << dates_[i-1] <<
00266                        " (t=" << this->times_[i-1] << ")");
00267             #endif
00268         }
00269 
00270         this->interpolation_ =
00271             this->interpolator_.interpolate(this->times_.begin(),
00272                                             this->times_.end(),
00273                                             this->data_.begin());
00274         this->interpolation_.update();
00275     }
