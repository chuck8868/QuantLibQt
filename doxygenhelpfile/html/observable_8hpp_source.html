00118     inline void Observable::notifyObservers() {
00119         bool successful = true;
00120         std::string errMsg;
00121         for (iterator i=observers_.begin(); i!=observers_.end(); ++i) {
00122             try {
00123                 (*i)->update();
00124             } catch (std::exception& e) {
00125                 // quite a dilemma. If we don't catch the exception,
00126                 // other observers will not receive the notification
00127                 // and might be left in an incorrect state. If we do
00128                 // catch it and continue the loop (as we do here) we
00129                 // lose the exception. The least evil might be to try
00130                 // and notify all observers, while raising an
00131                 // exception if something bad happened.
00132                 successful = false;
00133                 errMsg = e.what();
00134             } catch (...) {
00135                 successful = false;
00136             }
00137         }
00138         QL_ENSURE(successful,
00139                   "could not notify one or more observers: " << errMsg);
00140     }
