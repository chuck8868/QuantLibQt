00060     void BrownianBridge::initialize() {
00061 
00062         sqrtdt_[0] = std::sqrt(t_[0]);
00063         for (Size i=1; i<size_; ++i)
00064             sqrtdt_[i] = std::sqrt(t_[i]-t_[i-1]);
00065 
00066         // map is used to indicate which points are already constructed.
00067         // If map[i] is zero, path point i is yet unconstructed.
00068         // map[i]-1 is the index of the variate that constructs
00069         // the path point # i.
00070         std::vector<Size> map(size_, 0);
00071 
00072         //  The first point in the construction is the global step.
00073         map[size_-1] = 1;
00074         //  The global step is constructed from the first variate.
00075         bridgeIndex_[0] = size_-1;
00076         //  The variance of the global step
00077         stdDev_[0] = std::sqrt(t_[size_-1]);
00078         //  The global step to the last point in time is special.
00079         leftWeight_[0] = rightWeight_[0] = 0.0;
00080         for (Size j=0, i=1; i<size_; ++i) {
00081             // Find the next unpopulated entry in the map.
00082             while (map[j])
00083                 ++j;
00084             Size k = j;
00085             // Find the next populated entry in the map from there.
00086             while (!map[k])
00087                 ++k;
00088             // l-1 is now the index of the point to be constructed next.
00089             Size l = j + ((k-1-j)>>1);
00090             map[l] = i;
00091             // The i-th Gaussian variate will be used to set point l-1.
00092             bridgeIndex_[i] = l;
00093             leftIndex_[i]   = j;
00094             rightIndex_[i]  = k;
00095             if (j != 0) {
00096                 leftWeight_[i]= (t_[k]-t_[l])/(t_[k]-t_[j-1]);
00097                 rightWeight_[i] = (t_[l]-t_[j-1])/(t_[k]-t_[j-1]);
00098                 stdDev_[i] =
00099                     std::sqrt(((t_[l]-t_[j-1])*(t_[k]-t_[l]))
00100                               /(t_[k]-t_[j-1]));
00101             } else {
00102                 leftWeight_[i]  = (t_[k]-t_[l])/t_[k];
00103                 rightWeight_[i] =  t_[l]/t_[k];
00104                 stdDev_[i] = std::sqrt(t_[l]*(t_[k]-t_[l])/t_[k]);
00105             }
00106             j=k+1;
00107             if (j>=size_)
00108                 j=0;    //  wrap around
00109         }
00110     }
