00047         FastFourierTransform(std::size_t order)
00048         : cs_(order), sn_(order) {
00049             std::size_t m = 1 << order;
00050             cs_[order - 1] = std::cos (2 * M_PI / m);
00051             sn_[order - 1] = std::sin (2 * M_PI / m);
00052             for (std::size_t i = order - 1; i > 0; --i) {
00053                 cs_ [i - 1] = cs_[i]*cs_[i] - sn_[i]*sn_[i];
00054                 sn_ [i - 1] = 2*sn_[i]*cs_[i];
00055             }
00056         }
00057 
00059         std::size_t output_size() const {
00060             return (1 << cs_.size());
00061         }

00082         template<typename InputIterator, typename RandomAccessIterator>
00083         void transform_impl(InputIterator inBegin, InputIterator inEnd,
00084                             RandomAccessIterator out,
00085                             bool inverse) const {
00086             typedef
00087                 typename std::iterator_traits<RandomAccessIterator>::value_type
00088                                                                        complex;
00089             const std::size_t order = cs_.size();
00090             const std::size_t N = 1 << order;
00091             std::size_t i = 0;
00092             for (; inBegin != inEnd; ++i, ++inBegin) {
00093                 *(out + bit_reverse(i, order)) = *inBegin;
00094             }
00095             QL_REQUIRE (i <= N, "FFT order is too small");
00096             for (std::size_t s = 1; s <= order; ++s) {
00097                 std::size_t m = 1 << s;
00098                 complex w(1.0);
00099                 complex wm(cs_[s-1], inverse ? sn_[s-1] : -sn_[s-1]);
00100                 for (std::size_t j = 0; j < m/2; ++j) {
00101                     for (std::size_t k = j; k < N; k += m) {
00102                         complex t = w * (*(out + k + m/2));
00103                         complex u = *(out + k);
00104                         *(out + k) = u + t;
00105                         *(out + k + m/2) = u - t;
00106                     }
00107                     w *= wm;
00108                 }
00109             }
00110         }

00112         static std::size_t bit_reverse(std::size_t x, std::size_t order) {
00113             std::size_t n = 0;
00114             for (std::size_t i = 0; i < order; ++i) {
00115                 n <<= 1;
00116                 n |= (x & 1);
