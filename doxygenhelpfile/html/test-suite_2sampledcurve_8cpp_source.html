00053     if (std::fabs(curve.value(1) - 3.0) > 1e-5) {
00054         BOOST_ERROR("curve value grid failed");
00055     }
00056 
00057     curve.shiftGrid(10.0);
00058     if (std::fabs(curve.gridValue(0) - 0.0) > 1e-5) {
00059         BOOST_ERROR("sample curve shift grid failed");
00060     }
00061     if (std::fabs(curve.value(0) - 2.0) > 1e-5) {
00062         BOOST_ERROR("sample curve shift grid - value failed");
00063     }
00064 
00065     curve.sample(f2);
00066     curve.regrid(BoundedGrid(0.0,20.0,200));
00067     Real tolerance = 1.0e-2;
00068     for (Size i=0; i < curve.size(); i++) {
00069         Real grid = curve.gridValue(i);
00070         Real value = curve.value(i);
00071         Real expected = f2(grid);
00072         if (std::fabs(value - expected) > tolerance) {
00073             BOOST_ERROR("sample curve regriding failed" <<
00074                         "\n    at " << io::ordinal(i+1) << " point " << "(x = " << grid << ")" <<
00075                         "\n    grid value: " << value <<
00076                         "\n    expected:   " << expected);
00077         }
00078     }
