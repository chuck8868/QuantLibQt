00044     // compute derivative of swap-rate to underlying forward rate
00045     Real SwapForwardMappings::swapDerivative(const CurveState& cs,
00046                                              Size startIndex,
00047                                              Size endIndex,
00048                                              Size forwardIndex)
00049     {
00050         if (forwardIndex < startIndex)
00051             return 0.0;
00052         if (forwardIndex >= endIndex)
00053             return 0.0;
00054 
00055         Real numerator = cs.discountRatio(startIndex, endIndex)-1;
00056         Real swapAnnuity = annuity(cs, startIndex, endIndex,endIndex);
00057 
00058         Real ratio = cs.rateTaus()[forwardIndex] /
00059             (1 + cs.rateTaus()[forwardIndex] * cs.forwardRate(forwardIndex));
00060 
00061         Real part1 = ratio*(numerator+1)/swapAnnuity;
00062         Real part2 = numerator/(swapAnnuity*swapAnnuity);
00063 
00064         if (forwardIndex >=1)
00065             part2 *= ratio* annuity(cs, startIndex, forwardIndex, endIndex);
00066         else 
00067             part2 = 0.0;
00068 
00069         return part1-part2;
00070     }
