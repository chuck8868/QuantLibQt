00115             for (Size i=1; i<size_; ++i) {

00117                 Size k = rightIndex_[i];
00118                 Size l = bridgeIndex_[i];

00121                         leftWeight_[i] * output[j-1] +

00125                     output[l] =
00126                         rightWeight_[i] * output[k]   +

00129             }
00130             // ...after which, we calculate the variations and

00133                 output[i] -= output[i-1];
