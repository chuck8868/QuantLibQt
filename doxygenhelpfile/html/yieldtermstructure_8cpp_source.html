00072     void YieldTermStructure::setJumps() {
00073         if (jumpDates_.empty() && !jumps_.empty()) { // turn of year dates
00074             jumpDates_.resize(nJumps_);
00075             jumpTimes_.resize(nJumps_);
00076             Year y = referenceDate().year();
00077             for (Size i=0; i<nJumps_; ++i)
00078                 jumpDates_[i] = Date(31, December, y+i);
00079         } else { // fixed dats
00080             QL_REQUIRE(jumpDates_.size()==nJumps_,
00081                        "mismatch between number of jumps (" << nJumps_ <<
00082                        ") and jump dates (" << jumpDates_.size() << ")");
00083         }
00084         for (Size i=0; i<nJumps_; ++i)
00085             jumpTimes_[i] = timeFromReference(jumpDates_[i]);
00086         latestReference_ = referenceDate();
00087     }

00089     DiscountFactor YieldTermStructure::discount(Time t,
00090                                                 bool extrapolate) const {
00091         checkRange(t, extrapolate);
00092 
00093         if (jumps_.empty())
00094             return discountImpl(t);
00095 
00096         DiscountFactor jumpEffect = 1.0;
00097         for (Size i=0; i<nJumps_; ++i) {
00098             if (jumpTimes_[i]>0 && jumpTimes_[i]<t) {
00099                 QL_REQUIRE(jumps_[i]->isValid(),
00100                            "invalid " << io::ordinal(i+1) << " jump quote");
00101                 DiscountFactor thisJump = jumps_[i]->value();
00102                 QL_REQUIRE(thisJump>0.0 && thisJump<=1.0,
00103                            "invalid " << io::ordinal(i+1) << " jump value: " <<
00104                            thisJump);
00105                 jumpEffect *= thisJump;
00106             }
00107         }
00108         return jumpEffect * discountImpl(t);

00142     InterestRate YieldTermStructure::forwardRate(const Date& d1,
00143                                                  const Date& d2,
00144                                                  const DayCounter& dayCounter,
00145                                                  Compounding comp,
00146                                                  Frequency freq,
00147                                                  bool extrapolate) const {
00148         if (d1==d2) {
00149             checkRange(d1, extrapolate);
00150             Time t1 = std::max(timeFromReference(d1) - dt/2.0, 0.0);
00151             Time t2 = t1 + dt;
00152             Real compound =
00153                 discount(t1, true)/discount(t2, true);
00154             // times have been calculated with a possibly different daycounter
00155             // but the difference should not matter for very small times
00156             return InterestRate::impliedRate(compound,
00157                                              dayCounter, comp, freq,
00158                                              dt);
00159         }
00160         QL_REQUIRE(d1 < d2,  d1 << " later than " << d2);
00161         Real compound = discount(d1, extrapolate)/discount(d2, extrapolate);
00162         return InterestRate::impliedRate(compound,
00163                                          dayCounter, comp, freq,
00164                                          d1, d2);
00165     }

00168                                                  Time t2,
00169                                                  Compounding comp,
00170                                                  Frequency freq,
00171                                                  bool extrapolate) const {

00174             checkRange(t1, extrapolate);
