00082         const std::vector<Time>& taus = cs.rateTaus();

00086         for (Size k=0; k<PjPnWk_.rows(); ++k) {

00093                 double sr = cs.cmSwapRate(j+1,spanningFwds_);

00097                 * (sr+displacements_[j+1])

00100                 PjPnWk_[k][j+1] = first

00102                 + third;
00103 
00104                 if (j>=static_cast<Integer>(alive_))
00105                 {
00106                     wkaj_[k][j] = wkaj_[k][j+1] + PjPnWk_[k][j+1]*taus[j];

00108                     if (j+spanningFwds_+1 <= numberOfRates_)
00109                         wkaj_[k][j] -= PjPnWk_[k][endIndex]*taus[endIndex-1];
00110                 }
00111 
00112             }
00113         }
00114 
00115         Real PnOverPN = cs.discountRatio(numberOfRates_, numeraire_);
00116         //Real PnOverPN = 1.0;
00117 
00118         for (Size j=alive_; j<numberOfRates_; ++j)
00119             for (Size k=0; k<numberOfFactors_; ++k)
00120                 wkajN_[k][j] =  wkaj_[k][j]*PnOverPN
00121                     -PjPnWk_[k][numeraire_]*PnOverPN*cs.cmSwapAnnuity(numeraire_,j,spanningFwds_);
