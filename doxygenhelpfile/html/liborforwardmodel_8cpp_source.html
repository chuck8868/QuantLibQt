00063     Real LiborForwardModel::discountBondOption(Option::Type type,
00064                                                Real strike, Time maturity,
00065                                                Time bondMaturity) const {
00066 
00067         const std::vector<Time> & accrualStartTimes
00068             = process_->accrualStartTimes();
00069         const std::vector<Time> & accrualEndTimes
00070             = process_->accrualEndTimes();
00071 
00072         QL_REQUIRE(   accrualStartTimes.front()<= maturity
00073                    && accrualStartTimes.back() >= maturity,
00074                    "capet maturity does not fit to the process");
00075 
00076         const Size i = std::lower_bound(accrualStartTimes.begin(),
00077                                         accrualStartTimes.end(),
00078                                         maturity) - accrualStartTimes.begin();
00079 
00080         QL_REQUIRE(   i<process_->size()
00081                    && std::fabs(maturity - accrualStartTimes[i])
00082                         < 100*std::numeric_limits<Real>::epsilon()
00083                    && std::fabs(bondMaturity - accrualEndTimes[i])
00084                         < 100*std::numeric_limits<Real>::epsilon(),
00085                    "irregular fixings are not (yet) supported");
00086 
00087         const Real tenor     = accrualEndTimes[i] - accrualStartTimes[i];
00088         const Real forward   = process_->initialValues()[i];
00089         const Real capRate   = (1.0/strike - 1.0)/tenor;
00090         const Volatility var = covarProxy_
00091             ->integratedCovariance(i, i, process_->fixingTimes()[i]);
00092         const DiscountFactor dis =
00093             process_->index()->forwardingTermStructure()->discount(bondMaturity);
00094 
00095         const Real black = blackFormula(
00096             (type==Option::Put ? Option::Call : Option::Put),
00097             capRate, forward, std::sqrt(var));
00098 
00099         const Real npv = dis * tenor * black;
00100 
00101         return npv / (1.0 + capRate*tenor);
00102     }

00104     Disposable<Array> LiborForwardModel::w_0(Size alpha, Size beta) const {
00105         Array omega(beta+1,0.0);
00106         QL_REQUIRE(alpha<beta, "alpha needs to be smaller than beta");
00107 
00108         Real s=0.0;
00109         for (Size k=alpha+1; k<=beta; ++k) {
00110             Real b=accrualPeriod_[k];
00111             for (Size j=alpha+1; j<=k; ++j) {
00112                 b*=f_[j];
00113             }
00114             s+=b;
00115         }
00116 
00117         for (Size i=alpha+1; i<=beta; ++i) {
00118             Real a=accrualPeriod_[i];
00119             for (Size j=alpha+1; j<=i; ++j) {
00120                 a*=f_[j];
00121             }
00122 
00123             omega[i] = a/s;
00124         }
00125 
00126         return omega;

00141     // calculating swaption volatility matrix using
00142     // Rebonatos approx. formula. Be aware that this
00143     // matrix is valid only for regular fixings and
00144     // assumes that the fix and floating leg have the
00145     // same frequency
00146     boost::shared_ptr<SwaptionVolatilityMatrix>
00147         LiborForwardModel::getSwaptionVolatilityMatrix() const {
00148         if (swaptionVola) {
00149             return swaptionVola;
00150         }
00151 
00152         const boost::shared_ptr<IborIndex> index = process_->index();
00153         const Date today = process_->fixingDates()[0];
00154 
00155         const Size size=process_->size()/2;
00156         Matrix volatilities(size, size);
00157 
00158         std::vector<Date> exercises(process_->fixingDates().begin() + 1,
00159                                     process_->fixingDates().begin() + size+1);
00160 
00161         std::vector<Period> lengths(size);
00162         for (Size i=0; i < size; ++i) {
00163             lengths[i] = (i+1)*index->tenor();
00164         }
00165 
00166         const Array f = process_->initialValues();
00167         for (Size k=0; k < size; ++k) {
00168             const Size alpha  =k;
00169             const Time t_alpha=process_->fixingTimes()[alpha+1];
00170 
00171             Matrix var(size, size);
00172             for (Size i=alpha+1; i <= k+size; ++i) {
00173                 for (Size j=i; j <= k+size; ++j) {
00174                     var[i-alpha-1][j-alpha-1] = var[j-alpha-1][i-alpha-1] =
00175                         covarProxy_->integratedCovariance(i, j, t_alpha);
00176                 }
00177             }
00178 
00179             for (Size l=1; l <= size; ++l) {
00180                 const Size beta =l + k;
00181                 const Array w = w_0(alpha, beta);
00182 
00183                 Real sum=0.0;
00184                 for (Size i=alpha+1; i <= beta; ++i) {
00185                     for (Size j=alpha+1; j <= beta; ++j) {
00186                         sum+=w[i]*w[j]*f[i]*f[j]*var[i-alpha-1][j-alpha-1];
00187                     }
00188                 }
00189                 volatilities[k][l-1] =
00190                     std::sqrt(sum/t_alpha)/S_0(alpha, beta);
00191             }
00192         }
00193 
00194         return swaptionVola = boost::shared_ptr<SwaptionVolatilityMatrix>(
00195              new SwaptionVolatilityMatrix(today, exercises, lengths,
00196                                           volatilities,
00197                                           index->dayCounter()));
00198     }
