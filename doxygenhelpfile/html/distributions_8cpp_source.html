00036     Real gaussian(Real x) {
00037         Real normFact = sigma*std::sqrt(2*M_PI);
00038         Real dx = x-average;
00039         return std::exp(-dx*dx/(2.0*sigma*sigma))/normFact;
00040     }
00041 
00042     Real gaussianDerivative(Real x) {
00043         Real normFact = sigma*sigma*sigma*std::sqrt(2*M_PI);
00044         Real dx = x-average;
00045         return -dx*std::exp(-dx*dx/(2.0*sigma*sigma))/normFact;
00046     }

00190          xMax = average + numberOfStandardDeviation*sigma;

00201                    std::ptr_fun(gaussianDerivative));
