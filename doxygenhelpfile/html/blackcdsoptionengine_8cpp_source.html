00042     void BlackCdsOptionEngine::calculate() const {
00043 
00044         Date maturityDate = arguments_.swap->coupons().front()->date();
00045         Date exerciseDate = arguments_.exercise->date(0);
00046         QL_REQUIRE(maturityDate > exerciseDate,
00047                    "Underlying CDS should start after option maturity");
00048         Date settlement   = termStructure_->referenceDate();
00049 
00050         Rate spotFwdSpread = arguments_.swap->fairSpread();
00051         Rate swapSpread    = arguments_.swap->runningSpread();
00052 
00053         DayCounter tSDc = termStructure_->dayCounter();
00054 
00055         // The sense of the underlying/option has to be sent this way
00056         // to the Black formula, no sign.
00057         Real riskyAnnuity =
00058             std::fabs(arguments_.swap->couponLegNPV() / swapSpread);
00059         results_.riskyAnnuity = riskyAnnuity;
00060 
00061         Time T = tSDc.yearFraction(settlement, exerciseDate);
00062 
00063         Real stdDev = volatility_->value()  * std::sqrt(T);
00064         Option::Type callPut = (arguments_.side == Protection::Buyer) ?
00065                                                    Option::Call : Option::Put;
00066 
00067         results_.value =
00068             blackFormula(callPut, swapSpread, spotFwdSpread,
00069                          stdDev, riskyAnnuity);
00070 
00071         // if a non knock-out payer option, add front end protection value
00072         if (arguments_.side == Protection::Buyer && !arguments_.knocksOut) {
00073             Real frontEndProtection =
00074                 callPut * arguments_.swap->notional()
00075                 * (1.-recoveryRate_)
00076                 * probability_->defaultProbability(exerciseDate)
00077                 * termStructure_->discount(exerciseDate);
00078             results_.value += frontEndProtection;
00079         }
00080     }
