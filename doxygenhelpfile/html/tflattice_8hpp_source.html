00083     template <class T>
00084     void TsiveriotisFernandesLattice<T>::stepback(
00085                                           Size i,
00086                                           const Array& values,
00087                                           const Array& conversionProbability,
00088                                           const Array& spreadAdjustedRate,
00089                                           Array& newValues,
00090                                           Array& newConversionProbability,
00091                                           Array& newSpreadAdjustedRate) const {
00092 
00093         for (Size j=0; j<this->size(i); j++) {
00094 
00095             // new conversion probability is calculated via backward
00096             // induction using up and down probabilities on tree on
00097             // previous conversion probabilities, ie weighted average
00098             // of previous probabilities.
00099             newConversionProbability[j] =
00100                 this->pd_*conversionProbability[j] +
00101                 this->pu_*conversionProbability[j+1];
00102 
00103             // Use blended discounting rate
00104             newSpreadAdjustedRate[j] =
00105                 newConversionProbability[j] * this->riskFreeRate_ +
00106                 (1-newConversionProbability[j])*(this->riskFreeRate_+creditSpread_);
00107 
00108             newValues[j] =
00109                 (this->pd_*values[j]/(1+(spreadAdjustedRate[j]*this->dt_)))
00110               + (this->pu_*values[j+1]/(1+(spreadAdjustedRate[j+1]*this->dt_)));
00111 
00112         }
00113     }
