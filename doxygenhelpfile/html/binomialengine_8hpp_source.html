00138         // Rollback to second-last step, and get option value (p1) at

00140         option.rollback(grid[1]);

00142         QL_ENSURE(va.size() == 2, "Expect 2 nodes in grid at first step");
00143         Real p1 = va[1];

00146         option.rollback(0.0);

00151         Real delta0 = (p1-p0)/(s1-s0);   // dp/ds
