00038     Date CPIVolatilitySurface::baseDate() const {
00039         // Depends on interpolation, or not, of observed index
00040         // and observation lag with which it was built.
00041         // We want this to work even if the index does not
00042         // have a term structure.
00043         if (indexIsInterpolated()) {
00044             return referenceDate() - observationLag();
00045         } else {
00046             return inflationPeriod(referenceDate() - observationLag(),
00047                                    frequency()).first;
00048         }
00049     }

00080     Volatility CPIVolatilitySurface::volatility(const Date& maturityDate,
00081                                                 Rate strike,
00082                                                 const Period& obsLag,
00083                                                 bool extrapolate) const {
00084 
00085         Period useLag = obsLag;
00086         if (obsLag==Period(-1,Days)) {
00087             useLag = observationLag();
00088         }
00089 
00090         if (indexIsInterpolated()) {
00091             checkRange(maturityDate-useLag, strike, extrapolate);
00092             Time t = timeFromReference(maturityDate-useLag);
00093             return volatilityImpl(t,strike);
00094         } else {
00095             std::pair<Date,Date> dd =
00096                 inflationPeriod(maturityDate-useLag, frequency());
00097             checkRange(dd.first, strike, extrapolate);
00098             Time t = timeFromReference(dd.first);
00099             return volatilityImpl(t,strike);
00100         }
00101     }

00114     Time CPIVolatilitySurface::timeFromBase(const Date& maturityDate,

00120 
00121         Date useDate;
00122         if (indexIsInterpolated()) {
