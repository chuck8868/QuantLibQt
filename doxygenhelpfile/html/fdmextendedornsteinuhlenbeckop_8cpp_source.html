00057     void FdmExtendedOrnsteinUhlenbackOp::setTime(Time t1, Time t2) {
00058         const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
00059 
00060         const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
00061         const FdmLinearOpIterator endIter = layout->end();
00062 
00063         Array drift(layout->size());
00064         for (FdmLinearOpIterator iter = layout->begin();
00065              iter!=endIter; ++iter) {
00066             const Size i = iter.index();
00067             drift[i] = process_->drift(0.5*(t1+t2), x_[i]);
00068         }
00069         mapX_.axpyb(drift, dxMap_, dxxMap_, Array(1, -r));
00070     }
