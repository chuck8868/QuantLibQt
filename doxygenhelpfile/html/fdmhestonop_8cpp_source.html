00030     FdmHestonEquityPart::FdmHestonEquityPart(
00031         const boost::shared_ptr<FdmMesher>& mesher,
00032         const boost::shared_ptr<YieldTermStructure>& rTS,
00033         const boost::shared_ptr<YieldTermStructure>& qTS,
00034         const boost::shared_ptr<FdmQuantoHelper>& quantoHelper)
00035     : varianceValues_(0.5*mesher->locations(1)),
00036       dxMap_ (FirstDerivativeOp(0, mesher)),
00037       dxxMap_(SecondDerivativeOp(0, mesher).mult(0.5*mesher->locations(1))),
00038       mapT_  (0, mesher),
00039       mesher_(mesher),
00040       rTS_(rTS),
00041       qTS_(qTS),
00042       quantoHelper_(quantoHelper) {
00043 
00044         // on the boundary s_min and s_max the second derivative
00045         // d^2V/dS^2 is zero and due to Ito's Lemma the variance term
00046         // in the drift should vanish.
00047         boost::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
00048         FdmLinearOpIterator endIter = layout->end();
00049         for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
00050             ++iter) {
00051             if (   iter.coordinates()[0] == 0
00052                 || iter.coordinates()[0] == layout->dim()[0]-1) {
00053                 varianceValues_[iter.index()] = 0.0;
00054             }
00055         }
00056         volatilityValues_ = Sqrt(2*varianceValues_);
00057     }

00059     void FdmHestonEquityPart::setTime(Time t1, Time t2) {
00060         const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
00061         const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
00062 
00063         if (quantoHelper_) {
00064             mapT_.axpyb(r - q - varianceValues_
00065                 - quantoHelper_->quantoAdjustment(
00066                 volatilityValues_, t1, t2),
00067                 dxMap_, dxxMap_, Array(1, -0.5*r));
00068         }
00069         else {
00070             mapT_.axpyb(r - q - varianceValues_, dxMap_, dxxMap_, Array(1, -0.5*r));
00071         }
00072     }
