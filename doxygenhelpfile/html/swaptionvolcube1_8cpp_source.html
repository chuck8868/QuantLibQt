00496                    cube ATM volatilities corresponding to not quoted smile sections (that
00497                    is ATM volatilities in dense cube) are no more exactly the quoted values,

00503                    The extent of this error depends on the quality of the fit: in case of
00504                    good fits it is negligibile.                  
00505                 */
00506             }
00507         }
00508 
00509         for (Size k=0; k<nStrikes_; k++){
00510             const Real strike = atmForward + strikeSpreads_[k];
00511             const Real moneyness = atmForward/strike;
00512 
00513             Matrix strikes(2,2,0.);
00514             Matrix spreadVols(2,2,0.);
00515             for (Size i=0; i<2; i++){
00516                 for (Size j=0; j<2; j++){
00517                     strikes[i][j] = atmForwards[i][j]/moneyness;
00518                     spreadVols[i][j] =
00519                         smiles[i][j]->volatility(strikes[i][j]) - atmVols[i][j];
00520                 }
00521             }
00522            Cube localInterpolator(optionsDateNodes, swapTenorNodes,
00523                                   optionsNodes, swapLengthsNodes, 1);
00524            localInterpolator.setLayer(0, spreadVols);
00525            localInterpolator.updateInterpolators();
00526 
00527            result.push_back(localInterpolator(atmOptionTime, atmTimeLength)[0]);
00528         }
00529         return result;
