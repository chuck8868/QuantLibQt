00063     void GaussianRandomDefaultModel::nextSequence(Real tmax) {
00064         const std::vector<Real>& values = rsg_.nextSequence().value;
00065         Real a = sqrt(copula_->correlation());
00066         for (Size j = 0; j < pool_->size(); j++) {
00067             const string name = pool_->names()[j];
00068             const Handle<DefaultProbabilityTermStructure>&
00069                 dts = pool_->get(name).defaultProbability(defaultKeys_[j]);
00070 
00071             Real y = a * values[0] + sqrt(1-a*a) * values[j+1];
00072             Real p = CumulativeNormalDistribution()(y);
00073 
00074             if (dts->defaultProbability(tmax) < p)
00075                 pool_->setTime(name, tmax+1);
00076             else
00077                 pool_->setTime(name, Brent().solve(Root(dts,p),accuracy_,0,1));
00078         }
00079     }
