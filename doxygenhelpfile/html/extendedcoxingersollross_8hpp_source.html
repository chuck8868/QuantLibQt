00122             Real value(const Array&, Time t) const {
00123                 Rate forwardRate =
00124                     termStructure_->forwardRate(t, t, Continuous, NoFrequency);
00125                 Real h = std::sqrt(k_*k_ + 2.0*sigma_*sigma_);
