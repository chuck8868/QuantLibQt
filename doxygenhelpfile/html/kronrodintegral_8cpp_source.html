00025     static Real rescaleError(Real err,
00026                              const Real resultAbs,
00027                              const Real resultAsc) {
00028         err = std::fabs(err) ;
00029         if (resultAsc != 0 && err != 0){
00030             Real scale = std::pow((200 * err / resultAsc), 1.5) ;
00031             if (scale < 1)
00032                 err = resultAsc * scale ;
00033             else
00034                 err = resultAsc ;
00035             }
00036         if (resultAbs > QL_MIN_POSITIVE_REAL / (50 * QL_EPSILON )){
00037             Real min_err = 50 * QL_EPSILON  * resultAbs ;
00038             if (min_err > err)
00039                 err = min_err ;
00040             }
00041         return err ;
00042     }

00229     GaussKronrodNonAdaptive::integrate(const boost::function<Real (Real)>& f,
00230                                        Real a,
00231                                        Real b) const {
00232         Real result;
00233         //Size neval;
00234         Real fv1[5], fv2[5], fv3[5], fv4[5];
00235         Real savfun[21];  /* array of function values which have been computed */
00236         Real res10, res21, res43, res87;    /* 10, 21, 43 and 87 point results */
00237         Real err;
00238         Real resAbs; /* approximation to the integral of abs(f) */
00239         Real resasc; /* approximation to the integral of abs(f-i/(b-a)) */
00240         int k ;
00241 
00242         QL_REQUIRE(a<b, "b must be greater than a)");
00243 
00244         const Real halfLength = 0.5 * (b - a);
00245         const Real center = 0.5 * (b + a);
00246         const Real fCenter = f(center);
00247 
00248         // Compute the integral using the 10- and 21-point formula.
00249 
00250         res10 = 0;
00251         res21 = w21b[5] * fCenter;
00252         resAbs = w21b[5] * std::fabs(fCenter);
00253 
00254         for (k = 0; k < 5; k++) {
00255             Real abscissa = halfLength * x1[k];
00256             Real fval1 = f(center + abscissa);
00257             Real fval2 = f(center - abscissa);
00258             Real fval = fval1 + fval2;
00259             res10 += w10[k] * fval;
00260             res21 += w21a[k] * fval;
00261             resAbs += w21a[k] * (std::fabs(fval1) + std::fabs(fval2));
00262             savfun[k] = fval;
00263             fv1[k] = fval1;
00264             fv2[k] = fval2;
00265         }
00266 
00267         for (k = 0; k < 5; k++) {
00268             Real abscissa = halfLength * x2[k];
00269             Real fval1 = f(center + abscissa);
00270             Real fval2 = f(center - abscissa);
00271             Real fval = fval1 + fval2;
00272             res21 += w21b[k] * fval;
00273             resAbs += w21b[k] * (std::fabs(fval1) + std::fabs(fval2));
00274             savfun[k + 5] = fval;
00275             fv3[k] = fval1;
00276             fv4[k] = fval2;
00277         }
00278 
00279         result = res21 * halfLength;
00280         resAbs *= halfLength ;
00281         Real mean = 0.5 * res21;
00282         resasc = w21b[5] * std::fabs(fCenter - mean);
00283 
00284         for (k = 0; k < 5; k++)
00285             resasc += (w21a[k] * (std::fabs(fv1[k] - mean)
00286                         + std::fabs(fv2[k] - mean))
00287                         + w21b[k] * (std::fabs(fv3[k] - mean)
00288                         + std::fabs(fv4[k] - mean)));
00289 
00290         err = rescaleError ((res21 - res10) * halfLength, resAbs, resasc) ;
00291         resasc *= halfLength ;
00292 
00293         // test for convergence.
00294         if (err < absoluteAccuracy() || err < relativeAccuracy() * std::fabs(result)){
00295             setAbsoluteError(err);
00296             setNumberOfEvaluations(21);
00297             return result;
00298         }
00299 
00300         /* compute the integral using the 43-point formula. */
00301 
00302         res43 = w43b[11] * fCenter;
00303 
00304         for (k = 0; k < 10; k++)
00305             res43 += savfun[k] * w43a[k];
00306 
00307         for (k = 0; k < 11; k++){
00308             Real abscissa = halfLength * x3[k];
00309             Real fval = (f(center + abscissa)
00310                 + f(center - abscissa));
00311             res43 += fval * w43b[k];
00312             savfun[k + 10] = fval;
00313             }
00314 
00315         // test for convergence.
00316 
00317         result = res43 * halfLength;
00318         err = rescaleError ((res43 - res21) * halfLength, resAbs, resasc);
00319 
00320        if (err < absoluteAccuracy() || err < relativeAccuracy() * std::fabs(result)){
00321             setAbsoluteError(err);
00322             setNumberOfEvaluations(43);
00323             return result;
00324         }
00325 
00326         /* compute the integral using the 87-point formula. */
00327 
00328         res87 = w87b[22] * fCenter;
00329 
00330         for (k = 0; k < 21; k++)
00331             res87 += savfun[k] * w87a[k];
00332 
00333         for (k = 0; k < 22; k++){
00334             Real abscissa = halfLength * x4[k];
00335             res87 += w87b[k] * (f(center + abscissa)
00336                 + f(center - abscissa));
00337         }
00338 
00339         // test for convergence.
00340         result = res87 * halfLength ;
00341         err = rescaleError ((res87 - res43) * halfLength, resAbs, resasc);
00342 
00343         setAbsoluteError(err);
00344         setNumberOfEvaluations(87);
00345         return result;

00381     Real GaussKronrodAdaptive::integrateRecursively(
00382                                     const boost::function<Real (Real)>& f,
00383                                     Real a,
00384                                     Real b,
00385                                     Real tolerance) const {
00386 
00387             Real halflength = (b - a) / 2;
00388             Real center = (a + b) / 2;
00389 
00390             Real g7; // will be result of G7 integral
00391             Real k15; // will be result of K15 integral
00392 
00393             Real t, fsum; // t (abscissa) and f(t)
00394             Real fc = f(center);
00395             g7 = fc * g7w[0];
00396             k15 = fc * k15w[0];
00397 
00398             // calculate g7 and half of k15
00399             Integer j, j2;
00400             for (j = 1, j2 = 2; j < 4; j++, j2 += 2) {
00401                 t = halflength * k15t[j2];
00402                 fsum = f(center - t) + f(center + t);
00403                 g7  += fsum * g7w[j];
00404                 k15 += fsum * k15w[j2];
00405             }
00406 
00407             // calculate other half of k15
00408             for (j2 = 1; j2 < 8; j2 += 2) {
00409                 t = halflength * k15t[j2];
00410                 fsum = f(center - t) + f(center + t);
00411                 k15 += fsum * k15w[j2];
00412             }
00413 
00414             // multiply by (a - b) / 2
00415             g7 = halflength * g7;
00416             k15 = halflength * k15;
00417 
00418             // 15 more function evaluations have been used
00419             increaseNumberOfEvaluations(15);
00420 
00421             // error is <= k15 - g7
00422             // if error is larger than tolerance then split the interval
00423             // in two and integrate recursively
00424             if (std::fabs(k15 - g7) < tolerance) {
00425                 return k15;
00426             } else {
00427                 QL_REQUIRE(numberOfEvaluations()+30 <=
00428                            maxEvaluations(),
00429                            "maximum number of function evaluations "
00430                            "exceeded");
00431                 return integrateRecursively(f, a, center, tolerance/2)
00432                     + integrateRecursively(f, center, b, tolerance/2);
00433             }
00434         }
