00040     void CMSwapCurveState::setOnCMSwapRates(const std::vector<Rate>& rates,
00041                                             Size firstValidIndex) {
00042         QL_REQUIRE(rates.size()==numberOfRates_,
00043                    "rates mismatch: " <<
00044                    numberOfRates_ << " required, " <<
00045                    rates.size() << " provided");
00046         QL_REQUIRE(firstValidIndex<numberOfRates_,
00047                    "first valid index must be less than " <<
00048                    numberOfRates_ << ": " <<
00049                    firstValidIndex << " not allowed");
00050 
00051         // first copy input...
00052         first_ = firstValidIndex;
00053         std::copy(rates.begin()+first_, rates.end(),
00054                   cmSwapRates_.begin()+first_);
00055         // ...then calculate discount ratios and annuities
00056 
00057         // taken care at constructor time
00058         //discRatios_[numberOfRates_] = 1.0;
00059         //cmSwapAnnuities_[numberOfRates_-1] = rateTaus_[numberOfRates_-1];
00060 
00061         // assume i+1 and do i
00062         Integer oldAnnuityEndIndex = numberOfRates_;
00063         for (Size i=numberOfRates_-1; i>first_; --i)
00064         {
00065             // formula 6.1 Joshi Liesch
00066             Integer endIndex = std::min(i + spanningFwds_,numberOfRates_);
00067             Integer annuityEndIndex = std::min(i + spanningFwds_-1,numberOfRates_);
00068 
00069             discRatios_[i] = discRatios_[endIndex] +
00070             cmSwapRates_[i]*cmSwapAnnuities_[i];
00071             cmSwapAnnuities_[i-1]= cmSwapAnnuities_[i]
00072             +discRatios_[i] * rateTaus_[i-1];
00073 
00074             if (annuityEndIndex < oldAnnuityEndIndex)
00075               cmSwapAnnuities_[i-1]-=discRatios_[oldAnnuityEndIndex] * rateTaus_[oldAnnuityEndIndex-1 ];
00076 
00077             oldAnnuityEndIndex = annuityEndIndex;
00078         }
00079         Integer endIndex = std::min(first_ + spanningFwds_,numberOfRates_);
00080 
00081         discRatios_[first_]= discRatios_[endIndex] +
00082         cmSwapRates_[first_]*cmSwapAnnuities_[first_];
00083 
00084         // lazy evaluation of forward and coterminal swap rates&annuities
00085     }

00087     Real CMSwapCurveState::discountRatio(Size i, Size j) const {
00088         QL_REQUIRE(first_<numberOfRates_, "curve state not initialized yet");
00089         QL_REQUIRE(std::min(i, j)>=first_, "invalid index");
00090         QL_REQUIRE(std::max(i, j)<=numberOfRates_, "invalid index");
00091         return discRatios_[i]/discRatios_[j];
