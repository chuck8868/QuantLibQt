00180     Rate DigitalCoupon::callOptionRate() const {
00181 
00182         Rate callOptionRate = Rate(0.);
00183         if(hasCallStrike_) {
00184             // Step function
00185             callOptionRate = isCallCashOrNothing_ ? callDigitalPayoff_ : callStrike_;
00186             CappedFlooredCoupon next(underlying_, callStrike_ + callRightEps_);
00187             CappedFlooredCoupon previous(underlying_, callStrike_ - callLeftEps_);
00188             callOptionRate *= (next.rate() - previous.rate())
00189                             / (callLeftEps_ + callRightEps_);
00190             if (!isCallCashOrNothing_) {
00191                 // Call
00192                 CappedFlooredCoupon atStrike(underlying_, callStrike_);
00193                 Rate call = underlying_->rate() - atStrike.rate();
00194                 // Sum up
00195                 callOptionRate += call;
00196             }
00197         }
00198         return callOptionRate;
00199     }

00222     Rate DigitalCoupon::rate() const {
00223 
00224         QL_REQUIRE(underlying_->pricer(), "pricer not set");
00225 
00226         Date fixingDate = underlying_->fixingDate();
00227         Date today = Settings::instance().evaluationDate();
00228         bool enforceTodaysHistoricFixings =
00229             Settings::instance().enforcesTodaysHistoricFixings();
00230         Rate underlyingRate = underlying_->rate();
00231         if (fixingDate < today ||
00232             ((fixingDate == today) && enforceTodaysHistoricFixings)) {
00233             // must have been fixed
00234             return underlyingRate + callCsi_ * callPayoff() + putCsi_  * putPayoff();
00235         }
00236         if (fixingDate == today) {
00237             // might have been fixed
00238             Rate pastFixing =
00239                 IndexManager::instance().getHistory((underlying_->index())->name())[fixingDate];
00240             if (pastFixing != Null<Real>()) {
00241                 return underlyingRate + callCsi_ * callPayoff() + putCsi_  * putPayoff();
00242             } else
00243                 return underlyingRate + callCsi_ * callOptionRate() + putCsi_ * putOptionRate();
00244         }
00245         return underlyingRate + callCsi_ * callOptionRate() + putCsi_ * putOptionRate();
00246     }

00294     Rate DigitalCoupon::callPayoff() const {
00295         // to use only if index has fixed
00296         Rate payoff(0.);
00297         if(hasCallStrike_) {
00298             Rate underlyingRate = underlying_->rate();
00299             if ( (underlyingRate - callStrike_) > 1.e-16 ) {
00300                 payoff = isCallCashOrNothing_ ? callDigitalPayoff_ : underlyingRate;
00301             } else {
00302                 if (isCallATMIncluded_) {
00303                     if ( std::abs(callStrike_ - underlyingRate) <= 1.e-16 )
00304                         payoff = isCallCashOrNothing_ ? callDigitalPayoff_ : underlyingRate;
00305                 }
00306             }
00307         }
00308         return payoff;
00309     }
