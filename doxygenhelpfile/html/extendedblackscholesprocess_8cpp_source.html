00034     Real ExtendedBlackScholesMertonProcess::drift(Time t, Real x) const {
00035         Real sigma = diffusion(t,x);
00036         // we could be more anticipatory if we know the right dt
00037         // for which the drift will be used
00038         Time t1 = t + 0.0001;
00039         return riskFreeRate()->forwardRate(t,t1,Continuous,NoFrequency,true)
00040              - dividendYield()->forwardRate(t,t1,Continuous,NoFrequency,true)
00041              - 0.5 * sigma * sigma;
00042     }

00048     Real ExtendedBlackScholesMertonProcess::evolve(Time t0, Real x0,
00049                                                    Time dt, Real dw) const {
00050         Real predictor, sigma0, sigma1;
00051         Time t1;
00052         Rate rate0, rate1;
00053         Real driftterm, diffusionterm, corrector;
00054         switch (discretization_) {
00055           case Milstein:
00056             // Milstein scheme
00057             return apply(x0, drift(t0, x0)*dt
00058                            + 0.5*std::pow(diffusion(t0, x0),2)*(dw*dw-1)*dt
00059                            + diffusion(t0,x0)*std::sqrt(dt)*dw);
00060             break;
00061           case Euler:
00062             // Usual Euler scheme
00063             return apply(expectation(t0,x0,dt), stdDeviation(t0,x0,dt)*dw);
00064             break;
00065           case PredictorCorrector:
00066             // Predictor-Corrector scheme with equal weighting
00067             predictor =
00068                 apply(expectation(t0,x0,dt), stdDeviation(t0,x0,dt)*dw);
00069             t1 = t0 + 0.0001;
00070             sigma0 = diffusion(t0,x0);
00071             sigma1 = diffusion(t0+dt,predictor);
00072             rate0 =
00073                 riskFreeRate()->forwardRate(t0,t1,Continuous,NoFrequency,true)
00074               - dividendYield()->forwardRate(t0,t1,Continuous,NoFrequency,true)
00075               - 0.5*std::pow(sigma0,2);
00076             rate1 =
00077                 riskFreeRate()->forwardRate(t0+dt,t1+dt,Continuous,
00078                                             NoFrequency,true)
00079               - dividendYield()->forwardRate(t0+dt,t1+dt,
00080                                              Continuous,NoFrequency,true)
00081               - 0.5*std::pow(sigma1,2);
00082             driftterm = 0.5*rate1+0.5*rate0;
00083             diffusionterm = 0.5*(sigma1+sigma0);
00084             corrector =
00085                 apply(x0,driftterm*dt+diffusionterm*std::sqrt(dt)*dw);
00086             return corrector;
00087             break;
00088           default:
00089             QL_FAIL("unknown discretization scheme");
00090         }
00091     }
