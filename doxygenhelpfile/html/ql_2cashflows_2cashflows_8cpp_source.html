00179         template<typename Iter>
00180         Rate aggregateRate(const Leg& leg,
00181                            Iter first,
00182                            Iter last) {
00183             if (first==last) return 0.0;
00184 
00185             Date paymentDate = (*first)->date();
00186             bool firstCouponFound = false;
00187             Real nominal = 0.0;
00188             Time accrualPeriod = 0.0;
00189             DayCounter dc;
00190             Rate result = 0.0;
00191             for (; first<last && (*first)->date()==paymentDate; ++first) {
00192                 shared_ptr<Coupon> cp = dynamic_pointer_cast<Coupon>(*first);
00193                 if (cp) {
00194                     if (firstCouponFound) {
00195                         QL_REQUIRE(nominal       == cp->nominal() &&
00196                                    accrualPeriod == cp->accrualPeriod() &&
00197                                    dc            == cp->dayCounter(),
00198                                    "cannot aggregate two different coupons on "
00199                                    << paymentDate);
00200                     } else {
00201                         firstCouponFound = true;
00202                         nominal = cp->nominal();
00203                         accrualPeriod = cp->accrualPeriod();
00204                         dc = cp->dayCounter();
00205                     }
00206                     result += cp->rate();
00207                 }
00208             }
00209             QL_ENSURE(firstCouponFound,
00210                       "no coupon paid at cashflow date " << paymentDate);
00211             return result;
00212         }

00606

00610                         dPdy -= c * B*B * t;

00632                 return 0.0;

00745     Real CashFlows::npv(const Leg& leg,
00746                         const InterestRate& y,
00747                         bool includeSettlementDateFlows,
00748                         Date settlementDate,
00749                         Date npvDate) {
00750 
00751         if (leg.empty())
00752             return 0.0;
00753 
00754         if (settlementDate == Date())
00755             settlementDate = Settings::instance().evaluationDate();
00756 
00757         if (npvDate == Date())
00758             npvDate = settlementDate;
00759 
00760         Real npv = 0.0;
00761         DiscountFactor discount = 1.0;
00762         Date lastDate = Date();
00763 
00764         for (Size i=0; i<leg.size(); ++i) {
00765             if (leg[i]->hasOccurred(settlementDate,
00766                                     includeSettlementDateFlows))
00767                 continue;
00768 
00769             Date couponDate = leg[i]->date();
00770             Real amount = leg[i]->amount();
00771             if (lastDate == Date()) {
00772                 // first not-expired coupon
00773                 if (i > 0) {
00774                     lastDate = leg[i-1]->date();
00775                 } else {
00776                     shared_ptr<Coupon> coupon =
00777                         boost::dynamic_pointer_cast<Coupon>(leg[i]);
00778                     if (coupon)
00779                         lastDate = coupon->accrualStartDate();
00780                     else
00781                         lastDate = couponDate - 1*Years;
00782                 }
00783                 discount *= y.discountFactor(npvDate, couponDate,
00784                                              lastDate, couponDate);
00785             } else  {
00786                 discount *= y.discountFactor(lastDate, couponDate);
00787             }
00788             lastDate = couponDate;
00789 
00790             npv += amount * discount;
00791         }
00792 
00793         return npv;
00794     }

00915     Real CashFlows::convexity(const Leg& leg,
00916                               const InterestRate& y,
00917                               bool includeSettlementDateFlows,
00918                               Date settlementDate,
00919                               Date npvDate) {
00920         if (leg.empty())
00921             return 0.0;
00922 
00923         if (settlementDate == Date())
00924             settlementDate = Settings::instance().evaluationDate();
00925 
00926         if (npvDate == Date())
00927             npvDate = settlementDate;
00928 
00929         const DayCounter& dc = y.dayCounter();
00930 
00931         Real P = 0.0;
00932         Real d2Pdy2 = 0.0;
00933         Rate r = y.rate();
00934         Natural N = y.frequency();
00935         for (Size i=0; i<leg.size(); ++i) {
00936             if (!leg[i]->hasOccurred(settlementDate,
00937                                      includeSettlementDateFlows)) {
00938                 Time t = dc.yearFraction(npvDate,
00939                                          leg[i]->date());
00940                 Real c = leg[i]->amount();
00941                 DiscountFactor B = y.discountFactor(t);
00942                 P += c * B;
00943                 switch (y.compounding()) {
00944                   case Simple:
00945                     d2Pdy2 += c * 2.0*B*B*B*t*t;
00946                     break;
00947                   case Compounded:
00948                     d2Pdy2 += c * B*t*(N*t+1)/(N*(1+r/N)*(1+r/N));
00949                     break;
00950                   case Continuous:
00951                     d2Pdy2 += c * B*t*t;
00952                     break;
00953                   case SimpleThenCompounded:
00954                     if (t<=1.0/N)
00955                         d2Pdy2 += c * 2.0*B*B*B*t*t;
00956                     else
00957                         d2Pdy2 += c * B*t*(N*t+1)/(N*(1+r/N)*(1+r/N));
00958                     break;
00959                   default:
00960                     QL_FAIL("unknown compounding convention (" <<
00961                             Integer(y.compounding()) << ")");
00962                 }
00963             }
00964         }
00965 
00966         if (P == 0.0)
00967             // no cashflows
00968             return 0.0;
00969 
00970         return d2Pdy2/P;
00971     }
