00154     ObjectiveFunction::operator()(Volatility guess) const {
00155 
00156         vvec_[1] = guess;
00157         vvec_[0] = guess - slope_ * (tvec_[1] - tvec_[0]) * guess;
00158         // could have Interpolator1D instead of Linear
00159         boost::shared_ptr<InterpolatedYoYOptionletVolatilityCurve<Linear> >
00160         vCurve(
00161             new InterpolatedYoYOptionletVolatilityCurve<Linear>(
00162                                                0, TARGET(), ModifiedFollowing,
00163                                                Actual365Fixed(), lag_,
00164                                                frequency_, indexIsInterpolated_,
00165                                                dvec_, vvec_,
00166                                                -1.0, 3.0) ); // strike limits
00167         Handle<YoYOptionletVolatilitySurface> hCurve(vCurve);
00168         p_->setVolatility(hCurve);
00169         // hopefully this gets to the pricer ... then
00170         return priceToMatch_ - capfloor_.NPV();
