00105     Real GaussJacobiPolynomial::alpha(Size i) const {
00106         Real num = beta_*beta_ - alpha_*alpha_;
00107         Real denom = (2.0*i+alpha_+beta_)*(2.0*i+alpha_+beta_+2);
00108 
00109         if (!denom) {
00110             if (num) {
00111                 QL_FAIL("can't compute a_k for jacobi integration\n");
00112             }
00113             else {
00114                 // l'Hospital
00115                 num  = 2*beta_;
00116                 denom= 2*(2.0*i+alpha_+beta_+1);
00117 
00118                 QL_ASSERT(denom, "can't compute a_k for jacobi integration\n");
00119             }
00120         }
00121

00125     Real GaussJacobiPolynomial::beta(Size i) const {
00126         Real num = 4.0*i*(i+alpha_)*(i+beta_)*(i+alpha_+beta_);
00127         Real denom = (2.0*i+alpha_+beta_)*(2.0*i+alpha_+beta_)
00128                    * ((2.0*i+alpha_+beta_)*(2.0*i+alpha_+beta_)-1);
00129 
00130         if (!denom) {
00131             if (num) {
00132                 QL_FAIL("can't compute b_k for jacobi integration\n");
00133             } else {
00134                 // l'Hospital
00135                 num  = 4.0*i*(i+beta_)* (2.0*i+2*alpha_+beta_);
00136                 denom= 2.0*(2.0*i+alpha_+beta_);
00137                 denom*=denom-1;
00138                 QL_ASSERT(denom, "can't compute b_k for jacobi integration\n");
00139             }
00140         }
