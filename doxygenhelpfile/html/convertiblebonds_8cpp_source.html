00305     /* a zero-coupon convertible bond with no credit spread is

00308     BOOST_MESSAGE(

00310 
00311     CommonVars vars;
00312

00323             new BinomialVanillaEngine<CoxRossRubinstein>(vars.process,

00330                       new PlainVanillaPayoff(Option::Call, conversionStrike));

00332     Schedule schedule = MakeSchedule().from(vars.issueDate)
00333                                       .to(vars.maturityDate)
00334                                       .withFrequency(Once)
00335                                       .withCalendar(vars.calendar)

00339                                      vars.no_dividends, vars.no_callability,
00340                                      vars.creditSpread,

00342                                      vars.dayCounter, schedule,
00343                                      vars.redemption);

00345 
00346     VanillaOption euOption(payoff, euExercise);

00350 
00351     Real expected = vars.faceAmount/100.0 *
