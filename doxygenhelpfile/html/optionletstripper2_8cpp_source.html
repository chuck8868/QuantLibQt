00122     std::vector<Volatility> OptionletStripper2::spreadsVolImplied() const {
00123 
00124         Brent solver;
00125         std::vector<Volatility> result(nOptionExpiries_);
00126         Volatility guess = 0.0001, minSpread = -0.1, maxSpread = 0.1;
00127         for (Size j=0; j<nOptionExpiries_; ++j) {
00128             ObjectiveFunction f(stripper1_, caps_[j], atmCapFloorPrices_[j]);
00129             solver.setMaxEvaluations(maxEvaluations_);
00130             Volatility root = solver.solve(f, accuracy_, guess,
00131                                            minSpread, maxSpread);
00132             result[j] = root;
00133         }
00134         return result;
00135     }
