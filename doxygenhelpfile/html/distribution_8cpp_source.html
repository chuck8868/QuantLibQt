00031     Distribution::Distribution (int nBuckets, Real xmin, Real xmax)
00032     //-------------------------------------------------------------------------
00033         : size_(nBuckets),
00034           xmin_(xmin), xmax_(xmax), count_(nBuckets),
00035           x_(nBuckets,0), dx_(nBuckets,0),
00036           density_(nBuckets,0),
00037           cumulativeDensity_(nBuckets,0),
00038           excessProbability_(nBuckets,0),
00039           cumulativeExcessProbability_(nBuckets,0),
00040           average_(nBuckets,0),
00041           overFlow_(0), underFlow_(0),
00042           isNormalized_(false) {
00043         for (int i = 0; i < nBuckets; i++) {
00044             dx_[i] = (xmax - xmin) / nBuckets;
00045             x_[i] = (i == 0 ? xmin : x_[i-1] + dx_[i-1]);
00046         }
00047         // ensure we match exactly the domain, otherwise we might fail the
00048         //   locate test because of precission mismatches
00049         dx_.back() = xmax - x_.back();
00050     }
