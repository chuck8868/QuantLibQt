00031     Fdm2DimSolver::Fdm2DimSolver(
00032                              const FdmSolverDesc& solverDesc,
00033                              const FdmSchemeDesc& schemeDesc,
00034                              const boost::shared_ptr<FdmLinearOpComposite>& op)
00035     : solverDesc_(solverDesc),
00036       schemeDesc_(schemeDesc),
00037       op_(op),
00038       thetaCondition_(new FdmSnapshotCondition(
00039         0.99*std::min(1.0/365.0,
00040            solverDesc.condition->stoppingTimes().empty()
00041                     ? solverDesc.maturity
00042                     : solverDesc.condition->stoppingTimes().front()))),
00043       conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
00044                                                          solverDesc.condition)),
00045       initialValues_(solverDesc.mesher->layout()->size()),
00046       resultValues_ (solverDesc.mesher->layout()->dim()[1],
00047                      solverDesc.mesher->layout()->dim()[0]) {
00048 
00049         const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
00050         const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
00051 
00052         x_.reserve(layout->dim()[0]);
00053         y_.reserve(layout->dim()[1]);
00054 
00055         const FdmLinearOpIterator endIter = layout->end();
00056         for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
00057              ++iter) {
00058             initialValues_[iter.index()]
00059                  = solverDesc_.calculator->avgInnerValue(iter,
00060                                                          solverDesc.maturity);
00061 
00062             if (!iter.coordinates()[1]) {
00063                 x_.push_back(mesher->location(iter, 0));
00064             }
00065             if (!iter.coordinates()[0]) {
00066                 y_.push_back(mesher->location(iter, 1));
00067             }
00068         }
00069     }

00072     void Fdm2DimSolver::performCalculations() const {
00073         Array rhs(initialValues_.size());
00074         std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
00075 
00076         FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
00077             .rollback(rhs, solverDesc_.maturity, 0.0,
00078                       solverDesc_.timeSteps, solverDesc_.dampingSteps);
00079 
00080         std::copy(rhs.begin(), rhs.end(), resultValues_.begin());
00081         interpolation_ = boost::shared_ptr<BicubicSpline> (
00082             new BicubicSpline(x_.begin(), x_.end(),
00083                               y_.begin(), y_.end(),
00084                               resultValues_));
00085     }
