00151                    "Stochastic process array required");

00154             boost::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(

00157 
00158         boost::shared_ptr<EarlyExercise> exercise =

00160                 this->arguments_.exercise);
00161         QL_REQUIRE(exercise, "wrong exercise given");
00162         QL_REQUIRE(!exercise->payoffAtExpiry(),
00163                    "payoff at expiry not handled");
00164 
00165         boost::shared_ptr<AmericanBasketPathPricer> earlyExercisePathPricer(
00166             new AmericanBasketPathPricer(processArray->size(),
00167                                          this->arguments_.payoff));

00169         return boost::shared_ptr<LongstaffSchwartzPathPricer<MultiPath> > (
00170              new LongstaffSchwartzPathPricer<MultiPath>(
