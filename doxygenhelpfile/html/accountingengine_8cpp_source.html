00053     Real AccountingEngine::singlePathValues(std::vector<Real>& values) {
00054         std::fill(numerairesHeld_.begin(), numerairesHeld_.end(), 0.0);
00055         Real weight = evolver_->startNewPath();
00056         product_->reset();
00057         Real principalInNumerairePortfolio = 1.0;
00058 
00059         bool done = false;
00060         do {
00061             Size thisStep = evolver_->currentStep();
00062             weight *= evolver_->advanceStep();
00063             done = product_->nextTimeStep(evolver_->currentState(),
00064                                           numberCashFlowsThisStep_,
00065                                           cashFlowsGenerated_);
00066             Size numeraire =
00067                 evolver_->numeraires()[thisStep];
00068 
00069             // for each product...
00070             for (Size i=0; i<numberProducts_; ++i) {
00071                 // ...and each cash flow...
00072                 const std::vector<MarketModelMultiProduct::CashFlow>& cashflows =
00073                     cashFlowsGenerated_[i];
00074                 for (Size j=0; j<numberCashFlowsThisStep_[i]; ++j) {
00075                     // ...convert the cash flow to numeraires.
00076                     // This is done by calculating the number of
00077                     // numeraire bonds corresponding to such cash flow...
00078                     const MarketModelDiscounter& discounter =
00079                         discounters_[cashflows[j].timeIndex];
00080 
00081                     Real bonds = cashflows[j].amount *
00082                         discounter.numeraireBonds(evolver_->currentState(),
00083                                                   numeraire);
00084 
00085                     // ...and adding the newly bought bonds to the number
00086                     // of numeraires held.
00087                     numerairesHeld_[i] += bonds/principalInNumerairePortfolio;
00088                 }
00089             }
00090 
00091             if (!done) {
00092 
00093                 // The numeraire might change between steps. This implies
00094                 // that we might have to convert the numeraire bonds for
00095                 // this step into a corresponding amount of numeraire
00096                 // bonds for the next step. This can be done by changing
00097                 // the principal of the numeraire and updating the number
00098                 // of bonds in the numeraire portfolio accordingly.
00099 
00100                 Size nextNumeraire = evolver_->numeraires()[thisStep+1];
00101 
00102                 principalInNumerairePortfolio *=
00103                     evolver_->currentState().discountRatio(numeraire,
00104                                                            nextNumeraire);
00105             }
00106 
00107         } while (!done);
00108 
00109         for (Size i=0; i<numerairesHeld_.size(); ++i)
00110             values[i] = numerairesHeld_[i] * initialNumeraireValue_;
00111 
00112         return weight;
00113     }
