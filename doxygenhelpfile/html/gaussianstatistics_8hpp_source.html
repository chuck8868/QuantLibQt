00130     template<class Stat>
00131     inline
00132     Real GenericGaussianStatistics<Stat>::gaussianRegret(Real target) const {
00133         Real m = this->mean();
00134         Real std = this->standardDeviation();
00135         Real variance = std*std;
00136         CumulativeNormalDistribution gIntegral(m, std);
00137         NormalDistribution g(m, std);
00138         Real firstTerm = variance + m*m - 2.0*target*m + target*target;
00139         Real alfa = gIntegral(target);
00140         Real secondTerm = m - target;
00141         Real beta = variance*g(target);
00142         Real result = alfa*firstTerm - beta*secondTerm;
00143         return result/alfa;
00144     }

00186                                                     Real percentile) const {
00187 
00188         QL_REQUIRE(percentile<1.0 && percentile>=0.9,

00200     template<class Stat>
00201     inline Real GenericGaussianStatistics<Stat>::gaussianExpectedShortfall(
00202                                                     Real percentile) const {
00203         QL_REQUIRE(percentile<1.0 && percentile>=0.9,
00204                    "percentile (" << percentile << ") out of range [0.9, 1)");
00205 
00206         Real m = this->mean();
00207         Real std = this->standardDeviation();
00208         InverseCumulativeNormal gInverse(m, std);
00209         Real var = gInverse(1.0-percentile);
00210         NormalDistribution g(m, std);
00211         Real result = m - std*std*g(var)/(1.0-percentile);
00212         // expectedShortfall must be a loss
00213         // this means that it has to be MIN(result, 0.0)
00214         // expectedShortfall must also be a positive quantity, so -MIN(*)
00215         return -std::min<Real>(result, 0.0);
00216     }
