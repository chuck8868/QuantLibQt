00029     LMMDriftCalculator::LMMDriftCalculator(
00030                                     const Matrix& pseudo,
00031                                     const std::vector<Spread>& displacements,
00032                                     const std::vector<Time>& taus,
00033                                     Size numeraire,
00034                                     Size alive)
00035     : numberOfRates_(taus.size()), numberOfFactors_(pseudo.columns()),
00036       isFullFactor_(numberOfFactors_==numberOfRates_ ? true : false),
00037       numeraire_(numeraire), alive_(alive),
00038       displacements_(displacements), oneOverTaus_(taus.size()),
00039       pseudo_(pseudo), tmp_(taus.size(), 0.0),
00040       e_(pseudo_.columns(), pseudo_.rows(), 0.0),
00041       downs_(taus.size()), ups_(taus.size()) {
00042 
00043         // Check requirements
00044         QL_REQUIRE(numberOfRates_>0, "Dim out of range");
00045         QL_REQUIRE(displacements.size() == numberOfRates_,
00046             "Displacements out of range");
00047         QL_REQUIRE(pseudo.rows()==numberOfRates_,
00048             "pseudo.rows() not consistent with dim");
00049         QL_REQUIRE(pseudo.columns()>0 && pseudo.columns()<=numberOfRates_,
00050             "pseudo.rows() not consistent with pseudo.columns()");
00051         QL_REQUIRE(alive<numberOfRates_, "Alive out of bounds");
00052         QL_REQUIRE(numeraire_<=numberOfRates_, "Numeraire larger than dim");
00053         QL_REQUIRE(numeraire_>=alive, "Numeraire smaller than alive");
00054 
00055         // Precompute 1/taus
00056         for (Size i=0; i<taus.size(); ++i)
00057             oneOverTaus_[i] = 1.0/taus[i];
00058 
00059         // Compute covariance matrix from pseudoroot
00060         const Disposable<Matrix> pT = transpose(pseudo_);
00061         C_ = pseudo_*pT;
00062 
00063         // Compute lower and upper extrema for (non reduced) drift calculation
00064         for (Size i=alive_; i<numberOfRates_; ++i) {
00065             downs_[i] = std::min(i+1, numeraire_);
00066             ups_[i]   = std::max(i+1, numeraire_);
00067         }
00068     }

00093     void LMMDriftCalculator::computePlain(const std::vector<Rate>& forwards,
00094                                           std::vector<Real>& drifts) const {
00095 
00096         // Compute drifts without factor reduction,
00097         // using directly the covariance matrix.
00098 
00099         // Precompute forwards factor
00100         Size i;
00101         for(i=alive_; i<numberOfRates_; ++i)
00102             tmp_[i] = (forwards[i]+displacements_[i]) /
00103                       (oneOverTaus_[i]+forwards[i]);
00104 
00105         // Compute drifts
00106         for (i=alive_; i<numberOfRates_; ++i) {
00107             drifts[i] = std::inner_product(tmp_.begin()+downs_[i],
00108                                            tmp_.begin()+ups_[i],
00109                                            C_.row_begin(i)+downs_[i], 0.0);
00110             if (numeraire_>i+1)
00111                 drifts[i] = -drifts[i];
00112         }
00113     }

00123         // Compute drifts with factor reduction,
00124         // using the pseudo square root of the covariance matrix.
00125 
00126         // Precompute forwards factor
00127         for (Size i=alive_; i<numberOfRates_; ++i)
00128             tmp_[i] = (forwards[i]+displacements_[i]) /
00129                 (oneOverTaus_[i]+forwards[i]);

00131         // Enforce initialization
00132         for (Size r=0; r<numberOfFactors_; ++r)
00133             e_[r][std::max(0,static_cast<Integer>(numeraire_)-1)] = 0.0;
00134 
00135         // Now compute drifts: take the numeraire P_N (numeraire_=N)
00136         // as the reference point, divide the summation into 3 steps,
00137         // et impera:
00138 
00139         // 1st step: the drift corresponding to the numeraire P_N is zero.
00140         // (if N=0 no drift is null, if N=numberOfRates_ the last drift is null).
00141         if (numeraire_>0) drifts[numeraire_-1] = 0.0;
00142 
00143         // 2nd step: then, move backward from N-2 (included) back to
00144         // alive (included) (if N=0 jumps to 3rd step, if N=numberOfRates_ the
00145         // e_[r][N-1] are correctly initialized):
00146 
00147         for (Integer i=static_cast<Integer>(numeraire_)-2;
00148              i>=static_cast<Integer>(alive_); --i) {
00149             drifts[i] = 0.0;
00150             for (Size r=0; r<numberOfFactors_; ++r) {
00151                 e_[r][i] = e_[r][i+1] + tmp_[i+1] * pseudo_[i+1][r];
00152                 drifts[i] -= e_[r][i]*pseudo_[i][r];
00153             }
00154 
00155             /*

00157             Matrix::column_iterator end = e_.column_end(i);

00171         // (if N=0 this is the only relevant computation):
