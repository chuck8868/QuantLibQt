00035         struct remapper {
00036             F f;
00037             Time T;
00038             remapper(const F& f, Time T) : f(f), T(T) {}
00039             // This remaps [-1,1] to [0,T]. No differential included.
00040             Real operator()(Real x) const {
00041                 const Real arg = (x+1.0)*T/2.0;

00075     Probability HazardRateStructure::survivalProbabilityImpl(Time t) const {
00076         static GaussChebyshevIntegration integral(48);
00077         // this stores the address of the method to integrate (so that
00078         // we don't have to insert its full expression inside the
00079         // integral below--it's long enough already)
00080         Real (HazardRateStructure::*f)(Time) const =
00081             &HazardRateStructure::hazardRateImpl;
00082         // the Gauss-Chebyshev quadratures integrate over [-1,1],
00083         // hence the remapping (and the Jacobian term t/2)
00084         return std::exp(-integral(remap(bind(f,this,_1), t)) * t/2.0);
00085     }
