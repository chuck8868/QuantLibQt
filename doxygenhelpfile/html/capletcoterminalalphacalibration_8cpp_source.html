00069             const EvolutionDescription& evolution,
00070             const PiecewiseConstantCorrelation& corr,
00071             const std::vector<boost::shared_ptr<
00072                 PiecewiseConstantVariance> >&
00073                     displacedSwapVariances,
00074             const std::vector<Volatility>& capletVols,
00075             const CurveState& cs,

00080             const std::vector<Real>& alphaMin,

00090             std::vector<Real>& b,
00091 
00092             std::vector<Matrix>& swapCovariancePseudoRoots) {
00093 
00094         CTSMMCapletCalibration::performChecks(evolution, corr,
00095             displacedSwapVariances, capletVols, cs);
00096 
00097         Size numberOfSteps = evolution.numberOfSteps();
00098         Size numberOfRates = evolution.numberOfRates();
00099         const std::vector<Time>& rateTimes = evolution.rateTimes();
00100 
00101         QL_REQUIRE(numberOfFactors<=numberOfRates,
00102                    "number of factors (" << numberOfFactors <<
00103                    ") cannot be greater than numberOfRates (" <<
00104                    numberOfRates << ")");
00105         QL_REQUIRE(numberOfFactors>0,
00106                    "number of factors (" << numberOfFactors <<
00107                    ") must be greater than zero");
00108 
00109         Natural failures=0;
00110 
00111         alpha.resize(numberOfRates);
00112         a.resize(numberOfRates);
00113         b.resize(numberOfRates);
00114 
00115         // factor reduction
00116         std::vector<Matrix> corrPseudo(corr.times().size());
00117         for (Size i=0; i<corrPseudo.size(); ++i)
00118             corrPseudo[i] = rankReducedSqrt(corr.correlation(i),
00119                                             numberOfFactors, 1.0,
00120                                             SalvagingAlgorithm::None);
00121 
00122         // get Zinverse, we can get wj later
00123         Matrix zedMatrix =
00124             SwapForwardMappings::coterminalSwapZedMatrix(cs, displacement);
00125         Matrix invertedZedMatrix = inverse(zedMatrix);
00126 
00127         // vectors for new vol
00128         std::vector<std::vector<Volatility> > newVols;
00129         std::vector<Volatility> theseNewVols(numberOfRates);
00130         std::vector<Volatility> firstRateVols(numberOfRates);
00131         firstRateVols[0] = std::sqrt(displacedSwapVariances[0]->variances()[0]);
00132         std::vector<Volatility> secondRateVols(numberOfRates);
00133         std::vector<Real> correlations(numberOfRates);
00134         newVols.push_back(firstRateVols);
00135 
00136 
00137         alpha[0] = alphaInitial[0]; // has no effect on anything in any case
00138         a[0] = b[0] = 1.0; // no modifications to swap vol for first rate
00139 
00140         AlphaFinder solver(parametricForm);
00141 
00142         // final caplet and swaption are the same, so we skip that case
00143         for (Size i=0; i<numberOfRates-1; ++i) {
00144             // we will calibrate caplet on forward rate i,
00145             // we will do this by modifying the vol of swap rate i+1
00146             const std::vector<Real>& var =
00147                                     displacedSwapVariances[i+1]->variances();
00148             for (Size j =0; j < i+2; ++j)
00149                 secondRateVols[j] = std::sqrt(var[j]);
00150 
00151             for (Size k=0; k < i+1; k++) {
00152                 Real correlation=0.0;
00153                 for (Size l=0; l < numberOfFactors; ++l) {
00154                     Real term1 = corrPseudo[k][i][l];
00155                     Real term2 = corrPseudo[k][i+1][l];
00156                     correlation += term1*term2;
00157                 }
00158                 correlations[k] = correlation;
00159             }
00160 
00161             Real w0 = invertedZedMatrix[i][i];
00162             Real w1 = invertedZedMatrix[i][i+1];
00163             // w0 adjustment
00164             for (Size k = i+2; k <invertedZedMatrix.columns(); ++k)
00165                 w0+= invertedZedMatrix[i][k];
00166 
00167             Real targetVariance= capletVols[i]*capletVols[i]*rateTimes[i];
00168 
00169             bool success;
00170             if (maximizeHomogeneity)
00171                 success = solver.solveWithMaxHomogeneity(
00172                                                     alphaInitial[i+1] ,
00173                                                     i,
00174                                                     firstRateVols,
00175                                                     secondRateVols,
00176                                                     correlations,
00177                                                     w0,
00178                                                     w1,
00179                                                     targetVariance,
00180                                                     tolerance,
00181                                                     alphaMax[i+1],
00182                                                     alphaMin[i+1],
00183                                                     maxIterations,
00184                                                     alpha[i+1],
00185                                                     a[i+1],
00186                                                     b[i+1],
00187                                                     theseNewVols);
00188             else
00189                 success = solver.solve(alphaInitial[i+1] ,
00190                                        i,
00191                                        firstRateVols,
00192                                        secondRateVols,
00193                                        correlations,
00194                                        w0,
00195                                        w1,
00196                                        targetVariance,
00197                                        tolerance,
00198                                        alphaMax[i+1],
00199                                        alphaMin[i+1],
00200                                        maxIterations,
00201                                        alpha[i+1],
00202                                        a[i+1],
00203                                        b[i+1],
00204                                        theseNewVols);
00205 
00206             if (!success) {
00207                 //++failures;
00208                 QL_FAIL("alpha form failure");
00209             }
00210 
00211             newVols.push_back(theseNewVols);
00212             firstRateVols = theseNewVols;
00213         }
00214 
00215         swapCovariancePseudoRoots.resize(numberOfSteps);
00216         for (Size k=0; k<numberOfSteps; ++k) {
00217             swapCovariancePseudoRoots[k] = corrPseudo[k];
00218             for (Size j=0; j<numberOfRates; ++j) {
00219                 Real coeff =newVols[j][k];
00220                 for (Size i=0; i<numberOfFactors; ++i)
00221                     swapCovariancePseudoRoots[k][j][i]*=coeff;
00222             }
00223             QL_ENSURE(swapCovariancePseudoRoots[k].rows()==numberOfRates,
00224                 "step " << k
00225                 << " abcd vol wrong number of rows: "
00226                 << swapCovariancePseudoRoots[k].rows()
00227                 << " instead of " << numberOfRates);
00228             QL_ENSURE(swapCovariancePseudoRoots[k].columns()==numberOfFactors,
00229                 "step " << k
00230                 << " abcd vol wrong number of columns: "
00231                 << swapCovariancePseudoRoots[k].columns()
00232                 << " instead of " << numberOfFactors);
00233         }
00234 
00235         return failures;
00236     }
