00053     void MarketModelComposite::add(
00054                                 const Clone<MarketModelMultiProduct>& product,
00055                                 Real multiplier) {
00056         QL_REQUIRE(!finalized_, "product already finalized");
00057         EvolutionDescription d = product->evolution();
00058         if (!components_.empty()) {
00059             // enforce preconditions
00060             EvolutionDescription d1 =
00061                 components_.front().product->evolution();
00062             const std::vector<Time>& rateTimes1 = d1.rateTimes();
00063             const std::vector<Time>& rateTimes2 = d.rateTimes();
00064             QL_REQUIRE(rateTimes1.size() == rateTimes2.size() &&
00065                        std::equal(rateTimes1.begin(), rateTimes1.end(),
00066                                   rateTimes2.begin()),
00067                        "incompatible rate times");
00068         }
00069         components_.push_back(SubProduct());
00070         components_.back().product = product;
00071         components_.back().multiplier = multiplier;
00072         components_.back().done = false;
00073         allEvolutionTimes_.push_back(d.evolutionTimes());
00074     }
