00233         class base_cubic_spline : public std::unary_function<Real,Real> {
00234           public:
00235             typedef base_data data;
00236             typedef base_data_table data_table;
00237             typedef base_output_data output_data;
00238             base_cubic_spline(const data &d, const data &d2,
00239                               const data_table& y, data_table &y2,
00240                               output_data &v) {
00241                 Size dim = d.first.size();
00242                 Size j = 1, k = 2, l = 3;
00243                 result_type  &w = ((y2[dim] = y[1]) -= y[0]) /= d[0],
00244                     &u = ((y2[0] = y[2]) -= y[1]) /= d[1], &t = v[dim];
00245                 y2[1] = -d[1] / d2[0], v[1] = 6.0 * (u - w) / d2[0];
00246                 for(; k < dim; u = w, j = k, k = l, ++l) {
00247                     w = (y[l]-y[k])/d[k];
00248                     u = (u-w)*6.0;
00249                     (y2[k] = d[k]) /= ((t = -y2[j]) *= d[j]) -= d2[j];
00250                     (v[k] = (u += d[j] * v[j])) /= t;
00251                 }
00252                 y2[0] = y2[dim] = 0.0;
00253                 while (k) {
00254                     (y2[k-1] *= y2[l-1]) += v[k-1];
00255                     --k; --l;
00256                 }
00257             }
00258         };

00505     // the data is checked and, in case of insufficient number of points,
00506     // exception is thrown BEFORE the main body of interpolation begins
00507     template <Size i>
00508     void MultiCubicSpline<i>::set_shared_increments() const {
00509         SplineGrid x(i), y(i);
00510         Size k = 0, dim = 0;
00511         for(Size j = 0; j < i; k = 0, ++j) {
00512             const std::vector<Real> &v = grid_[j];
00513             if((dim = v.size() - 1) > 2) {
00514                 std::vector<Real> tmp1(dim);
00515                 x[j].swap(tmp1);
00516                 std::vector<Real> tmp2(dim - 1);
00517                 y[j].swap(tmp2);
00518                 for(; k < dim; ++k) {
00519                     if((x[j][k] = v[k + 1] - v[k]) <= 0.0) break;
00520                     if(k) y[j][k - 1] = 2.0 * (v[k + 1] - v[k - 1]);
00521                 }
00522             }
00523             QL_REQUIRE(dim >= 3,
00524                        "Dimension " << j
00525                        << " : not enough points for interpolation");
00526             QL_REQUIRE(k >= dim,
00527                        "Dimension " << j << " : invalid data");
00528         }
00529 
00530         typename c_splint::data tmp1(x), tmp2(y);
00531         d_.swap(tmp1);
00532         d2_.swap(tmp2);
