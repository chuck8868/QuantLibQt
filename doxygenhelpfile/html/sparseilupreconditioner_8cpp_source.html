00034                                                      Integer lfil)

00036       U_(A.size1(),A.size2()) {
00037 
00038         QL_REQUIRE(A.size1() == A.size2(),
00039                    "sparse ILU preconditioner works only with square matrices");
00040 
00041         for (SparseMatrix::size_type i=0; i < L_.size1(); ++i)
00042             L_(i,i) = 1.0;
00043 
00044         const Integer n = A.size1();
00045         std::set<Integer> lBandSet, uBandSet;
00046 
00047         compressed_matrix<Integer> levs(n,n);
00048         Integer lfilp = lfil + 1;

00050         for (Integer ii=0; ii<n; ++ii) {
00051             Array w(n);
00052             for(Integer k=0; k<n; ++k) {
00053                 w[k] = A(ii,k);
00054             }

00057             for (Integer i=0; i<n; ++i) {
00058                 if(   w[i] > QL_EPSILON
00059                       || w[i] < -1.0*QL_EPSILON) levii[i] = 1;

00063                 for (Integer k=jj+1; k<n; ++k) {
00064                     if(levii[k]) {
00065                         jj = k;
00066                         break;
00067                     }
00068                 }
00069                 if (jj >= ii) {
00070                     break;
00071                 }

00079                     if(entry > QL_EPSILON || entry < -1.0*QL_EPSILON) {
00080                         nonZeros.push_back(jj);
00081                         nonZeroEntries.push_back(entry);
00082                     }

00087                         if(entry > QL_EPSILON || entry < -1.0*QL_EPSILON) {
00088                             nonZeros.push_back(jj+*iter);
00089                             nonZeroEntries.push_back(entry);
00090                         }
00091                     }
00092                     Real fact = w[jj];
00093                     if(!nonZeroEntries.empty()) {
00094                         fact /= nonZeroEntries[0];
00095                     }
00096                     for (Size k=0; k<nonZeros.size(); ++k) {
00097                         const Integer j = nonZeros[k] ;
00098                         const Integer temp = levs(jj,j) + jlev ;
00099                         if (levii[j] == 0) {
00100                             if (temp <= lfilp) {
00101                                 w[j] =  - fact*nonZeroEntries[k];

00103                             }
00104                         }

00106                             w[j] -= fact*nonZeroEntries[k];

00109                     }
00110                     w[jj] = fact;
00111                 }
00112             }

00114             std::vector<Real> wNonZeroEntries;
00115             wNonZeros.reserve(w.size());

00118                 const Real entry = w[i];

00120                     wNonZeros.push_back(i);
00121                     wNonZeroEntries.push_back(entry);
00122                 }
00123             }

00125             leviiNonZeroEntries.reserve(levii.size());

00127                 const Integer entry = levii[i];
00128                 if(entry > QL_EPSILON || entry < -1.0*QL_EPSILON) {
00129                     leviiNonZeroEntries.push_back(entry);
00130                 }
00131             }
00132             for (Size k=0; k<wNonZeros.size(); ++k) {
00133                 Integer j = wNonZeros[k];
00134                 if (j < ii) {
00135                     L_(ii,j) = wNonZeroEntries[k];
00136                     lBandSet.insert(ii-j);
00137                 }
00138                 else {
00139                     U_(ii,j) = wNonZeroEntries[k];
00140                     levs(ii,j) = leviiNonZeroEntries[k];

00142                         uBandSet.insert(j-ii);
00143                     }
00144                 }
00145             }
00146         }
00147         lBands_.resize(lBandSet.size());

00150         std::copy(uBandSet.begin(), uBandSet.end(), uBands_.begin());

00165     Disposable<Array> SparseILUPreconditioner::forwardSolve(
00166                                                        const Array& b) const {
00167         Size n = b.size();
00168         Array y(n, 0.0);
00169         y[0]=b[0]/L_(0,0);
00170         for (Size i=1; i<=n-1; ++i) {
00171             y[i] = b[i]/L_(i,i);
00172             for (Integer j=lBands_.size()-1;
00173                  j>=0 && i-lBands_[j] <= i-1; --j) {
00174                 y[i]-=L_(i,i-lBands_[j])*y[i-lBands_[j]]/L_(i,i);
00175             }
00176         }
00177         return y;
00178     }

00180     Disposable<Array> SparseILUPreconditioner::backwardSolve(
00181                                                        const Array& y) const {
00182         Size n = y.size();

00184         x[n-1] = y[n-1]/U_(n-1,n-1);
00185         for (Integer i=n-2; i>=0; --i) {
00186             x[i] = y[i]/U_(i,i);
00187             for (Size j=0; j<uBands_.size() && i+uBands_[j] <= n-1; ++j) {
00188                 x[i] -= U_(i,i+uBands_[j])*x[i+uBands_[j]]/U_(i,i);
