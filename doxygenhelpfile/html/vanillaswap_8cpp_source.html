00181     void VanillaSwap::fetchResults(const PricingEngine::results* r) const {
00182         static const Spread basisPoint = 1.0e-4;
00183 
00184         Swap::fetchResults(r);
00185 
00186         const VanillaSwap::results* results =
00187             dynamic_cast<const VanillaSwap::results*>(r);
00188         if (results) { // might be a swap engine, so no error is thrown
00189             fairRate_ = results->fairRate;
00190             fairSpread_ = results->fairSpread;
00191         } else {
00192             fairRate_ = Null<Rate>();
00193             fairSpread_ = Null<Spread>();
00194         }
00195 
00196         if (fairRate_ == Null<Rate>()) {
00197             // calculate it from other results
00198             if (legBPS_[0] != Null<Real>())
00199                 fairRate_ = fixedRate_ - NPV_/(legBPS_[0]/basisPoint);
00200         }
00201         if (fairSpread_ == Null<Spread>()) {
00202             // ditto
00203             if (legBPS_[1] != Null<Real>())
00204                 fairSpread_ = spread_ - NPV_/(legBPS_[1]/basisPoint);
00205         }
00206     }
