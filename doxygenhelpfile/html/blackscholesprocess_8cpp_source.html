00051     Real GeneralizedBlackScholesProcess::drift(Time t, Real x) const {
00052         Real sigma = diffusion(t,x);
00053         // we could be more anticipatory if we know the right dt
00054         // for which the drift will be used
00055         Time t1 = t + 0.0001;
00056         return riskFreeRate_->forwardRate(t,t1,Continuous,NoFrequency,true)
00057              - dividendYield_->forwardRate(t,t1,Continuous,NoFrequency,true)
00058              - 0.5 * sigma * sigma;
00059     }

00112     GeneralizedBlackScholesProcess::localVolatility() const {

00117                 boost::dynamic_pointer_cast<BlackConstantVol>(
00118                                                           *blackVolatility());

00120                 // ok, the local vol is constant too.
00121                 localVolatility_.linkTo(
00122                     boost::shared_ptr<LocalVolTermStructure>(new

00124                                          constVol->blackVol(0.0, x0_->value()),
00125                                          constVol->dayCounter())));
00126                 updated_ = true;
00127                 return localVolatility_;
00128             }
00129

00131             boost::shared_ptr<BlackVarianceCurve> volCurve =
00132                 boost::dynamic_pointer_cast<BlackVarianceCurve>(
00133                                                           *blackVolatility());

00135                 // ok, we can use the optimized algorithm
00136                 localVolatility_.linkTo(
00137                     boost::shared_ptr<LocalVolTermStructure>(
00138                         new LocalVolCurve(
00139                                       Handle<BlackVarianceCurve>(volCurve))));
00140                 updated_ = true;
00141                 return localVolatility_;
00142             }
00143 
00144             // ok, so it's strike-dependent. Never mind.
00145             localVolatility_.linkTo(
00146                       boost::shared_ptr<LocalVolTermStructure>(
00147                           new LocalVolSurface(blackVolatility_, riskFreeRate_,
00148                                               dividendYield_, x0_->value())));
00149             updated_ = true;
