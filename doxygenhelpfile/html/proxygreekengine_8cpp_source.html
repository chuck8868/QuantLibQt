00117                                                std::vector<Real>& values,
00118                                                bool storeRates) {

00120         std::fill(numerairesHeld_.begin(), numerairesHeld_.end(), 0.0);
00121         Real weight = evolver.startNewPath();
00122         product_->reset();
00123         Real principalInNumerairePortfolio = 1.0;
00124 
00125         if (storeRates)
00126             constraintsActive_ =false;
00127         //            std::fill(constraintsActive_.begin(),
00128         //                    constraintsActive_.end(),
00129         //                  false);
00130         //  }
00131 
00132         bool done = false;
00133         do {
00134             Size thisStep = evolver.currentStep();
00135             weight *= evolver.advanceStep();
00136             done = product_->nextTimeStep(evolver.currentState(),
00137                                           numberCashFlowsThisStep_,
00138                                           cashFlowsGenerated_);
00139             if (storeRates) {
00140                 constraints_[thisStep] = evolver.currentState().swapRate(
00141                                         startIndexOfConstraint_[thisStep],
00142                                         endIndexOfConstraint_[thisStep]);
00143                 constraintsActive_[thisStep] = true;
00144             }
00145 
00146             Size numeraire =
00147                 evolver.numeraires()[thisStep];
00148 
00149             // for each product...
00150             for (Size i=0; i<numberProducts_; ++i) {
00151                 // ...and each cash flow...
00152                 const std::vector<MarketModelMultiProduct::CashFlow>& cashflows =
00153                     cashFlowsGenerated_[i];
00154                 for (Size j=0; j<numberCashFlowsThisStep_[i]; ++j) {
00155                     // ...convert the cash flow to numeraires.
00156                     // This is done by calculating the number of
00157                     // numeraire bonds corresponding to such cash flow...
00158                     const MarketModelDiscounter& discounter =
00159                         discounters_[cashflows[j].timeIndex];
00160 
00161                     Real bonds = cashflows[j].amount *
00162                         discounter.numeraireBonds(evolver.currentState(),
00163                                                   numeraire);
00164 
00165                     // ...and adding the newly bought bonds to the number
00166                     // of numeraires held.
00167                     numerairesHeld_[i] +=
00168                         weight*bonds/principalInNumerairePortfolio;
00169                 }
00170             }
00171 
00172             if (!done) {
00173 
00174                 // The numeraire might change between steps. This implies
00175                 // that we might have to convert the numeraire bonds for
00176                 // this step into a corresponding amount of numeraire
00177                 // bonds for the next step. This can be done by changing
00178                 // the principal of the numeraire and updating the number
00179                 // of bonds in the numeraire portfolio accordingly.
00180 
00181                 Size nextNumeraire = evolver.numeraires()[thisStep+1];
00182 
00183                 principalInNumerairePortfolio *=
00184                     evolver.currentState().discountRatio(numeraire,
00185                                                          nextNumeraire);
00186             }
00187 
00188         } while (!done);
00189 
00190         for (Size i=0; i<numerairesHeld_.size(); ++i)
00191             values[i] = numerairesHeld_[i] * initialNumeraireValue_;
