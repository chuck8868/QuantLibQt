00030                                                  const Array& sub,
00031                                                  EigenVectorCalculation calc,
00032                                                  ShiftStrategy strategy)

00034       ev_((calc == WithEigenVector)? d_.size() :
00035           (calc == WithoutEigenVector)? 0 : 1, d_.size(), 0)
00036     {
00037         Size n = diag.size();
00038 
00039         QL_REQUIRE(n == sub.size()+1, "Wrong dimensions");
00040 
00041         Array e(n, 0.0);
00042         std::copy(sub.begin(),sub.end(),e.begin()+1);
00043         Size i;
00044         for (i=0; i < ev_.rows(); ++i) {
00045             ev_[i][i] = 1.0;
00046         }

00048         for (Size k=n-1; k >=1; --k) {
00049             while (!offDiagIsZero(k, e)) {
00050                 Size l = k;
00051                 while (--l > 0 && !offDiagIsZero(l,e));
00052                 iter_++;
00053 
00054                 Real q = d_[l];
00055                 if (strategy != NoShift) {
00056                     // calculated eigenvalue of 2x2 sub matrix of
00057                     // [ d_[k-1] e_[k] ]
00058                     // [  e_[k]  d_[k] ]
00059                     // which is closer to d_[k+1].
00060                     // FLOATING_POINT_EXCEPTION
00061                     const Real t1 = std::sqrt(
00062                                           0.25*(d_[k]*d_[k] + d_[k-1]*d_[k-1])
00063                                           - 0.5*d_[k-1]*d_[k] + e[k]*e[k]);
00064                     const Real t2 = 0.5*(d_[k]+d_[k-1]);
00065 
00066                     const Real lambda =
00067                         (std::fabs(t2+t1 - d_[k]) < std::fabs(t2-t1 - d_[k]))?
00068                         t2+t1 : t2-t1;
00069 
00070                     if (strategy == CloseEigenValue) {
00071                         q-=lambda;
00072                     } else {
00073                         q-=((k==n-1)? 1.25 : 1.0)*lambda;
00074                     }
00075                 }
00076 
00077                 // the QR transformation
00078                 Real sine = 1.0;
00079                 Real cosine = 1.0;
00080                 Real u = 0.0;
00081 
00082                 bool recoverUnderflow = false;
00083                 for (Size i=l+1; i <= k && !recoverUnderflow; ++i) {
00084                     const Real h = cosine*e[i];
00085                     const Real p = sine*e[i];
00086 
00087                     e[i-1] = std::sqrt(p*p+q*q);
00088                     if (e[i-1] != 0.0) {
00089                         sine = p/e[i-1];
00090                         cosine = q/e[i-1];
00091 
00092                         const Real g = d_[i-1]-u;
00093                         const Real t = (d_[i]-g)*sine+2*cosine*h;
00094 
00095                         u = sine*t;
00096                         d_[i-1] = g + u;
00097                         q = cosine*t - h;
00098 
00099                         for (Size j=0; j < ev_.rows(); ++j) {
00100                             const Real tmp = ev_[j][i-1];
00101                             ev_[j][i-1] = sine*ev_[j][i] + cosine*tmp;
00102                             ev_[j][i] = cosine*ev_[j][i] - sine*tmp;
00103                         }
00104                     } else {
00105                         // recover from underflow
00106                         d_[i-1] -= u;
00107                         e[l] = 0.0;
00108                         recoverUnderflow = true;
00109                     }
00110                 }
00111 
00112                 if (!recoverUnderflow) {
00113                     d_[k] -= u;
00114                     e[k] = q;
00115                     e[l] = 0.0;
00116                 }
00117             }
00118         }
00119 
00120         // sort (eigenvalues, eigenvectors),
00121         // code taken from symmetricSchureDecomposition.cpp
00122         std::vector<std::pair<Real, std::vector<Real> > > temp(n);
00123         std::vector<Real> eigenVector(ev_.rows());
00124         for (i=0; i<n; i++) {
00125             if (ev_.rows() > 0)
00126                 std::copy(ev_.column_begin(i),
00127                           ev_.column_end(i), eigenVector.begin());
00128             temp[i] = std::make_pair(d_[i], eigenVector);
00129         }
00130         std::sort(temp.begin(), temp.end(),
00131                   std::greater<std::pair<Real, std::vector<Real> > >());
00132         // first element is positive
00133         for (i=0; i<n; i++) {
00134             d_[i] = temp[i].first;
00135             Real sign = 1.0;
00136             if (ev_.rows() > 0 && temp[i].second[0]<0.0)
00137                 sign = -1.0;
00138             for (Size j=0; j<ev_.rows(); ++j) {
00139                 ev_[j][i] = sign * temp[i].second[j];
00140             }
00141         }
00142     }
