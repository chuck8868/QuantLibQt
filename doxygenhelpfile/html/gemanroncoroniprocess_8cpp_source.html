00056     Real GemanRoncoroniProcess::drift(Time t, Real x) const {
00057         const Real mu = alpha_ + beta_*t + gamma_*std::cos(eps_+2*M_PI*t)
00058                                          + delta_*std::cos(zeta_+4*M_PI*t);
00059         const Real muPrime = beta_ - gamma_*2*M_PI*std::sin(eps_+2*M_PI*t)
00060                                    - delta_*4*M_PI*std::sin(zeta_+4*M_PI*t);
00061 
00062         return muPrime + theta1_*(mu-x);
