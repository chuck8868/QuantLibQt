00095             const Real growthFactor = 1.6;
00096             Integer flipflop = -1;
00097

00100 
00101             // monotonically crescent bias, as in optionValue(volatility)
00102             if (close(fxMax_,0.0))
00103                 return root_;

00105                 xMin_ = enforceBounds_(root_ - step);
00106                 fxMin_ = f(xMin_);
00107                 xMax_ = root_;
00108             } else {
00109                 xMin_ = root_;
00110                 fxMin_ = fxMax_;
00111                 xMax_ = enforceBounds_(root_+step);
00112                 fxMax_ = f(xMax_);
00113             }
00114 
00115             evaluationNumber_ = 2;
00116             while (evaluationNumber_ <= maxEvaluations_) {
00117                 if (fxMin_*fxMax_ <= 0.0) {
00118                     if (close(fxMin_, 0.0))
00119                         return xMin_;
00120                     if (close(fxMax_, 0.0))
00121                         return xMax_;

00123                     return this->impl().solveImpl(f, accuracy);
00124                 }
00125                 if (std::fabs(fxMin_) < std::fabs(fxMax_)) {
00126                     xMin_ = enforceBounds_(xMin_+growthFactor*(xMin_ - xMax_));
00127                     fxMin_= f(xMin_);
00128                 } else if (std::fabs(fxMin_) > std::fabs(fxMax_)) {
00129                     xMax_ = enforceBounds_(xMax_+growthFactor*(xMax_ - xMin_));
00130                     fxMax_= f(xMax_);

00132                     xMin_ = enforceBounds_(xMin_+growthFactor*(xMin_ - xMax_));
00133                     fxMin_= f(xMin_);
00134                     evaluationNumber_++;
00135                     flipflop = 1;
00136                 } else if (flipflop == 1) {
00137                     xMax_ = enforceBounds_(xMax_+growthFactor*(xMax_ - xMin_));
00138                     fxMax_= f(xMax_);
00139                     flipflop = -1;
00140                 }
00141                 evaluationNumber_++;
00142             }
