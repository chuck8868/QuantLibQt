00068                         const boost::shared_ptr<StochasticProcess1D>& process,

00070                         Size steps)
00071         : BinomialTree<T>(process, end, steps) {}
00072         Real underlying(Size i, Size index) const {
00073             BigInteger j = 2*BigInteger(index) - BigInteger(i);
