00082     Rate IborCoupon::indexFixing() const {
00083 
00084         /* instead of just returning index_->fixing(fixingValueDate_)
00085            its logic is duplicated here using a specialized iborIndex
00086            forecastFixing overload which
00087            1) allows to save date/time recalculations, and
00088            2) takes into account par coupon needs
00089         */
00090         Date today = Settings::instance().evaluationDate();
00091 
00092         if (fixingDate_>today)
00093             return iborIndex_->forecastFixing(fixingValueDate_,
00094                                               fixingEndDate_,
00095                                               spanningTime_);
00096 
00097         if (fixingDate_<today ||
00098             Settings::instance().enforcesTodaysHistoricFixings()) {
00099             // do not catch exceptions
00100             Rate result = index_->pastFixing(fixingDate_);
00101             QL_REQUIRE(result != Null<Real>(),
00102                        "Missing " << index_->name() << " fixing for " << fixingDate_);
00103             return result;
00104         }
00105 
00106         try {
00107             Rate result = index_->pastFixing(fixingDate_);
00108             if (result!=Null<Real>())
00109                 return result;
00110             else
00111                 ;   // fall through and forecast
00112         } catch (Error&) {
00113                 ;   // fall through and forecast
00114         }
00115         return iborIndex_->forecastFixing(fixingValueDate_,
00116                                           fixingEndDate_,
00117                                           spanningTime_);
00118     }
