00031       strikes_(strikes), currentStep_(0), rateIndex_(exerciseTimes.size()),
00032       evolution_(rateTimes, exerciseTimes) {
00033         Size j = 0;

00035             while (j < rateTimes.size() && rateTimes[j] < exerciseTimes[i])
