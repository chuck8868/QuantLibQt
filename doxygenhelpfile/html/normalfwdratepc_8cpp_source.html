00098     Real NormalFwdRatePc::advanceStep()
00099     {
00100         // we're going from T1 to T2
00101 
00102         // a) compute drifts D1 at T1;
00103         if (currentStep_ > initialStep_) {
00104             calculators_[currentStep_].compute(forwards_, drifts1_);
00105         } else {
00106             std::copy(initialDrifts_.begin(), initialDrifts_.end(),
00107                       drifts1_.begin());
00108         }
00109 
00110         // b) evolve forwards up to T2 using D1;
00111         Real weight = generator_->nextStep(brownians_);
00112         const Matrix& A = marketModel_->pseudoRoot(currentStep_);
00113 
00114         Size i, alive = alive_[currentStep_];
00115         for (i=alive; i<numberOfRates_; ++i) {
00116             forwards_[i] += drifts1_[i] ;
00117             forwards_[i] +=
00118                 std::inner_product(A.row_begin(i), A.row_end(i),
00119                                    brownians_.begin(), 0.0);
00120         }
00121 
00122         // c) recompute drifts D2 using the predicted forwards;
00123         calculators_[currentStep_].compute(forwards_, drifts2_);
00124 
00125         // d) correct forwards using both drifts
00126         for (i=alive; i<numberOfRates_; ++i) {
00127             forwards_[i] += (drifts2_[i]-drifts1_[i])/2.0;
00128         }
00129 
00130         // e) update curve state
00131         curveState_.setOnForwardRates(forwards_);
00132 
00133         ++currentStep_;
00134 
00135         return weight;
00136     }
